def generate_eda_report(df, target=None):
    """
    EDA ë³´ê³ ì„œ ìë™ ìƒì„± í•¨ìˆ˜

    Parameters:
    -----------
    df : DataFrame
        ë¶„ì„í•  ë°ì´í„°í”„ë ˆì„
    target : str, optional
        íƒ€ê²Ÿ ë³€ìˆ˜ëª…
    """
    print("="*60)
    print("           íƒìƒ‰ì  ë°ì´í„° ë¶„ì„ (EDA) ë³´ê³ ì„œ")
    print("="*60)

    # 1. ê¸°ë³¸ ì •ë³´
    print("\n1. ë°ì´í„° ê¸°ë³¸ ì •ë³´")
    print("-"*40)
    print(f"   í–‰ ìˆ˜: {len(df):,}")
    print(f"   ì—´ ìˆ˜: {len(df.columns)}")
    print(f"   ë©”ëª¨ë¦¬ ì‚¬ìš©ëŸ‰: {df.memory_usage(deep=True).sum() / 1024**2:.2f} MB")

    # 2. ë°ì´í„° íƒ€ì…
    print("\n2. ë°ì´í„° íƒ€ì… ìš”ì•½")
    print("-"*40)
    dtype_counts = df.dtypes.value_counts()
    for dtype, count in dtype_counts.items():
        print(f"   {dtype}: {count}ê°œ")

    # 3. ê²°ì¸¡ì¹˜
    print("\n3. ê²°ì¸¡ì¹˜ í˜„í™©")
    print("-"*40)
    missing = df.isnull().sum()
    missing_pct = (missing / len(df) * 100)
    for col, (cnt, pct) in zip(df.columns, zip(missing, missing_pct)):
        if cnt > 0:
            print(f"   {col}: {cnt}ê°œ ({pct:.1f}%)")
    if missing.sum() == 0:
        print("   ê²°ì¸¡ì¹˜ ì—†ìŒ")

    # 4. ìˆ˜ì¹˜í˜• ë³€ìˆ˜
    print("\n4. ìˆ˜ì¹˜í˜• ë³€ìˆ˜ í†µê³„")
    print("-"*40)
    numeric_cols = df.select_dtypes(include=[np.number]).columns
    if len(numeric_cols) > 0:
        for col in numeric_cols[:5]:  # ìƒìœ„ 5ê°œë§Œ
            print(f"\n   [{col}]")
            print(f"   í‰ê· : {df[col].mean():.2f}, ì¤‘ì•™ê°’: {df[col].median():.2f}")
            print(f"   í‘œì¤€í¸ì°¨: {df[col].std():.2f}")
            print(f"   ë²”ìœ„: [{df[col].min():.2f}, {df[col].max():.2f}]")

    # 5. ë²”ì£¼í˜• ë³€ìˆ˜
    print("\n5. ë²”ì£¼í˜• ë³€ìˆ˜ ìš”ì•½")
    print("-"*40)
    categorical_cols = df.select_dtypes(include=['object', 'category']).columns
    for col in categorical_cols[:5]:  # ìƒìœ„ 5ê°œë§Œ
        print(f"\n   [{col}]")
        print(f"   ê³ ìœ ê°’ ìˆ˜: {df[col].nunique()}")
        print(f"   ìµœë¹ˆê°’: {df[col].mode().values[0]}")

    # 6. íƒ€ê²Ÿ ë³€ìˆ˜ (ìˆëŠ” ê²½ìš°)
    if target and target in df.columns:
        print(f"\n6. íƒ€ê²Ÿ ë³€ìˆ˜ ({target}) ë¶„ì„")
        print("-"*40)
        if df[target].dtype in ['int64', 'float64']:
            print(f"   í‰ê· : {df[target].mean():.2f}")
            print(f"   ë¶„í¬: ì—°ì†í˜•")
        else:
            print(f"   í´ë˜ìŠ¤ ë¶„í¬:")
            for val, cnt in df[target].value_counts().items():
                print(f"   - {val}: {cnt} ({cnt/len(df)*100:.1f}%)")

    print("\n" + "="*60)
    print("                     ë³´ê³ ì„œ ë")
    print("="*60)
