#include <iostream>

// =============================================================================
// 1. Union-Find (Disjoint Set Union)
// =============================================================================

class UnionFind {
private:
    vector<int> parent, rank_;

public:
    UnionFind(int n) : parent(n), rank_(n, 0) {
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);  // ê²½ë¡œ ì••ì¶•
        }
        return parent[x];
    }

    bool unite(int x, int y) {
        int px = find(x), py = find(y);
        if (px == py) return false;

        // ë­í¬ ê¸°ë°˜ í•©ì¹˜ê¸°
        if (rank_[px] < rank_[py]) swap(px, py);
        parent[py] = px;
        if (rank_[px] == rank_[py]) rank_[px]++;

        return true;
    }

    bool connected(int x, int y) {
        return find(x) == find(y);
    }
};

// =============================================================================
// 2. Kruskal ì•Œê³ ë¦¬ì¦˜
// =============================================================================

struct Edge {
    int u, v, weight;
    bool operator<(const Edge& other) const {
        return weight < other.weight;
    }
};

pair<int, vector<Edge>> kruskal(int n, vector<Edge>& edges) {
    sort(edges.begin(), edges.end());

    UnionFind uf(n);
    vector<Edge> mst;
    int totalWeight = 0;

    for (const auto& e : edges) {
        if (uf.unite(e.u, e.v)) {
            mst.push_back(e);
            totalWeight += e.weight;

            if ((int)mst.size() == n - 1) break;
        }
    }

    return {totalWeight, mst};
}

// =============================================================================
// 3. Prim ì•Œê³ ë¦¬ì¦˜
// =============================================================================

pair<int, vector<pair<int, int>>> prim(int n, const vector<vector<pair<int, int>>>& adj) {
    vector<bool> visited(n, false);
    vector<pair<int, int>> mst;  // {u, v} ê°„ì„ 
    int totalWeight = 0;

    // {weight, vertex, parent}
    priority_queue<tuple<int, int, int>, vector<tuple<int, int, int>>, greater<>> pq;
    pq.push({0, 0, -1});

    while (!pq.empty() && (int)mst.size() < n) {
        auto [w, u, parent] = pq.top();
        pq.pop();

        if (visited[u]) continue;
        visited[u] = true;
        totalWeight += w;

        if (parent != -1) {
            mst.push_back({parent, u});
        }

        for (auto [v, weight] : adj[u]) {
            if (!visited[v]) {
                pq.push({weight, v, u});
            }
        }
    }

    return {totalWeight, mst};
}

int secondMST(int n, vector<Edge>& edges) {
    // ë¨¼ì € MST êµ¬í•˜ê¸°
    auto [mstWeight, mst] = kruskal(n, edges);

    int secondBest = INT_MAX;

    // MSTì˜ ê° ê°„ì„ ì„ ì œê±°í•˜ê³  ë‹¤ì‹œ MST êµ¬í•˜ê¸°
    for (int i = 0; i < (int)mst.size(); i++) {
        vector<Edge> filtered;
        for (const auto& e : edges) {
            if (!(e.u == mst[i].u && e.v == mst[i].v && e.weight == mst[i].weight)) {
                filtered.push_back(e);
            }
        }

        auto [newWeight, newMst] = kruskal(n, filtered);

        if ((int)newMst.size() == n - 1) {
            secondBest = min(secondBest, newWeight);
        }
    }

    return secondBest;
}

pair<int, vector<Edge>> maxSpanningTree(int n, vector<Edge>& edges) {
    // ê°€ì¤‘ì¹˜ë¥¼ ìŒìˆ˜ë¡œ ë°”ê¿”ì„œ Kruskal ì ìš©
    for (auto& e : edges) {
        e.weight = -e.weight;
    }

    auto [weight, mst] = kruskal(n, edges);

    // ì›ë˜ ê°€ì¤‘ì¹˜ë¡œ ë³µì›
    for (auto& e : edges) {
        e.weight = -e.weight;
    }
    for (auto& e : mst) {
        e.weight = -e.weight;
    }

    return {-weight, mst};
}

// 6. í¬ëŸ¬ìŠ¤ì»¬ ì‘ìš©: ì—°ê²° ë¹„ìš© ìµœì†Œí™”

int minCostToConnect(int n, vector<vector<int>>& connections) {
    vector<Edge> edges;
    for (const auto& conn : connections) {
        edges.push_back({conn[0] - 1, conn[1] - 1, conn[2]});
    }

    auto [cost, mst] = kruskal(n, edges);

    if ((int)mst.size() != n - 1) {
        return -1;  // ì—°ê²° ë¶ˆê°€
    }

    return cost;
}

class FriendNetwork {
private:
    unordered_map<string, string> parent;
    unordered_map<string, int> size_;

    string find(const string& x) {
        if (parent.find(x) == parent.end()) {
            parent[x] = x;
            size_[x] = 1;
        }
        if (parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }

public:
    int unite(const string& a, const string& b) {
        string pa = find(a), pb = find(b);

        if (pa == pb) {
            return size_[pa];
        }

        if (size_[pa] < size_[pb]) swap(pa, pb);
        parent[pb] = pa;
        size_[pa] += size_[pb];

        return size_[pa];
    }
};

vector<int> findRedundantConnection(vector<vector<int>>& edges) {
    int n = edges.size();
    UnionFind uf(n + 1);

    for (const auto& e : edges) {
        if (!uf.unite(e[0], e[1])) {
            return e;  // ì¤‘ë³µ ì—°ê²°
        }
    }

    return {};
}
