const int INF = INT_MAX / 2;

int tsp(int n, const vector<vector<int>>& dist) {
    // dp[mask][i]: mask ì§‘í•©ì„ ë°©ë¬¸í•˜ê³  í˜„ì¬ iì— ìˆì„ ë•Œ ìµœì†Œ ë¹„ìš©
    vector<vector<int>> dp(1 << n, vector<int>(n, INF));

    dp[1][0] = 0;  // ì‹œì‘ì  0ì—ì„œ ì¶œë°œ

    for (int mask = 1; mask < (1 << n); mask++) {
        for (int u = 0; u < n; u++) {
            if (!(mask & (1 << u))) continue;
            if (dp[mask][u] == INF) continue;

            for (int v = 0; v < n; v++) {
                if (mask & (1 << v)) continue;  // ì´ë¯¸ ë°©ë¬¸
                if (dist[u][v] == INF) continue;

                int newMask = mask | (1 << v);
                dp[newMask][v] = min(dp[newMask][v], dp[mask][u] + dist[u][v]);
            }
        }
    }

    // ëª¨ë“  ë„ì‹œ ë°©ë¬¸ í›„ ì‹œì‘ì ìœ¼ë¡œ ë³µê·€
    int fullMask = (1 << n) - 1;
    int result = INF;
    for (int i = 0; i < n; i++) {
        if (dp[fullMask][i] != INF && dist[i][0] != INF) {
            result = min(result, dp[fullMask][i] + dist[i][0]);
        }
    }

    return result;
}

// TSP ê²½ë¡œ ë³µì›
pair<int, vector<int>> tspWithPath(int n, const vector<vector<int>>& dist) {
    vector<vector<int>> dp(1 << n, vector<int>(n, INF));
    vector<vector<int>> parent(1 << n, vector<int>(n, -1));

    dp[1][0] = 0;

    for (int mask = 1; mask < (1 << n); mask++) {
        for (int u = 0; u < n; u++) {
            if (!(mask & (1 << u))) continue;
            if (dp[mask][u] == INF) continue;

            for (int v = 0; v < n; v++) {
                if (mask & (1 << v)) continue;
                if (dist[u][v] == INF) continue;

                int newMask = mask | (1 << v);
                if (dp[mask][u] + dist[u][v] < dp[newMask][v]) {
                    dp[newMask][v] = dp[mask][u] + dist[u][v];
                    parent[newMask][v] = u;
                }
            }
        }
    }

    // ìµœì¢… ê²°ê³¼
    int fullMask = (1 << n) - 1;
    int result = INF;
    int lastCity = -1;

    for (int i = 0; i < n; i++) {
        if (dp[fullMask][i] != INF && dist[i][0] != INF) {
            if (dp[fullMask][i] + dist[i][0] < result) {
                result = dp[fullMask][i] + dist[i][0];
                lastCity = i;
            }
        }
    }

    // ê²½ë¡œ ë³µì›
    vector<int> path;
    int mask = fullMask;
    int curr = lastCity;

    while (curr != -1) {
        path.push_back(curr);
        int prev = parent[mask][curr];
        mask ^= (1 << curr);
        curr = prev;
    }

    reverse(path.begin(), path.end());
    path.push_back(0);  // ì‹œì‘ì ìœ¼ë¡œ ë³µê·€

    return {result, path};
}

int minCostAssignment(int n, const vector<vector<int>>& cost) {
    // dp[mask]: maskì— í•´ë‹¹í•˜ëŠ” ì‘ì—…ë“¤ì´ í• ë‹¹ë˜ì—ˆì„ ë•Œ ìµœì†Œ ë¹„ìš©
    vector<int> dp(1 << n, INF);
    dp[0] = 0;

    for (int mask = 0; mask < (1 << n); mask++) {
        int person = __builtin_popcount(mask);  // í˜„ì¬ í• ë‹¹í•  ì‚¬ëŒ
        if (person >= n) continue;

        for (int job = 0; job < n; job++) {
            if (mask & (1 << job)) continue;  // ì´ë¯¸ í• ë‹¹ëœ ì‘ì—…

            int newMask = mask | (1 << job);
            dp[newMask] = min(dp[newMask], dp[mask] + cost[person][job]);
        }
    }

    return dp[(1 << n) - 1];
}

bool subsetSum(const vector<int>& nums, int target) {
    int n = nums.size();

    for (int mask = 0; mask < (1 << n); mask++) {
        int sum = 0;
        for (int i = 0; i < n; i++) {
            if (mask & (1 << i)) {
                sum += nums[i];
            }
        }
        if (sum == target) return true;
    }

    return false;
}

// ìµœì í™”: Meet in the Middle
bool subsetSumMITM(const vector<int>& nums, int target) {
    int n = nums.size();
    int half = n / 2;

    // ì•ìª½ ì ˆë°˜ì˜ ëª¨ë“  ë¶€ë¶„ì§‘í•© í•©
    vector<int> leftSums;
    for (int mask = 0; mask < (1 << half); mask++) {
        int sum = 0;
        for (int i = 0; i < half; i++) {
            if (mask & (1 << i)) sum += nums[i];
        }
        leftSums.push_back(sum);
    }
    sort(leftSums.begin(), leftSums.end());

    // ë’¤ìª½ ì ˆë°˜ í™•ì¸
    int rightHalf = n - half;
    for (int mask = 0; mask < (1 << rightHalf); mask++) {
        int sum = 0;
        for (int i = 0; i < rightHalf; i++) {
            if (mask & (1 << i)) sum += nums[half + i];
        }

        // target - sumì´ leftSumsì— ìˆëŠ”ì§€ í™•ì¸
        if (binary_search(leftSums.begin(), leftSums.end(), target - sum)) {
            return true;
        }
    }

    return false;
}

int countHamiltonianPaths(int n, const vector<vector<int>>& adj) {
    // dp[mask][i]: mask ì§‘í•©ì„ ë°©ë¬¸í•˜ê³  iì—ì„œ ëë‚˜ëŠ” ê²½ë¡œ ìˆ˜
    vector<vector<int>> dp(1 << n, vector<int>(n, 0));

    // ì‹œì‘ì  ì´ˆê¸°í™”
    for (int i = 0; i < n; i++) {
        dp[1 << i][i] = 1;
    }

    for (int mask = 1; mask < (1 << n); mask++) {
        for (int u = 0; u < n; u++) {
            if (!(mask & (1 << u))) continue;
            if (dp[mask][u] == 0) continue;

            for (int v : adj[u]) {
                if (mask & (1 << v)) continue;

                int newMask = mask | (1 << v);
                dp[newMask][v] += dp[mask][u];
            }
        }
    }

    // ëª¨ë“  ì •ì  ë°©ë¬¸í•œ ê²½ë¡œ ìˆ˜
    int fullMask = (1 << n) - 1;
    int total = 0;
    for (int i = 0; i < n; i++) {
        total += dp[fullMask][i];
    }

    return total;
}

// 7. SOS DP (Sum over Subsets)
// =============================================================================

void sosDP(vector<int>& dp, int n) {
    // dp[mask]ì— maskì˜ ëª¨ë“  ë¶€ë¶„ì§‘í•©ì˜ í•© ì €ì¥
    for (int i = 0; i < n; i++) {
        for (int mask = 0; mask < (1 << n); mask++) {
            if (mask & (1 << i)) {
                dp[mask] += dp[mask ^ (1 << i)];
            }
        }
    }
}
