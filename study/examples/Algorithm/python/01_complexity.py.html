import time

def measure_time(func, *args):
    """í•¨ìˆ˜ ì‹¤í–‰ ì‹œê°„ ì¸¡ì •"""
    start = time.perf_counter()
    result = func(*args)
    end = time.perf_counter()
    return end - start, result


# =============================================================================
# O(1) - ìƒìˆ˜ ì‹œê°„
# =============================================================================
def constant_time(arr):
    """ë°°ì—´ì˜ ì²« ë²ˆì§¸ ìš”ì†Œ ë°˜í™˜ - O(1)"""
    if arr:
        return arr[0]
    return None

# O(log n) - ë¡œê·¸ ì‹œê°„
# =============================================================================
def binary_search(arr, target):
    """ì´ë¶„ íƒìƒ‰ - O(log n)"""
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1


# =============================================================================
# O(n) - ì„ í˜• ì‹œê°„
# =============================================================================
def linear_search(arr, target):
    """ì„ í˜• íƒìƒ‰ - O(n)"""
    for i, val in enumerate(arr):
        if val == target:
            return i
    return -1


def find_max(arr):
    """ìµœëŒ“ê°’ ì°¾ê¸° - O(n)"""
    if not arr:
        return None
    max_val = arr[0]
    for val in arr:
        if val > max_val:
            max_val = val
    return max_val


# =============================================================================
# O(n log n) - ì„ í˜• ë¡œê·¸ ì‹œê°„
# =============================================================================
def merge_sort(arr):
    """ë³‘í•© ì •ë ¬ - O(n log n)"""
    if len(arr) <= 1:
        return arr

    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])

    result = []
    i = j = 0
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    result.extend(left[i:])
    result.extend(right[j:])
    return result

# O(nÂ²) - ì œê³± ì‹œê°„

def bubble_sort(arr):
    """ë²„ë¸” ì •ë ¬ - O(nÂ²)"""
    arr = arr.copy()
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr


def has_duplicate_naive(arr):
    """ì¤‘ë³µ ê²€ì‚¬ (naive) - O(nÂ²)"""
    n = len(arr)
    for i in range(n):
        for j in range(i + 1, n):
            if arr[i] == arr[j]:
                return True
    return False


# =============================================================================
# O(2^n) - ì§€ìˆ˜ ì‹œê°„
# =============================================================================
def fibonacci_recursive(n):
    """í”¼ë³´ë‚˜ì¹˜ (ì¬ê·€) - O(2^n)"""
    if n <= 1:
        return n
    return fibonacci_recursive(n - 1) + fibonacci_recursive(n - 2)


def fibonacci_dp(n):
    """í”¼ë³´ë‚˜ì¹˜ (DP) - O(n)"""
    if n <= 1:
        return n
    dp = [0] * (n + 1)
    dp[1] = 1
    for i in range(2, n + 1):
        dp[i] = dp[i - 1] + dp[i - 2]
    return dp[n]
