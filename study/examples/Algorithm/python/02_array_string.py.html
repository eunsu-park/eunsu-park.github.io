def two_sum_sorted(arr: List[int], target: int) -> Optional[Tuple[int, int]]:

    ì‹œê°„ë³µì¡ë„: O(n), ê³µê°„ë³µì¡ë„: O(1)

    while left < right:

        else:

def three_sum(arr: List[int]) -> List[List[int]]:

    í•©ì´ 0ì¸ ì„¸ ìˆ˜ì˜ ì¡°í•© ëª¨ë‘ ì°¾ê¸° (ì¤‘ë³µ ì œê±°)
    ì‹œê°„ë³µì¡ë„: O(nÂ²), ê³µê°„ë³µì¡ë„: O(1)
    """
    arr.sort()
    result = []
    n = len(arr)

    for i in range(n - 2):
        # ì¤‘ë³µ ê±´ë„ˆë›°ê¸°
        if i > 0 and arr[i] == arr[i - 1]:
            continue

        left, right = i + 1, n - 1

        while left < right:
            total = arr[i] + arr[left] + arr[right]

            if total == 0:
                result.append([arr[i], arr[left], arr[right]])
                # ì¤‘ë³µ ê±´ë„ˆë›°ê¸°
                while left < right and arr[left] == arr[left + 1]:
                    left += 1
                while left < right and arr[right] == arr[right - 1]:
                    right -= 1
                left += 1
                right -= 1
            elif total < 0:
                left += 1
            else:
                right -= 1

    return result

def max_water(heights: List[int]) -> int:
    """
    ë‘ ë²½ ì‚¬ì´ì— ë‹´ì„ ìˆ˜ ìˆëŠ” ìµœëŒ€ ë¬¼ì˜ ì–‘
    ì‹œê°„ë³µì¡ë„: O(n), ê³µê°„ë³µì¡ë„: O(1)

    left, right = 0, len(heights) - 1
    max_area = 0

    while left < right:
        # í˜„ì¬ ë©´ì  ê³„ì‚°
        width = right - left
        height = min(heights[left], heights[right])

        max_area = max(max_area, area)

        # ë” ë‚®ì€ ìª½ì˜ í¬ì¸í„° ì´ë™
        if heights[left] < heights[right]:
            left += 1
        else:
            right -= 1

    return max_area

def merge_sorted_arrays(arr1: List[int], arr2: List[int]) -> List[int]:
    """
    ì •ë ¬ëœ ë‘ ë°°ì—´ì„ í•˜ë‚˜ì˜ ì •ë ¬ëœ ë°°ì—´ë¡œ í•©ë³‘
    ì‹œê°„ë³µì¡ë„: O(n + m), ê³µê°„ë³µì¡ë„: O(n + m)
    """
    result = []
    i, j = 0, 0

    while i < len(arr1) and j < len(arr2):
        if arr1[i] <= arr2[j]:
            result.append(arr1[i])
            i += 1
        else:
            result.append(arr2[j])
            j += 1

    # ë‚¨ì€ ìš”ì†Œ ì¶”ê°€
    result.extend(arr1[i:])
    result.extend(arr2[j:])

    return result

def remove_duplicates(arr: List[int]) -> int:
    """
    ì •ë ¬ëœ ë°°ì—´ì—ì„œ ì¤‘ë³µ ì œê±° (in-place)
    ë°˜í™˜ê°’: ê³ ìœ í•œ ìš”ì†Œì˜ ê°œìˆ˜
    ì‹œê°„ë³µì¡ë„: O(n), ê³µê°„ë³µì¡ë„: O(1)
    """
    if not arr:
        return 0

    write_idx = 1  # ë‹¤ìŒ ê³ ìœ  ê°’ì„ ì“¸ ìœ„ì¹˜

    for read_idx in range(1, len(arr)):
        if arr[read_idx] != arr[write_idx - 1]:
            arr[write_idx] = arr[read_idx]
            write_idx += 1

    return write_idx

def is_palindrome(s: str) -> bool:
    """
    ë¬¸ìì—´ì´ íšŒë¬¸ì¸ì§€ ê²€ì‚¬ (ì•ŒíŒŒë²³/ìˆ«ìë§Œ ë¹„êµ)
    ì‹œê°„ë³µì¡ë„: O(n), ê³µê°„ë³µì¡ë„: O(1)
    """
    left, right = 0, len(s) - 1

    while left < right:
        # ì•ŒíŒŒë²³/ìˆ«ìê°€ ì•„ë‹ˆë©´ ê±´ë„ˆë›°ê¸°
        while left < right and not s[left].isalnum():
            left += 1
        while left < right and not s[right].isalnum():
            right -= 1

        if s[left].lower() != s[right].lower():
            return False

        left += 1
        right -= 1

    return True

def max_sum_subarray(arr: List[int], k: int) -> int:
    """
    í¬ê¸° kì¸ ì—°ì† ë¶€ë¶„ ë°°ì—´ì˜ ìµœëŒ€ í•©
    ì‹œê°„ë³µì¡ë„: O(n), ê³µê°„ë³µì¡ë„: O(1)
    """
    if len(arr) < k:
        return 0

    # ì´ˆê¸° ìœˆë„ìš° í•©
    window_sum = sum(arr[:k])
    max_sum = window_sum

    # ìœˆë„ìš° ìŠ¬ë¼ì´ë”©
    for i in range(k, len(arr)):
        window_sum = window_sum - arr[i - k] + arr[i]
        max_sum = max(max_sum, window_sum)

    return max_sum
