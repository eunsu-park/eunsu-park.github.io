def bubble_sort(arr: List[int]) -> List[int]:
    """

    ì‹œê°„: O(nÂ²), ê³µê°„: O(1), ì•ˆì • ì •ë ¬

            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        if not swapped:

    return arr

def selection_sort(arr: List[int]) -> List[int]:
    """
    ì„ íƒ ì •ë ¬
    ì‹œê°„: O(nÂ²), ê³µê°„: O(1), ë¶ˆì•ˆì • ì •ë ¬
    """
    arr = arr.copy()
    n = len(arr)

    for i in range(n):
        min_idx = i
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        arr[i], arr[min_idx] = arr[min_idx], arr[i]

    return arr

def insertion_sort(arr: List[int]) -> List[int]:
    """
    ì‚½ì… ì •ë ¬
    ì‹œê°„: O(nÂ²), ìµœì„  O(n), ê³µê°„: O(1), ì•ˆì • ì •ë ¬
    ê±°ì˜ ì •ë ¬ëœ ë°ì´í„°ì— íš¨ìœ¨ì 
    """
    arr = arr.copy()
    n = len(arr)

    for i in range(1, n):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key

    return arr

def merge_sort(arr: List[int]) -> List[int]:
    """
    ë³‘í•© ì •ë ¬
    ì‹œê°„: O(n log n), ê³µê°„: O(n), ì•ˆì • ì •ë ¬
    """
    if len(arr) <= 1:
        return arr

    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])

    return merge(left, right)


def merge(left: List[int], right: List[int]) -> List[int]:
    """ë‘ ì •ë ¬ëœ ë°°ì—´ ë³‘í•©"""
    result = []
    i = j = 0

    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1

    result.extend(left[i:])
    result.extend(right[j:])
    return result

# 5. í€µ ì •ë ¬ (Quick Sort)
# =============================================================================
def quick_sort(arr: List[int]) -> List[int]:
    """
    í€µ ì •ë ¬
    ì‹œê°„: í‰ê·  O(n log n), ìµœì•… O(nÂ²), ê³µê°„: O(log n), ë¶ˆì•ˆì • ì •ë ¬
    """
    if len(arr) <= 1:
        return arr

    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]

    return quick_sort(left) + middle + quick_sort(right)


def quick_sort_inplace(arr: List[int], low: int = 0, high: int = None) -> List[int]:
    """í€µ ì •ë ¬ (in-place)"""
    if high is None:
        arr = arr.copy()
        high = len(arr) - 1

    if low < high:
        pivot_idx = partition(arr, low, high)
        quick_sort_inplace(arr, low, pivot_idx - 1)
        quick_sort_inplace(arr, pivot_idx + 1, high)

    return arr


def partition(arr: List[int], low: int, high: int) -> int:
    """íŒŒí‹°ì…˜ (Lomuto scheme)"""
    pivot = arr[high]
    i = low - 1

    for j in range(low, high):
        if arr[j] <= pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i]

    arr[i + 1], arr[high] = arr[high], arr[i + 1]
    return i + 1

def heap_sort(arr: List[int]) -> List[int]:
    """
    í™ ì •ë ¬
    ì‹œê°„: O(n log n), ê³µê°„: O(1), ë¶ˆì•ˆì • ì •ë ¬

    arr = arr.copy()

    # í•˜ë‚˜ì”© ì¶”ì¶œ

def heapify(arr: List[int], n: int, i: int):
    """ìµœëŒ€ í™ ì†ì„± ìœ ì§€"""
    largest = i
    left = 2 * i + 1
    right = 2 * i + 2

    if right < n and arr[right] > arr[largest]:
        largest = right

    if largest != i:
        arr[i], arr[largest] = arr[largest], arr[i]

def counting_sort(arr: List[int]) -> List[int]:
    """
    ê³„ìˆ˜ ì •ë ¬
    ì‹œê°„: O(n + k), ê³µê°„: O(k), ì•ˆì • ì •ë ¬

    """
    if not arr:

    min_val, max_val = min(arr), max(arr)
    range_val = max_val - min_val + 1

    # ì¹´ìš´íŠ¸
    for num in arr:
        count[num - min_val] += 1

    # ëˆ„ì

    """
    ê¸°ìˆ˜ ì •ë ¬ (LSD)

def counting_sort_by_digit(arr: List[int], exp: int) -> List[int]:
    """íŠ¹ì • ìë¦¿ìˆ˜ ê¸°ì¤€ ê³„ìˆ˜ ì •ë ¬"""
    n = len(arr)
    output = [0] * n
    count = [0] * 10

    for num in arr:
        digit = (num // exp) % 10
        count[digit] += 1

    for i in range(1, 10):
        count[i] += count[i - 1]

    for i in range(n - 1, -1, -1):
        digit = (arr[i] // exp) % 10
        count[digit] -= 1
        output[count[digit]] = arr[i]

    return output
