def binary_search(arr: List[int], target: int) -> int:
    """
    ì •ë ¬ëœ ë°°ì—´ì—ì„œ targetì˜ ì¸ë±ìŠ¤ ì°¾ê¸°
    ì—†ìœ¼ë©´ -1 ë°˜í™˜
    ì‹œê°„ë³µì¡ë„: O(log n)
    """
    left, right = 0, len(arr) - 1

    while left <= right:
        mid = left + (right - left) // 2  # ì˜¤ë²„í”Œë¡œìš° ë°©ì§€

        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1

    return -1


def binary_search_recursive(arr: List[int], target: int, left: int, right: int) -> int:
    """ì´ë¶„ íƒìƒ‰ (ì¬ê·€ ë²„ì „)"""
    if left > right:
        return -1

    mid = left + (right - left) // 2

    if arr[mid] == target:
        return mid
    elif arr[mid] < target:
        return binary_search_recursive(arr, target, mid + 1, right)
    else:
        return binary_search_recursive(arr, target, left, mid - 1)

def lower_bound(arr: List[int], target: int) -> int:
    """
    target ì´ìƒì¸ ì²« ë²ˆì§¸ ìš”ì†Œì˜ ì¸ë±ìŠ¤
    ëª¨ë“  ìš”ì†Œê°€ targetë³´ë‹¤ ì‘ìœ¼ë©´ len(arr) ë°˜í™˜
    """
    left, right = 0, len(arr)

    while left < right:
        mid = left + (right - left) // 2
        if arr[mid] < target:
            left = mid + 1
        else:
            right = mid

    return left


def upper_bound(arr: List[int], target: int) -> int:
    """
    target ì´ˆê³¼ì¸ ì²« ë²ˆì§¸ ìš”ì†Œì˜ ì¸ë±ìŠ¤
    ëª¨ë“  ìš”ì†Œê°€ target ì´í•˜ì´ë©´ len(arr) ë°˜í™˜
    """
    left, right = 0, len(arr)

    while left < right:
        mid = left + (right - left) // 2
        if arr[mid] <= target:
            left = mid + 1
        else:
            right = mid

    return left


def count_occurrences(arr: List[int], target: int) -> int:
    """íŠ¹ì • ê°’ì˜ ë“±ì¥ íšŸìˆ˜"""
    return upper_bound(arr, target) - lower_bound(arr, target)

def find_first_position(arr: List[int], target: int) -> int:
    """targetì´ ì²˜ìŒ ë“±ì¥í•˜ëŠ” ì¸ë±ìŠ¤ (ì—†ìœ¼ë©´ -1)"""
    idx = lower_bound(arr, target)
    if idx < len(arr) and arr[idx] == target:

    return -1


def find_last_position(arr: List[int], target: int) -> int:
    """targetì´ ë§ˆì§€ë§‰ìœ¼ë¡œ ë“±ì¥í•˜ëŠ” ì¸ë±ìŠ¤ (ì—†ìœ¼ë©´ -1)"""

def search_rotated(arr: List[int], target: int) -> int:
    """
    íšŒì „ëœ ì •ë ¬ ë°°ì—´ì—ì„œ ê²€ìƒ‰
    ì˜ˆ: [4, 5, 6, 7, 0, 1, 2] - ì›ë˜ [0,1,2,4,5,6,7]ì„ íšŒì „
    ì‹œê°„ë³µì¡ë„: O(log n)
    """
    if not arr:
        return -1

    left, right = 0, len(arr) - 1

    while left <= right:
        mid = left + (right - left) // 2

        if arr[mid] == target:
            return mid

        # ì™¼ìª½ ì ˆë°˜ì´ ì •ë ¬ë˜ì–´ ìˆëŠ” ê²½ìš°
        if arr[left] <= arr[mid]:
            if arr[left] <= target < arr[mid]:
                right = mid - 1
            else:
                left = mid + 1
        # ì˜¤ë¥¸ìª½ ì ˆë°˜ì´ ì •ë ¬ë˜ì–´ ìˆëŠ” ê²½ìš°
        else:
            if arr[mid] < target <= arr[right]:
                left = mid + 1
            else:
                right = mid - 1

    return -1

def find_minimum_rotated(arr: List[int]) -> int:
    """íšŒì „ ì •ë ¬ ë°°ì—´ì˜ ìµœì†Ÿê°’ ì°¾ê¸°"""
    left, right = 0, len(arr) - 1

        if arr[mid] > arr[right]:
            left = mid + 1

            right = mid

def integer_sqrt(n: int) -> int:
    """
    nì˜ ì •ìˆ˜ ì œê³±ê·¼ (ë²„ë¦¼)
    ì˜ˆ: sqrt(8) = 2
    """
    if n < 0:
        raise ValueError("ìŒìˆ˜ì˜ ì œê³±ê·¼ì€ ì •ì˜ë˜ì§€ ì•ŠìŠµë‹ˆë‹¤")
    if n == 0:
        return 0

    left, right = 1, n

    while left <= right:
        mid = left + (right - left) // 2
        square = mid * mid

        if square == n:
            return mid
        elif square < n:
            left = mid + 1
        else:
            right = mid - 1

    return right  # floor(sqrt(n))

def can_split(arr: List[int], m: int, max_sum: int) -> bool:
    """
    ë°°ì—´ì„ mê°œ ì´í•˜ì˜ ê·¸ë£¹ìœ¼ë¡œ ë‚˜ëˆŒ ë•Œ
    ê° ê·¸ë£¹ í•©ì´ max_sum ì´í•˜ì¸ì§€ í™•ì¸
    """
    count = 1
    current_sum = 0

    for num in arr:
        if current_sum + num > max_sum:
            count += 1
            current_sum = num
            if count > m:
                return False
        else:
            current_sum += num

    return True


def split_array_min_largest_sum(arr: List[int], m: int) -> int:
    """
    ë°°ì—´ì„ mê°œì˜ ì—°ì† ë¶€ë¶„ ë°°ì—´ë¡œ ë‚˜ëˆŒ ë•Œ
    ê° ë¶€ë¶„ ë°°ì—´ í•©ì˜ ìµœëŒ“ê°’ì„ ìµœì†Œí™”
    ì‹œê°„ë³µì¡ë„: O(n log(sum))
    """
    left = max(arr)      # ìµœì†Œ ê°€ëŠ¥í•œ ê°’: ê°€ì¥ í° ìš”ì†Œ
    right = sum(arr)     # ìµœëŒ€ ê°€ëŠ¥í•œ ê°’: ì „ì²´ í•©

    while left < right:
        mid = left + (right - left) // 2
        if can_split(arr, m, mid):
            right = mid
        else:
            left = mid + 1

    return left

def find_peak_element(arr: List[int]) -> int:
    """
    ë°°ì—´ì—ì„œ peak elementì˜ ì¸ë±ìŠ¤ ì°¾ê¸°

            right = mid
        else:
            left = mid + 1
