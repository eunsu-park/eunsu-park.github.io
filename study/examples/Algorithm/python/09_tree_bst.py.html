from typing import List, Optional, Generator

class TreeNode:
    """ì´ì§„ íŠ¸ë¦¬ ë…¸ë“œ"""

    def __init__(self, val: int, left: 'TreeNode' = None, right: 'TreeNode' = None):
        self.val = val
        self.left = left
        self.right = right

    def __repr__(self):
        return f"TreeNode({self.val})"

# 2. íŠ¸ë¦¬ ìˆœíšŒ (Tree Traversal)

def preorder_recursive(root: TreeNode) -> List[int]:
    """ì „ìœ„ ìˆœíšŒ (ì¬ê·€) - O(n)"""
    result = []

    def traverse(node):
        if not node:
            return
        result.append(node.val)
        traverse(node.left)
        traverse(node.right)

    traverse(root)
    return result


def preorder_iterative(root: TreeNode) -> List[int]:
    """ì „ìœ„ ìˆœíšŒ (ë°˜ë³µ) - O(n)"""
    if not root:
        return []

    result = []
    stack = [root]

    while stack:
        node = stack.pop()
        result.append(node.val)

        # ì˜¤ë¥¸ìª½ ë¨¼ì € push (ì™¼ìª½ì´ ë¨¼ì € ì²˜ë¦¬ë˜ë„ë¡)
        if node.right:
            stack.append(node.right)
        if node.left:
            stack.append(node.left)

    return result


def inorder_recursive(root: TreeNode) -> List[int]:
    """ì¤‘ìœ„ ìˆœíšŒ (ì¬ê·€) - O(n)"""
    result = []

    def traverse(node):
        if not node:
            return
        traverse(node.left)
        result.append(node.val)
        traverse(node.right)

    traverse(root)
    return result


def inorder_iterative(root: TreeNode) -> List[int]:
    """ì¤‘ìœ„ ìˆœíšŒ (ë°˜ë³µ) - O(n)"""
    result = []
    stack = []
    current = root

    while stack or current:
        # ì™¼ìª½ ëê¹Œì§€ ì´ë™
        while current:
            stack.append(current)
            current = current.left

        current = stack.pop()
        result.append(current.val)
        current = current.right

    return result


def postorder_recursive(root: TreeNode) -> List[int]:
    """í›„ìœ„ ìˆœíšŒ (ì¬ê·€) - O(n)"""
    result = []

    def traverse(node):
        if not node:
            return
        traverse(node.left)
        traverse(node.right)
        result.append(node.val)

    traverse(root)
    return result


def postorder_iterative(root: TreeNode) -> List[int]:
    """í›„ìœ„ ìˆœíšŒ (ë°˜ë³µ) - O(n)"""
    if not root:
        return []

    result = []
    stack = [root]

    while stack:
        node = stack.pop()
        result.append(node.val)

        if node.left:
            stack.append(node.left)
        if node.right:
            stack.append(node.right)

    return result[::-1]  # ì—­ìˆœ


def level_order(root: TreeNode) -> List[List[int]]:
    """ë ˆë²¨ ìˆœíšŒ (BFS) - O(n)"""
    if not root:
        return []

    result = []
    queue = deque([root])

    while queue:
        level_size = len(queue)
        level = []

        for _ in range(level_size):
            node = queue.popleft()
            level.append(node.val)

            if node.left:
                queue.append(node.left)
            if node.right:
                queue.append(node.right)

        result.append(level)

    return result

class BST:
    """ì´ì§„ íƒìƒ‰ íŠ¸ë¦¬"""

    def __init__(self):
        self.root: Optional[TreeNode] = None

    def insert(self, val: int) -> None:
        """ë…¸ë“œ ì‚½ì… - í‰ê·  O(log n), ìµœì•… O(n)"""
        if not self.root:
            self.root = TreeNode(val)
            return

        current = self.root
        while True:
            if val < current.val:
                if current.left is None:
                    current.left = TreeNode(val)
                    return
                current = current.left
            else:
                if current.right is None:
                    current.right = TreeNode(val)
                    return
                current = current.right

    def search(self, val: int) -> Optional[TreeNode]:
        """ë…¸ë“œ ê²€ìƒ‰ - í‰ê·  O(log n), ìµœì•… O(n)"""
        current = self.root

        while current:
            if val == current.val:
                return current
            elif val < current.val:
                current = current.left
            else:
                current = current.right

        return None

    def delete(self, val: int) -> bool:
        """ë…¸ë“œ ì‚­ì œ - í‰ê·  O(log n), ìµœì•… O(n)"""

        def find_min(node: TreeNode) -> TreeNode:
            while node.left:
                node = node.left
            return node

        def delete_recursive(node: TreeNode, val: int) -> Optional[TreeNode]:
            if not node:
                return None

            if val < node.val:
                node.left = delete_recursive(node.left, val)
            elif val > node.val:
                node.right = delete_recursive(node.right, val)
            else:
                # ì‚­ì œí•  ë…¸ë“œ ë°œê²¬

                # Case 1: ë¦¬í”„ ë…¸ë“œ
                if not node.left and not node.right:
                    return None

                # Case 2: ìì‹ì´ í•˜ë‚˜
                if not node.left:
                    return node.right
                if not node.right:
                    return node.left

                # Case 3: ìì‹ì´ ë‘˜ - í›„ê³„ì(ì˜¤ë¥¸ìª½ ì„œë¸ŒíŠ¸ë¦¬ì˜ ìµœì†Œê°’)ë¡œ ëŒ€ì²´
                successor = find_min(node.right)
                node.val = successor.val
                node.right = delete_recursive(node.right, successor.val)

            return node

        old_root = self.root
        self.root = delete_recursive(self.root, val)
        return old_root != self.root or (self.root and old_root.val != val if old_root else False)

    def inorder(self) -> List[int]:
        """ì¤‘ìœ„ ìˆœíšŒ (ì •ë ¬ëœ ìˆœì„œ)"""
        return inorder_recursive(self.root)

    def find_min(self) -> Optional[int]:
        """ìµœì†Ÿê°’ ì°¾ê¸° - O(h)"""
        if not self.root:
            return None

        current = self.root
        while current.left:
            current = current.left
        return current.val

    def find_max(self) -> Optional[int]:
        """ìµœëŒ“ê°’ ì°¾ê¸° - O(h)"""
        if not self.root:
            return None

        current = self.root
        while current.right:
            current = current.right
        return current.val

def tree_height(root: TreeNode) -> int:
    """íŠ¸ë¦¬ ë†’ì´ ê³„ì‚° - O(n)"""
    if not root:
        return -1  # ë¹ˆ íŠ¸ë¦¬ëŠ” ë†’ì´ -1, ë…¸ë“œ 1ê°œëŠ” ë†’ì´ 0

    return 1 + max(tree_height(root.left), tree_height(root.right))


def is_balanced(root: TreeNode) -> bool:
    """ê· í˜• íŠ¸ë¦¬ ê²€ì‚¬ - O(n)"""

    def check(node: TreeNode) -> int:
        if not node:
            return 0

        left_height = check(node.left)
        if left_height == -1:
            return -1

        right_height = check(node.right)
        if right_height == -1:
            return -1

        if abs(left_height - right_height) > 1:
            return -1

        return 1 + max(left_height, right_height)

    return check(root) != -1


def is_valid_bst(root: TreeNode) -> bool:
    """ìœ íš¨í•œ BST ê²€ì‚¬ - O(n)"""

    def validate(node: TreeNode, min_val: float, max_val: float) -> bool:
        if not node:
            return True

        if node.val <= min_val or node.val >= max_val:
            return False

        return (validate(node.left, min_val, node.val) and
                validate(node.right, node.val, max_val))

    return validate(root, float('-inf'), float('inf'))


def count_nodes(root: TreeNode) -> int:
    """ë…¸ë“œ ê°œìˆ˜ - O(n)"""
    if not root:
        return 0
    return 1 + count_nodes(root.left) + count_nodes(root.right)

def build_tree_from_list(values: List[Optional[int]]) -> Optional[TreeNode]:
    """ë ˆë²¨ ìˆœì„œ ë¦¬ìŠ¤íŠ¸ë¡œ íŠ¸ë¦¬ êµ¬ì„± - O(n)"""
    if not values or values[0] is None:
        return None

    root = TreeNode(values[0])
    queue = deque([root])
    i = 1

    while queue and i < len(values):
        node = queue.popleft()

        if i < len(values) and values[i] is not None:
            node.left = TreeNode(values[i])
            queue.append(node.left)
        i += 1

        if i < len(values) and values[i] is not None:
            node.right = TreeNode(values[i])
            queue.append(node.right)
        i += 1

    return root


def sorted_array_to_bst(nums: List[int]) -> Optional[TreeNode]:
    """ì •ë ¬ëœ ë°°ì—´ë¡œ ê· í˜• BST êµ¬ì„± - O(n)"""
    if not nums:
        return None

    def build(left: int, right: int) -> Optional[TreeNode]:
        if left > right:
            return None

        mid = (left + right) // 2
        node = TreeNode(nums[mid])
        node.left = build(left, mid - 1)
        node.right = build(mid + 1, right)
        return node

    return build(0, len(nums) - 1)


def invert_tree(root: TreeNode) -> TreeNode:
    """íŠ¸ë¦¬ ì¢Œìš° ë°˜ì „ - O(n)"""
    if not root:
        return None

    root.left, root.right = invert_tree(root.right), invert_tree(root.left)
    return root

def lowest_common_ancestor(root: TreeNode, p: int, q: int) -> Optional[TreeNode]:
    """BSTì—ì„œ ìµœì†Œ ê³µí†µ ì¡°ìƒ (LCA) - O(h)"""
    current = root

    while current:
        if p < current.val and q < current.val:
            current = current.left
        elif p > current.val and q > current.val:
            current = current.right
        else:
            return current

    return None


def kth_smallest(root: TreeNode, k: int) -> int:
    """BSTì—ì„œ kë²ˆì§¸ ì‘ì€ ê°’ - O(h + k)"""
    stack = []
    current = root
    count = 0

    while stack or current:
        while current:
            stack.append(current)
            current = current.left

        current = stack.pop()
        count += 1

        if count == k:
            return current.val

        current = current.right

    return -1


def path_sum(root: TreeNode, target: int) -> bool:
    """ë£¨íŠ¸~ë¦¬í”„ ê²½ë¡œ í•© í™•ì¸ - O(n)"""
    if not root:
        return False

    if not root.left and not root.right:
        return root.val == target

    remaining = target - root.val
    return path_sum(root.left, remaining) or path_sum(root.right, remaining)


def serialize(root: TreeNode) -> str:
    """íŠ¸ë¦¬ ì§ë ¬í™” - O(n)"""
    if not root:
        return "[]"

    result = []
    queue = deque([root])

    while queue:
        node = queue.popleft()
        if node:
            result.append(str(node.val))
            queue.append(node.left)
            queue.append(node.right)
        else:
            result.append("null")

    # ëì˜ null ì œê±°
    while result and result[-1] == "null":
        result.pop()

    return "[" + ",".join(result) + "]"

def print_tree(root: TreeNode, prefix: str = "", is_left: bool = True) -> None:
    """íŠ¸ë¦¬ ASCII ì¶œë ¥"""
    if not root:
        return

    print(prefix + ("â”œâ”€â”€ " if is_left else "â””â”€â”€ ") + str(root.val))

    children = []
    if root.left:
        children.append((root.left, True))
    if root.right:
        children.append((root.right, False))

    for i, (child, is_left_child) in enumerate(children):
        extension = "â”‚   " if is_left and i < len(children) - 1 else "    "
        print_tree(child, prefix + extension, is_left_child)
