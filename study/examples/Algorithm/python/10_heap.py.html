# 1. ìµœì†Œ í™ (Min Heap) ì§ì ‘ êµ¬í˜„

class MinHeap:
    """
    ìµœì†Œ í™ (ì™„ì „ ì´ì§„ íŠ¸ë¦¬)
    - ë¶€ëª¨ ë…¸ë“œ â‰¤ ìì‹ ë…¸ë“œ
    - ë°°ì—´ë¡œ êµ¬í˜„: parent(i) = (i-1)//2, left(i) = 2i+1, right(i) = 2i+2
    """

    def __init__(self):
        self.heap: List[int] = []

    def __len__(self) -> int:
        return len(self.heap)

    def __bool__(self) -> bool:
        return len(self.heap) > 0

    def _parent(self, i: int) -> int:
        return (i - 1) // 2

    def _left(self, i: int) -> int:
        return 2 * i + 1

    def _right(self, i: int) -> int:
        return 2 * i + 2

    def _swap(self, i: int, j: int) -> None:
        self.heap[i], self.heap[j] = self.heap[j], self.heap[i]

    def _sift_up(self, i: int) -> None:
        """ì‚½ì… í›„ ìœ„ë¡œ ì¬ì •ë ¬ - O(log n)"""
        while i > 0:
            parent = self._parent(i)
            if self.heap[i] < self.heap[parent]:
                self._swap(i, parent)
                i = parent
            else:
                break

    def _sift_down(self, i: int) -> None:
        """ì‚­ì œ í›„ ì•„ë˜ë¡œ ì¬ì •ë ¬ - O(log n)"""
        n = len(self.heap)

        while True:
            smallest = i
            left = self._left(i)
            right = self._right(i)

            if left < n and self.heap[left] < self.heap[smallest]:
                smallest = left
            if right < n and self.heap[right] < self.heap[smallest]:
                smallest = right

            if smallest != i:
                self._swap(i, smallest)
                i = smallest
            else:
                break

    def push(self, val: int) -> None:
        """ì‚½ì… - O(log n)"""
        self.heap.append(val)
        self._sift_up(len(self.heap) - 1)

    def pop(self) -> int:
        """ìµœì†Ÿê°’ ì œê±° ë° ë°˜í™˜ - O(log n)"""
        if not self.heap:
            raise IndexError("pop from empty heap")

        result = self.heap[0]
        last = self.heap.pop()

        if self.heap:
            self.heap[0] = last
            self._sift_down(0)

        return result

    def peek(self) -> int:
        """ìµœì†Ÿê°’ ì¡°íšŒ - O(1)"""
        if not self.heap:
            raise IndexError("peek from empty heap")
        return self.heap[0]

    def heapify(self, arr: List[int]) -> None:
        """ë°°ì—´ì„ í™ìœ¼ë¡œ ë³€í™˜ - O(n)"""
        self.heap = arr[:]
        # ë§ˆì§€ë§‰ ë¹„-ë¦¬í”„ ë…¸ë“œë¶€í„° sift_down
        for i in range(len(self.heap) // 2 - 1, -1, -1):
            self._sift_down(i)

class MaxHeap:
    """ìµœëŒ€ í™ - ë¶€ëª¨ ë…¸ë“œ â‰¥ ìì‹ ë…¸ë“œ"""

    def __init__(self):
        self.heap: List[int] = []

    def __len__(self) -> int:

    def push(self, val: int) -> None:
        """ì‚½ì… - O(log n)"""
        # ìµœì†Œ í™ì— ìŒìˆ˜ë¡œ ì €ì¥
        heapq.heappush(self.heap, -val)

    def pop(self) -> int:
        """ìµœëŒ“ê°’ ì œê±° ë° ë°˜í™˜ - O(log n)"""
        return -heapq.heappop(self.heap)

        return -self.heap[0]

def heap_sort(arr: List[int]) -> List[int]:
    """
    í™ ì •ë ¬
    ì‹œê°„ë³µì¡ë„: O(n log n)
    ê³µê°„ë³µì¡ë„: O(1) (in-place)
    """
    n = len(arr)
    result = arr[:]

    def sift_down(arr: List[int], n: int, i: int) -> None:
        while True:
            largest = i
            left = 2 * i + 1
            right = 2 * i + 2

            if left < n and arr[left] > arr[largest]:
                largest = left
            if right < n and arr[right] > arr[largest]:
                largest = right

            if largest != i:
                arr[i], arr[largest] = arr[largest], arr[i]
                i = largest
            else:
                break

    # 1. ìµœëŒ€ í™ êµ¬ì„± - O(n)
    for i in range(n // 2 - 1, -1, -1):
        sift_down(result, n, i)

    # 2. í•˜ë‚˜ì”© ì¶”ì¶œ - O(n log n)
    for i in range(n - 1, 0, -1):
        result[0], result[i] = result[i], result[0]
        sift_down(result, i, 0)

    return result

def kth_largest(arr: List[int], k: int) -> int:
    """
    Kë²ˆì§¸ë¡œ í° ìš”ì†Œ - ìµœì†Œ í™ ì‚¬ìš©
    ì‹œê°„ë³µì¡ë„: O(n log k)
    ê³µê°„ë³µì¡ë„: O(k)
    """
    min_heap = []

    for num in arr:
        heapq.heappush(min_heap, num)
        if len(min_heap) > k:
            heapq.heappop(min_heap)

    return min_heap[0]


def kth_smallest(arr: List[int], k: int) -> int:
    """
    Kë²ˆì§¸ë¡œ ì‘ì€ ìš”ì†Œ - ìµœëŒ€ í™ ì‚¬ìš©
    ì‹œê°„ë³µì¡ë„: O(n log k)
    ê³µê°„ë³µì¡ë„: O(k)
    """
    max_heap = []

    for num in arr:
        heapq.heappush(max_heap, -num)
        if len(max_heap) > k:
            heapq.heappop(max_heap)

    return -max_heap[0]

class MedianFinder:
    """
    ë°ì´í„° ìŠ¤íŠ¸ë¦¼ì˜ ì¤‘ì•™ê°’
    - ìµœëŒ€ í™ (ì™¼ìª½ ì ˆë°˜): ì‘ì€ ê°’ë“¤
    - ìµœì†Œ í™ (ì˜¤ë¥¸ìª½ ì ˆë°˜): í° ê°’ë“¤
    """

    def __init__(self):
        self.max_heap: List[int] = []  # ì™¼ìª½ (ì‘ì€ ê°’ë“¤)
        self.min_heap: List[int] = []  # ì˜¤ë¥¸ìª½ (í° ê°’ë“¤)

    def add_num(self, num: int) -> None:
        """ìˆ«ì ì¶”ê°€ - O(log n)"""
        # ì™¼ìª½ í™ì— ì¶”ê°€
        heapq.heappush(self.max_heap, -num)

        # ì™¼ìª½ ìµœëŒ“ê°’ì„ ì˜¤ë¥¸ìª½ìœ¼ë¡œ ì´ë™
        heapq.heappush(self.min_heap, -heapq.heappop(self.max_heap))

        # í¬ê¸° ê· í˜• ìœ ì§€ (ì™¼ìª½ â‰¥ ì˜¤ë¥¸ìª½)
        if len(self.min_heap) > len(self.max_heap):
            heapq.heappush(self.max_heap, -heapq.heappop(self.min_heap))

    def find_median(self) -> float:
        """ì¤‘ì•™ê°’ ë°˜í™˜ - O(1)"""
        if len(self.max_heap) > len(self.min_heap):
            return -self.max_heap[0]
        return (-self.max_heap[0] + self.min_heap[0]) / 2

def schedule_tasks(tasks: List[Tuple[int, int, str]]) -> List[str]:
    """
    ìš°ì„ ìˆœìœ„ ê¸°ë°˜ ì‘ì—… ìŠ¤ì¼€ì¤„ë§
    tasks: [(ìš°ì„ ìˆœìœ„, ë„ì°©ì‹œê°„, ì‘ì—…ëª…), ...]
    ë‚®ì€ ìš°ì„ ìˆœìœ„ ë²ˆí˜¸ = ë†’ì€ ìš°ì„ ìˆœìœ„
    """
    # (ìš°ì„ ìˆœìœ„, ë„ì°©ì‹œê°„, ì‘ì—…ëª…) í˜•íƒœë¡œ í™ì— ì¶”ê°€
    task_heap = []
    for priority, arrival, name in tasks:
        heapq.heappush(task_heap, (priority, arrival, name))

    schedule = []
    while task_heap:
        _, _, name = heapq.heappop(task_heap)
        schedule.append(name)

    return schedule

def merge_k_sorted_lists(lists: List[List[int]]) -> List[int]:
    """
    Kê°œì˜ ì •ë ¬ëœ ë¦¬ìŠ¤íŠ¸ ë³‘í•©
    ì‹œê°„ë³µì¡ë„: O(N log k), N = ì „ì²´ ì›ì†Œ ìˆ˜, k = ë¦¬ìŠ¤íŠ¸ ìˆ˜
    """
    result = []
    min_heap = []

    # ê° ë¦¬ìŠ¤íŠ¸ì˜ ì²« ë²ˆì§¸ ì›ì†Œë¥¼ í™ì— ì¶”ê°€
    for i, lst in enumerate(lists):
        if lst:
            heapq.heappush(min_heap, (lst[0], i, 0))  # (ê°’, ë¦¬ìŠ¤íŠ¸ ì¸ë±ìŠ¤, ì›ì†Œ ì¸ë±ìŠ¤)

    while min_heap:
        val, list_idx, elem_idx = heapq.heappop(min_heap)
        result.append(val)

        # ë‹¤ìŒ ì›ì†Œê°€ ìˆìœ¼ë©´ í™ì— ì¶”ê°€
        if elem_idx + 1 < len(lists[list_idx]):
            next_val = lists[list_idx][elem_idx + 1]
            heapq.heappush(min_heap, (next_val, list_idx, elem_idx + 1))

    return result

def top_k_frequent(nums: List[int], k: int) -> List[int]:
    """
    ê°€ì¥ ë¹ˆë²ˆí•œ Kê°œ ìš”ì†Œ
    ì‹œê°„ë³µì¡ë„: O(n log k)
    """
    from collections import Counter

    freq = Counter(nums)

    # ìµœì†Œ í™ìœ¼ë¡œ Kê°œë§Œ ìœ ì§€
    min_heap = []
    for num, count in freq.items():
        heapq.heappush(min_heap, (count, num))
        if len(min_heap) > k:
            heapq.heappop(min_heap)

    return [num for count, num in min_heap]

def k_closest_points(points: List[Tuple[int, int]], k: int) -> List[Tuple[int, int]]:
    """
    ì›ì ì—ì„œ ê°€ì¥ ê°€ê¹Œìš´ Kê°œ ì 
    ì‹œê°„ë³µì¡ë„: O(n log k)
    """
    # ìµœëŒ€ í™ìœ¼ë¡œ Kê°œë§Œ ìœ ì§€ (ê±°ë¦¬ì˜ ìŒìˆ˜ ì €ì¥)
    max_heap = []

    for x, y in points:
        dist = x * x + y * y
        heapq.heappush(max_heap, (-dist, x, y))
        if len(max_heap) > k:
            heapq.heappop(max_heap)

    return [(x, y) for _, x, y in max_heap]

def min_meeting_rooms(intervals: List[Tuple[int, int]]) -> int:
    """
    í•„ìš”í•œ ìµœì†Œ íšŒì˜ì‹¤ ìˆ˜
    intervals: [(ì‹œì‘ì‹œê°„, ì¢…ë£Œì‹œê°„), ...]
    ì‹œê°„ë³µì¡ë„: O(n log n)
    """
    if not intervals:
        return 0

    # ì‹œì‘ ì‹œê°„ìœ¼ë¡œ ì •ë ¬
    intervals.sort(key=lambda x: x[0])

    # ì¢…ë£Œ ì‹œê°„ì„ ì €ì¥í•˜ëŠ” ìµœì†Œ í™
    end_times = []
    heapq.heappush(end_times, intervals[0][1])

    for start, end in intervals[1:]:
        # ê°€ì¥ ë¹¨ë¦¬ ëë‚˜ëŠ” íšŒì˜ê°€ í˜„ì¬ íšŒì˜ ì‹œì‘ ì „ì— ëë‚˜ë©´ ì¬ì‚¬ìš©
        if end_times[0] <= start:
            heapq.heappop(end_times)
        heapq.heappush(end_times, end)

    return len(end_times)
