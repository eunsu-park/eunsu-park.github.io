class TrieNode:
    """íŠ¸ë¼ì´ ë…¸ë“œ"""

    def __init__(self):
        self.children: Dict[str, 'TrieNode'] = {}
        self.is_end: bool = False
        self.count: int = 0  # ì´ ì ‘ë‘ì‚¬ë¡œ ì‹œì‘í•˜ëŠ” ë‹¨ì–´ ìˆ˜


class Trie:
    """
    íŠ¸ë¼ì´ (ì ‘ë‘ì‚¬ íŠ¸ë¦¬)
    - ì‚½ì…: O(m), m = ë‹¨ì–´ ê¸¸ì´
    - ê²€ìƒ‰: O(m)
    - ì ‘ë‘ì‚¬ ê²€ìƒ‰: O(m)
    """

    def __init__(self):
        self.root = TrieNode()

    def insert(self, word: str) -> None:
        """ë‹¨ì–´ ì‚½ì… - O(m)"""
        node = self.root

        for char in word:
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]
            node.count += 1

        node.is_end = True

    def search(self, word: str) -> bool:
        """ë‹¨ì–´ ì¡´ì¬ ì—¬ë¶€ ê²€ìƒ‰ - O(m)"""
        node = self._find_node(word)
        return node is not None and node.is_end

    def starts_with(self, prefix: str) -> bool:
        """ì ‘ë‘ì‚¬ë¡œ ì‹œì‘í•˜ëŠ” ë‹¨ì–´ ì¡´ì¬ ì—¬ë¶€ - O(m)"""
        return self._find_node(prefix) is not None

    def count_prefix(self, prefix: str) -> int:
        """ì ‘ë‘ì‚¬ë¡œ ì‹œì‘í•˜ëŠ” ë‹¨ì–´ ê°œìˆ˜ - O(m)"""
        node = self._find_node(prefix)
        return node.count if node else 0

    def _find_node(self, prefix: str) -> Optional[TrieNode]:
        """ì ‘ë‘ì‚¬ì— í•´ë‹¹í•˜ëŠ” ë…¸ë“œ ì°¾ê¸°"""
        node = self.root

        for char in prefix:
            if char not in node.children:
                return None
            node = node.children[char]

        return node

    def delete(self, word: str) -> bool:
        """ë‹¨ì–´ ì‚­ì œ - O(m)"""

        def _delete(node: TrieNode, word: str, depth: int) -> bool:
            if depth == len(word):
                if not node.is_end:
                    return False
                node.is_end = False
                return len(node.children) == 0

            char = word[depth]
            if char not in node.children:
                return False

            should_delete = _delete(node.children[char], word, depth + 1)

            if should_delete:
                del node.children[char]
                return len(node.children) == 0 and not node.is_end

            node.children[char].count -= 1
            return False

        return _delete(self.root, word, 0)

class AutocompleteSystem:
    """ìë™ì™„ì„± ì‹œìŠ¤í…œ"""

    def __init__(self):
        self.root = TrieNode()

    def add_word(self, word: str, weight: int = 1) -> None:
        """ë‹¨ì–´ ì¶”ê°€ (ê°€ì¤‘ì¹˜ í¬í•¨)"""
        node = self.root

        for char in word:
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]

        node.is_end = True
        node.count = weight  # ì—¬ê¸°ì„œ countëŠ” ë¹ˆë„/ê°€ì¤‘ì¹˜

    def autocomplete(self, prefix: str, limit: int = 5) -> List[str]:
        """ì ‘ë‘ì‚¬ë¡œ ì‹œì‘í•˜ëŠ” ë‹¨ì–´ ëª©ë¡ - O(m + k), k = ê²°ê³¼ ìˆ˜"""
        node = self.root

        # ì ‘ë‘ì‚¬ ë…¸ë“œ ì°¾ê¸°
        for char in prefix:
            if char not in node.children:
                return []
            node = node.children[char]

        # DFSë¡œ ëª¨ë“  ë‹¨ì–´ ìˆ˜ì§‘
        results = []
        self._collect_words(node, prefix, results)

        # ë¹ˆë„ìˆœ ì •ë ¬ í›„ ìƒìœ„ limitê°œ
        results.sort(key=lambda x: x[1], reverse=True)
        return [word for word, _ in results[:limit]]

    def _collect_words(self, node: TrieNode, prefix: str, results: List) -> None:
        """ë…¸ë“œì—ì„œ ëª¨ë“  ì™„ì„± ë‹¨ì–´ ìˆ˜ì§‘"""
        if node.is_end:
            results.append((prefix, node.count))

        for char, child in node.children.items():
            self._collect_words(child, prefix + char, results)

class WildcardTrie:
    """ì™€ì¼ë“œì¹´ë“œ '.'ë¥¼ ì§€ì›í•˜ëŠ” íŠ¸ë¼ì´"""

    def __init__(self):
        self.root = TrieNode()

    def add_word(self, word: str) -> None:
        node = self.root
        for char in word:
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]
        node.is_end = True

    def search(self, word: str) -> bool:
        """'.'ëŠ” ì„ì˜ì˜ í•œ ë¬¸ìì™€ ë§¤ì¹­"""
        return self._search(self.root, word, 0)

    def _search(self, node: TrieNode, word: str, index: int) -> bool:
        if index == len(word):
            return node.is_end

        char = word[index]

        if char == '.':
            # ëª¨ë“  ìì‹ ë…¸ë“œ ì‹œë„
            for child in node.children.values():
                if self._search(child, word, index + 1):
                    return True
            return False
        else:
            if char not in node.children:
                return False
            return self._search(node.children[char], word, index + 1)

def longest_common_prefix(words: List[str]) -> str:
    """
    ë‹¨ì–´ ë°°ì—´ì˜ ìµœì¥ ê³µí†µ ì ‘ë‘ì‚¬
    íŠ¸ë¼ì´ í™œìš©
    """
    if not words:
        return ""

    # íŠ¸ë¼ì´ êµ¬ì„±
    trie = Trie()
    for word in words:
        trie.insert(word)

    # ë£¨íŠ¸ì—ì„œ ë¶„ê¸°ì ê¹Œì§€ ì´ë™
    prefix = []
    node = trie.root

    while node:
        # ìì‹ì´ í•˜ë‚˜ì´ê³  ì¢…ë£Œ ë…¸ë“œê°€ ì•„ë‹ ë•Œë§Œ ê³„ì†
        if len(node.children) == 1 and not node.is_end:
            char = list(node.children.keys())[0]
            prefix.append(char)
            node = node.children[char]
        else:
            break

    return ''.join(prefix)

class WordDictionary:
    """
    ë‹¨ì–´ ì¶”ê°€/ê²€ìƒ‰ ì‚¬ì „
    - ì •í™•í•œ ê²€ìƒ‰
    - ì ‘ë‘ì‚¬ ê²€ìƒ‰
    - ì™€ì¼ë“œì¹´ë“œ ê²€ìƒ‰
    """

    def __init__(self):
        self.trie = Trie()
        self.wildcard_trie = WildcardTrie()

    def add_word(self, word: str) -> None:
        self.trie.insert(word)
        self.wildcard_trie.add_word(word)

    def search_exact(self, word: str) -> bool:
        """ì •í™•í•œ ë‹¨ì–´ ê²€ìƒ‰"""
        return self.trie.search(word)

    def search_prefix(self, prefix: str) -> bool:
        """ì ‘ë‘ì‚¬ ê²€ìƒ‰"""
        return self.trie.starts_with(prefix)

    def search_pattern(self, pattern: str) -> bool:
        """íŒ¨í„´ ê²€ìƒ‰ (. = ì„ì˜ ë¬¸ì)"""
        return self.wildcard_trie.search(pattern)

class XORTrie:
    """
    ë¹„íŠ¸ íŠ¸ë¼ì´ë¡œ ìµœëŒ€ XOR ìŒ ì°¾ê¸°
    ê° ìˆ«ìë¥¼ ì´ì§„ìˆ˜ë¡œ ì €ì¥
    """

    def __init__(self, max_bits: int = 31):
        self.root = {}
        self.max_bits = max_bits

    def insert(self, num: int) -> None:
        """ìˆ«ì ì‚½ì… - O(max_bits)"""
        node = self.root

        for i in range(self.max_bits, -1, -1):
            bit = (num >> i) & 1
            if bit not in node:
                node[bit] = {}
            node = node[bit]

    def find_max_xor(self, num: int) -> int:
        """numê³¼ XORí–ˆì„ ë•Œ ìµœëŒ€ê°€ ë˜ëŠ” ê°’ ë°˜í™˜ - O(max_bits)"""
        node = self.root
        result = 0

        for i in range(self.max_bits, -1, -1):
            bit = (num >> i) & 1
            # XOR ìµœëŒ€í™”ë¥¼ ìœ„í•´ ë°˜ëŒ€ ë¹„íŠ¸ ì„ íƒ
            toggle_bit = 1 - bit

            if toggle_bit in node:
                result |= (1 << i)
                node = node[toggle_bit]
            elif bit in node:
                node = node[bit]
            else:
                break

        return result


def find_maximum_xor(nums: List[int]) -> int:
    """
    ë°°ì—´ì—ì„œ ë‘ ìˆ˜ì˜ XOR ìµœëŒ“ê°’
    ì‹œê°„ë³µì¡ë„: O(n * max_bits)
    """
    if len(nums) < 2:
        return 0

    xor_trie = XORTrie()
    max_xor = 0

    for num in nums:
        xor_trie.insert(num)
        max_xor = max(max_xor, xor_trie.find_max_xor(num))

    return max_xor

# 7. ì‹¤ì „ ë¬¸ì œ: ë‹¨ì–´ ê²€ìƒ‰ II (Word Search II)

def find_words(board: List[List[str]], words: List[str]) -> List[str]:
    """
    2D ë³´ë“œì—ì„œ ë‹¨ì–´ ëª©ë¡ ì°¾ê¸°
    íŠ¸ë¼ì´ + DFS
    """
    # íŠ¸ë¼ì´ êµ¬ì„±
    trie = Trie()
    for word in words:
        trie.insert(word)

    rows, cols = len(board), len(board[0])
    result = set()

    def dfs(r: int, c: int, node: TrieNode, path: str) -> None:
        if node.is_end:
            result.add(path)

        if r < 0 or r >= rows or c < 0 or c >= cols:
            return

        char = board[r][c]
        if char == '#' or char not in node.children:
            return

        board[r][c] = '#'  # ë°©ë¬¸ í‘œì‹œ
        next_node = node.children[char]

        for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            dfs(r + dr, c + dc, next_node, path + char)

        board[r][c] = char  # ë³µì›

    for r in range(rows):
        for c in range(cols):
            dfs(r, c, trie.root, "")

    return list(result)

class PrefixSuffixTrie:
    """ì ‘ë‘ì‚¬ì™€ ì ‘ë¯¸ì‚¬ë¥¼ ë™ì‹œì— ê²€ìƒ‰í•˜ëŠ” íŠ¸ë¼ì´"""

    def __init__(self, words: List[str]):
        self.trie = {}

        for idx, word in enumerate(words):
            # ëª¨ë“  ì ‘ë¯¸ì‚¬#ë‹¨ì–´ í˜•íƒœë¡œ ì €ì¥
            key = word + '#' + word
            for i in range(len(word) + 1):
                node = self.trie
                for char in key[i:]:
                    if char not in node:
                        node[char] = {'idx': -1}
                    node = node[char]
                    node['idx'] = idx  # ê°€ì¥ í° ì¸ë±ìŠ¤ ì €ì¥

    def search(self, prefix: str, suffix: str) -> int:
        """ì ‘ë‘ì‚¬ì™€ ì ‘ë¯¸ì‚¬ ëª¨ë‘ ë§Œì¡±í•˜ëŠ” ë‹¨ì–´ì˜ ì¸ë±ìŠ¤"""
        key = suffix + '#' + prefix
        node = self.trie

        for char in key:
            if char not in node:
                return -1
            node = node[char]

        return node.get('idx', -1)
