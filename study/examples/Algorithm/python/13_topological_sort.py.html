def topological_sort_kahn(n: int, edges: List[Tuple[int, int]]) -> List[int]:
    """
    Kahn's ì•Œê³ ë¦¬ì¦˜ (ì§„ì… ì°¨ìˆ˜ ê¸°ë°˜)
    ì‹œê°„ë³µì¡ë„: O(V + E)
    ê³µê°„ë³µì¡ë„: O(V + E)

    edges: [(from, to), ...] - from â†’ to ì˜ì¡´ ê´€ê³„
    ë°˜í™˜: ìœ„ìƒ ì •ë ¬ëœ ìˆœì„œ, ì‚¬ì´í´ ìˆìœ¼ë©´ ë¹ˆ ë¦¬ìŠ¤íŠ¸
    """
    # ê·¸ë˜í”„ ë° ì§„ì… ì°¨ìˆ˜ êµ¬ì„±
    graph = defaultdict(list)
    in_degree = [0] * n

    for u, v in edges:
        graph[u].append(v)
        in_degree[v] += 1

    # ì§„ì… ì°¨ìˆ˜ 0ì¸ ë…¸ë“œë¡œ ì‹œì‘
    queue = deque([i for i in range(n) if in_degree[i] == 0])
    result = []

    while queue:
        node = queue.popleft()
        result.append(node)

        for neighbor in graph[node]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)

    # ëª¨ë“  ë…¸ë“œ ë°©ë¬¸ ì—¬ë¶€ë¡œ ì‚¬ì´í´ í™•ì¸
    return result if len(result) == n else []

def topological_sort_dfs(n: int, edges: List[Tuple[int, int]]) -> List[int]:
    """
    DFS ê¸°ë°˜ ìœ„ìƒ ì •ë ¬
    ì‹œê°„ë³µì¡ë„: O(V + E)
    ê³µê°„ë³µì¡ë„: O(V + E)
    """
    graph = defaultdict(list)
    for u, v in edges:
        graph[u].append(v)

    WHITE, GRAY, BLACK = 0, 1, 2
    color = [WHITE] * n
    result = []
    has_cycle = False

    def dfs(node: int) -> None:
        nonlocal has_cycle

        if has_cycle:
            return

        color[node] = GRAY  # ë°©ë¬¸ ì¤‘

        for neighbor in graph[node]:
            if color[neighbor] == GRAY:  # ì‚¬ì´í´ ë°œê²¬
                has_cycle = True
                return
            if color[neighbor] == WHITE:
                dfs(neighbor)

        color[node] = BLACK  # ë°©ë¬¸ ì™„ë£Œ
        result.append(node)

    for i in range(n):
        if color[i] == WHITE:
            dfs(i)

    return result[::-1] if not has_cycle else []

def has_cycle(n: int, edges: List[Tuple[int, int]]) -> bool:
    """
    ë°©í–¥ ê·¸ë˜í”„ì—ì„œ ì‚¬ì´í´ ì¡´ì¬ ì—¬ë¶€
    ì‹œê°„ë³µì¡ë„: O(V + E)
    """
    graph = defaultdict(list)
    for u, v in edges:
        graph[u].append(v)

    WHITE, GRAY, BLACK = 0, 1, 2
    color = [WHITE] * n

    def dfs(node: int) -> bool:
        color[node] = GRAY

        for neighbor in graph[node]:
            if color[neighbor] == GRAY:
                return True  # ì‚¬ì´í´
            if color[neighbor] == WHITE and dfs(neighbor):
                return True

        color[node] = BLACK
        return False

    for i in range(n):
        if color[i] == WHITE and dfs(i):
            return True

    return False

def all_topological_sorts(n: int, edges: List[Tuple[int, int]]) -> List[List[int]]:
    """
    ëª¨ë“  ê°€ëŠ¥í•œ ìœ„ìƒ ì •ë ¬ ìˆœì„œ ì°¾ê¸°
    ì‹œê°„ë³µì¡ë„: O(V! * (V + E)) - ìµœì•…ì˜ ê²½ìš°
    """
    graph = defaultdict(list)
    in_degree = [0] * n

    for u, v in edges:
        graph[u].append(v)
        in_degree[v] += 1

    result = []
    path = []
    visited = [False] * n

    def backtrack():
        # ëª¨ë“  ë…¸ë“œ ë°©ë¬¸ ì™„ë£Œ
        if len(path) == n:
            result.append(path.copy())
            return

        for i in range(n):
            if not visited[i] and in_degree[i] == 0:
                # ì„ íƒ
                visited[i] = True
                path.append(i)
                for neighbor in graph[i]:
                    in_degree[neighbor] -= 1

                backtrack()

                # ë³µì›
                visited[i] = False
                path.pop()
                for neighbor in graph[i]:
                    in_degree[neighbor] += 1

    backtrack()
    return result

def can_finish(num_courses: int, prerequisites: List[List[int]]) -> bool:
    """
    ëª¨ë“  ê³¼ëª© ìˆ˜ê°• ê°€ëŠ¥ ì—¬ë¶€ (ì‚¬ì´í´ ì—†ìŒ = ê°€ëŠ¥)
    prerequisites: [course, prereq] - prereq â†’ course

def find_order(num_courses: int, prerequisites: List[List[int]]) -> List[int]:
    """
    ê³¼ëª© ìˆ˜ê°• ìˆœì„œ ë°˜í™˜
    """
    edges = [(prereq, course) for course, prereq in prerequisites]
    return topological_sort_kahn(num_courses, edges)

def alien_order(words: List[str]) -> str:
    """
    ì™¸ê³„ì¸ ì•ŒíŒŒë²³ ìˆœì„œ ê²°ì •
    ë‹¨ì–´ ëª©ë¡ì´ ì‚¬ì „ìˆœìœ¼ë¡œ ì •ë ¬ë˜ì–´ ìˆë‹¤ê³  ê°€ì •
    """
    # ëª¨ë“  ë¬¸ì ìˆ˜ì§‘
    chars = set()
    for word in words:
        chars.update(word)

    # ê·¸ë˜í”„ êµ¬ì„±
    graph = defaultdict(set)
    in_degree = defaultdict(int)
    for char in chars:
        in_degree[char] = 0

    for i in range(len(words) - 1):
        word1, word2 = words[i], words[i + 1]

        # ìœ íš¨í•˜ì§€ ì•Šì€ ìˆœì„œ ì²´í¬
        if len(word1) > len(word2) and word1.startswith(word2):
            return ""

        # ì²« ë²ˆì§¸ ë‹¤ë¥¸ ë¬¸ìë¡œ ìˆœì„œ ê²°ì •
        for c1, c2 in zip(word1, word2):
            if c1 != c2:
                if c2 not in graph[c1]:
                    graph[c1].add(c2)
                    in_degree[c2] += 1
                break

    # ìœ„ìƒ ì •ë ¬
    queue = deque([c for c in chars if in_degree[c] == 0])
    result = []

    while queue:
        char = queue.popleft()
        result.append(char)

        for neighbor in graph[char]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)

    return ''.join(result) if len(result) == len(chars) else ""

def minimum_semesters(n: int, relations: List[List[int]]) -> int:
    """
    ëª¨ë“  ê³¼ëª©ì„ ìˆ˜ê°•í•˜ëŠ”ë° í•„ìš”í•œ ìµœì†Œ í•™ê¸° ìˆ˜
    ë³‘ë ¬ ìˆ˜ê°• ê°€ëŠ¥, relations: [prev, next]
    """
    graph = defaultdict(list)
    in_degree = [0] * (n + 1)

    for prev_course, next_course in relations:
        graph[prev_course].append(next_course)
        in_degree[next_course] += 1

    # ì§„ì… ì°¨ìˆ˜ 0ì¸ ë…¸ë“œë¡œ ì‹œì‘ (1-indexed)
    queue = deque([i for i in range(1, n + 1) if in_degree[i] == 0])
    semesters = 0
    completed = 0

    while queue:
        semesters += 1
        next_queue = deque()

        while queue:
            course = queue.popleft()
            completed += 1

            for next_course in graph[course]:
                in_degree[next_course] -= 1
                if in_degree[next_course] == 0:
                    next_queue.append(next_course)

        queue = next_queue

    return semesters if completed == n else -1

def longest_path_dag(n: int, edges: List[Tuple[int, int, int]]) -> List[int]:
    """
    DAGì—ì„œ ê° ë…¸ë“œê¹Œì§€ì˜ ìµœì¥ ê²½ë¡œ (ê°€ì¤‘ì¹˜ í¬í•¨)
    edges: [(from, to, weight), ...]
    """
    graph = defaultdict(list)
    in_degree = [0] * n

    for u, v, w in edges:
        graph[u].append((v, w))
        in_degree[v] += 1

    # ìœ„ìƒ ì •ë ¬
    topo_order = []
    queue = deque([i for i in range(n) if in_degree[i] == 0])

    while queue:
        node = queue.popleft()
        topo_order.append(node)
        for neighbor, _ in graph[node]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)

    # ìµœì¥ ê²½ë¡œ ê³„ì‚°
    dist = [0] * n

    for node in topo_order:
        for neighbor, weight in graph[node]:
            dist[neighbor] = max(dist[neighbor], dist[node] + weight)

    return dist

def build_order(projects: List[str], dependencies: List[Tuple[str, str]]) -> List[str]:
    """
    ë¹Œë“œ ìˆœì„œ ê²°ì •
    dependencies: [(proj, depends_on), ...] - projê°€ depends_onì— ì˜ì¡´
    """
    # í”„ë¡œì íŠ¸ ì¸ë±ìŠ¤ ë§¤í•‘
    proj_to_idx = {p: i for i, p in enumerate(projects)}
    n = len(projects)

    # ê°„ì„  ë³€í™˜ (depends_on â†’ proj)
    edges = [(proj_to_idx[dep], proj_to_idx[proj]) for proj, dep in dependencies]

    # ìœ„ìƒ ì •ë ¬
    order = topological_sort_kahn(n, edges)

    if not order:
        return []  # ì‚¬ì´í´ ì¡´ì¬

    return [projects[i] for i in order]
