# ê°€ì¤‘ì¹˜ ê·¸ë˜í”„ í‘œí˜„

def create_weighted_graph(edges: List[Tuple[int, int, int]], directed: bool = False) -> Dict[int, List[Tuple[int, int]]]:
    """
    ê°„ì„  ë¦¬ìŠ¤íŠ¸ (u, v, weight)ë¡œë¶€í„° ê°€ì¤‘ì¹˜ ì¸ì ‘ ë¦¬ìŠ¤íŠ¸ ìƒì„±
    graph[u] = [(v, weight), ...]
    """
    graph = defaultdict(list)
    for u, v, w in edges:
        graph[u].append((v, w))
        if not directed:
            graph[v].append((u, w))
    return graph

def dijkstra(graph: Dict[int, List[Tuple[int, int]]], start: int, n: int) -> List[int]:
    """
    ë‹¤ìµìŠ¤íŠ¸ë¼ ì•Œê³ ë¦¬ì¦˜ (ìš°ì„ ìˆœìœ„ í ì‚¬ìš©)
    ì‹œê°„ë³µì¡ë„: O((V + E) log V)

        start: ì‹œì‘ ë…¸ë“œ

    Returns:
        ê° ë…¸ë“œê¹Œì§€ì˜ ìµœë‹¨ ê±°ë¦¬ ë°°ì—´ (ë„ë‹¬ ë¶ˆê°€ì‹œ ë¬´í•œëŒ€)
    """
    INF = float('inf')
    dist = [INF] * n
    dist[start] = 0

    # (ê±°ë¦¬, ë…¸ë“œ) íŠœí”Œì„ ì €ì¥í•˜ëŠ” ìµœì†Œ í™
    pq = [(0, start)]

    while pq:
        d, u = heapq.heappop(pq)

        # ì´ë¯¸ ì²˜ë¦¬ëœ ê±°ë¦¬ë³´ë‹¤ í¬ë©´ ìŠ¤í‚µ
        if d > dist[u]:
            continue

        # ì¸ì ‘ ë…¸ë“œ í™•ì¸
        for v, weight in graph[u]:
            new_dist = dist[u] + weight
            if new_dist < dist[v]:
                dist[v] = new_dist
                heapq.heappush(pq, (new_dist, v))

    return dist

# 2. ë‹¤ìµìŠ¤íŠ¸ë¼ + ê²½ë¡œ ì¶”ì 
# =============================================================================
def dijkstra_with_path(
    graph: Dict[int, List[Tuple[int, int]]],
    start: int,
    end: int,
    n: int
) -> Tuple[int, List[int]]:
    """
    ìµœë‹¨ ê±°ë¦¬ì™€ í•¨ê»˜ ì‹¤ì œ ê²½ë¡œë„ ë°˜í™˜
    """
    INF = float('inf')
    dist = [INF] * n
    dist[start] = 0
    parent = [-1] * n

    pq = [(0, start)]

    while pq:
        d, u = heapq.heappop(pq)

        if d > dist[u]:
            continue

        if u == end:
            break

        for v, weight in graph[u]:
            new_dist = dist[u] + weight
            if new_dist < dist[v]:
                dist[v] = new_dist
                parent[v] = u
                heapq.heappush(pq, (new_dist, v))

    # ê²½ë¡œ ë³µì›
    if dist[end] == INF:
        return INF, []

    path = []
    node = end
    while node != -1:
        path.append(node)
        node = parent[node]
    path.reverse()

    return dist[end], path

def floyd_warshall(n: int, edges: List[Tuple[int, int, int]]) -> List[List[int]]:
    """
    í”Œë¡œì´ë“œ-ì›Œì…œ ì•Œê³ ë¦¬ì¦˜
    ëª¨ë“  ì •ì  ìŒ ì‚¬ì´ì˜ ìµœë‹¨ ê±°ë¦¬ ê³„ì‚°
    ì‹œê°„ë³µì¡ë„: O(VÂ³)
    """
    INF = float('inf')
    dist = [[INF] * n for _ in range(n)]

    # ìê¸° ìì‹ ìœ¼ë¡œì˜ ê±°ë¦¬ëŠ” 0
    for i in range(n):
        dist[i][i] = 0

    # ê°„ì„  ì •ë³´ ë°˜ì˜
    for u, v, w in edges:
        dist[u][v] = w

    # kë¥¼ ê²½ìœ í•˜ëŠ” ê²½ë¡œ ê³ ë ¤
    for k in range(n):
        for i in range(n):
            for j in range(n):
                if dist[i][k] + dist[k][j] < dist[i][j]:
                    dist[i][j] = dist[i][k] + dist[k][j]

    return dist

def bellman_ford(n: int, edges: List[Tuple[int, int, int]], start: int) -> Tuple[List[int], bool]:
    """
    ë²¨ë§Œ-í¬ë“œ ì•Œê³ ë¦¬ì¦˜
    ìŒì˜ ê°€ì¤‘ì¹˜ë¥¼ í—ˆìš©í•˜ë©°, ìŒì˜ ì‚¬ì´í´ ê²€ì¶œ ê°€ëŠ¥
    ì‹œê°„ë³µì¡ë„: O(VE)

    Returns:
        (ìµœë‹¨ ê±°ë¦¬ ë°°ì—´, ìŒì˜ ì‚¬ì´í´ ì¡´ì¬ ì—¬ë¶€)
    """
    INF = float('inf')
    dist = [INF] * n
    dist[start] = 0

    # V-1ë²ˆ ë°˜ë³µ
    for _ in range(n - 1):
        for u, v, w in edges:
            if dist[u] != INF and dist[u] + w < dist[v]:
                dist[v] = dist[u] + w

    # ìŒì˜ ì‚¬ì´í´ ê²€ì¶œ (í•œ ë²ˆ ë” ë°˜ë³µí•´ì„œ ê°±ì‹ ë˜ë©´ ìŒì˜ ì‚¬ì´í´)
    has_negative_cycle = False
    for u, v, w in edges:
        if dist[u] != INF and dist[u] + w < dist[v]:
            has_negative_cycle = True
            break

    return dist, has_negative_cycle

def network_delay_time(times: List[List[int]], n: int, k: int) -> int:
    """
    ë…¸ë“œ kì—ì„œ ëª¨ë“  ë…¸ë“œë¡œ ì‹ í˜¸ê°€ ë„ë‹¬í•˜ëŠ” ìµœì†Œ ì‹œê°„
    times[i] = [source, target, time]
    ë„ë‹¬ ë¶ˆê°€ëŠ¥í•˜ë©´ -1 ë°˜í™˜
    """
    graph = defaultdict(list)
    for u, v, w in times:
        graph[u].append((v, w))

    dist = dijkstra(graph, k, n + 1)  # ë…¸ë“œ ë²ˆí˜¸ê°€ 1ë¶€í„° ì‹œì‘

    # 1~në²ˆ ë…¸ë“œ ì¤‘ ìµœëŒ€ ê±°ë¦¬
    max_time = max(dist[1:n + 1])

    return max_time if max_time != float('inf') else -1

# 6. Kë²ˆì§¸ ìµœë‹¨ ê²½ë¡œ
# =============================================================================
def kth_shortest_path(
    graph: Dict[int, List[Tuple[int, int]]],
    start: int,
    end: int,
    k: int,
    n: int
) -> int:
    """
    Kë²ˆì§¸ë¡œ ì§§ì€ ê²½ë¡œì˜ ê¸¸ì´ ë°˜í™˜
    ì°¾ì„ ìˆ˜ ì—†ìœ¼ë©´ -1 ë°˜í™˜
    """
    INF = float('inf')
    count = [0] * n  # ê° ë…¸ë“œì— ë„ì°©í•œ íšŸìˆ˜
    pq = [(0, start)]

    while pq:
        d, u = heapq.heappop(pq)
        count[u] += 1

        if u == end and count[u] == k:
            return d

        # kë²ˆ ì´ìƒ ë°©ë¬¸í•œ ë…¸ë“œëŠ” ë” ì´ìƒ í™•ì¥í•˜ì§€ ì•ŠìŒ
        if count[u] > k:
            continue

        for v, weight in graph[u]:
            heapq.heappush(pq, (d + weight, v))

    return -1
