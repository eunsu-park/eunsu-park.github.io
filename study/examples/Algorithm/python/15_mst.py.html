class UnionFind:
    """
    ê¸°ë³¸ ìœ ë‹ˆì˜¨ íŒŒì¸ë“œ êµ¬í˜„
    - ê²½ë¡œ ì••ì¶• (Path Compression)

    """

    def __init__(self, n: int):

        """
        self.parent = list(range(n))  # ìê¸° ìì‹ ì„ ë¶€ëª¨ë¡œ
        self.rank = [0] * n           # íŠ¸ë¦¬ì˜ ë†’ì´ (ê·¼ì‚¬ê°’)
        self.count = n                # ì§‘í•©ì˜ ê°œìˆ˜

    def find(self, x: int) -> int:

        ê²½ë¡œ ì••ì¶•ìœ¼ë¡œ ê±°ì˜ O(1)

        if self.parent[x] != x:

        return self.parent[x]

    def union(self, x: int, y: int) -> bool:
        """

        """
        root_x = self.find(x)

        # ë­í¬ ê¸°ë°˜ í•©ì¹˜ê¸° (ì‘ì€ íŠ¸ë¦¬ë¥¼ í° íŠ¸ë¦¬ì— ë¶™ì„)
        if self.rank[root_x] < self.rank[root_y]:
            self.parent[root_x] = root_y
        elif self.rank[root_x] > self.rank[root_y]:
            self.parent[root_y] = root_x
        else:
            self.parent[root_y] = root_x
            self.rank[root_x] += 1

        return True

    def connected(self, x: int, y: int) -> bool:

        return self.find(x) == self.find(y)

        return self.count

class UnionFindWithSize:
    """
    ì§‘í•©ì˜ í¬ê¸°ë¥¼ ì¶”ì í•˜ëŠ” ìœ ë‹ˆì˜¨ íŒŒì¸ë“œ
    """

    def __init__(self, n: int):
        self.parent = list(range(n))
        self.size = [1] * n  # ê° ì§‘í•©ì˜ í¬ê¸°

    def find(self, x: int) -> int:
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x: int, y: int) -> bool:
        root_x = self.find(x)
        root_y = self.find(y)

        if root_x == root_y:
            return False

        # í¬ê¸° ê¸°ë°˜ í•©ì¹˜ê¸° (ì‘ì€ ì§‘í•©ì„ í° ì§‘í•©ì—)
        if self.size[root_x] < self.size[root_y]:
            self.parent[root_x] = root_y
            self.size[root_y] += self.size[root_x]
        else:
            self.parent[root_y] = root_x
            self.size[root_x] += self.size[root_y]

        return True

    def get_size(self, x: int) -> int:
        """xê°€ ì†í•œ ì§‘í•©ì˜ í¬ê¸°"""
        return self.size[self.find(x)]

def count_components(n: int, edges: List[List[int]]) -> int:
    """
    nê°œì˜ ë…¸ë“œì™€ ê°„ì„  ëª©ë¡ì´ ì£¼ì–´ì§ˆ ë•Œ ì—°ê²° ìš”ì†Œ ê°œìˆ˜

    uf = UnionFind(n)
    for u, v in edges:
        uf.union(u, v)
    return uf.get_count()

def has_cycle(n: int, edges: List[List[int]]) -> bool:
    """
    ë¬´ë°©í–¥ ê·¸ë˜í”„ì—ì„œ ì‚¬ì´í´ ì¡´ì¬ ì—¬ë¶€
    ê°„ì„ ì„ ì¶”ê°€í•  ë•Œ ì´ë¯¸ ê°™ì€ ì§‘í•©ì´ë©´ ì‚¬ì´í´

    uf = UnionFind(n)

            return True  # ì´ë¯¸ ì—°ê²°ë¨ = ì‚¬ì´í´
    return False

def kruskal_mst(n: int, edges: List[Tuple[int, int, int]]) -> Tuple[int, List[Tuple[int, int, int]]]:
    """
    í¬ë£¨ìŠ¤ì¹¼ ì•Œê³ ë¦¬ì¦˜ìœ¼ë¡œ ìµœì†Œ ì‹ ì¥ íŠ¸ë¦¬ êµ¬í•˜ê¸°
    edges: [(u, v, weight), ...]
    ë°˜í™˜: (ì´ ê°€ì¤‘ì¹˜, MST ê°„ì„  ë¦¬ìŠ¤íŠ¸)
    """
    # ê°€ì¤‘ì¹˜ ê¸°ì¤€ ì •ë ¬
    edges = sorted(edges, key=lambda x: x[2])

    uf = UnionFind(n)
    mst_weight = 0
    mst_edges = []

    for u, v, w in edges:
        if uf.union(u, v):
            mst_weight += w
            mst_edges.append((u, v, w))

            # n-1ê°œì˜ ê°„ì„ ì„ ì„ íƒí•˜ë©´ ì™„ë£Œ
            if len(mst_edges) == n - 1:
                break

    return mst_weight, mst_edges

def merge_accounts(accounts: List[List[str]]) -> List[List[str]]:
    """
    ê°™ì€ ì´ë©”ì¼ì„ ê°€ì§„ ê³„ì • ë³‘í•©
    accounts[i] = [ì´ë¦„, ì´ë©”ì¼1, ì´ë©”ì¼2, ...]
    """
    from collections import defaultdict

    # ì´ë©”ì¼ -> ê³„ì • ì¸ë±ìŠ¤ ë§¤í•‘
    email_to_id = {}
    email_to_name = {}

    for i, account in enumerate(accounts):
        name = account[0]
        for email in account[1:]:
            if email in email_to_id:
                pass  # ë‚˜ì¤‘ì— union
            email_to_id[email] = i
            email_to_name[email] = name

    # ìœ ë‹ˆì˜¨ íŒŒì¸ë“œë¡œ ê°™ì€ ì‚¬ëŒì˜ ê³„ì • ì—°ê²°
    n = len(accounts)
    uf = UnionFind(n)

    email_first_account = {}
    for i, account in enumerate(accounts):
        for email in account[1:]:
            if email in email_first_account:
                uf.union(i, email_first_account[email])
            else:
                email_first_account[email] = i

    # ê²°ê³¼ ì§‘ê³„
    root_to_emails = defaultdict(set)
    for i, account in enumerate(accounts):
        root = uf.find(i)
        for email in account[1:]:
            root_to_emails[root].add(email)

    # ê²°ê³¼ í¬ë§·íŒ…
    result = []
    for root, emails in root_to_emails.items():
        name = accounts[root][0]
        result.append([name] + sorted(emails))

    return result

def num_islands_union_find(grid: List[List[str]]) -> int:
    """
    '1'ì€ ë•…, '0'ì€ ë¬¼
    ì—°ê²°ëœ ë•… ë©ì–´ë¦¬(ì„¬)ì˜ ê°œìˆ˜
    """
    if not grid or not grid[0]:
        return 0

    rows, cols = len(grid), len(grid[0])

    # 2D -> 1D ì¢Œí‘œ ë³€í™˜
    def get_index(r, c):
        return r * cols + c

    uf = UnionFind(rows * cols)
    land_count = 0

    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == '1':
                land_count += 1
                # ì˜¤ë¥¸ìª½, ì•„ë˜ ë°©í–¥ë§Œ í™•ì¸ (ì¤‘ë³µ ë°©ì§€)
                for dr, dc in [(0, 1), (1, 0)]:
                    nr, nc = r + dr, c + dc
                    if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == '1':
                        if uf.union(get_index(r, c), get_index(nr, nc)):
                            land_count -= 1

    return land_count
