from typing import List, Tuple, Set

# 1. Kosaraju ì•Œê³ ë¦¬ì¦˜
# =============================================================================

def kosaraju_scc(n: int, edges: List[Tuple[int, int]]) -> List[List[int]]:
    """
    Kosaraju ì•Œê³ ë¦¬ì¦˜ìœ¼ë¡œ SCC ì°¾ê¸°
    ì‹œê°„ë³µì¡ë„: O(V + E)
    ê³µê°„ë³µì¡ë„: O(V + E)

    1. ì •ë°©í–¥ DFSë¡œ ì¢…ë£Œ ìˆœì„œ ìŠ¤íƒ êµ¬ì„±
    2. ì—­ë°©í–¥ ê·¸ë˜í”„ì—ì„œ ìŠ¤íƒ ìˆœì„œëŒ€ë¡œ DFS
    """
    # ì •ë°©í–¥/ì—­ë°©í–¥ ê·¸ë˜í”„
    graph = defaultdict(list)
    reverse_graph = defaultdict(list)

    for u, v in edges:
        graph[u].append(v)
        reverse_graph[v].append(u)

    # 1ë‹¨ê³„: ì •ë°©í–¥ DFSë¡œ ì¢…ë£Œ ìˆœì„œ ê¸°ë¡
    visited = [False] * n
    finish_stack = []

    def dfs1(node: int):
        visited[node] = True
        for neighbor in graph[node]:
            if not visited[neighbor]:
                dfs1(neighbor)
        finish_stack.append(node)

    for i in range(n):
        if not visited[i]:
            dfs1(i)

    # 2ë‹¨ê³„: ì—­ë°©í–¥ DFSë¡œ SCC ì°¾ê¸°
    visited = [False] * n
    sccs = []

    def dfs2(node: int, component: List[int]):
        visited[node] = True
        component.append(node)
        for neighbor in reverse_graph[node]:
            if not visited[neighbor]:
                dfs2(neighbor, component)

    while finish_stack:
        node = finish_stack.pop()
        if not visited[node]:
            component = []
            dfs2(node, component)
            sccs.append(component)

    return sccs

def tarjan_scc(n: int, edges: List[Tuple[int, int]]) -> List[List[int]]:
    """
    Tarjan ì•Œê³ ë¦¬ì¦˜ìœ¼ë¡œ SCC ì°¾ê¸°
    ì‹œê°„ë³µì¡ë„: O(V + E)
    ê³µê°„ë³µì¡ë„: O(V)

    low[v] = vì—ì„œ ë„ë‹¬ ê°€ëŠ¥í•œ ìµœì†Œ ë°œê²¬ ì‹œê°„
    """
    graph = defaultdict(list)
    for u, v in edges:
        graph[u].append(v)

    disc = [-1] * n  # ë°œê²¬ ì‹œê°„
    low = [-1] * n   # low-link ê°’
    on_stack = [False] * n
    stack = []
    sccs = []
    time = [0]  # ì „ì—­ ì‹œê°„

    def dfs(node: int):
        disc[node] = low[node] = time[0]
        time[0] += 1
        stack.append(node)
        on_stack[node] = True

        for neighbor in graph[node]:
            if disc[neighbor] == -1:  # ë¯¸ë°©ë¬¸
                dfs(neighbor)
                low[node] = min(low[node], low[neighbor])
            elif on_stack[neighbor]:  # ìŠ¤íƒì— ìˆìŒ (back edge)
                low[node] = min(low[node], disc[neighbor])

        # SCC ë£¨íŠ¸ ë°œê²¬
        if low[node] == disc[node]:
            component = []
            while True:
                v = stack.pop()
                on_stack[v] = False
                component.append(v)
                if v == node:
                    break
            sccs.append(component)

    for i in range(n):
        if disc[i] == -1:
            dfs(i)

    return sccs

def build_scc_dag(n: int, edges: List[Tuple[int, int]]) -> Tuple[List[List[int]], List[List[int]], List[int]]:
    """
    SCCë¥¼ ì°¾ê³  ì¶•ì•½ ê·¸ë˜í”„(DAG) êµ¬ì„±
    ë°˜í™˜: (sccs, scc_graph, node_to_scc)
    """
    sccs = tarjan_scc(n, edges)

    # ê° ë…¸ë“œê°€ ì†í•œ SCC ë§¤í•‘
    node_to_scc = [-1] * n
    for i, component in enumerate(sccs):
        for node in component:
            node_to_scc[node] = i

    # SCC ê°„ì˜ ê°„ì„  (DAG)
    scc_edges = set()
    for u, v in edges:
        scc_u = node_to_scc[u]
        scc_v = node_to_scc[v]
        if scc_u != scc_v:
            scc_edges.add((scc_u, scc_v))

    scc_graph = defaultdict(list)
    for u, v in scc_edges:
        scc_graph[u].append(v)

    return sccs, dict(scc_graph), node_to_scc

class TwoSAT:
    """
    2-SAT ë¬¸ì œ í•´ê²°ê¸°
    nê°œì˜ ë¶ˆë¦° ë³€ìˆ˜ì™€ 2-CNF ì¡°ê±´ì‹

    ë³€ìˆ˜ x_i: ë…¸ë“œ 2*i (true), ë…¸ë“œ 2*i+1 (false)
    ì¡°ê±´ (a âˆ¨ b): Â¬a â†’ b, Â¬b â†’ a
    """

    def __init__(self, n: int):
        self.n = n
        self.graph = defaultdict(list)
        self.reverse_graph = defaultdict(list)

    def _var(self, x: int, negated: bool) -> int:
        """ë³€ìˆ˜ë¥¼ ê·¸ë˜í”„ ë…¸ë“œë¡œ ë³€í™˜"""
        return 2 * x + (1 if negated else 0)

    def _neg(self, node: int) -> int:
        """ë…¸ë“œì˜ ë¶€ì •"""
        return node ^ 1

    def add_clause(self, x: int, neg_x: bool, y: int, neg_y: bool):
        """
        ì¡°ê±´ ì¶”ê°€: (x âˆ¨ y)
        neg_x: xê°€ ë¶€ì •ì¸ì§€
        neg_y: yê°€ ë¶€ì •ì¸ì§€
        """
        # (x âˆ¨ y) â‰¡ (Â¬x â†’ y) âˆ§ (Â¬y â†’ x)
        node_x = self._var(x, neg_x)
        node_y = self._var(y, neg_y)

        # Â¬x â†’ y
        self.graph[self._neg(node_x)].append(node_y)
        self.reverse_graph[node_y].append(self._neg(node_x))

        # Â¬y â†’ x
        self.graph[self._neg(node_y)].append(node_x)
        self.reverse_graph[node_x].append(self._neg(node_y))

    def solve(self) -> Tuple[bool, List[bool]]:
        """
        2-SAT í•´ê²°
        ë°˜í™˜: (ë§Œì¡± ê°€ëŠ¥ ì—¬ë¶€, ê° ë³€ìˆ˜ì˜ ê°’)
        """
        total_nodes = 2 * self.n

        # Kosarajuë¡œ SCC ì°¾ê¸°
        visited = [False] * total_nodes
        finish_stack = []

        def dfs1(node: int):
            visited[node] = True
            for neighbor in self.graph[node]:
                if not visited[neighbor]:
                    dfs1(neighbor)
            finish_stack.append(node)

        for i in range(total_nodes):
            if not visited[i]:
                dfs1(i)

        visited = [False] * total_nodes
        scc_id = [-1] * total_nodes
        current_scc = 0

        def dfs2(node: int):
            visited[node] = True
            scc_id[node] = current_scc
            for neighbor in self.reverse_graph[node]:
                if not visited[neighbor]:
                    dfs2(neighbor)

        while finish_stack:
            node = finish_stack.pop()
            if not visited[node]:
                dfs2(node)
                current_scc += 1

        # ë§Œì¡± ê°€ëŠ¥ì„± ê²€ì‚¬
        for i in range(self.n):
            if scc_id[2 * i] == scc_id[2 * i + 1]:
                return False, []

        # í•´ êµ¬ì„± (ë‚˜ì¤‘ì— ë°œê²¬ëœ SCC = ë” ì‘ì€ SCC ID = ë” ë†’ì€ ìœ„ìƒ ìˆœì„œ)
        assignment = [False] * self.n
        for i in range(self.n):
            # scc_idê°€ ì‘ìœ¼ë©´ ìœ„ìƒ ìˆœì„œì—ì„œ ë’¤ì— ìˆìŒ â†’ ê·¸ ê°’ì´ true
            assignment[i] = scc_id[2 * i] > scc_id[2 * i + 1]

        return True, assignment

def min_roads_to_connect(n: int, roads: List[Tuple[int, int]]) -> int:
    """
    ëª¨ë“  í•™êµê°€ ì„œë¡œ ë„ë‹¬ ê°€ëŠ¥í•˜ê²Œ í•˜ë ¤ë©´ ì¶”ê°€í•´ì•¼ í•  ìµœì†Œ ë„ë¡œ ìˆ˜
    = max(ì§„ì… ì°¨ìˆ˜ 0ì¸ SCC ìˆ˜, ì§„ì¶œ ì°¨ìˆ˜ 0ì¸ SCC ìˆ˜)
    (SCCê°€ 1ê°œë©´ 0)
    """
    if not roads:
        return n - 1 if n > 1 else 0

    sccs, scc_graph, node_to_scc = build_scc_dag(n, roads)

    if len(sccs) == 1:
        return 0

    # ê° SCCì˜ ì§„ì…/ì§„ì¶œ ì°¨ìˆ˜
    in_degree = [0] * len(sccs)
    out_degree = [0] * len(sccs)

    for scc_u, neighbors in scc_graph.items():
        out_degree[scc_u] = len(neighbors)
        for scc_v in neighbors:
            in_degree[scc_v] += 1

    # ì§„ì…/ì§„ì¶œ ì°¨ìˆ˜ 0ì¸ SCC ê°œìˆ˜
    sources = sum(1 for d in in_degree if d == 0)
    sinks = sum(1 for d in out_degree if d == 0)

    return max(sources, sinks)

def find_critical_nodes(n: int, edges: List[Tuple[int, int]]) -> List[int]:
    """
    ì œê±°í•˜ë©´ SCC ê°œìˆ˜ê°€ ì¦ê°€í•˜ëŠ” ë…¸ë“œë“¤ ì°¾ê¸°
    (ê°„ë‹¨í•œ brute force êµ¬í˜„)
    """
    original_scc_count = len(tarjan_scc(n, edges))
    critical = []

    for remove_node in range(n):
        # í•´ë‹¹ ë…¸ë“œ ì œì™¸í•œ ê·¸ë˜í”„
        new_edges = [(u, v) for u, v in edges if u != remove_node and v != remove_node]

        # ë…¸ë“œ ì¬ë§¤í•‘
        remaining = [i for i in range(n) if i != remove_node]
        if not remaining:
            continue

        node_map = {old: new for new, old in enumerate(remaining)}
        remapped_edges = [(node_map[u], node_map[v]) for u, v in new_edges
                          if u in node_map and v in node_map]

        new_scc_count = len(tarjan_scc(len(remaining), remapped_edges))

        if new_scc_count > original_scc_count - 1:  # -1ì€ ì œê±°ëœ ë…¸ë“œì˜ SCC
            critical.append(remove_node)

    return critical
