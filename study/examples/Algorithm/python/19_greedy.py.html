def activity_selection(activities: List[Tuple[int, int]]) -> List[Tuple[int, int]]:
    """
    ê²¹ì¹˜ì§€ ì•ŠëŠ” ìµœëŒ€ í™œë™ ìˆ˜ ì„ íƒ
    activities: [(ì‹œì‘, ì¢…ë£Œ), ...]
    íƒìš• ì „ëµ: ì¢…ë£Œ ì‹œê°„ì´ ë¹ ë¥¸ ìˆœì„œë¡œ ì„ íƒ
    ì‹œê°„ë³µì¡ë„: O(n log n)
    """
    # ì¢…ë£Œ ì‹œê°„ìœ¼ë¡œ ì •ë ¬
    sorted_activities = sorted(activities, key=lambda x: x[1])

    selected = []
    last_end = 0

    for start, end in sorted_activities:
        if start >= last_end:
            selected.append((start, end))
            last_end = end

    return selected

def fractional_knapsack(capacity: int, items: List[Tuple[int, int]]) -> float:
    """
    ë¶„í•  ê°€ëŠ¥í•œ ë°°ë‚­ ë¬¸ì œ
    items: [(ê°€ì¹˜, ë¬´ê²Œ), ...]
    íƒìš• ì „ëµ: ë‹¨ìœ„ ë¬´ê²Œë‹¹ ê°€ì¹˜ê°€ ë†’ì€ ìˆœì„œë¡œ ì„ íƒ
    ì‹œê°„ë³µì¡ë„: O(n log n)
    """
    # ë‹¨ìœ„ ë¬´ê²Œë‹¹ ê°€ì¹˜ë¡œ ì •ë ¬ (ë‚´ë¦¼ì°¨ìˆœ)
    sorted_items = sorted(items, key=lambda x: x[0] / x[1], reverse=True)

    total_value = 0
    remaining = capacity

    for value, weight in sorted_items:
        if remaining >= weight:
            total_value += value
            remaining -= weight
        else:
            # ë¶„í• í•´ì„œ ë‹´ê¸°
            total_value += value * (remaining / weight)
            break

    return total_value

# 3. íšŒì˜ì‹¤ ë°°ì • (Meeting Rooms)
# =============================================================================

def min_meeting_rooms(intervals: List[Tuple[int, int]]) -> int:
    """
    í•„ìš”í•œ ìµœì†Œ íšŒì˜ì‹¤ ìˆ˜
    ì‹œê°„ë³µì¡ë„: O(n log n)
    """
    if not intervals:
        return 0

    # ì‹œì‘/ì¢…ë£Œ ì´ë²¤íŠ¸ë¡œ ë¶„ë¦¬
    events = []
    for start, end in intervals:
        events.append((start, 1))   # ì‹œì‘: +1
        events.append((end, -1))    # ì¢…ë£Œ: -1

    events.sort()

    max_rooms = current_rooms = 0
    for _, delta in events:
        current_rooms += delta
        max_rooms = max(max_rooms, current_rooms)

    return max_rooms

def job_scheduling(jobs: List[Tuple[int, int, int]]) -> Tuple[int, List[int]]:
    """
    ë§ˆê° ê¸°í•œ ë‚´ ìµœëŒ€ ì´ìµ ìŠ¤ì¼€ì¤„ë§
    jobs: [(ì‘ì—…ID, ë§ˆê°ì¼, ì´ìµ), ...]
    íƒìš• ì „ëµ: ì´ìµì´ ë†’ì€ ìˆœì„œë¡œ, ê°€ëŠ¥í•œ ëŠ¦ì€ ìŠ¬ë¡¯ì— ë°°ì •
    """
    # ì´ìµ ë‚´ë¦¼ì°¨ìˆœ ì •ë ¬
    sorted_jobs = sorted(jobs, key=lambda x: x[2], reverse=True)

    max_deadline = max(job[1] for job in jobs)
    slots = [None] * (max_deadline + 1)  # ê° ì‹œê°„ ìŠ¬ë¡¯
    total_profit = 0
    scheduled = []

    for job_id, deadline, profit in sorted_jobs:
        # ë§ˆê°ì¼ë¶€í„° ì—­ìˆœìœ¼ë¡œ ë¹ˆ ìŠ¬ë¡¯ ì°¾ê¸°
        for slot in range(deadline, 0, -1):
            if slots[slot] is None:
                slots[slot] = job_id
                total_profit += profit
                scheduled.append(job_id)
                break

    return total_profit, scheduled

# 5. í—ˆí”„ë§Œ ì½”ë”© (Huffman Coding)

class HuffmanNode:
    def __init__(self, char: str = None, freq: int = 0):
        self.char = char
        self.freq = freq
        self.left = None
        self.right = None

    def __lt__(self, other):
        return self.freq < other.freq


def huffman_encoding(text: str) -> Tuple[dict, str]:
    """
    í—ˆí”„ë§Œ ì¸ì½”ë”©
    ë°˜í™˜: (ë¬¸ìâ†’ì½”ë“œ ë”•ì…”ë„ˆë¦¬, ì¸ì½”ë”©ëœ ë¬¸ìì—´)
    ì‹œê°„ë³µì¡ë„: O(n log n)
    """
    if not text:
        return {}, ""

    # ë¹ˆë„ ê³„ì‚°
    freq = {}
    for char in text:
        freq[char] = freq.get(char, 0) + 1

    # ìš°ì„ ìˆœìœ„ íë¡œ íŠ¸ë¦¬ êµ¬ì„±
    heap = [HuffmanNode(char, f) for char, f in freq.items()]
    heapq.heapify(heap)

    while len(heap) > 1:
        left = heapq.heappop(heap)
        right = heapq.heappop(heap)

        merged = HuffmanNode(freq=left.freq + right.freq)
        merged.left = left
        merged.right = right

        heapq.heappush(heap, merged)

    # ì½”ë“œ ìƒì„±
    codes = {}

    def generate_codes(node: HuffmanNode, code: str):
        if node is None:
            return

        if node.char is not None:
            codes[node.char] = code if code else "0"
            return

        generate_codes(node.left, code + "0")
        generate_codes(node.right, code + "1")

    if heap:
        generate_codes(heap[0], "")

    # ì¸ì½”ë”©
    encoded = ''.join(codes[char] for char in text)

    return codes, encoded


def huffman_decoding(encoded: str, codes: dict) -> str:
    """í—ˆí”„ë§Œ ë””ì½”ë”©"""
    reverse_codes = {v: k for k, v in codes.items()}
    decoded = []
    current = ""

    for bit in encoded:
        current += bit
        if current in reverse_codes:
            decoded.append(reverse_codes[current])
            current = ""

    return ''.join(decoded)

def coin_change_greedy(coins: List[int], amount: int) -> List[int]:
    """
    ë™ì „ ê±°ìŠ¤ë¦„ëˆ (íƒìš•, íŠ¹ì • ë™ì „ ì§‘í•©ì—ì„œë§Œ ìµœì )
    coins: ë‚´ë¦¼ì°¨ìˆœ ì •ë ¬ëœ ë™ì „
    ì£¼ì˜: ì¼ë°˜ì ì¸ ë™ì „ ì¡°í•©ì—ì„œëŠ” DP í•„ìš”
    """
    coins = sorted(coins, reverse=True)
    result = []

    for coin in coins:
        while amount >= coin:
            result.append(coin)
            amount -= coin

    return result if amount == 0 else []

def interval_partitioning(intervals: List[Tuple[int, int]]) -> List[List[Tuple[int, int]]]:
    """
    êµ¬ê°„ë“¤ì„ ìµœì†Œ ê·¸ë£¹ ìˆ˜ë¡œ ë¶„í•  (ê° ê·¸ë£¹ ë‚´ ê²¹ì¹¨ ì—†ìŒ)
    ì‹œê°„ë³µì¡ë„: O(n log n)
    """
    if not intervals:
        return []

    # ì‹œì‘ ì‹œê°„ìœ¼ë¡œ ì •ë ¬
    sorted_intervals = sorted(enumerate(intervals), key=lambda x: x[1][0])

    # ê° ê·¸ë£¹ì˜ ì¢…ë£Œ ì‹œê°„ì„ í™ìœ¼ë¡œ ê´€ë¦¬
    groups = []  # [(ì¢…ë£Œì‹œê°„, ê·¸ë£¹ì¸ë±ìŠ¤)]
    assignment = [[] for _ in range(len(intervals))]

    for idx, (start, end) in sorted_intervals:
        if groups and groups[0][0] <= start:
            # ê¸°ì¡´ ê·¸ë£¹ì— ë°°ì •
            _, group_idx = heapq.heappop(groups)
            assignment[group_idx].append(intervals[idx])
            heapq.heappush(groups, (end, group_idx))
        else:
            # ìƒˆ ê·¸ë£¹ ìƒì„±
            new_group = len(groups)
            assignment.append([intervals[idx]])
            heapq.heappush(groups, (end, new_group))

    return [g for g in assignment if g]

def can_jump(nums: List[int]) -> bool:
    """
    ë§ˆì§€ë§‰ ì¸ë±ìŠ¤ ë„ë‹¬ ê°€ëŠ¥ ì—¬ë¶€
    nums[i] = iì—ì„œ ìµœëŒ€ ì í”„ ê±°ë¦¬
    ì‹œê°„ë³µì¡ë„: O(n)
    """
    max_reach = 0

    for i, jump in enumerate(nums):
        if i > max_reach:
            return False
        max_reach = max(max_reach, i + jump)

    return True


def min_jumps(nums: List[int]) -> int:
    """
    ë§ˆì§€ë§‰ ì¸ë±ìŠ¤ê¹Œì§€ ìµœì†Œ ì í”„ íšŸìˆ˜
    ì‹œê°„ë³µì¡ë„: O(n)
    """
    if len(nums) <= 1:
        return 0

    jumps = 0
    current_end = 0
    farthest = 0

    for i in range(len(nums) - 1):
        farthest = max(farthest, i + nums[i])

        if i == current_end:
            jumps += 1
            current_end = farthest

            if current_end >= len(nums) - 1:
                break

    return jumps

def can_complete_circuit(gas: List[int], cost: List[int]) -> int:
    """
    ì›í˜• ê²½ë¡œ ì™„ì£¼ ê°€ëŠ¥í•œ ì‹œì‘ì  ì°¾ê¸°
    gas[i] = iì—ì„œ ì–»ëŠ” ê¸°ë¦„
    cost[i] = iâ†’i+1 ì´ë™ ë¹„ìš©
    ì‹œê°„ë³µì¡ë„: O(n)
    """
    total_tank = 0
    current_tank = 0
    start = 0

    for i in range(len(gas)):
        gain = gas[i] - cost[i]
        total_tank += gain
        current_tank += gain

        if current_tank < 0:
            start = i + 1
            current_tank = 0

    return start if total_tank >= 0 else -1

def min_arrows(points: List[List[int]]) -> int:
    """
    ìˆ˜í‰ì„  í’ì„ ì„ í„°íŠ¸ë¦¬ëŠ” ìµœì†Œ í™”ì‚´ ìˆ˜
    points[i] = [ì‹œì‘, ë] ë²”ìœ„
    ì‹œê°„ë³µì¡ë„: O(n log n)

    if not points:
        return 0

    # ëì ìœ¼ë¡œ ì •ë ¬
    points.sort(key=lambda x: x[1])

    arrows = 1
    arrow_pos = points[0][1]

    for start, end in points[1:]:
        if start > arrow_pos:
            arrows += 1
            arrow_pos = end

    return arrows
