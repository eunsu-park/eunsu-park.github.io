def bit_operations_demo():

    n = 5  # ì§‘í•© í¬ê¸°

    # ë¹ˆ ì§‘í•©
    empty = 0

    # ì „ì²´ ì§‘í•© {0, 1, 2, 3, 4}
    full = (1 << n) - 1  # 11111 (ì´ì§„ìˆ˜)

    # ië²ˆì§¸ ì›ì†Œ ì¶”ê°€
    def add(mask: int, i: int) -> int:
        return mask | (1 << i)

    # ië²ˆì§¸ ì›ì†Œ ì œê±°
    def remove(mask: int, i: int) -> int:
        return mask & ~(1 << i)

    # ië²ˆì§¸ ì›ì†Œ í† ê¸€
    def toggle(mask: int, i: int) -> int:
        return mask ^ (1 << i)

    # ië²ˆì§¸ ì›ì†Œ í¬í•¨ ì—¬ë¶€
    def contains(mask: int, i: int) -> bool:
        return bool(mask & (1 << i))

    # ì›ì†Œ ê°œìˆ˜
    def count(mask: int) -> int:
        return bin(mask).count('1')

    # ìµœí•˜ìœ„ ë¹„íŠ¸ (ê°€ì¥ ì‘ì€ ì›ì†Œ)
    def lowest_bit(mask: int) -> int:
        return mask & (-mask)

    # ë¶€ë¶„ì§‘í•© ìˆœíšŒ
    def subsets(mask: int):
        """maskì˜ ëª¨ë“  ë¶€ë¶„ì§‘í•©ì„ ìˆœíšŒ"""
        subset = mask
        while True:
            yield subset
            if subset == 0:
                break

    return {
        'empty': empty,
        'full': full,
        'add': add,

        'toggle': toggle,

def tsp(dist: List[List[int]]) -> int:
    """
    ì™¸íŒì› ë¬¸ì œ (TSP)
    ëª¨ë“  ë„ì‹œë¥¼ ë°©ë¬¸í•˜ê³  ì‹œì‘ì ìœ¼ë¡œ ëŒì•„ì˜¤ëŠ” ìµœì†Œ ë¹„ìš©

    ì‹œê°„ë³µì¡ë„: O(nÂ² * 2^n)
    ê³µê°„ë³µì¡ë„: O(n * 2^n)
    """
    n = len(dist)
    INF = float('inf')

    # dp[mask][i] = maskì— í¬í•¨ëœ ë„ì‹œë¥¼ ë°©ë¬¸í•˜ê³  í˜„ì¬ iì— ìˆì„ ë•Œ ìµœì†Œ ë¹„ìš©
    dp = [[INF] * n for _ in range(1 << n)]
    dp[1][0] = 0  # ì‹œì‘ì  (ë„ì‹œ 0)

    for mask in range(1 << n):
        for last in range(n):
            if dp[mask][last] == INF:
                continue
            if not (mask & (1 << last)):
                continue

            for next_city in range(n):
                if mask & (1 << next_city):
                    continue

                new_mask = mask | (1 << next_city)
                dp[new_mask][next_city] = min(
                    dp[new_mask][next_city],
                    dp[mask][last] + dist[last][next_city]
                )

    # ëª¨ë“  ë„ì‹œ ë°©ë¬¸ í›„ ì‹œì‘ì ìœ¼ë¡œ
    full_mask = (1 << n) - 1
    result = min(dp[full_mask][i] + dist[i][0] for i in range(n))

    return result if result != INF else -1


def tsp_path(dist: List[List[int]]) -> Tuple[int, List[int]]:
    """TSP ìµœì†Œ ë¹„ìš©ê³¼ ê²½ë¡œ ë°˜í™˜"""
    n = len(dist)
    INF = float('inf')

    dp = [[INF] * n for _ in range(1 << n)]
    parent = [[-1] * n for _ in range(1 << n)]

    dp[1][0] = 0

    for mask in range(1 << n):
        for last in range(n):
            if dp[mask][last] == INF:
                continue

            for next_city in range(n):
                if mask & (1 << next_city):
                    continue

                new_mask = mask | (1 << next_city)
                new_cost = dp[mask][last] + dist[last][next_city]

                if new_cost < dp[new_mask][next_city]:
                    dp[new_mask][next_city] = new_cost
                    parent[new_mask][next_city] = last

    full_mask = (1 << n) - 1
    min_cost = INF
    last_city = -1

    for i in range(n):
        cost = dp[full_mask][i] + dist[i][0]
        if cost < min_cost:
            min_cost = cost
            last_city = i

    # ê²½ë¡œ ë³µì›
    path = []
    mask = full_mask
    city = last_city

    while city != -1:
        path.append(city)
        prev_city = parent[mask][city]
        mask ^= (1 << city)
        city = prev_city

    path.reverse()
    path.append(0)  # ì‹œì‘ì ìœ¼ë¡œ ë³µê·€

    return min_cost, path

def can_partition_k_subsets(nums: List[int], k: int) -> bool:
    """
    ë°°ì—´ì„ í•©ì´ ê°™ì€ kê°œì˜ ë¶€ë¶„ì§‘í•©ìœ¼ë¡œ ë¶„í•  ê°€ëŠ¥í•œì§€
    ì‹œê°„ë³µì¡ë„: O(n * 2^n)
    """
    total = sum(nums)
    if total % k != 0:
        return False

    target = total // k
    n = len(nums)

    # dp[mask] = mask ì§‘í•©ì„ ì‚¬ìš©í–ˆì„ ë•Œ í˜„ì¬ ë²„í‚·ì˜ í•© (targetìœ¼ë¡œ ë‚˜ëˆˆ ë‚˜ë¨¸ì§€)
    dp = [-1] * (1 << n)
    dp[0] = 0

    for mask in range(1 << n):
        if dp[mask] == -1:
            continue

        for i in range(n):
            if mask & (1 << i):
                continue

            if dp[mask] + nums[i] <= target:
                new_mask = mask | (1 << i)
                dp[new_mask] = (dp[mask] + nums[i]) % target

    return dp[(1 << n) - 1] == 0

def min_cost_assignment(cost: List[List[int]]) -> int:
    """
    nëª…ì˜ ì‚¬ëŒì—ê²Œ nê°œì˜ ì‘ì—…ì„ 1:1 í• ë‹¹í•˜ëŠ” ìµœì†Œ ë¹„ìš©
    cost[i][j] = ì‚¬ëŒ iê°€ ì‘ì—… jë¥¼ ìˆ˜í–‰í•˜ëŠ” ë¹„ìš©

    ì‹œê°„ë³µì¡ë„: O(n * 2^n)
    """
    n = len(cost)

    @lru_cache(maxsize=None)
    def dp(mask: int) -> int:
        person = bin(mask).count('1')

        if person == n:
            return 0

        min_cost = float('inf')
        for job in range(n):
            if mask & (1 << job):
                continue

            min_cost = min(min_cost, cost[person][job] + dp(mask | (1 << job)))

        return min_cost

    return dp(0)

def hamiltonian_path_count(adj: List[List[int]]) -> int:
    """
    í•´ë°€í„´ ê²½ë¡œì˜ ê°œìˆ˜ (ëª¨ë“  ì •ì ì„ í•œ ë²ˆì”© ë°©ë¬¸í•˜ëŠ” ê²½ë¡œ)
    adj: ì¸ì ‘ í–‰ë ¬ (adj[i][j] = 1ì´ë©´ iâ†’j ê°„ì„  ì¡´ì¬)

    ì‹œê°„ë³µì¡ë„: O(nÂ² * 2^n)
    """
    n = len(adj)

    # dp[mask][i] = mask ì •ì ë“¤ì„ ë°©ë¬¸í•˜ê³  iì—ì„œ ëë‚˜ëŠ” ê²½ë¡œ ìˆ˜
    dp = [[0] * n for _ in range(1 << n)]

    # ì´ˆê¸°í™”: ê° ì •ì ì—ì„œ ì‹œì‘
    for i in range(n):
        dp[1 << i][i] = 1

    for mask in range(1 << n):
        for last in range(n):
            if dp[mask][last] == 0:
                continue
            if not (mask & (1 << last)):
                continue

            for next_v in range(n):
                if mask & (1 << next_v):
                    continue
                if not adj[last][next_v]:
                    continue

                new_mask = mask | (1 << next_v)
                dp[new_mask][next_v] += dp[mask][last]

    # ëª¨ë“  ì •ì  ë°©ë¬¸í•œ ê²½ë¡œ í•©
    full_mask = (1 << n) - 1
    return sum(dp[full_mask])

def sos_dp(arr: List[int]) -> List[int]:
    """
    Sum over Subsets DP
    ê° ë§ˆìŠ¤í¬ì— ëŒ€í•´ ë¶€ë¶„ì§‘í•©ë“¤ì˜ ê°’ í•© ê³„ì‚°

    result[mask] = sum(arr[subset]) for all subset of mask

    ì‹œê°„ë³µì¡ë„: O(n * 2^n)
    """
    n = len(arr).bit_length()
    dp = arr.copy()

    # 0~(len(arr)-1)ê¹Œì§€ í™•ì¥
    while len(dp) < (1 << n):
        dp.append(0)

    for i in range(n):
        for mask in range(1 << n):
            if mask & (1 << i):
                dp[mask] += dp[mask ^ (1 << i)]

    return dp

def max_independent_set(adj: List[List[int]]) -> int:
    """
    ê·¸ë˜í”„ì—ì„œ ìµœëŒ€ ë…ë¦½ ì§‘í•© í¬ê¸° (ì„œë¡œ ì¸ì ‘í•˜ì§€ ì•Šì€ ì •ì  ì§‘í•©)
    ì‘ì€ ê·¸ë˜í”„ì—ì„œ ë¹„íŠ¸ë§ˆìŠ¤í¬ë¡œ brute force

    ì‹œê°„ë³µì¡ë„: O(2^n * nÂ²)
    """
    n = len(adj)
    max_size = 0

    for mask in range(1 << n):
        # maskê°€ ë…ë¦½ ì§‘í•©ì¸ì§€ í™•ì¸
        valid = True
        for i in range(n):
            if not (mask & (1 << i)):
                continue
            for j in range(i + 1, n):
                if not (mask & (1 << j)):
                    continue
                if adj[i][j]:
                    valid = False
                    break
            if not valid:
                break

        if valid:
            max_size = max(max_size, bin(mask).count('1'))

    return max_size

def domino_tiling(m: int, n: int) -> int:
    """
    mÃ—n ê²©ìë¥¼ 1Ã—2 ë„ë¯¸ë…¸ë¡œ ì±„ìš°ëŠ” ê²½ìš°ì˜ ìˆ˜
    ë¹„íŠ¸ë§ˆìŠ¤í¬ DP (profile ë°©ì‹)

    ì‹œê°„ë³µì¡ë„: O(n * 2^m * 2^m)
    """
    if m > n:
        m, n = n, m

    # dp[col][profile] = í˜„ì¬ ì—´ê¹Œì§€ ì±„ìš°ê³  í”„ë¡œíŒŒì¼ì´ profileì¸ ê²½ìš°ì˜ ìˆ˜
    dp = {0: 1}

    for col in range(n):
        for row in range(m):
            new_dp = {}

            for profile, count in dp.items():
                # í˜„ì¬ ì…€ì´ ì´ë¯¸ ì±„ì›Œì§„ ê²½ìš°
                if profile & (1 << row):
                    new_profile = profile ^ (1 << row)
                    new_dp[new_profile] = new_dp.get(new_profile, 0) + count
                else:
                    # ìˆ˜í‰ ë„ë¯¸ë…¸ (ë‹¤ìŒ ì—´ë¡œ í™•ì¥)
                    new_profile = profile | (1 << row)
                    new_dp[new_profile] = new_dp.get(new_profile, 0) + count

                    # ìˆ˜ì§ ë„ë¯¸ë…¸ (ì•„ë˜ ì…€ê³¼ í•¨ê»˜)
                    if row + 1 < m and not (profile & (1 << (row + 1))):
                        new_dp[profile] = new_dp.get(profile, 0) + count

            dp = new_dp

    return dp.get(0, 0)
