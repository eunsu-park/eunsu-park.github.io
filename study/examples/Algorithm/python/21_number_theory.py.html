def gcd(a: int, b: int) -> int:
    """
    ìµœëŒ€ê³µì•½ìˆ˜ (ìœ í´ë¦¬ë“œ í˜¸ì œë²•)
    ì‹œê°„ë³µì¡ë„: O(log(min(a, b)))

    while b:

    return a


def lcm(a: int, b: int) -> int:
    """ìµœì†Œê³µë°°ìˆ˜"""
    return a * b // gcd(a, b)


def gcd_multiple(numbers: List[int]) -> int:
    """ì—¬ëŸ¬ ìˆ˜ì˜ GCD"""
    return reduce(gcd, numbers)


def lcm_multiple(numbers: List[int]) -> int:
    """ì—¬ëŸ¬ ìˆ˜ì˜ LCM"""
    return reduce(lcm, numbers)


def extended_gcd(a: int, b: int) -> Tuple[int, int, int]:
    """
    í™•ì¥ ìœ í´ë¦¬ë“œ ì•Œê³ ë¦¬ì¦˜
    gcd(a, b) = a*x + b*yë¥¼ ë§Œì¡±í•˜ëŠ” (gcd, x, y) ë°˜í™˜
    """
    if b == 0:
        return a, 1, 0

    g, x1, y1 = extended_gcd(b, a % b)
    x = y1
    y = x1 - (a // b) * y1

    return g, x, y

def is_prime(n: int) -> bool:
    """
    ì†Œìˆ˜ íŒë³„
    ì‹œê°„ë³µì¡ë„: O(âˆšn)
    """
    if n < 2:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False

    for i in range(3, int(n ** 0.5) + 1, 2):
        if n % i == 0:
            return False
    return True


def sieve_of_eratosthenes(n: int) -> List[int]:
    """
    ì—ë¼í† ìŠ¤í…Œë„¤ìŠ¤ì˜ ì²´
    n ì´í•˜ì˜ ëª¨ë“  ì†Œìˆ˜ ë°˜í™˜
    ì‹œê°„ë³µì¡ë„: O(n log log n)
    """
    if n < 2:
        return []

    is_prime_arr = [True] * (n + 1)
    is_prime_arr[0] = is_prime_arr[1] = False

    for i in range(2, int(n ** 0.5) + 1):
        if is_prime_arr[i]:
            for j in range(i * i, n + 1, i):
                is_prime_arr[j] = False

    return [i for i in range(n + 1) if is_prime_arr[i]]


def prime_factorization(n: int) -> List[Tuple[int, int]]:
    """
    ì†Œì¸ìˆ˜ë¶„í•´
    ë°˜í™˜: [(ì†Œìˆ˜, ì§€ìˆ˜), ...]
    ì‹œê°„ë³µì¡ë„: O(âˆšn)
    """
    factors = []
    d = 2

    while d * d <= n:
        if n % d == 0:
            count = 0
            while n % d == 0:
                n //= d
                count += 1
            factors.append((d, count))
        d += 1

    if n > 1:
        factors.append((n, 1))

    return factors

def mod_pow(base: int, exp: int, mod: int) -> int:
    """
    ë¹ ë¥¸ ê±°ë“­ì œê³± (ëª¨ë“ˆëŸ¬)
    ì‹œê°„ë³µì¡ë„: O(log exp)
    """
    result = 1
    base %= mod

    while exp > 0:
        if exp % 2 == 1:
            result = (result * base) % mod
        exp //= 2
        base = (base * base) % mod

    return result


def mod_inverse(a: int, mod: int) -> int:
    """
    ëª¨ë“ˆëŸ¬ ì—­ì› (modê°€ ì†Œìˆ˜ì¼ ë•Œ)
    a^(-1) mod p = a^(p-2) mod p (í˜ë¥´ë§ˆ ì†Œì •ë¦¬)
    """
    return mod_pow(a, mod - 2, mod)


def mod_inverse_extended(a: int, mod: int) -> int:
    """
    ëª¨ë“ˆëŸ¬ ì—­ì› (í™•ì¥ ìœ í´ë¦¬ë“œ)
    gcd(a, mod) = 1ì¼ ë•Œë§Œ ì¡´ì¬
    """
    g, x, _ = extended_gcd(a, mod)
    if g != 1:
        return -1  # ì—­ì› ì—†ìŒ
    return x % mod

def factorial_mod(n: int, mod: int) -> int:
    """n! mod p"""
    result = 1
    for i in range(2, n + 1):
        result = (result * i) % mod
    return result


class Combination:
    """
    ì¡°í•© ê³„ì‚° (ëª¨ë“ˆëŸ¬)
    ì „ì²˜ë¦¬: O(n)
    ì¿¼ë¦¬: O(1)
    """

    def __init__(self, n: int, mod: int):
        self.mod = mod
        self.fact = [1] * (n + 1)
        self.inv_fact = [1] * (n + 1)

        # íŒ©í† ë¦¬ì–¼ ê³„ì‚°
        for i in range(1, n + 1):
            self.fact[i] = self.fact[i - 1] * i % mod

        # ì—­ íŒ©í† ë¦¬ì–¼ ê³„ì‚°
        self.inv_fact[n] = mod_pow(self.fact[n], mod - 2, mod)
        for i in range(n - 1, -1, -1):
            self.inv_fact[i] = self.inv_fact[i + 1] * (i + 1) % mod

    def nCr(self, n: int, r: int) -> int:
        """nCr mod p"""
        if r < 0 or r > n:
            return 0
        return self.fact[n] * self.inv_fact[r] % self.mod * self.inv_fact[n - r] % self.mod

    def nPr(self, n: int, r: int) -> int:
        """nPr mod p"""
        if r < 0 or r > n:
            return 0
        return self.fact[n] * self.inv_fact[n - r] % self.mod

def euler_phi(n: int) -> int:
    """
    Ï†(n) = në³´ë‹¤ ì‘ê±°ë‚˜ ê°™ê³  nê³¼ ì„œë¡œì†Œì¸ ì–‘ì˜ ì •ìˆ˜ì˜ ê°œìˆ˜
    ì‹œê°„ë³µì¡ë„: O(âˆšn)
    """
    result = n

    d = 2
    while d * d <= n:
        if n % d == 0:
            while n % d == 0:
                n //= d
            result -= result // d
        d += 1

    if n > 1:
        result -= result // n

    return result


def euler_phi_sieve(n: int) -> List[int]:
    """1~nê¹Œì§€ì˜ ì˜¤ì¼ëŸ¬ í”¼ í•¨ìˆ˜ ê°’ (ì²´)"""
    phi = list(range(n + 1))

    for i in range(2, n + 1):
        if phi[i] == i:  # iê°€ ì†Œìˆ˜
            for j in range(i, n + 1, i):
                phi[j] -= phi[j] // i

    return phi

def chinese_remainder_theorem(remainders: List[int], moduli: List[int]) -> int:
    """
    x â‰¡ r_i (mod m_i)ë¥¼ ë§Œì¡±í•˜ëŠ” ìµœì†Œ ì–‘ì˜ ì •ìˆ˜ x
    ëª¨ë“  m_iëŠ” ì„œë¡œì†Œì—¬ì•¼ í•¨
    """
    M = 1
    for m in moduli:
        M *= m

    result = 0
    for r, m in zip(remainders, moduli):
        Mi = M // m
        yi = mod_inverse_extended(Mi, m)
        result += r * Mi * yi

    return result % M

def lucas(n: int, r: int, p: int) -> int:
    """
    ë£¨ì¹´ìŠ¤ ì •ë¦¬ë¡œ nCr mod p ê³„ì‚°
    pê°€ ì†Œìˆ˜ì¼ ë•Œ, n, rì´ ë§¤ìš° í´ ë•Œ ìœ ìš©

    if r == 0:
        return 1
    return lucas(n // p, r // p, p) * nCr_small(n % p, r % p, p) % p


def nCr_small(n: int, r: int, p: int) -> int:
    """ì‘ì€ nCr mod p"""
    if r > n:
        return 0
    if r == 0 or r == n:
        return 1

    num = den = 1
    for i in range(r):
        num = num * (n - i) % p
        den = den * (i + 1) % p

    return num * mod_pow(den, p - 2, p) % p

def divisors(n: int) -> List[int]:
    """
    nì˜ ëª¨ë“  ì•½ìˆ˜
    ì‹œê°„ë³µì¡ë„: O(âˆšn)
    """
    result = []

    for i in range(1, int(n ** 0.5) + 1):
        if n % i == 0:
            result.append(i)
            if i != n // i:
                result.append(n // i)

    return sorted(result)


def divisor_count(n: int) -> int:
    """ì•½ìˆ˜ì˜ ê°œìˆ˜"""
    factors = prime_factorization(n)
    count = 1
    for _, exp in factors:
        count *= (exp + 1)
    return count


def divisor_sum(n: int) -> int:
    """ì•½ìˆ˜ì˜ í•©"""
    factors = prime_factorization(n)
    total = 1
    for p, e in factors:
        total *= (pow(p, e + 1) - 1) // (p - 1)
    return total

def solve_linear_diophantine(a: int, b: int, c: int) -> Tuple[bool, int, int]:
    """
    ax + by = cì˜ ì •ìˆ˜í•´ (x, y) ì°¾ê¸°
    ë°˜í™˜: (í•´ ì¡´ì¬ ì—¬ë¶€, x, y)
    """
    g, x0, y0 = extended_gcd(abs(a), abs(b))

    if c % g != 0:
        return False, 0, 0

    x0 *= c // g
    y0 *= c // g

    if a < 0:
        x0 = -x0
    if b < 0:
        y0 = -y0

    return True, x0, y0
