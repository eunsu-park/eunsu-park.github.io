"""
ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨ (Segment Tree)
Segment Tree for Range Queries

Íµ¨Í∞Ñ ÏøºÎ¶¨ÏôÄ Ï†ê ÏóÖÎç∞Ïù¥Ìä∏Î•º Ìö®Ïú®Ï†ÅÏúºÎ°ú Ï≤òÎ¶¨ÌïòÎäî ÏûêÎ£åÍµ¨Ï°∞ÏûÖÎãàÎã§.
"""

from typing import List, Callable, Optional


# =============================================================================
# 1. Í∏∞Î≥∏ ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨ (Íµ¨Í∞Ñ Ìï©)
# =============================================================================

class SegmentTree:
    """
    ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨ (Íµ¨Í∞Ñ Ìï©)
    - Ï†ê ÏóÖÎç∞Ïù¥Ìä∏: O(log n)
    - Íµ¨Í∞Ñ ÏøºÎ¶¨: O(log n)
    - Í≥µÍ∞Ñ: O(n)
    """

    def __init__(self, arr: List[int]):
        self.n = len(arr)
        self.tree = [0] * (4 * self.n)
        if self.n > 0:
            self._build(arr, 1, 0, self.n - 1)

    def _build(self, arr: List[int], node: int, start: int, end: int):
        """Ìä∏Î¶¨ Íµ¨ÏÑ± - O(n)"""
        if start == end:
            self.tree[node] = arr[start]
        else:
            mid = (start + end) // 2
            self._build(arr, 2 * node, start, mid)
            self._build(arr, 2 * node + 1, mid + 1, end)
            self.tree[node] = self.tree[2 * node] + self.tree[2 * node + 1]

    def update(self, idx: int, val: int):
        """Ï†ê ÏóÖÎç∞Ïù¥Ìä∏ - O(log n)"""
        self._update(1, 0, self.n - 1, idx, val)

    def _update(self, node: int, start: int, end: int, idx: int, val: int):
        if start == end:
            self.tree[node] = val
        else:
            mid = (start + end) // 2
            if idx <= mid:
                self._update(2 * node, start, mid, idx, val)
            else:
                self._update(2 * node + 1, mid + 1, end, idx, val)
            self.tree[node] = self.tree[2 * node] + self.tree[2 * node + 1]

    def query(self, left: int, right: int) -> int:
        """Íµ¨Í∞Ñ Ìï© ÏøºÎ¶¨ - O(log n)"""
        return self._query(1, 0, self.n - 1, left, right)

    def _query(self, node: int, start: int, end: int, left: int, right: int) -> int:
        if right < start or end < left:
            return 0  # Î≤îÏúÑ Î≤óÏñ¥ÎÇ®
        if left <= start and end <= right:
            return self.tree[node]  # ÏôÑÏ†Ñ Ìè¨Ìï®

        mid = (start + end) // 2
        left_sum = self._query(2 * node, start, mid, left, right)
        right_sum = self._query(2 * node + 1, mid + 1, end, left, right)
        return left_sum + right_sum

# 2. ÏùºÎ∞ò ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨ (ÏûÑÏùò Ïó∞ÏÇ∞)
# =============================================================================

class GenericSegmentTree:
    """
    ÏùºÎ∞ò ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨ (ÏûÑÏùòÏùò Í≤∞Ìï© Ïó∞ÏÇ∞)
    - Í≤∞Ìï© Î≤ïÏπôÏùÑ ÎßåÏ°±ÌïòÎäî Ïó∞ÏÇ∞Ïù¥Î©¥ ÏÇ¨Ïö© Í∞ÄÎä•
    """

    def __init__(self, arr: List[int], func: Callable[[int, int], int], identity: int):
        """
        func: Í≤∞Ìï© Ïó∞ÏÇ∞ (Ïòà: min, max, gcd, +, *)
        identity: Ìï≠Îì±Ïõê (Ïòà: inf for min, 0 for +, 1 for *)
        """
        self.n = len(arr)
        self.func = func
        self.identity = identity
        self.tree = [identity] * (4 * self.n)
        if self.n > 0:
            self._build(arr, 1, 0, self.n - 1)

    def _build(self, arr: List[int], node: int, start: int, end: int):
        if start == end:
            self.tree[node] = arr[start]
        else:
            mid = (start + end) // 2
            self._build(arr, 2 * node, start, mid)
            self._build(arr, 2 * node + 1, mid + 1, end)
            self.tree[node] = self.func(self.tree[2 * node], self.tree[2 * node + 1])

    def update(self, idx: int, val: int):
        self._update(1, 0, self.n - 1, idx, val)

    def _update(self, node: int, start: int, end: int, idx: int, val: int):
        if start == end:
            self.tree[node] = val
        else:
            mid = (start + end) // 2
            if idx <= mid:
                self._update(2 * node, start, mid, idx, val)
            else:
                self._update(2 * node + 1, mid + 1, end, idx, val)
            self.tree[node] = self.func(self.tree[2 * node], self.tree[2 * node + 1])

    def query(self, left: int, right: int) -> int:
        return self._query(1, 0, self.n - 1, left, right)

    def _query(self, node: int, start: int, end: int, left: int, right: int) -> int:
        if right < start or end < left:
            return self.identity
        if left <= start and end <= right:
            return self.tree[node]

        mid = (start + end) // 2
        left_val = self._query(2 * node, start, mid, left, right)
        right_val = self._query(2 * node + 1, mid + 1, end, left, right)
        return self.func(left_val, right_val)

# 3. Lazy Propagation (Íµ¨Í∞Ñ ÏóÖÎç∞Ïù¥Ìä∏)
# =============================================================================

class LazySegmentTree:
    """
    Lazy Propagation ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨
    - Íµ¨Í∞Ñ ÏóÖÎç∞Ïù¥Ìä∏: O(log n)
    - Íµ¨Í∞Ñ ÏøºÎ¶¨: O(log n)
    """

    def __init__(self, arr: List[int]):
        self.n = len(arr)
        self.tree = [0] * (4 * self.n)
        self.lazy = [0] * (4 * self.n)
        if self.n > 0:
            self._build(arr, 1, 0, self.n - 1)

    def _build(self, arr: List[int], node: int, start: int, end: int):
        if start == end:
            self.tree[node] = arr[start]
        else:
            mid = (start + end) // 2
            self._build(arr, 2 * node, start, mid)
            self._build(arr, 2 * node + 1, mid + 1, end)
            self.tree[node] = self.tree[2 * node] + self.tree[2 * node + 1]

    def _push_down(self, node: int, start: int, end: int):
        """lazy Í∞í Ï†ÑÌåå"""
        if self.lazy[node] != 0:
            mid = (start + end) // 2

            # ÏôºÏ™Ω ÏûêÏãù
            self.tree[2 * node] += self.lazy[node] * (mid - start + 1)
            self.lazy[2 * node] += self.lazy[node]

            # Ïò§Î•∏Ï™Ω ÏûêÏãù
            self.tree[2 * node + 1] += self.lazy[node] * (end - mid)
            self.lazy[2 * node + 1] += self.lazy[node]

            self.lazy[node] = 0

    def update_range(self, left: int, right: int, val: int):
        """Íµ¨Í∞Ñ [left, right]Ïóê val ÎçîÌïòÍ∏∞"""
        self._update_range(1, 0, self.n - 1, left, right, val)

    def _update_range(self, node: int, start: int, end: int, left: int, right: int, val: int):
        if right < start or end < left:
            return

        if left <= start and end <= right:
            self.tree[node] += val * (end - start + 1)
            self.lazy[node] += val
            return

        self._push_down(node, start, end)

        mid = (start + end) // 2
        self._update_range(2 * node, start, mid, left, right, val)
        self._update_range(2 * node + 1, mid + 1, end, left, right, val)
        self.tree[node] = self.tree[2 * node] + self.tree[2 * node + 1]

    def query(self, left: int, right: int) -> int:
        """Íµ¨Í∞Ñ Ìï© ÏøºÎ¶¨"""
        return self._query(1, 0, self.n - 1, left, right)

    def _query(self, node: int, start: int, end: int, left: int, right: int) -> int:
        if right < start or end < left:
            return 0

        if left <= start and end <= right:
            return self.tree[node]

        self._push_down(node, start, end)

        mid = (start + end) // 2
        left_sum = self._query(2 * node, start, mid, left, right)
        right_sum = self._query(2 * node + 1, mid + 1, end, left, right)
        return left_sum + right_sum

# 4. Î∞òÎ≥µ ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨ (Iterative)
# =============================================================================

class IterativeSegmentTree:
    """
    Î∞òÎ≥µ ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨ (ÎπÑÏû¨Í∑Ä)
    Î©îÎ™®Î¶¨ Ìö®Ïú®Ï†Å, Ï∫êÏãú ÏπúÌôîÏ†Å
    """

    def __init__(self, arr: List[int]):
        self.n = len(arr)
        self.tree = [0] * (2 * self.n)

        # Î¶¨ÌîÑ ÎÖ∏Îìú Ï±ÑÏö∞Í∏∞
        for i in range(self.n):
            self.tree[self.n + i] = arr[i]

        # ÎÇ¥Î∂Ä ÎÖ∏Îìú Íµ¨ÏÑ±
        for i in range(self.n - 1, 0, -1):
            self.tree[i] = self.tree[2 * i] + self.tree[2 * i + 1]

    def update(self, idx: int, val: int):
        """Ï†ê ÏóÖÎç∞Ïù¥Ìä∏"""
        idx += self.n
        self.tree[idx] = val

        while idx > 1:
            idx //= 2
            self.tree[idx] = self.tree[2 * idx] + self.tree[2 * idx + 1]

    def query(self, left: int, right: int) -> int:
        """Íµ¨Í∞Ñ [left, right] Ìï©"""
        left += self.n
        right += self.n + 1
        result = 0

        while left < right:
            if left % 2 == 1:
                result += self.tree[left]
                left += 1
            if right % 2 == 1:
                right -= 1
                result += self.tree[right]
            left //= 2
            right //= 2

        return result

# 5. 2D ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨
# =============================================================================

class SegmentTree2D:
    """
    2D ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨ (Íµ¨Í∞Ñ Ìï©)
    - ÏøºÎ¶¨/ÏóÖÎç∞Ïù¥Ìä∏: O(log n * log m)
    """

    def __init__(self, matrix: List[List[int]]):
        if not matrix or not matrix[0]:
            self.n = self.m = 0
            return

        self.n = len(matrix)
        self.m = len(matrix[0])
        self.tree = [[0] * (4 * self.m) for _ in range(4 * self.n)]
        self._build_x(matrix, 1, 0, self.n - 1)

    def _build_x(self, matrix: List[List[int]], node_x: int, lx: int, rx: int):
        if lx == rx:
            self._build_y(matrix, node_x, lx, rx, 1, 0, self.m - 1, lx)
        else:
            mid = (lx + rx) // 2
            self._build_x(matrix, 2 * node_x, lx, mid)
            self._build_x(matrix, 2 * node_x + 1, mid + 1, rx)
            self._merge_y(node_x, 1, 0, self.m - 1)

    def _build_y(self, matrix, node_x, lx, rx, node_y, ly, ry, row):
        if ly == ry:
            self.tree[node_x][node_y] = matrix[row][ly]
        else:
            mid = (ly + ry) // 2
            self._build_y(matrix, node_x, lx, rx, 2 * node_y, ly, mid, row)
            self._build_y(matrix, node_x, lx, rx, 2 * node_y + 1, mid + 1, ry, row)
            self.tree[node_x][node_y] = self.tree[node_x][2 * node_y] + self.tree[node_x][2 * node_y + 1]

    def _merge_y(self, node_x, node_y, ly, ry):
        if ly == ry:
            self.tree[node_x][node_y] = self.tree[2 * node_x][node_y] + self.tree[2 * node_x + 1][node_y]
        else:
            mid = (ly + ry) // 2
            self._merge_y(node_x, 2 * node_y, ly, mid)
            self._merge_y(node_x, 2 * node_y + 1, mid + 1, ry)
            self.tree[node_x][node_y] = self.tree[node_x][2 * node_y] + self.tree[node_x][2 * node_y + 1]

    def query(self, x1: int, y1: int, x2: int, y2: int) -> int:
        """(x1,y1) ~ (x2,y2) ÏßÅÏÇ¨Í∞ÅÌòï Íµ¨Í∞Ñ Ìï©"""
        return self._query_x(1, 0, self.n - 1, x1, x2, y1, y2)

    def _query_x(self, node_x, lx, rx, x1, x2, y1, y2):
        if x2 < lx or rx < x1:
            return 0
        if x1 <= lx and rx <= x2:
            return self._query_y(node_x, 1, 0, self.m - 1, y1, y2)

        mid = (lx + rx) // 2
        left = self._query_x(2 * node_x, lx, mid, x1, x2, y1, y2)
        right = self._query_x(2 * node_x + 1, mid + 1, rx, x1, x2, y1, y2)
        return left + right

    def _query_y(self, node_x, node_y, ly, ry, y1, y2):
        if y2 < ly or ry < y1:
            return 0
        if y1 <= ly and ry <= y2:
            return self.tree[node_x][node_y]

        mid = (ly + ry) // 2
        left = self._query_y(node_x, 2 * node_y, ly, mid, y1, y2)
        right = self._query_y(node_x, 2 * node_y + 1, mid + 1, ry, y1, y2)
        return left + right

# 6. ÏùëÏö©: Ïó≠Ïàú Ïåç Í∞úÏàò (Inversion Count)
# =============================================================================

def count_inversions_segtree(arr: List[int]) -> int:
    """
    Ïó≠Ïàú Ïåç Í∞úÏàò (ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨ ÌôúÏö©)
    ÏãúÍ∞ÑÎ≥µÏû°ÎèÑ: O(n log n)
    """
    if not arr:
        return 0

    # Ï¢åÌëú ÏïïÏ∂ï
    sorted_arr = sorted(set(arr))
    rank = {v: i for i, v in enumerate(sorted_arr)}
    n = len(sorted_arr)

    # ÏÑ∏Í∑∏Î®ºÌä∏ Ìä∏Î¶¨ (ÎπàÎèÑ Ï†ÄÏû•)
    tree = [0] * (4 * n)

    def update(node, start, end, idx):
        if start == end:
            tree[node] += 1
        else:
            mid = (start + end) // 2
            if idx <= mid:
                update(2 * node, start, mid, idx)
            else:
                update(2 * node + 1, mid + 1, end, idx)
            tree[node] = tree[2 * node] + tree[2 * node + 1]

    def query(node, start, end, left, right):
        if right < start or end < left:
            return 0
        if left <= start and end <= right:
            return tree[node]
        mid = (start + end) // 2
        return query(2 * node, start, mid, left, right) + \
               query(2 * node + 1, mid + 1, end, left, right)

    inversions = 0
    for val in arr:
        r = rank[val]
        # rÎ≥¥Îã§ ÌÅ∞ Í∞íÏùò Í∞úÏàò (Ïù¥ÎØ∏ ÏÇΩÏûÖÎêú Í≤É Ï§ë)
        inversions += query(1, 0, n - 1, r + 1, n - 1)
        update(1, 0, n - 1, r)

    return inversions
