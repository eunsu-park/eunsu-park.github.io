class FenwickTree:
    """
    íœìœ… íŠ¸ë¦¬ (Binary Indexed Tree)
    - ì  ì—…ë°ì´íŠ¸: O(log n)
    - ì ‘ë‘ì‚¬ í•©: O(log n)
    - êµ¬ê°„ í•©: O(log n)
    - ê³µê°„: O(n)
    """

    def __init__(self, n: int):
        """í¬ê¸° nì˜ ë¹ˆ íœìœ… íŠ¸ë¦¬ ìƒì„±"""
        self.n = n
        self.tree = [0] * (n + 1)  # 1-indexed

    @classmethod
    def from_array(cls, arr: List[int]) -> 'FenwickTree':
        """ë°°ì—´ë¡œë¶€í„° íœìœ… íŠ¸ë¦¬ ìƒì„± - O(n)"""
        ft = cls(len(arr))

        # íš¨ìœ¨ì ì¸ êµ¬ì„± (O(n))
        for i, val in enumerate(arr):
            ft.tree[i + 1] += val
            parent = i + 1 + (ft._lowbit(i + 1))
            if parent <= ft.n:
                ft.tree[parent] += ft.tree[i + 1]

        return ft

    def _lowbit(self, x: int) -> int:
        """ìµœí•˜ìœ„ ë¹„íŠ¸ (x & -x)"""
        return x & (-x)

    def update(self, idx: int, delta: int):
        """idx ìœ„ì¹˜ì— delta ë”í•˜ê¸° (0-indexed) - O(log n)"""
        idx += 1  # 1-indexedë¡œ ë³€í™˜

        while idx <= self.n:
            self.tree[idx] += delta
            idx += self._lowbit(idx)

    def prefix_sum(self, idx: int) -> int:
        """[0, idx] êµ¬ê°„ í•© (0-indexed) - O(log n)"""
        idx += 1
        result = 0

        while idx > 0:
            result += self.tree[idx]
            idx -= self._lowbit(idx)

        return result

    def range_sum(self, left: int, right: int) -> int:
        """[left, right] êµ¬ê°„ í•© (0-indexed) - O(log n)"""
        if left == 0:
            return self.prefix_sum(right)
        return self.prefix_sum(right) - self.prefix_sum(left - 1)

    def get(self, idx: int) -> int:
        """idx ìœ„ì¹˜ì˜ ê°’ (0-indexed)"""
        return self.range_sum(idx, idx)

    def set(self, idx: int, val: int):
        """idx ìœ„ì¹˜ì˜ ê°’ì„ valë¡œ ì„¤ì •"""
        current = self.get(idx)
        self.update(idx, val - current)

class FenwickTreeRangeUpdate:
    """
    êµ¬ê°„ ì—…ë°ì´íŠ¸ + ì  ì¿¼ë¦¬
    ì°¨ë¶„ ë°°ì—´ ê¸°ë²• í™œìš©
    """

    def __init__(self, n: int):
        self.n = n
        self.tree = [0] * (n + 1)

    def _lowbit(self, x: int) -> int:
        return x & (-x)

    def _update(self, idx: int, delta: int):
        idx += 1
        while idx <= self.n:
            self.tree[idx] += delta
            idx += self._lowbit(idx)

    def update_range(self, left: int, right: int, delta: int):
        """[left, right] êµ¬ê°„ì— delta ë”í•˜ê¸°"""
        self._update(left, delta)
        if right + 1 < self.n:
            self._update(right + 1, -delta)

    def query(self, idx: int) -> int:
        """idx ìœ„ì¹˜ì˜ ê°’"""
        idx += 1
        result = 0
        while idx > 0:
            result += self.tree[idx]
            idx -= self._lowbit(idx)
        return result

class FenwickTreeRangeUpdateRangeQuery:
    """
    êµ¬ê°„ ì—…ë°ì´íŠ¸ + êµ¬ê°„ ì¿¼ë¦¬
    ë‘ ê°œì˜ BIT ì‚¬ìš©
    """

    def __init__(self, n: int):
        self.n = n
        self.tree1 = [0] * (n + 1)  # B[i]
        self.tree2 = [0] * (n + 1)  # B[i] * i

    def _lowbit(self, x: int) -> int:
        return x & (-x)

    def _update(self, tree: List[int], idx: int, delta: int):
        while idx <= self.n:
            tree[idx] += delta
            idx += self._lowbit(idx)

    def _prefix_sum(self, tree: List[int], idx: int) -> int:
        result = 0
        while idx > 0:
            result += tree[idx]
            idx -= self._lowbit(idx)
        return result

    def update_range(self, left: int, right: int, delta: int):
        """[left, right] êµ¬ê°„ì— delta ë”í•˜ê¸° (1-indexed)"""
        self._update(self.tree1, left, delta)
        self._update(self.tree1, right + 1, -delta)
        self._update(self.tree2, left, delta * (left - 1))
        self._update(self.tree2, right + 1, -delta * right)

    def prefix_sum(self, idx: int) -> int:
        """[1, idx] êµ¬ê°„ í•© (1-indexed)"""
        return self._prefix_sum(self.tree1, idx) * idx - self._prefix_sum(self.tree2, idx)

    def range_sum(self, left: int, right: int) -> int:
        """[left, right] êµ¬ê°„ í•© (1-indexed)"""
        return self.prefix_sum(right) - self.prefix_sum(left - 1)

class FenwickTree2D:
    """
    2D íœìœ… íŠ¸ë¦¬
    - ì  ì—…ë°ì´íŠ¸: O(log n * log m)
    - ì§ì‚¬ê°í˜• í•©: O(log n * log m)
    """

    def __init__(self, n: int, m: int):
        self.n = n
        self.m = m
        self.tree = [[0] * (m + 1) for _ in range(n + 1)]

    def _lowbit(self, x: int) -> int:
        return x & (-x)

    def update(self, x: int, y: int, delta: int):
        """(x, y)ì— delta ë”í•˜ê¸° (0-indexed)"""
        x += 1
        while x <= self.n:
            y_idx = y + 1
            while y_idx <= self.m:
                self.tree[x][y_idx] += delta
                y_idx += self._lowbit(y_idx)
            x += self._lowbit(x)

    def prefix_sum(self, x: int, y: int) -> int:
        """(0,0) ~ (x,y) í•© (0-indexed)"""
        x += 1
        result = 0
        while x > 0:
            y_idx = y + 1
            while y_idx > 0:
                result += self.tree[x][y_idx]
                y_idx -= self._lowbit(y_idx)
            x -= self._lowbit(x)
        return result

    def range_sum(self, x1: int, y1: int, x2: int, y2: int) -> int:
        """(x1,y1) ~ (x2,y2) ì§ì‚¬ê°í˜• í•© (0-indexed)"""
        result = self.prefix_sum(x2, y2)
        if x1 > 0:
            result -= self.prefix_sum(x1 - 1, y2)
        if y1 > 0:
            result -= self.prefix_sum(x2, y1 - 1)
        if x1 > 0 and y1 > 0:
            result += self.prefix_sum(x1 - 1, y1 - 1)
        return result

def count_inversions(arr: List[int]) -> int:
    """
    ì—­ìˆœ ìŒ ê°œìˆ˜ (íœìœ… íŠ¸ë¦¬ í™œìš©)
    ì‹œê°„ë³µì¡ë„: O(n log n)
    """
    if not arr:
        return 0

    # ì¢Œí‘œ ì••ì¶•
    sorted_arr = sorted(set(arr))
    rank = {v: i for i, v in enumerate(sorted_arr)}
    n = len(sorted_arr)

    ft = FenwickTree(n)
    inversions = 0

    for val in arr:
        r = rank[val]
        # rë³´ë‹¤ í° ì¸ë±ìŠ¤ì˜ ê°œìˆ˜ (ì´ë¯¸ ì‚½ì…ëœ ê²ƒ ì¤‘)
        inversions += ft.prefix_sum(n - 1) - ft.prefix_sum(r)
        ft.update(r, 1)

    return inversions

class FenwickTreeKth:
    """Kë²ˆì§¸ ì›ì†Œ ì°¾ê¸°ë¥¼ ì§€ì›í•˜ëŠ” íœìœ… íŠ¸ë¦¬"""

    def __init__(self, n: int):
        self.n = n
        self.tree = [0] * (n + 1)

    def _lowbit(self, x: int) -> int:
        return x & (-x)

    def update(self, idx: int, delta: int):
        """idxì— delta ë”í•˜ê¸° (1-indexed)"""
        while idx <= self.n:
            self.tree[idx] += delta
            idx += self._lowbit(idx)

    def prefix_sum(self, idx: int) -> int:
        """[1, idx] í•©"""
        result = 0
        while idx > 0:
            result += self.tree[idx]
            idx -= self._lowbit(idx)
        return result

    def find_kth(self, k: int) -> int:
        """
        kë²ˆì§¸ ì›ì†Œì˜ ì¸ë±ìŠ¤ ì°¾ê¸° (1-indexed)
        prefix_sum(idx) >= kì¸ ìµœì†Œ idx
        ì‹œê°„ë³µì¡ë„: O(log n)
        """
        idx = 0
        bit_mask = 1

        while bit_mask <= self.n:
            bit_mask <<= 1
        bit_mask >>= 1

        while bit_mask > 0:
            next_idx = idx + bit_mask
            if next_idx <= self.n and self.tree[next_idx] < k:
                idx = next_idx
                k -= self.tree[idx]
            bit_mask >>= 1

        return idx + 1

# 7. ì‘ìš©: êµ¬ê°„ì—ì„œ Kë³´ë‹¤ ì‘ì€ ì›ì†Œ ê°œìˆ˜

def count_smaller_in_range(arr: List[int], queries: List[tuple]) -> List[int]:
    """
    ì¿¼ë¦¬: (left, right, k) - arr[left:right+1]ì—ì„œ kë³´ë‹¤ ì‘ì€ ì›ì†Œ ê°œìˆ˜
    ì˜¤í”„ë¼ì¸ ì¿¼ë¦¬ + íœìœ… íŠ¸ë¦¬
    ì‹œê°„ë³µì¡ë„: O((n + q) log n)
    """
    n = len(arr)
    q = len(queries)

    # ì¢Œí‘œ ì••ì¶•
    all_vals = sorted(set(arr) | set(k for _, _, k in queries))
    val_to_idx = {v: i + 1 for i, v in enumerate(all_vals)}
    m = len(all_vals)

    # (ê°’, ì¸ë±ìŠ¤, íƒ€ì…) ì´ë²¤íŠ¸ ìƒì„±
    events = []
    for i, val in enumerate(arr):
        events.append((val_to_idx[val], i, 'arr', None))

    for qi, (left, right, k) in enumerate(queries):
        k_idx = val_to_idx.get(k, m + 1)
        events.append((k_idx, right, 'query_end', (qi, left, right)))

    events.sort()

    # ê²°ê³¼
    results = [0] * q
    ft = FenwickTree(n)

    for val_idx, pos, event_type, data in events:
        if event_type == 'arr':
            ft.update(pos, 1)
        else:
            qi, left, right = data
            results[qi] = ft.range_sum(left, right)

    return results
