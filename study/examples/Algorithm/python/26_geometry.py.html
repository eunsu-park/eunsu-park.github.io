"""
Í∏∞Ìïò ÏïåÍ≥†Î¶¨Ï¶ò (Computational Geometry)
Computational Geometry Algorithms

Ï†ê, ÏÑ†, Îã§Í∞ÅÌòï Îì±Ïùò Í∏∞ÌïòÌïôÏ†Å Í∞ùÏ≤¥Î•º Îã§Î£®Îäî ÏïåÍ≥†Î¶¨Ï¶òÏûÖÎãàÎã§.
"""

from typing import List, Tuple, Optional
from math import sqrt, atan2, pi, inf
from functools import cmp_to_key


# =============================================================================
# 1. Í∏∞Î≥∏ Í∏∞Ìïò Ïó∞ÏÇ∞
# =============================================================================

class Point:
    """2D Ï†ê"""
    def __init__(self, x: float, y: float):
        self.x = x
        self.y = y

    def __add__(self, other: 'Point') -> 'Point':
        return Point(self.x + other.x, self.y + other.y)

    def __sub__(self, other: 'Point') -> 'Point':
        return Point(self.x - other.x, self.y - other.y)

    def __mul__(self, scalar: float) -> 'Point':
        return Point(self.x * scalar, self.y * scalar)

    def __eq__(self, other: 'Point') -> bool:
        return abs(self.x - other.x) < 1e-9 and abs(self.y - other.y) < 1e-9

    def __repr__(self) -> str:
        return f"({self.x}, {self.y})"

    def dot(self, other: 'Point') -> float:
        """ÎÇ¥Ï†Å"""
        return self.x * other.x + self.y * other.y

    def cross(self, other: 'Point') -> float:
        """Ïô∏Ï†Å (z ÏÑ±Î∂Ñ)"""
        return self.x * other.y - self.y * other.x

    def norm(self) -> float:
        """Î≤°ÌÑ∞ ÌÅ¨Í∏∞"""
        return sqrt(self.x ** 2 + self.y ** 2)

    def normalize(self) -> 'Point':
        """Îã®ÏúÑ Î≤°ÌÑ∞"""
        n = self.norm()
        return Point(self.x / n, self.y / n) if n > 0 else Point(0, 0)


def distance(p1: Point, p2: Point) -> float:
    """Îëê Ï†ê ÏÇ¨Ïù¥ Í±∞Î¶¨"""
    return (p2 - p1).norm()

# 2. CCW (Counter-Clockwise)
# =============================================================================

def ccw(a: Point, b: Point, c: Point) -> int:
    """
    ÏÑ∏ Ï†êÏùò Î∞©Ìñ• ÌåêÎ≥Ñ
    Î∞òÌôò: 1 (Î∞òÏãúÍ≥Ñ), -1 (ÏãúÍ≥Ñ), 0 (ÏùºÏßÅÏÑ†)
    """
    cross = (b - a).cross(c - a)
    if cross > 1e-9:
        return 1   # Î∞òÏãúÍ≥Ñ
    elif cross < -1e-9:
        return -1  # ÏãúÍ≥Ñ
    return 0       # ÏùºÏßÅÏÑ†


def ccw_tuple(a: Tuple[float, float], b: Tuple[float, float], c: Tuple[float, float]) -> int:
    """ÌäúÌîå Î≤ÑÏ†Ñ CCW"""
    cross = (b[0] - a[0]) * (c[1] - a[1]) - (b[1] - a[1]) * (c[0] - a[0])
    if cross > 1e-9:
        return 1
    elif cross < -1e-9:
        return -1
    return 0

# 3. ÏÑ†Î∂Ñ ÍµêÏ∞® ÌåêÏ†ï
# =============================================================================

def segments_intersect(p1: Point, p2: Point, p3: Point, p4: Point) -> bool:
    """
    ÏÑ†Î∂Ñ p1-p2ÏôÄ p3-p4Ïùò ÍµêÏ∞® Ïó¨Î∂Ä
    """
    d1 = ccw(p3, p4, p1)
    d2 = ccw(p3, p4, p2)
    d3 = ccw(p1, p2, p3)
    d4 = ccw(p1, p2, p4)

    # ÏùºÎ∞òÏ†ÅÏù∏ ÍµêÏ∞®
    if d1 * d2 < 0 and d3 * d4 < 0:
        return True

    # Í≤ΩÍ≥Ñ ÏºÄÏù¥Ïä§ (Ï†êÏù¥ ÏÑ†Î∂Ñ ÏúÑÏóê ÏûàÎäî Í≤ΩÏö∞)
    def on_segment(p: Point, q: Point, r: Point) -> bool:
        return (min(p.x, r.x) <= q.x <= max(p.x, r.x) and
                min(p.y, r.y) <= q.y <= max(p.y, r.y))

    if d1 == 0 and on_segment(p3, p1, p4):
        return True
    if d2 == 0 and on_segment(p3, p2, p4):
        return True
    if d3 == 0 and on_segment(p1, p3, p2):
        return True
    if d4 == 0 and on_segment(p1, p4, p2):
        return True

    return False


def line_intersection(p1: Point, p2: Point, p3: Point, p4: Point) -> Optional[Point]:
    """
    Îëê ÏßÅÏÑ†Ïùò ÍµêÏ†ê Í≥ÑÏÇ∞
    ÏßÅÏÑ† p1-p2ÏôÄ ÏßÅÏÑ† p3-p4
    """
    d1 = p2 - p1
    d2 = p4 - p3
    cross = d1.cross(d2)

    if abs(cross) < 1e-9:
        return None  # ÌèâÌñâ

    t = (p3 - p1).cross(d2) / cross
    return p1 + d1 * t

# 4. Î≥ºÎ°ù ÍªçÏßà (Convex Hull)
# =============================================================================

def convex_hull_graham(points: List[Point]) -> List[Point]:
    """
    Graham Scan ÏïåÍ≥†Î¶¨Ï¶ò
    ÏãúÍ∞ÑÎ≥µÏû°ÎèÑ: O(n log n)
    """
    if len(points) < 3:
        return points[:]

    # Í∞ÄÏû• ÏïÑÎûò, ÏôºÏ™Ω Ï†ê Ï∞æÍ∏∞
    start = min(points, key=lambda p: (p.y, p.x))

    # Í∞ÅÎèÑÎ°ú Ï†ïÎ†¨
    def polar_angle(p: Point) -> float:
        return atan2(p.y - start.y, p.x - start.x)

    def dist_sq(p: Point) -> float:
        return (p.x - start.x) ** 2 + (p.y - start.y) ** 2

    sorted_points = sorted(points, key=lambda p: (polar_angle(p), dist_sq(p)))

    # Ïä§ÌÉùÏúºÎ°ú Î≥ºÎ°ù ÍªçÏßà Íµ¨ÏÑ±
    hull = []
    for p in sorted_points:
        while len(hull) >= 2 and ccw(hull[-2], hull[-1], p) <= 0:
            hull.pop()
        hull.append(p)

    return hull


def convex_hull_monotone_chain(points: List[Tuple[float, float]]) -> List[Tuple[float, float]]:
    """
    Monotone Chain ÏïåÍ≥†Î¶¨Ï¶ò
    ÏãúÍ∞ÑÎ≥µÏû°ÎèÑ: O(n log n)
    """
    points = sorted(set(points))
    if len(points) <= 1:
        return points

    # ÌïòÎ∂Ä ÍªçÏßà
    lower = []
    for p in points:
        while len(lower) >= 2 and ccw_tuple(lower[-2], lower[-1], p) <= 0:
            lower.pop()
        lower.append(p)

    # ÏÉÅÎ∂Ä ÍªçÏßà
    upper = []
    for p in reversed(points):
        while len(upper) >= 2 and ccw_tuple(upper[-2], upper[-1], p) <= 0:
            upper.pop()
        upper.append(p)

    return lower[:-1] + upper[:-1]

def polygon_area(vertices: List[Point]) -> float:
    """
    Îã§Í∞ÅÌòï ÎÑìÏù¥ (Ïã†Î∞úÎÅà Í≥µÏãù)
    Ï†ïÏ†êÏù¥ Î∞òÏãúÍ≥Ñ Î∞©Ìñ•ÏúºÎ°ú Ï†ïÎ†¨ÎêòÏñ¥ ÏûàÏñ¥Ïïº Ìï®
    """
    n = len(vertices)
    if n < 3:
        return 0

    area = 0
    for i in range(n):
        j = (i + 1) % n
        area += vertices[i].cross(vertices[j])

    return abs(area) / 2


def polygon_perimeter(vertices: List[Point]) -> float:
    """Îã§Í∞ÅÌòï ÎëòÎ†à"""
    n = len(vertices)
    perimeter = 0
    for i in range(n):
        j = (i + 1) % n
        perimeter += distance(vertices[i], vertices[j])
    return perimeter


def point_in_polygon(point: Point, polygon: List[Point]) -> int:
    """
    Ï†êÏù¥ Îã§Í∞ÅÌòï ÎÇ¥Î∂ÄÏóê ÏûàÎäîÏßÄ ÌåêÏ†ï
    Î∞òÌôò: 1 (ÎÇ¥Î∂Ä), 0 (Í≤ΩÍ≥Ñ), -1 (Ïô∏Î∂Ä)
    """
    n = len(polygon)
    winding = 0

    for i in range(n):
        j = (i + 1) % n
        p1, p2 = polygon[i], polygon[j]

        # Í≤ΩÍ≥Ñ ÏúÑÏùò Ï†ê Í≤ÄÏÇ¨
        if ccw(p1, p2, point) == 0:
            if (min(p1.x, p2.x) <= point.x <= max(p1.x, p2.x) and
                min(p1.y, p2.y) <= point.y <= max(p1.y, p2.y)):
                return 0

        # Winding number Í≥ÑÏÇ∞
        if p1.y <= point.y:
            if p2.y > point.y and ccw(p1, p2, point) > 0:
                winding += 1
        else:
            if p2.y <= point.y and ccw(p1, p2, point) < 0:
                winding -= 1

    return 1 if winding != 0 else -1


def is_convex(polygon: List[Point]) -> bool:
    """Îã§Í∞ÅÌòïÏù¥ Î≥ºÎ°ùÏù∏ÏßÄ Í≤ÄÏÇ¨"""
    n = len(polygon)
    if n < 3:
        return False

    sign = 0
    for i in range(n):
        d = ccw(polygon[i], polygon[(i + 1) % n], polygon[(i + 2) % n])
        if d != 0:
            if sign == 0:
                sign = d
            elif sign != d:
                return False

    return True

def point_to_line_distance(point: Point, line_p1: Point, line_p2: Point) -> float:
    """Ï†êÏóêÏÑú ÏßÅÏÑ†ÍπåÏßÄÏùò Í±∞Î¶¨"""
    v = line_p2 - line_p1
    w = point - line_p1
    return abs(v.cross(w)) / v.norm()


def point_to_segment_distance(point: Point, seg_p1: Point, seg_p2: Point) -> float:
    """Ï†êÏóêÏÑú ÏÑ†Î∂ÑÍπåÏßÄÏùò Í±∞Î¶¨"""
    v = seg_p2 - seg_p1
    w = point - seg_p1

    c1 = w.dot(v)
    if c1 <= 0:
        return distance(point, seg_p1)

    c2 = v.dot(v)
    if c2 <= c1:
        return distance(point, seg_p2)

    t = c1 / c2
    proj = seg_p1 + v * t
    return distance(point, proj)

# 7. Í∞ÄÏû• Í∞ÄÍπåÏö¥ Ï†ê Ïåç (Closest Pair)
# =============================================================================

def closest_pair(points: List[Point]) -> Tuple[Point, Point, float]:
    """
    Í∞ÄÏû• Í∞ÄÍπåÏö¥ Ï†ê Ïåç Ï∞æÍ∏∞
    Î∂ÑÌï† Ï†ïÎ≥µ: O(n log n)
    """
    def brute_force(pts: List[Point]) -> Tuple[Point, Point, float]:
        min_dist = inf
        p1, p2 = None, None
        for i in range(len(pts)):
            for j in range(i + 1, len(pts)):
                d = distance(pts[i], pts[j])
                if d < min_dist:
                    min_dist = d
                    p1, p2 = pts[i], pts[j]
        return p1, p2, min_dist

    def closest_split(pts_y: List[Point], mid_x: float, delta: float) -> Tuple[Point, Point, float]:
        # Ï§ëÍ∞ÑÏÑ† Í∑ºÏ≤òÏùò Ï†êÎì§Îßå
        strip = [p for p in pts_y if abs(p.x - mid_x) < delta]

        min_dist = delta
        p1, p2 = None, None

        for i in range(len(strip)):
            j = i + 1
            while j < len(strip) and strip[j].y - strip[i].y < min_dist:
                d = distance(strip[i], strip[j])
                if d < min_dist:
                    min_dist = d
                    p1, p2 = strip[i], strip[j]
                j += 1

        return p1, p2, min_dist

    def divide_conquer(pts_x: List[Point], pts_y: List[Point]) -> Tuple[Point, Point, float]:
        if len(pts_x) <= 3:
            return brute_force(pts_x)

        mid = len(pts_x) // 2
        mid_point = pts_x[mid]

        # ÏôºÏ™Ω/Ïò§Î•∏Ï™Ω Î∂ÑÌï†
        left_x = pts_x[:mid]
        right_x = pts_x[mid:]

        left_set = set(id(p) for p in left_x)
        left_y = [p for p in pts_y if id(p) in left_set]
        right_y = [p for p in pts_y if id(p) not in left_set]

        # Ïû¨Í∑Ä Ìò∏Ï∂ú
        l1, l2, left_dist = divide_conquer(left_x, left_y)
        r1, r2, right_dist = divide_conquer(right_x, right_y)

        if left_dist < right_dist:
            best = (l1, l2, left_dist)
        else:
            best = (r1, r2, right_dist)

        # Î∂ÑÌï†ÏÑ† Í∑ºÏ≤ò Í≤ÄÏÇ¨
        s1, s2, split_dist = closest_split(pts_y, mid_point.x, best[2])
        if split_dist < best[2]:
            return s1, s2, split_dist

        return best

    if len(points) < 2:
        return None, None, inf

    pts_x = sorted(points, key=lambda p: p.x)
    pts_y = sorted(points, key=lambda p: p.y)

    return divide_conquer(pts_x, pts_y)

# 8. ÌöåÏ†ÑÌïòÎäî Ï∫òÎ¶¨ÌçºÏä§ (Rotating Calipers)
# =============================================================================

def rotating_calipers_diameter(hull: List[Point]) -> Tuple[Point, Point, float]:
    """
    Î≥ºÎ°ù ÍªçÏßàÏùò ÏßÄÎ¶Ñ (Í∞ÄÏû• Î®º Ï†ê Ïåç)
    ÏãúÍ∞ÑÎ≥µÏû°ÎèÑ: O(n)
    """
    n = len(hull)
    if n < 2:
        return None, None, 0
    if n == 2:
        return hull[0], hull[1], distance(hull[0], hull[1])

    # Í∞ÄÏû• Î®º Ï†ê Ïåç Ï∞æÍ∏∞
    max_dist = 0
    p1, p2 = None, None

    j = 1
    for i in range(n):
        # hull[i]-hull[(i+1)%n] ÏóêÏßÄÏóê ÎåÄÌï¥ Í∞ÄÏû• Î®º Ï†ê Ï∞æÍ∏∞
        while True:
            next_j = (j + 1) % n
            # ÏÇºÍ∞ÅÌòï ÎÑìÏù¥ ÎπÑÍµê
            area1 = abs((hull[(i + 1) % n] - hull[i]).cross(hull[j] - hull[i]))
            area2 = abs((hull[(i + 1) % n] - hull[i]).cross(hull[next_j] - hull[i]))
            if area2 > area1:
                j = next_j
            else:
                break

        d1 = distance(hull[i], hull[j])
        d2 = distance(hull[(i + 1) % n], hull[j])

        if d1 > max_dist:
            max_dist = d1
            p1, p2 = hull[i], hull[j]
        if d2 > max_dist:
            max_dist = d2
            p1, p2 = hull[(i + 1) % n], hull[j]

    return p1, p2, max_dist

class HalfPlane:
    """Î∞òÌèâÎ©¥: ax + by + c >= 0"""
    def __init__(self, a: float, b: float, c: float):
        self.a = a
        self.b = b
        self.c = c
        self.angle = atan2(b, a)

    @classmethod
    def from_points(cls, p1: Point, p2: Point) -> 'HalfPlane':
        """p1 ‚Üí p2 Î∞©Ìñ•Ïùò ÏôºÏ™Ω Î∞òÌèâÎ©¥"""
        a = p2.y - p1.y
        b = p1.x - p2.x
        c = -a * p1.x - b * p1.y
        return cls(a, b, c)

    def side(self, p: Point) -> float:
        """Ï†êÏù¥ Ïñ¥Îäê Ï™ΩÏóê ÏûàÎäîÏßÄ (ÏñëÏàò: ÎÇ¥Î∂Ä, ÏùåÏàò: Ïô∏Î∂Ä)"""
        return self.a * p.x + self.b * p.y + self.c


def half_plane_intersection_point(h1: HalfPlane, h2: HalfPlane) -> Optional[Point]:
    """Îëê Î∞òÌèâÎ©¥ Í≤ΩÍ≥ÑÏÑ†Ïùò ÍµêÏ†ê"""
    det = h1.a * h2.b - h2.a * h1.b
    if abs(det) < 1e-9:
        return None
    x = (h1.b * h2.c - h2.b * h1.c) / det
    y = (h2.a * h1.c - h1.a * h2.c) / det
    return Point(x, y)

def triangle_area(p1: Point, p2: Point, p3: Point) -> float:
    """ÏÇºÍ∞ÅÌòï ÎÑìÏù¥"""
    return abs((p2 - p1).cross(p3 - p1)) / 2


def triangle_circumcircle(p1: Point, p2: Point, p3: Point) -> Tuple[Point, float]:
    """ÏÇºÍ∞ÅÌòïÏùò Ïô∏Ï†ëÏõê (Ï§ëÏã¨, Î∞òÏßÄÎ¶Ñ)"""
    ax, ay = p1.x, p1.y
    bx, by = p2.x, p2.y
    cx, cy = p3.x, p3.y

    d = 2 * (ax * (by - cy) + bx * (cy - ay) + cx * (ay - by))
    if abs(d) < 1e-9:
        return None, 0

    ux = ((ax**2 + ay**2) * (by - cy) + (bx**2 + by**2) * (cy - ay) + (cx**2 + cy**2) * (ay - by)) / d
    uy = ((ax**2 + ay**2) * (cx - bx) + (bx**2 + by**2) * (ax - cx) + (cx**2 + cy**2) * (bx - ax)) / d

    center = Point(ux, uy)
    radius = distance(center, p1)

    return center, radius
