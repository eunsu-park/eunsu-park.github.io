def nim_xor(piles: List[int]) -> int:
    """
    ë‹˜ ê²Œì„ì˜ XOR ê°’ (Nim-sum)
    XOR != 0ì´ë©´ ì„ ê³µ ìŠ¹ë¦¬, XOR == 0ì´ë©´ í›„ê³µ ìŠ¹ë¦¬
    """
    result = 0
    for pile in piles:
        result ^= pile
    return result


def nim_winning_move(piles: List[int]) -> Optional[Tuple[int, int]]:
    """
    ë‹˜ ê²Œì„ì—ì„œ ìŠ¹ë¦¬í•˜ëŠ” ìˆ˜ ì°¾ê¸°
    ë°˜í™˜: (ë”ë¯¸ ì¸ë±ìŠ¤, ë‚¨ê¸¸ ê°œìˆ˜) ë˜ëŠ” None
    """
    xor = nim_xor(piles)

    if xor == 0:
        return None  # íŒ¨ë°° ìƒíƒœ, ìŠ¹ë¦¬ ìˆ˜ ì—†ìŒ

    for i, pile in enumerate(piles):
        target = pile ^ xor
        if target < pile:
            return (i, target)  # pileì—ì„œ targetê°œë§Œ ë‚¨ê¸°ê¸°

    return None


def nim_game_simulation(piles: List[int], verbose: bool = False) -> int:
    """
    ë‹˜ ê²Œì„ ì‹œë®¬ë ˆì´ì…˜
    ë°˜í™˜: ìŠ¹ì (0: ì„ ê³µ, 1: í›„ê³µ)
    """
    current = 0  # í˜„ì¬ í”Œë ˆì´ì–´

    while max(piles) > 0:
        move = nim_winning_move(piles)

        if move is None:
            # íŒ¨ë°° ìƒíƒœ: ì•„ë¬´ ìˆ˜ë‚˜
            for i, pile in enumerate(piles):
                if pile > 0:
                    move = (i, pile - 1)
                    break

        pile_idx, new_count = move
        if verbose:
            print(f"    Player {current}: ë”ë¯¸ {pile_idx}ì—ì„œ {piles[pile_idx]}â†’{new_count}")

        piles[pile_idx] = new_count
        current = 1 - current

    return 1 - current  # ë§ˆì§€ë§‰ì— ê°€ì ¸ê°„ ì‚¬ëŒì´ ìŠ¹ë¦¬

def mex(s: Set[int]) -> int:
    """
    Minimum Excludant
    ì§‘í•©ì— ì—†ëŠ” ê°€ì¥ ì‘ì€ ìŒì´ ì•„ë‹Œ ì •ìˆ˜
    """
    i = 0
    while i in s:
        i += 1
    return i


def calculate_grundy(position: int, moves: List[int], memo: Dict[int, int] = None) -> int:
    """
    ìŠ¤í”„ë¼ê·¸-ê·¸ëŸ°ë”” ìˆ˜ ê³„ì‚°
    position: í˜„ì¬ ìƒíƒœ (ì˜ˆ: ëŒì˜ ê°œìˆ˜)
    moves: ê°€ëŠ¥í•œ ì´ë™ëŸ‰ ë¦¬ìŠ¤íŠ¸
    """
    if memo is None:
        memo = {}

    if position in memo:
        return memo[position]

    if position == 0:
        memo[position] = 0
        return 0

    reachable = set()
    for move in moves:
        if position >= move:
            reachable.add(calculate_grundy(position - move, moves, memo))

    result = mex(reachable)
    memo[position] = result
    return result


def multi_pile_grundy(piles: List[int], moves: List[int]) -> int:
    """
    ì—¬ëŸ¬ ë”ë¯¸ ê²Œì„ì˜ ì „ì²´ ê·¸ëŸ°ë”” ìˆ˜
    ê° ë”ë¯¸ì˜ ê·¸ëŸ°ë”” ìˆ˜ë¥¼ XOR
    """
    memo = {}
    total_grundy = 0

    for pile in piles:
        grundy = calculate_grundy(pile, moves, memo)
        total_grundy ^= grundy

    return total_grundy

def staircase_nim(stairs: List[int]) -> int:
    """
    ê³„ë‹¨ ë‹˜ (Staircase Nim)
    í™€ìˆ˜ ë²ˆì§¸ ê³„ë‹¨ì˜ XOR = ê·¸ëŸ°ë”” ìˆ˜
    """
    xor = 0
    for i in range(0, len(stairs), 2):  # í™€ìˆ˜ ì¸ë±ìŠ¤ (0-indexedì˜ ì§ìˆ˜)
        xor ^= stairs[i]
    return xor

        ones = sum(1 for p in piles if p == 1)

        return xor != 0

def minimax(position, depth: int, is_maximizing: bool,
            evaluate, get_moves, is_terminal) -> int:
    """
    ë¯¸ë‹ˆë§¥ìŠ¤ ì•Œê³ ë¦¬ì¦˜
    position: í˜„ì¬ ê²Œì„ ìƒíƒœ
    depth: íƒìƒ‰ ê¹Šì´
    is_maximizing: ìµœëŒ€í™” í”Œë ˆì´ì–´ì˜ í„´ì¸ì§€
    evaluate: ìƒíƒœ í‰ê°€ í•¨ìˆ˜
    get_moves: ê°€ëŠ¥í•œ ìˆ˜ ë°˜í™˜ í•¨ìˆ˜
    is_terminal: ì¢…ë£Œ ìƒíƒœ í™•ì¸ í•¨ìˆ˜
    """
    if depth == 0 or is_terminal(position):
        return evaluate(position)

    moves = get_moves(position)

    if is_maximizing:
        max_eval = float('-inf')
        for move in moves:
            new_position = apply_move(position, move)
            eval_score = minimax(new_position, depth - 1, False,
                                evaluate, get_moves, is_terminal)
            max_eval = max(max_eval, eval_score)
        return max_eval
    else:
        min_eval = float('inf')
        for move in moves:
            new_position = apply_move(position, move)
            eval_score = minimax(new_position, depth - 1, True,
                                evaluate, get_moves, is_terminal)
            min_eval = min(min_eval, eval_score)
        return min_eval

def alpha_beta(position, depth: int, alpha: float, beta: float,
               is_maximizing: bool, evaluate, get_moves, is_terminal) -> int:
    """
    ì•ŒíŒŒ-ë² íƒ€ ê°€ì§€ì¹˜ê¸°
    alpha: ìµœëŒ€í™” í”Œë ˆì´ì–´ì˜ ìµœì„ ì˜ ë³´ì¥ê°’
    beta: ìµœì†Œí™” í”Œë ˆì´ì–´ì˜ ìµœì„ ì˜ ë³´ì¥ê°’
    """
    if depth == 0 or is_terminal(position):
        return evaluate(position)

    moves = get_moves(position)

    if is_maximizing:
        max_eval = float('-inf')
        for move in moves:
            new_position = apply_move(position, move)
            eval_score = alpha_beta(new_position, depth - 1, alpha, beta,
                                   False, evaluate, get_moves, is_terminal)
            max_eval = max(max_eval, eval_score)
            alpha = max(alpha, eval_score)
            if beta <= alpha:
                break  # ê°€ì§€ì¹˜ê¸°
        return max_eval
    else:
        min_eval = float('inf')
        for move in moves:
            new_position = apply_move(position, move)
            eval_score = alpha_beta(new_position, depth - 1, alpha, beta,
                                   True, evaluate, get_moves, is_terminal)
            min_eval = min(min_eval, eval_score)
            beta = min(beta, eval_score)
            if beta <= alpha:
                break  # ê°€ì§€ì¹˜ê¸°
        return min_eval

class TicTacToe:
    """í‹±íƒí†  ê²Œì„"""

    def __init__(self):
        self.board = [[' '] * 3 for _ in range(3)]
        self.current_player = 'X'

    def get_moves(self) -> List[Tuple[int, int]]:
        """ê°€ëŠ¥í•œ ìˆ˜ ë°˜í™˜"""
        moves = []
        for i in range(3):
            for j in range(3):
                if self.board[i][j] == ' ':
                    moves.append((i, j))
        return moves

    def make_move(self, row: int, col: int) -> bool:
        """ìˆ˜ ë‘ê¸°"""
        if self.board[row][col] != ' ':
            return False
        self.board[row][col] = self.current_player
        self.current_player = 'O' if self.current_player == 'X' else 'X'
        return True

    def undo_move(self, row: int, col: int):
        """ìˆ˜ ë˜ëŒë¦¬ê¸°"""
        self.board[row][col] = ' '
        self.current_player = 'O' if self.current_player == 'X' else 'X'

    def check_winner(self) -> Optional[str]:
        """ìŠ¹ì í™•ì¸"""
        # í–‰
        for row in self.board:
            if row[0] == row[1] == row[2] != ' ':
                return row[0]
        # ì—´
        for col in range(3):
            if self.board[0][col] == self.board[1][col] == self.board[2][col] != ' ':
                return self.board[0][col]
        # ëŒ€ê°ì„ 
        if self.board[0][0] == self.board[1][1] == self.board[2][2] != ' ':
            return self.board[0][0]
        if self.board[0][2] == self.board[1][1] == self.board[2][0] != ' ':
            return self.board[0][2]
        return None

    def is_terminal(self) -> bool:
        """ê²Œì„ ì¢…ë£Œ í™•ì¸"""
        return self.check_winner() is not None or len(self.get_moves()) == 0

    def evaluate(self) -> int:
        """ìƒíƒœ í‰ê°€ (X ê´€ì )"""
        winner = self.check_winner()
        if winner == 'X':
            return 10
        elif winner == 'O':
            return -10
        return 0

    def minimax(self, is_maximizing: bool) -> int:
        """ë¯¸ë‹ˆë§¥ìŠ¤"""
        if self.is_terminal():
            return self.evaluate()

        if is_maximizing:
            max_eval = float('-inf')
            for row, col in self.get_moves():
                self.make_move(row, col)
                eval_score = self.minimax(False)
                self.undo_move(row, col)
                max_eval = max(max_eval, eval_score)
            return max_eval
        else:
            min_eval = float('inf')
            for row, col in self.get_moves():
                self.make_move(row, col)
                eval_score = self.minimax(True)
                self.undo_move(row, col)
                min_eval = min(min_eval, eval_score)
            return min_eval

    def best_move(self) -> Tuple[int, int]:
        """ìµœì„ ì˜ ìˆ˜ ì°¾ê¸°"""
        best_score = float('-inf') if self.current_player == 'X' else float('inf')
        best_move = None

        for row, col in self.get_moves():
            self.make_move(row, col)
            score = self.minimax(self.current_player == 'X')
            self.undo_move(row, col)

            if self.current_player == 'X':
                if score > best_score:
                    best_score = score
                    best_move = (row, col)
            else:
                if score < best_score:
                    best_score = score
                    best_move = (row, col)

        return best_move

    def display(self):
        """ë³´ë“œ ì¶œë ¥"""
        for i, row in enumerate(self.board):
            print("    " + " | ".join(row))
            if i < 2:
                print("    " + "-" * 9)

@lru_cache(maxsize=None)
def stone_game_dp(piles: Tuple[int, ...], left: int, right: int) -> int:
    """
    ëŒ ê²Œì„: ì–‘ ëì—ì„œë§Œ ê°€ì ¸ê°ˆ ìˆ˜ ìˆìŒ
    ì„ ê³µì´ ì–»ì„ ìˆ˜ ìˆëŠ” ìµœëŒ€ ì ìˆ˜ ì°¨ì´ ë°˜í™˜
    """
    if left > right:
        return 0

    # ì„ ê³µì´ ì™¼ìª½ ì„ íƒ
    pick_left = piles[left] - stone_game_dp(piles, left + 1, right)
    # ì„ ê³µì´ ì˜¤ë¥¸ìª½ ì„ íƒ
    pick_right = piles[right] - stone_game_dp(piles, left, right - 1)

    return max(pick_left, pick_right)


def stone_game(piles: List[int]) -> bool:
    """
    ëŒ ê²Œì„: ì„ ê³µì´ ì´ê¸°ë©´ True
    """
    n = len(piles)
    diff = stone_game_dp(tuple(piles), 0, n - 1)
    return diff > 0

def bash_game(n: int, k: int) -> bool:
    """
    ë°”ì‹œ ê²Œì„: nê°œ ëŒì—ì„œ ìµœëŒ€ kê°œì”© ê°€ì ¸ê°

    ë°˜í™˜: ì„ ê³µì´ ì´ê¸°ë©´ True

def bash_game_optimal_move(n: int, k: int) -> int:
    """ë°”ì‹œ ê²Œì„ì—ì„œ ìµœì ì˜ ìˆ˜ (ê°€ì ¸ê°ˆ ëŒì˜ ê°œìˆ˜)"""
    if n % (k + 1) == 0:
        return 1  # íŒ¨ë°° ìƒíƒœ, ì•„ë¬´ ìˆ˜ë‚˜
    return n % (k + 1)

# 9. ìœ„ë”í”„ ê²Œì„ (Wythoff's Game)

def wythoff_game(a: int, b: int) -> bool:
    """
    ìœ„ë”í”„ ê²Œì„: ë‘ ë”ë¯¸ì—ì„œ ê°™ì€ ê°œìˆ˜ ë˜ëŠ” í•œ ë”ë¯¸ì—ì„œ ì„ì˜ ê°œìˆ˜
    ë§ˆì§€ë§‰ ëŒì„ ê°€ì ¸ê°€ëŠ” ì‚¬ëŒì´ ìŠ¹ë¦¬
    ë°˜í™˜: ì„ ê³µì´ ì´ê¸°ë©´ True
    """
    phi = (1 + 5 ** 0.5) / 2  # í™©ê¸ˆë¹„

    if a > b:
        a, b = b, a

    k = b - a
    ak = int(k * phi)

    return a != ak

# =============================================================================

def euclid_game(a: int, b: int) -> bool:
    """
    ìœ í´ë¦¬ë“œ ê²Œì„: í° ìˆ˜ì—ì„œ ì‘ì€ ìˆ˜ì˜ ë°°ìˆ˜ë¥¼ ëºŒ
    0ì„ ë§Œë“œëŠ” ì‚¬ëŒì´ ìŠ¹ë¦¬
    ë°˜í™˜: ì„ ê³µì´ ì´ê¸°ë©´ True
    """
    if a < b:
        a, b = b, a

    if b == 0:
        return False  # ì´ë¯¸ ëë‚¨

    # ì¬ê·€ì  ë¶„ì„
    turn = True  # True: ì„ ê³µì˜ í„´
    while b > 0:
        if a >= 2 * b or a == b:
            return turn
        a, b = b, a - b
        turn = not turn

    return not turn
