class ConvexHullTrick:
    """
    ë³¼ë¡ ê»ì§ˆ íŠ¸ë¦­
    ìµœì†Ÿê°’ ì¿¼ë¦¬: min(a[i] * x + b[i]) for all i
    ì¡°ê±´: a[i]ê°€ ë‹¨ì¡° ê°ì†Œ (ë˜ëŠ” ì¦ê°€)

    ì‹œê°„ë³µì¡ë„: ì‚½ì… O(1) í‰ê· , ì¿¼ë¦¬ O(log n) ë˜ëŠ” O(1)
    """

    def __init__(self):
        self.lines = deque()  # (ê¸°ìš¸ê¸°, yì ˆí¸)

    def is_bad(self, l1: Tuple[int, int], l2: Tuple[int, int], l3: Tuple[int, int]) -> bool:
        """l2ê°€ ë¶ˆí•„ìš”í•œì§€ í™•ì¸ (l1ê³¼ l3 ì‚¬ì´ì—ì„œ)"""
        # êµì  ë¹„êµ: (l1, l2) êµì  >= (l2, l3) êµì ì´ë©´ l2 ë¶ˆí•„ìš”
        # (b2 - b1) / (a1 - a2) >= (b3 - b2) / (a2 - a3)
        # (b2 - b1) * (a2 - a3) >= (b3 - b2) * (a1 - a2)
        a1, b1 = l1
        a2, b2 = l2
        a3, b3 = l3
        return (b2 - b1) * (a2 - a3) >= (b3 - b2) * (a1 - a2)

    def add_line(self, a: int, b: int):
        """
        ì§ì„  y = ax + b ì¶”ê°€
        aëŠ” ë‹¨ì¡° ê°ì†Œí•´ì•¼ í•¨
        """
        line = (a, b)

        while len(self.lines) >= 2 and self.is_bad(self.lines[-2], self.lines[-1], line):
            self.lines.pop()

        self.lines.append(line)

    def query_min(self, x: int) -> int:
        """
        xì—ì„œì˜ ìµœì†Ÿê°’ ì¿¼ë¦¬
        xê°€ ë‹¨ì¡° ì¦ê°€í•  ë•Œ O(1)
        """
        while len(self.lines) >= 2:
            a1, b1 = self.lines[0]
            a2, b2 = self.lines[1]
            if a1 * x + b1 >= a2 * x + b2:
                self.lines.popleft()
            else:
                break

        a, b = self.lines[0]
        return a * x + b


class LiChaoTree:
    """
    Li Chao Tree (ì„¸ê·¸ë¨¼íŠ¸ íŠ¸ë¦¬ ê¸°ë°˜ CHT)
    ì„ì˜ì˜ ì§ì„  ì¶”ê°€ì™€ ì„ì˜ì˜ xì—ì„œ ì¿¼ë¦¬ ì§€ì›

    ì‹œê°„ë³µì¡ë„: ì‚½ì… O(log C), ì¿¼ë¦¬ O(log C)
    C = ì¢Œí‘œ ë²”ìœ„
    """

    def __init__(self, lo: int, hi: int):
        self.lo = lo
        self.hi = hi
        self.tree = {}  # ë…¸ë“œë³„ ìš°ì„¸ ì§ì„  ì €ì¥

    def _eval(self, line: Tuple[int, int], x: int) -> int:
        """ì§ì„  ê°’ ê³„ì‚°"""
        if line is None:
            return inf
        a, b = line
        return a * x + b

    def add_line(self, a: int, b: int):
        """ì§ì„  ì¶”ê°€"""
        self._add_line_impl((a, b), self.lo, self.hi, 1)

    def _add_line_impl(self, new_line: Tuple[int, int], lo: int, hi: int, node: int):
        if lo > hi:
            return

        mid = (lo + hi) // 2
        cur_line = self.tree.get(node)

        # ì¤‘ê°„ì ì—ì„œ ë¹„êµ
        new_better_at_mid = self._eval(new_line, mid) < self._eval(cur_line, mid)

        if cur_line is None or new_better_at_mid:
            self.tree[node], new_line = new_line, cur_line

        if lo == hi or new_line is None:
            return

        # ì™¼ìª½/ì˜¤ë¥¸ìª½ ìì‹ìœ¼ë¡œ ì „íŒŒ
        new_better_at_lo = self._eval(new_line, lo) < self._eval(self.tree.get(node), lo)

        if new_better_at_lo:
            self._add_line_impl(new_line, lo, mid - 1, 2 * node)
        else:
            self._add_line_impl(new_line, mid + 1, hi, 2 * node + 1)

    def query(self, x: int) -> int:
        """xì—ì„œì˜ ìµœì†Ÿê°’"""
        return self._query_impl(x, self.lo, self.hi, 1)

    def _query_impl(self, x: int, lo: int, hi: int, node: int) -> int:
        if lo > hi:
            return inf

        result = self._eval(self.tree.get(node), x)

        if lo == hi:
            return result

        mid = (lo + hi) // 2
        if x <= mid:
            return min(result, self._query_impl(x, lo, mid - 1, 2 * node))
        else:
            return min(result, self._query_impl(x, mid + 1, hi, 2 * node + 1))

def dc_optimization(n: int, m: int, cost: Callable[[int, int], int]) -> List[List[int]]:
    """
    ë¶„í•  ì •ë³µ ìµœì í™”
    ì¡°ê±´: opt[k][i] <= opt[k][i+1] (ë‹¨ì¡°ì„±)
    ì í™”ì‹: dp[k][j] = min(dp[k-1][i] + cost(i, j)) for i < j

    ì‹œê°„ë³µì¡ë„: O(k * n log n) (ì¼ë°˜ O(k * n^2)ì—ì„œ ê°œì„ )

    n: ì›ì†Œ ê°œìˆ˜
    m: ë¶„í•  ê·¸ë£¹ ìˆ˜
    cost(i, j): i+1 ~ j êµ¬ê°„ì˜ ë¹„ìš©
    """
    INF = float('inf')
    dp = [[INF] * (n + 1) for _ in range(m + 1)]
    dp[0][0] = 0

    def compute(k: int, lo: int, hi: int, opt_lo: int, opt_hi: int):
        """dp[k][lo:hi+1] ê³„ì‚°, ìµœì ì˜ ë¶„í• ì ì€ opt_lo ~ opt_hi ë²”ìœ„"""
        if lo > hi:
            return

        mid = (lo + hi) // 2
        best_cost = INF
        best_opt = opt_lo

        for i in range(opt_lo, min(opt_hi, mid) + 1):
            curr_cost = dp[k - 1][i] + cost(i, mid)
            if curr_cost < best_cost:
                best_cost = curr_cost
                best_opt = i

        dp[k][mid] = best_cost

        # ë¶„í•  ì •ë³µ
        compute(k, lo, mid - 1, opt_lo, best_opt)
        compute(k, mid + 1, hi, best_opt, opt_hi)

    for k in range(1, m + 1):
        compute(k, k, n, k - 1, n - 1)

    return dp


def dc_optimization_example():
    """
    ì˜ˆì œ: ë°°ì—´ì„ kê°œ ê·¸ë£¹ìœ¼ë¡œ ë‚˜ëˆ„ê¸°
    ê° ê·¸ë£¹ì˜ ë¹„ìš© = êµ¬ê°„ ë‚´ ì›ì†Œ ì°¨ì´ì˜ ì œê³± í•©
    """
    arr = [1, 5, 2, 8, 3, 7, 4, 6]
    n = len(arr)
    k = 3

    # ì „ì²˜ë¦¬: prefix sum for cost calculation
    prefix = [0] * (n + 1)
    prefix_sq = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + arr[i]
        prefix_sq[i + 1] = prefix_sq[i] + arr[i] * arr[i]

    def cost(l: int, r: int) -> int:
        """êµ¬ê°„ [l+1, r]ì˜ ë¶„ì‚° (ì œê³±í•© - í‰ê· *í•©)"""
        if l >= r:
            return 0
        length = r - l
        s = prefix[r] - prefix[l]
        sq = prefix_sq[r] - prefix_sq[l]
        # ë¶„ì‚° = E[X^2] - E[X]^2, ì—¬ê¸°ì„œëŠ” ì œê³±í•© - í•©^2/n
        return sq * length - s * s

    dp = dc_optimization(n, k, cost)
    return dp[k][n]

def knuth_optimization(n: int, cost: List[List[int]]) -> Tuple[List[List[int]], List[List[int]]]:
    """
    Knuth ìµœì í™”
    ì¡°ê±´: costê°€ ì‚¬ê° ë¶€ë“±ì‹ ë§Œì¡± (Quadrangle Inequality)
          cost[a][c] + cost[b][d] <= cost[a][d] + cost[b][c] (a <= b <= c <= d)
    ì í™”ì‹: dp[i][j] = min(dp[i][k] + dp[k][j]) + cost[i][j] for i < k < j

    ì‹œê°„ë³µì¡ë„: O(n^2) (ì¼ë°˜ O(n^3)ì—ì„œ ê°œì„ )

    ì˜ˆ: ìµœì  ì´ì§„ íƒìƒ‰ íŠ¸ë¦¬, í–‰ë ¬ ì²´ì¸ ê³±ì…ˆ
    """
    INF = float('inf')
    dp = [[0] * n for _ in range(n)]
    opt = [[0] * n for _ in range(n)]

    # ê¸°ì €: ê¸¸ì´ 1
    for i in range(n):
        opt[i][i] = i

    # ê¸¸ì´ 2 ì´ìƒ
    for length in range(2, n + 1):
        for i in range(n - length + 1):
            j = i + length - 1
            dp[i][j] = INF

            # opt[i][j-1] <= opt[i][j] <= opt[i+1][j]
            lo = opt[i][j - 1] if j > 0 else i
            hi = opt[i + 1][j] if i + 1 < n else j

            for k in range(lo, min(hi, j) + 1):
                curr = dp[i][k] + dp[k + 1][j] + cost[i][j]
                if curr < dp[i][j]:
                    dp[i][j] = curr
                    opt[i][j] = k

    return dp, opt


def optimal_bst(keys: List[int], freq: List[int]) -> int:
    """
    ìµœì  ì´ì§„ íƒìƒ‰ íŠ¸ë¦¬
    freq[i] = keys[i]ì˜ ì ‘ê·¼ ë¹ˆë„
    """
    n = len(keys)

    # cost[i][j] = freq[i] + ... + freq[j]
    cost = [[0] * n for _ in range(n)]
    for i in range(n):
        total = 0
        for j in range(i, n):
            total += freq[j]
            cost[i][j] = total

    dp, opt = knuth_optimization(n, cost)
    return dp[0][n - 1]

def sliding_window_max(arr: List[int], k: int) -> List[int]:
    """
    ìŠ¬ë¼ì´ë”© ìœˆë„ìš° ìµœëŒ“ê°’
    ì‹œê°„ë³µì¡ë„: O(n)
    """
    result = []
    dq = deque()  # (ì¸ë±ìŠ¤, ê°’)

    for i, val in enumerate(arr):
        # ìœˆë„ìš° ë²”ìœ„ ë°– ì œê±°
        while dq and dq[0][0] <= i - k:
            dq.popleft()

        # í˜„ì¬ ê°’ë³´ë‹¤ ì‘ì€ ì›ì†Œ ì œê±°
        while dq and dq[-1][1] <= val:
            dq.pop()

        dq.append((i, val))

        if i >= k - 1:
            result.append(dq[0][1])

    return result


def dp_with_monotone_queue(arr: List[int], k: int) -> List[int]:
    """
    DP with Monotone Queue
    dp[i] = max(dp[j] + arr[i]) for i-k <= j < i
    ì‹œê°„ë³µì¡ë„: O(n)
    """
    n = len(arr)
    dp = [0] * n
    dq = deque()  # (ì¸ë±ìŠ¤, dpê°’)

    for i in range(n):
        # ë²”ìœ„ ë°– ì œê±°
        while dq and dq[0][0] < i - k:
            dq.popleft()

        # ìµœëŒ“ê°’ìœ¼ë¡œ dp ê³„ì‚°
        if dq:
            dp[i] = dq[0][1] + arr[i]
        else:
            dp[i] = arr[i]

        # í˜„ì¬ dpê°’ ì‚½ì…
        while dq and dq[-1][1] <= dp[i]:
            dq.pop()
        dq.append((i, dp[i]))

    return dp

class SlopeTrick:
    """
    Slope Trick
    ë³¼ë¡ í•¨ìˆ˜ì˜ íš¨ìœ¨ì ì¸ ê´€ë¦¬
    ì ˆëŒ€ê°’ í•¨ìˆ˜ì˜ í•© ìµœì í™”ì— ìœ ìš©
    """

    def __init__(self):
        import heapq
        self.left = []   # ìµœëŒ€ í™ (ìŒìˆ˜ë¡œ ì €ì¥)
        self.right = []  # ìµœì†Œ í™
        self.min_f = 0
        self.add_l = 0   # leftì— ë”í•  ê°’
        self.add_r = 0   # rightì— ë”í•  ê°’

    def add_abs(self, a: int):
        """
        f(x) += |x - a|
        """
        import heapq

        # a ìœ„ì¹˜ì—ì„œ ê¸°ìš¸ê¸° ë³€í™”: ì™¼ìª½ +1, ì˜¤ë¥¸ìª½ -1
        l = -self.left[0] + self.add_l if self.left else -inf
        r = self.right[0] + self.add_r if self.right else inf

        if a <= l:
            # aê°€ ì™¼ìª½ì— ìœ„ì¹˜
            self.min_f += l - a
            heapq.heappush(self.left, -(a - self.add_l))
            # ì™¼ìª½ ìµœëŒ“ê°’ì„ ì˜¤ë¥¸ìª½ìœ¼ë¡œ
            val = -heapq.heappop(self.left) + self.add_l
            heapq.heappush(self.right, val - self.add_r)
        elif a >= r:
            # aê°€ ì˜¤ë¥¸ìª½ì— ìœ„ì¹˜
            self.min_f += a - r
            heapq.heappush(self.right, a - self.add_r)
            # ì˜¤ë¥¸ìª½ ìµœì†Ÿê°’ì„ ì™¼ìª½ìœ¼ë¡œ
            val = heapq.heappop(self.right) + self.add_r
            heapq.heappush(self.left, -(val - self.add_l))
        else:
            # aê°€ í‰í‰í•œ êµ¬ê°„ì— ìœ„ì¹˜
            heapq.heappush(self.left, -(a - self.add_l))
            heapq.heappush(self.right, a - self.add_r)

    def shift(self, a: int, b: int):
        """
        f(x) â†’ f(x-a) (ì™¼ìª½ ì´ë™), f(x) â†’ f(x-b) (ì˜¤ë¥¸ìª½ ì´ë™)
        í‰í‰í•œ êµ¬ê°„ í™•ì¥
        """
        self.add_l += a
        self.add_r += b

    def get_min(self) -> int:
        """ìµœì†Ÿê°’ ë°˜í™˜"""
        return self.min_f

def alien_trick_example(arr: List[int], k: int) -> int:
    """
    Alien Trick (WQS Binary Search / Lagrange Relaxation)
    ì •í™•íˆ kê°œì˜ ì›ì†Œë¥¼ ì„ íƒí•˜ëŠ” ë¬¸ì œë¥¼ ì´ì™„

    ì˜ˆ: ë°°ì—´ì—ì„œ ì •í™•íˆ kê°œ ì„ íƒ, ì¸ì ‘í•œ ê²ƒ ë¶ˆê°€, í•© ìµœëŒ€í™”
    """

    def check(penalty: float) -> Tuple[float, int]:
        """
        penaltyë¥¼ ì‚¬ìš©í•œ ì´ì™„ ë¬¸ì œ í•´ê²°
        ë°˜í™˜: (ìµœì ê°’, ì„ íƒí•œ ê°œìˆ˜)
        """
        n = len(arr)
        # dp[i][0]: iê¹Œì§€, arr[i] ì„ íƒ ì•ˆí•¨
        # dp[i][1]: iê¹Œì§€, arr[i] ì„ íƒí•¨

        dp = [[-inf, -inf] for _ in range(n)]
        cnt = [[0, 0] for _ in range(n)]

        dp[0][0] = 0
        dp[0][1] = arr[0] - penalty
        cnt[0][1] = 1

        for i in range(1, n):
            # ì„ íƒ ì•ˆí•¨
            if dp[i - 1][0] > dp[i - 1][1]:
                dp[i][0] = dp[i - 1][0]
                cnt[i][0] = cnt[i - 1][0]
            else:
                dp[i][0] = dp[i - 1][1]
                cnt[i][0] = cnt[i - 1][1]

            # ì„ íƒí•¨ (ì´ì „ì— ì„ íƒ ì•ˆí•œ ìƒíƒœì—ì„œë§Œ)
            dp[i][1] = dp[i - 1][0] + arr[i] - penalty
            cnt[i][1] = cnt[i - 1][0] + 1

        if dp[n - 1][0] > dp[n - 1][1]:
            return dp[n - 1][0], cnt[n - 1][0]
        return dp[n - 1][1], cnt[n - 1][1]

    # ì´ë¶„ íƒìƒ‰
    lo, hi = -10**9, 10**9

    while hi - lo > 1e-6:
        mid = (lo + hi) / 2
        _, count = check(mid)
        if count >= k:
            lo = mid
        else:
            hi = mid

    result, _ = check(lo)
    return int(result + lo * k)
