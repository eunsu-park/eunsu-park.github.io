// ë¹„ì–´ìˆëŠ”ì§€ í™•ì¸

bool queue_isFull(CircularQueue *q) {

// Enqueue - ë’¤ì— ìš”ì†Œ ì¶”ê°€ (O(1))
bool queue_enqueue(CircularQueue *q, int value) {
    if (queue_isFull(q)) {
        printf("Queue is full!\n");
        return false;
    }

    q->rear = (q->rear + 1) % MAX_SIZE;  // ì›í˜•ìœ¼ë¡œ ìˆœí™˜
    q->data[q->rear] = value;
    q->count++;
    return true;
}

// Dequeue - ì•ì—ì„œ ìš”ì†Œ ì œê±° (O(1))
bool queue_dequeue(CircularQueue *q, int *value) {
    if (queue_isEmpty(q)) {
        printf("Queue is empty!\n");
        return false;
    }

    *value = q->data[q->front];
    q->front = (q->front + 1) % MAX_SIZE;  // ì›í˜•ìœ¼ë¡œ ìˆœí™˜
    q->count--;
    return true;
}

// Front - ì•ì˜ ê°’ í™•ì¸ (ì œê±° ì•ˆí•¨)
bool queue_front(CircularQueue *q, int *value) {
    if (queue_isEmpty(q)) {
        return false;
    }
    *value = q->data[q->front];
    return true;
}

// Rear - ë’¤ì˜ ê°’ í™•ì¸
bool queue_rear(CircularQueue *q, int *value) {
    if (queue_isEmpty(q)) {
        return false;
    }
    *value = q->data[q->rear];
    return true;
}

// í ì¶œë ¥ (frontë¶€í„° rearê¹Œì§€)
void queue_print(CircularQueue *q) {
    printf("Queue (count=%d): [", q->count);
    if (!queue_isEmpty(q)) {
        int i = q->front;
        for (int c = 0; c < q->count; c++) {
            printf("%d", q->data[i]);
            if (c < q->count - 1) printf(", ");
            i = (i + 1) % MAX_SIZE;
        }
    }
    printf("] (front=%d, rear=%d)\n", q->front, q->rear);
}

void queue_visualize(CircularQueue *q) {
    printf("ë°°ì—´ ìƒíƒœ: [");
    for (int i = 0; i < MAX_SIZE; i++) {
        if (q->count > 0) {
            int start = q->front;
            int end = q->rear;
            bool inRange = false;

            // ì›í˜• íì—ì„œ iê°€ ìœ íš¨ ë²”ìœ„ì¸ì§€ í™•ì¸
            if (start <= end) {
                inRange = (i >= start && i <= end);
            } else {
                inRange = (i >= start || i <= end);
            }

            if (inRange) {
                printf("%d", q->data[i]);
            } else {
                printf("-");
            }
        } else {
            printf("-");
        }
        if (i < MAX_SIZE - 1) printf(" ");
    }
    printf("]\n");

    // frontì™€ rear ìœ„ì¹˜ í‘œì‹œ
    printf("           ");
    for (int i = 0; i < MAX_SIZE; i++) {
        if (i == q->front && i == q->rear && q->count > 0) {
            printf("FR");
        } else if (i == q->front && q->count > 0) {
            printf("F ");
        } else if (i == q->rear && q->count > 0) {
            printf("R ");
        } else {
            printf("  ");
        }
    }
    printf("\n");
}
