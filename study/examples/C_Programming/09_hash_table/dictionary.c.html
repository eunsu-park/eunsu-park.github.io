 46unsigned int hash(const char *key) {
 47    unsigned int hash = 5381;
 48    while (*key) {
 49        hash = ((hash << 5) + hash) + tolower((unsigned char)*key++);
 50    }
 51    return hash % TABLE_SIZE;
 52}
