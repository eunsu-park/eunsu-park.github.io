 43unsigned int hash(const char *key) {
 44    unsigned int hash = 5381;
 45    int c;
 46    while ((c = *key++)) {
 47        hash = ((hash << 5) + hash) + c;
 48    }
 49    return hash % TABLE_SIZE;
