void snake_destroy(Snake* snake) {
    if (!snake) return;

    SnakeNode* current = snake->head;
    while (current) {
        SnakeNode* next = current->next;
        free(current);
        current = next;
    }
    free(snake);
}

Point snake_next_head_position(const Snake* snake) {
    Point next = snake->head->pos;

    switch (snake->dir) {
        case DIR_UP:
            next.y--;
            break;
        case DIR_DOWN:
            next.y++;
            break;
        case DIR_LEFT:
            next.x--;
            break;
        case DIR_RIGHT:
            next.x++;
            break;
    }

    return next;
}

    SnakeNode* current = snake->head->next;  // ë¨¸ë¦¬ ë‹¤ìŒ ë…¸ë“œë¶€í„° í™•ì¸

    SnakeNode* current = snake->head;

    }

    return false;
