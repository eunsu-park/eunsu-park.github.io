static struct termios orig_termios;

// í„°ë¯¸ë„ ì„¤ì • ë³µì›
void disable_raw_mode(void) {
    tcsetattr(STDIN_FILENO, TCSAFLUSH, &orig_termios);

// Raw ëª¨ë“œ í™œì„±í™” (non-blocking ì…ë ¥)
void enable_raw_mode(void) {
    tcgetattr(STDIN_FILENO, &orig_termios);
    atexit(disable_raw_mode);

    struct termios raw = orig_termios;
    raw.c_lflag &= ~(ECHO | ICANON);  // ì—ì½” ë„ê¸°, ë¼ì¸ ë²„í¼ë§ ë„ê¸°
    raw.c_cc[VMIN] = 0;   // ìµœì†Œ ì…ë ¥ ë¬¸ì ìˆ˜ 0
    raw.c_cc[VTIME] = 0;  // íƒ€ì„ì•„ì›ƒ 0 (ì¦‰ì‹œ ë°˜í™˜)

    tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw);
    printf(HIDE_CURSOR);

/**
 * í‚¤ë³´ë“œ ì…ë ¥ ì½ê¸° ë° ë°©í–¥ ë°˜í™˜
 * ESC sequence ì²˜ë¦¬ (ë°©í–¥í‚¤)
 */
Direction read_direction(Direction current) {
    int ch = getchar();
    if (ch == EOF) return current;

    // ESC sequence (ë°©í–¥í‚¤)
    if (ch == '\033') {
        if (getchar() == '[') {
            switch (getchar()) {
                case 'A': return (current != DOWN) ? UP : current;
                case 'B': return (current != UP) ? DOWN : current;
                case 'C': return (current != LEFT) ? RIGHT : current;
                case 'D': return (current != RIGHT) ? LEFT : current;
            }
        }
    }

    // WASD í‚¤
    switch (ch) {
        case 'w': case 'W': return (current != DOWN) ? UP : current;
        case 's': case 'S': return (current != UP) ? DOWN : current;
        case 'a': case 'A': return (current != RIGHT) ? LEFT : current;
        case 'd': case 'D': return (current != LEFT) ? RIGHT : current;
        case 'q': case 'Q': return -1;  // ì¢…ë£Œ ì‹ í˜¸
    }

    return current;
}

// ì¼ì‹œì •ì§€ í‚¤ í™•ì¸
int check_pause_key(void) {
    int ch = getchar();
    if (ch == 'p' || ch == 'P') return 1;
    if (ch == 'q' || ch == 'Q') return -1;
    return 0;
}

/**
 * íŠ¹ì • ìœ„ì¹˜ì— ë±€ì´ ìˆëŠ”ì§€ í™•ì¸
 */
bool snake_at(Node* head, int x, int y) {
    for (Node* n = head; n; n = n->next) {
        if (n->pos.x == x && n->pos.y == y) return true;
    }
    return false;
}

/**
 * ìŒì‹ ìƒì„± (ë±€ê³¼ ê²¹ì¹˜ì§€ ì•ŠëŠ” ìœ„ì¹˜)
 */
void spawn_food(Game* g) {
    do {
        g->food.x = 1 + rand() % (WIDTH - 2);
        g->food.y = 1 + rand() % (HEIGHT - 2);
    } while (snake_at(g->head, g->food.x, g->food.y));
}

/**
 * ê²Œì„ ìƒíƒœ ì´ˆê¸°í™”
 */
Game* game_init(int high_score) {
    Game* g = malloc(sizeof(Game));
    if (!g) return NULL;

    // ë±€ ì´ˆê¸°í™” (ê¸¸ì´ 3)
    g->head = NULL;
    g->tail = NULL;
    g->length = 0;

    for (int i = 0; i < 3; i++) {
        Node* n = malloc(sizeof(Node));
        if (!n) {
            // ë©”ëª¨ë¦¬ í• ë‹¹ ì‹¤íŒ¨ ì‹œ ì •ë¦¬
            while (g->head) {
                Node* temp = g->head;
                g->head = g->head->next;
                free(temp);
            }
            free(g);
            return NULL;
        }

        n->pos.x = WIDTH / 2 - i;
        n->pos.y = HEIGHT / 2;
        n->next = g->head;
        g->head = n;
        g->length++;
    }

    // ê¼¬ë¦¬ ì°¾ê¸°
    Node* curr = g->head;
    while (curr->next) curr = curr->next;
    g->tail = curr;

    // ê²Œì„ ìƒíƒœ ì´ˆê¸°í™”
    g->dir = RIGHT;
    g->score = 0;
    g->game_over = false;
    g->paused = false;
    g->speed = INITIAL_SPEED;
    g->high_score = high_score;

    spawn_food(g);
    return g;
}

/**
 * ê²Œì„ ë©”ëª¨ë¦¬ í•´ì œ
 */
void game_free(Game* g) {
    if (!g) return;

    Node* n = g->head;
    while (n) {
        Node* next = n->next;
        free(n);
        n = next;
    }
    free(g);
}

/**
 * ê²Œì„ ìƒíƒœ ì—…ë°ì´íŠ¸
 * ë°˜í™˜: true = ìŒì‹ì„ ë¨¹ìŒ, false = ë¨¹ì§€ ì•ŠìŒ
 */
bool game_update(Game* g) {
    if (g->paused || g->game_over) return false;

    // ë‹¤ìŒ ë¨¸ë¦¬ ìœ„ì¹˜ ê³„ì‚°
    Point next = g->head->pos;
    switch (g->dir) {
        case UP:    next.y--; break;
        case DOWN:  next.y++; break;
        case LEFT:  next.x--; break;
        case RIGHT: next.x++; break;
    }

    // ë²½ ì¶©ëŒ ê²€ì‚¬
    if (next.x <= 0 || next.x >= WIDTH - 1 ||
        next.y <= 0 || next.y >= HEIGHT - 1) {
        g->game_over = true;
        return false;
    }

    // ìê¸° ëª¸ ì¶©ëŒ ê²€ì‚¬
    if (snake_at(g->head, next.x, next.y)) {
        g->game_over = true;
        return false;
    }

    // ìƒˆ ë¨¸ë¦¬ ì¶”ê°€
    Node* new_head = malloc(sizeof(Node));
    if (!new_head) {
        g->game_over = true;
        return false;
    }

    new_head->pos = next;
    new_head->next = g->head;
    g->head = new_head;
    g->length++;

    // ìŒì‹ í™•ì¸
    if (next.x == g->food.x && next.y == g->food.y) {
        g->score += 10;
        spawn_food(g);

        // ì†ë„ ì¦ê°€ (ì ì  ë¹¨ë¼ì§)
        if (g->speed > MIN_SPEED) {
            g->speed -= 5000;
            if (g->speed < MIN_SPEED) g->speed = MIN_SPEED;
        }

        return true;
    }

    // ìŒì‹ì„ ë¨¹ì§€ ì•Šì•˜ìœ¼ë©´ ê¼¬ë¦¬ ì œê±°
    Node* curr = g->head;
    while (curr->next && curr->next->next) {
        curr = curr->next;
    }
    if (curr->next) {
        free(curr->next);
        curr->next = NULL;
        g->tail = curr;
        g->length--;
    }

    return false;
}

/**

void draw_game(Game* g) {
    printf(CLEAR HOME);

    draw_border();

    // ìŒì‹ ê·¸ë¦¬ê¸°
    MOVE(g->food.y + 1, g->food.x + 1);
    printf(RED "â—" RESET);

    // ë±€ ê·¸ë¦¬ê¸°
    bool is_head = true;
    for (Node* n = g->head; n; n = n->next) {
        MOVE(n->pos.y + 1, n->pos.x + 1);
        if (is_head) {
            printf(BOLD GREEN "â—†" RESET);  // ë¨¸ë¦¬
            is_head = false;
        } else {
            printf(GREEN "â– " RESET);       // ëª¸í†µ
        }

    // ì ìˆ˜ ë° ì •ë³´ í‘œì‹œ
    MOVE(HEIGHT + 1, 1);

           g->score, g->length, g->high_score);

    MOVE(HEIGHT + 2, 1);
    printf("ì¡°ì‘: â†‘â†“â†â†’ ë˜ëŠ” WASD  |  P: ì¼ì‹œì •ì§€  |  Q: ì¢…ë£Œ");

    if (g->paused) {

#define SCORE_FILE ".snake_highscore"

int load_high_score(void) {
    FILE* f = fopen(SCORE_FILE, "r");
    if (!f) return 0;

    int score = 0;
    fscanf(f, "%d", &score);
    fclose(f);
    return score;
}

void save_high_score(int score) {
    FILE* f = fopen(SCORE_FILE, "w");
    if (f) {
        fprintf(f, "%d", score);
        fclose(f);
    }
}

// ============ ë©”ì¸ í•¨ìˆ˜ ============

int main(void) {
    srand(time(NULL));
    enable_raw_mode();

    int high_score = load_high_score();
    Game* game = game_init(high_score);

    if (!game) {
        fprintf(stderr, "ê²Œì„ ì´ˆê¸°í™” ì‹¤íŒ¨\n");
        return 1;
    }

    draw_game(game);

    // ë©”ì¸ ê²Œì„ ë£¨í”„
    while (1) {
        if (!game->game_over) {
            // ì…ë ¥ ì²˜ë¦¬
            Direction new_dir = read_direction(game->dir);

            if (new_dir == (Direction)-1) {
                // Q í‚¤ë¡œ ì¢…ë£Œ
                break;
            }

            game->dir = new_dir;

            // ì¼ì‹œì •ì§€ ì²˜ë¦¬
            int pause_key = check_pause_key();
            if (pause_key == 1) {
                game->paused = !game->paused;
            } else if (pause_key == -1) {
                break;
            }

            // ê²Œì„ ì—…ë°ì´íŠ¸ ë° ê·¸ë¦¬ê¸°
            game_update(game);
            draw_game(game);

            if (game->game_over) {
                // ìµœê³  ì ìˆ˜ ì €ì¥
                if (game->score > game->high_score) {
                    save_high_score(game->score);
                }
                draw_game_over(game);
            }
        } else {
            // ê²Œì„ ì˜¤ë²„ ìƒíƒœì—ì„œ í‚¤ ì…ë ¥ ì²˜ë¦¬
            int ch = getchar();
            if (ch == 'r' || ch == 'R') {
                // ì¬ì‹œì‘
                int final_high = (game->score > game->high_score) ?
                                 game->score : game->high_score;
                game_free(game);
                game = game_init(final_high);
                if (!game) break;
                draw_game(game);
            } else if (ch == 'q' || ch == 'Q') {
                // ì¢…ë£Œ
                break;
            }
        }

        usleep(game->speed);
    }

    game_free(game);

    // í™”ë©´ ì •ë¦¬
    printf(CLEAR HOME SHOW_CURSOR);
    MOVE(1, 1);
    printf("ê²Œì„ì„ ì¢…ë£Œí•©ë‹ˆë‹¤. í”Œë ˆì´í•´ì£¼ì…”ì„œ ê°ì‚¬í•©ë‹ˆë‹¤!\n");

    return 0;
}
