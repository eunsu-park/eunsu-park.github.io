int builtin_echo(char** args) {
    int newline = 1;
    int start = 1;

    // -n ì˜µì…˜: ì¤„ë°”ê¿ˆ ì—†ì´ ì¶œë ¥
    if (args[1] && strcmp(args[1], "-n") == 0) {
        newline = 0;
        start = 2;

    for (int i = start; args[i]; i++) {
        printf("%s", args[i]);
        if (args[i + 1]) printf(" ");
    }

    if (newline) printf("\n");
    return 0;
}

int builtin_export(char** args) {
    if (args[1] == NULL) {
        // ì¸ì ì—†ìœ¼ë©´ í™˜ê²½ë³€ìˆ˜ ëª©ë¡ ì¶œë ¥
        extern char** environ;

    // VAR=value í˜•ì‹ íŒŒì‹±
    for (int i = 1; args[i]; i++) {

int execute_builtin(char** args) {

    if (strcmp(args[0], "cd") == 0) return builtin_cd(args);

    if (strcmp(args[0], "export") == 0) return builtin_export(args);
    if (strcmp(args[0], "env") == 0) return builtin_env(args);

    return -1;  // ë‚´ì¥ ëª…ë ¹ì–´ ì•„ë‹˜
