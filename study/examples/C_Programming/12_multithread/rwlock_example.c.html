// rwlock_example.c
// ÏùΩÍ∏∞-Ïì∞Í∏∞ Ïû†Í∏à (Read-Write Lock) ÏòàÏ†ú
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>
#include <time.h>

#define NUM_READERS 5
#define NUM_WRITERS 2

// Í≥µÏú† Îç∞Ïù¥ÌÑ∞
typedef struct {
    int data;
    pthread_rwlock_t lock;
} SharedData;

SharedData shared = { .data = 0 };

void* reader(void* arg) {
    int id = *(int*)arg;

    for (int i = 0; i < 5; i++) {
        pthread_rwlock_rdlock(&shared.lock);  // ÏùΩÍ∏∞ Ïû†Í∏à

        printf("[ÎèÖÏûê %d] Îç∞Ïù¥ÌÑ∞ ÏùΩÏùå: %d\n", id, shared.data);
        usleep(100000);  // ÏùΩÍ∏∞ Ï§ë...

        pthread_rwlock_unlock(&shared.lock);

        usleep(rand() % 200000);
    }

    return NULL;
}

void* writer(void* arg) {
    int id = *(int*)arg;

    for (int i = 0; i < 3; i++) {
        pthread_rwlock_wrlock(&shared.lock);  // Ïì∞Í∏∞ Ïû†Í∏à (Î∞∞ÌÉÄÏ†Å)

        shared.data = rand() % 1000;
        printf("[ÏûëÍ∞Ä %d] Îç∞Ïù¥ÌÑ∞ ÏîÄ: %d\n", id, shared.data);
        usleep(200000);  // Ïì∞Í∏∞ Ï§ë...

        pthread_rwlock_unlock(&shared.lock);

        usleep(rand() % 500000);
    }

    return NULL;
}

int main(void) {
    srand(time(NULL));

    pthread_rwlock_init(&shared.lock, NULL);

    pthread_t readers[NUM_READERS];
    pthread_t writers[NUM_WRITERS];
    int reader_ids[NUM_READERS];
    int writer_ids[NUM_WRITERS];

    // ÎèÖÏûê ÏÉùÏÑ±
    for (int i = 0; i < NUM_READERS; i++) {
        reader_ids[i] = i;
        pthread_create(&readers[i], NULL, reader, &reader_ids[i]);
    }

    // ÏûëÍ∞Ä ÏÉùÏÑ±
    for (int i = 0; i < NUM_WRITERS; i++) {
        writer_ids[i] = i;
        pthread_create(&writers[i], NULL, writer, &writer_ids[i]);
    }

    // ÎåÄÍ∏∞
    for (int i = 0; i < NUM_READERS; i++) {
        pthread_join(readers[i], NULL);
    }
    for (int i = 0; i < NUM_WRITERS; i++) {
        pthread_join(writers[i], NULL);
    }

    pthread_rwlock_destroy(&shared.lock);
    printf("ÏôÑÎ£å\n");

    return 0;
}
