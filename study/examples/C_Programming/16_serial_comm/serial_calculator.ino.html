16float calculate(float a, char op, float b) {
17    switch (op) {
18        case '+': return a + b;
19        case '-': return a - b;
20        case '*': return a * b;
21        case '/':
22            if (b == 0) {
23                Serial.println("Error: Division by zero");
24                return 0;
25            }
26            return a / b;
27        case '%':
28            return (int)a % (int)b;
29        default:
30            Serial.print("Unknown operator: ");
31            Serial.println(op);
32            return 0;
33    }
34}

36void processExpression(char* expr) {
