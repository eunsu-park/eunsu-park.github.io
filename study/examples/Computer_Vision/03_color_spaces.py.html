import cv2

def create_color_image():
    """ë‹¤ì–‘í•œ ìƒ‰ìƒì˜ í…ŒìŠ¤íŠ¸ ì´ë¯¸ì§€ ìƒì„±"""
    img = np.zeros((300, 400, 3), dtype=np.uint8)

    # ë¬´ì§€ê°œ ìƒ‰ìƒ
    colors = [
        [0, 0, 255],      # ë¹¨ê°•
        [0, 128, 255],    # ì£¼í™©
        [0, 255, 255],    # ë…¸ë‘
        [0, 255, 0],      # ì´ˆë¡
        [255, 255, 0],    # ì²­ë¡
        [255, 0, 0],      # íŒŒë‘
        [255, 0, 128],    # ë³´ë¼
    ]

    width = 400 // len(colors)
    for i, color in enumerate(colors):
        img[:, i*width:(i+1)*width] = color

    return img

def color_extraction_demo():
    """ìƒ‰ìƒ ê¸°ë°˜ ê°ì²´ ì¶”ì¶œ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("ìƒ‰ìƒ ê¸°ë°˜ ê°ì²´ ì¶”ì¶œ")
    print("=" * 50)

    # ë‹¤ì–‘í•œ ìƒ‰ìƒì˜ ì´ë¯¸ì§€ ìƒì„±
    img = np.zeros((300, 400, 3), dtype=np.uint8)
    cv2.circle(img, (100, 150), 50, (0, 0, 255), -1)   # ë¹¨ê°„ ì›
    cv2.circle(img, (200, 150), 50, (0, 255, 0), -1)   # ì´ˆë¡ ì›
    cv2.circle(img, (300, 150), 50, (255, 0, 0), -1)   # íŒŒë€ ì›

    # HSV ë³€í™˜
    hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)

    # ë¹¨ê°„ìƒ‰ ë²”ìœ„ (HSV)
    # ë¹¨ê°„ìƒ‰ì€ H=0 ë˜ëŠ” H=180 ê·¼ì²˜
    lower_red1 = np.array([0, 100, 100])
    upper_red1 = np.array([10, 255, 255])
    lower_red2 = np.array([170, 100, 100])
    upper_red2 = np.array([180, 255, 255])

    # ë§ˆìŠ¤í¬ ìƒì„±
    mask1 = cv2.inRange(hsv, lower_red1, upper_red1)
    mask2 = cv2.inRange(hsv, lower_red2, upper_red2)
    red_mask = cv2.bitwise_or(mask1, mask2)

    # ë¹¨ê°„ìƒ‰ ì¶”ì¶œ
    red_only = cv2.bitwise_and(img, img, mask=red_mask)

    # ì´ˆë¡ìƒ‰ ë²”ìœ„
    lower_green = np.array([40, 100, 100])
    upper_green = np.array([80, 255, 255])
    green_mask = cv2.inRange(hsv, lower_green, upper_green)
    green_only = cv2.bitwise_and(img, img, mask=green_mask)

    # íŒŒë€ìƒ‰ ë²”ìœ„
    lower_blue = np.array([100, 100, 100])
    upper_blue = np.array([130, 255, 255])
    blue_mask = cv2.inRange(hsv, lower_blue, upper_blue)
    blue_only = cv2.bitwise_and(img, img, mask=blue_mask)

    cv2.imwrite('original_circles.jpg', img)
    cv2.imwrite('red_extracted.jpg', red_only)
    cv2.imwrite('green_extracted.jpg', green_only)
    cv2.imwrite('blue_extracted.jpg', blue_only)
    print("ìƒ‰ìƒ ì¶”ì¶œ ì´ë¯¸ì§€ ì €ì¥ ì™„ë£Œ")

def grayscale_methods():
    """ê·¸ë ˆì´ìŠ¤ì¼€ì¼ ë³€í™˜ ë°©ë²•"""
    print("\n" + "=" * 50)
    print("ê·¸ë ˆì´ìŠ¤ì¼€ì¼ ë³€í™˜")
    print("=" * 50)

    img = create_color_image()

    # ë°©ë²• 1: cvtColor
    gray1 = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # ë°©ë²• 2: ê°€ì¤‘ì¹˜ í‰ê·  (ì§ì ‘ ê³„ì‚°)
    # Gray = 0.299*R + 0.587*G + 0.114*B
    b, g, r = cv2.split(img)
    gray2 = (0.299 * r + 0.587 * g + 0.114 * b).astype(np.uint8)

    # ë°©ë²• 3: ë‹¨ìˆœ í‰ê· 
    gray3 = np.mean(img, axis=2).astype(np.uint8)

    cv2.imwrite('gray_cvtcolor.jpg', gray1)
    cv2.imwrite('gray_weighted.jpg', gray2)
    cv2.imwrite('gray_average.jpg', gray3)
    print("ê·¸ë ˆì´ìŠ¤ì¼€ì¼ ë³€í™˜ ì´ë¯¸ì§€ ì €ì¥ ì™„ë£Œ")

    print(f"\në³€í™˜ ê²°ê³¼ ë¹„êµ (íŠ¹ì • í”½ì…€):")
    print(f"  cvtColor: {gray1[150, 25]}")
    print(f"  ê°€ì¤‘ì¹˜: {gray2[150, 25]}")
    print(f"  í‰ê· : {gray3[150, 25]}")
