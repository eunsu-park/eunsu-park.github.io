import cv2
import numpy as np


def create_test_image():
    """í…ŒìŠ¤íŠ¸ ì´ë¯¸ì§€ ìƒì„±"""
    img = np.zeros((300, 400, 3), dtype=np.uint8)
    img[:] = [200, 200, 200]  # ë°ì€ íšŒìƒ‰ ë°°ê²½

    # ì‚¬ê°í˜•
    cv2.rectangle(img, (50, 50), (150, 150), (0, 0, 255), -1)

    # í…ìŠ¤íŠ¸
    cv2.putText(img, 'OpenCV', (180, 100),
                cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 0, 0), 2)

    # ì›
    cv2.circle(img, (300, 200), 50, (255, 0, 0), -1)

    return img


def resize_demo():
    """í¬ê¸° ì¡°ì • ë°ëª¨"""
    print("=" * 50)
    print("í¬ê¸° ì¡°ì • (resize)")
    print("=" * 50)

    img = create_test_image()
    h, w = img.shape[:2]
    print(f"ì›ë³¸ í¬ê¸°: {w}x{h}")

    # ì ˆëŒ€ í¬ê¸°ë¡œ ì¡°ì •
    resized1 = cv2.resize(img, (200, 150))  # (width, height)
    print(f"ì ˆëŒ€ í¬ê¸° ì¡°ì •: {resized1.shape[1]}x{resized1.shape[0]}")

    # ë¹„ìœ¨ë¡œ ì¡°ì •
    resized2 = cv2.resize(img, None, fx=0.5, fy=0.5)
    print(f"50% ì¶•ì†Œ: {resized2.shape[1]}x{resized2.shape[0]}")

    resized3 = cv2.resize(img, None, fx=2, fy=2)
    print(f"200% í™•ëŒ€: {resized3.shape[1]}x{resized3.shape[0]}")

    # ë³´ê°„ë²• ë¹„êµ
    print("\në³´ê°„ë²• ì¢…ë¥˜:")
    print("  cv2.INTER_NEAREST: ìµœê·¼ì ‘ ì´ì›ƒ (ë¹ ë¦„, í’ˆì§ˆ ë‚®ìŒ)")
    print("  cv2.INTER_LINEAR: ì–‘ì„ í˜• (ê¸°ë³¸ê°’, ê· í˜•)")
    print("  cv2.INTER_CUBIC: ë°”ì´íë¹… (í’ˆì§ˆ ì¢‹ìŒ, ëŠë¦¼)")
    print("  cv2.INTER_AREA: ì˜ì—­ (ì¶•ì†Œì— ì í•©)")

    # ê° ë³´ê°„ë²• ì ìš©
    enlarged_nearest = cv2.resize(img, (800, 600), interpolation=cv2.INTER_NEAREST)
    enlarged_linear = cv2.resize(img, (800, 600), interpolation=cv2.INTER_LINEAR)
    enlarged_cubic = cv2.resize(img, (800, 600), interpolation=cv2.INTER_CUBIC)

    cv2.imwrite('resize_half.jpg', resized2)
    cv2.imwrite('resize_double.jpg', resized3)
    cv2.imwrite('resize_nearest.jpg', enlarged_nearest)
    cv2.imwrite('resize_cubic.jpg', enlarged_cubic)
    print("\ní¬ê¸° ì¡°ì • ì´ë¯¸ì§€ ì €ì¥ ì™„ë£Œ")


def rotate_demo():
    """íšŒì „ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("íšŒì „ (rotate)")
    print("=" * 50)

    img = create_test_image()

    # ê°„ë‹¨í•œ 90ë„ íšŒì „ (ë‚´ì¥ í•¨ìˆ˜)
    rotated_90 = cv2.rotate(img, cv2.ROTATE_90_CLOCKWISE)
    rotated_180 = cv2.rotate(img, cv2.ROTATE_180)
    rotated_270 = cv2.rotate(img, cv2.ROTATE_90_COUNTERCLOCKWISE)

    print("ê°„ë‹¨í•œ íšŒì „:")
    print(f"  90ë„ ì‹œê³„ë°©í–¥: {rotated_90.shape[1]}x{rotated_90.shape[0]}")
    print(f"  180ë„: {rotated_180.shape[1]}x{rotated_180.shape[0]}")
    print(f"  270ë„(ë°˜ì‹œê³„): {rotated_270.shape[1]}x{rotated_270.shape[0]}")

    cv2.imwrite('rotate_90.jpg', rotated_90)
    cv2.imwrite('rotate_180.jpg', rotated_180)
    cv2.imwrite('rotate_270.jpg', rotated_270)

    # ì„ì˜ ê°ë„ íšŒì „ (getRotationMatrix2D ì‚¬ìš©)
    h, w = img.shape[:2]
    center = (w // 2, h // 2)

    # 45ë„ íšŒì „, ìŠ¤ì¼€ì¼ 1.0
    M = cv2.getRotationMatrix2D(center, 45, 1.0)
    rotated_45 = cv2.warpAffine(img, M, (w, h))

    # 30ë„ íšŒì „, ìŠ¤ì¼€ì¼ 0.8
    M = cv2.getRotationMatrix2D(center, 30, 0.8)
    rotated_30_scaled = cv2.warpAffine(img, M, (w, h))

    cv2.imwrite('rotate_45.jpg', rotated_45)
    cv2.imwrite('rotate_30_scaled.jpg', rotated_30_scaled)
    print("\níšŒì „ ì´ë¯¸ì§€ ì €ì¥ ì™„ë£Œ")


def flip_demo():
    """ë’¤ì§‘ê¸° ë°ëª¨"""

    print("\në’¤ì§‘ê¸° ì´ë¯¸ì§€ ì €ì¥ ì™„ë£Œ")


def translation_demo():
    """ì´ë™ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("ì´ë™ (Translation)")
    print("=" * 50)

    img = create_test_image()

    # ì´ë™ í–‰ë ¬: [[1, 0, tx], [0, 1, ty]]

def affine_transform_demo():
    """ì–´íŒŒì¸ ë³€í™˜ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("ì–´íŒŒì¸ ë³€í™˜ (Affine Transform)")
    print("=" * 50)

    img = create_test_image()
    h, w = img.shape[:2]

    # ì–´íŒŒì¸ ë³€í™˜: 3ê°œì˜ ì  ëŒ€ì‘
    # ì›ë³¸ ì´ë¯¸ì§€ì˜ 3ì

def perspective_transform_demo():
    """ì›ê·¼ ë³€í™˜ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("ì›ê·¼ ë³€í™˜ (Perspective Transform)")
    print("=" * 50)

    img = create_test_image()
    h, w = img.shape[:2]

    # ì›ê·¼ ë³€í™˜: 4ê°œì˜ ì  ëŒ€ì‘
    # ì›ë³¸ ì´ë¯¸ì§€ì˜ 4ì  (ì‚¬ê°í˜• ëª¨ì„œë¦¬)
    pts1 = np.float32([[0, 0], [w-1, 0], [0, h-1], [w-1, h-1]])

    # ë³€í™˜ í›„ 4ì  (ì›ê·¼ê° ì ìš©)
    pts2 = np.float32([[50, 50], [w-50, 20], [30, h-30], [w-30, h-50]])

    # ë³€í™˜ í–‰ë ¬ ê³„ì‚°
    M = cv2.getPerspectiveTransform(pts1, pts2)

    # ë³€í™˜ ì ìš©
    perspective = cv2.warpPerspective(img, M, (w, h))

    print("ì›ê·¼ ë³€í™˜ íŠ¹ì„±:")
    print("  - í‰í–‰ì„ ì´ í•œ ì ì—ì„œ ë§Œë‚¨ (ì›ê·¼ê°)")
    print("  - 4ì  ëŒ€ì‘ìœ¼ë¡œ ì •ì˜")
    print("  - ë¬¸ì„œ ìŠ¤ìº” ë³´ì •ì— í™œìš©")

    cv2.imwrite('perspective.jpg', perspective)

    # ì—­ë³€í™˜ (ë¬¸ì„œ ë³´ì • ì‹œë®¬ë ˆì´ì…˜)
    pts_doc = np.float32([[50, 50], [350, 30], [60, 280], [340, 270]])
    pts_rect = np.float32([[0, 0], [300, 0], [0, 200], [300, 200]])

    M_rect = cv2.getPerspectiveTransform(pts_doc, pts_rect)
    rectified = cv2.warpPerspective(img, M_rect, (300, 200))

    cv2.imwrite('perspective_rectified.jpg', rectified)
    print("ì›ê·¼ ë³€í™˜ ì´ë¯¸ì§€ ì €ì¥ ì™„ë£Œ")


def combined_transforms_demo():

    print("\n" + "=" * 50)
    print("ë³µí•© ë³€í™˜")

    h, w = img.shape[:2]

    # ì´ë™ -> íšŒì „ -> ìŠ¤ì¼€ì¼ì„ í•œ ë²ˆì—

    cv2.imwrite('combined_transform.jpg', result)
