192def auto_canny_threshold():
193    """ìë™ Canny ì„ê³„ê°’"""
194    print("\n" + "=" * 50)
195    print("ìë™ Canny ì„ê³„ê°’")
196    print("=" * 50)
197
198    img = create_test_image()
199
200    # ì¤‘ì•™ê°’ ê¸°ë°˜ ìë™ ì„ê³„ê°’
201    median = np.median(img)
202    sigma = 0.33
203
204    lower = int(max(0, (1.0 - sigma) * median))
205    upper = int(min(255, (1.0 + sigma) * median))
206
207    auto_canny = cv2.Canny(img, lower, upper)
208
209    print(f"ì´ë¯¸ì§€ ì¤‘ì•™ê°’: {median}")
210    print(f"ìë™ ì„ê³„ê°’: lower={lower}, upper={upper}")
211    print(f"ê³µì‹: lower = (1-sigma)*median, upper = (1+sigma)*median")
212
213    cv2.imwrite('canny_auto.jpg', auto_canny)
214
215    return lower, upper
