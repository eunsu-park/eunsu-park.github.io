"""
09. Ïú§Í≥ΩÏÑ† Í≤ÄÏ∂ú
- findContours, drawContours

import cv2
import numpy as np


def create_shapes_image():
    """ÎèÑÌòïÏù¥ ÏûàÎäî Ïù¥ÏßÑ Ïù¥ÎØ∏ÏßÄ"""
    img = np.zeros((400, 500), dtype=np.uint8)

    # ÏÇ¨Í∞ÅÌòï
    cv2.rectangle(img, (50, 50), (150, 150), 255, -1)

    # Ïõê
    cv2.circle(img, (300, 100), 50, 255, -1)

    # ÏÇºÍ∞ÅÌòï
    pts = np.array([[400, 50], [350, 150], [450, 150]], np.int32)
    cv2.fillPoly(img, [pts], 255)

    # Ï§ëÏ≤©Îêú ÏÇ¨Í∞ÅÌòï (Í≥ÑÏ∏µ Íµ¨Ï°∞)
    cv2.rectangle(img, (50, 200), (200, 350), 255, -1)
    cv2.rectangle(img, (80, 230), (170, 320), 0, -1)  # ÎÇ¥Î∂Ä Íµ¨Î©ç
    cv2.rectangle(img, (100, 250), (150, 290), 255, -1)  # Íµ¨Î©ç ÏïàÏùò Í∞ùÏ≤¥

    # Î≥Ñ Î™®Ïñë
    pts_star = np.array([
        [350, 200], [365, 250], [420, 250], [375, 280],
        [390, 330], [350, 300], [310, 330], [325, 280],
        [280, 250], [335, 250]
    ], np.int32)
    cv2.fillPoly(img, [pts_star], 255)

    return img


def find_contours_demo():
    """Ïú§Í≥ΩÏÑ† Ï∞æÍ∏∞ Îç∞Î™®"""
    print("=" * 50)
    print("Ïú§Í≥ΩÏÑ† Ï∞æÍ∏∞ (findContours)")
    print("=" * 50)

    img = create_shapes_image()

    # Ïú§Í≥ΩÏÑ† Ï∞æÍ∏∞
    # RETR_EXTERNAL: Ïô∏Î∂Ä Ïú§Í≥ΩÏÑ†Îßå
    # RETR_LIST: Î™®Îì† Ïú§Í≥ΩÏÑ† (Í≥ÑÏ∏µ Î¨¥Ïãú)
    # RETR_TREE: Î™®Îì† Ïú§Í≥ΩÏÑ† (Í≥ÑÏ∏µ Ìè¨Ìï®)
    # RETR_CCOMP: 2Îã®Í≥Ñ Í≥ÑÏ∏µ

    contours, hierarchy = cv2.findContours(
        img, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE
    )

    print(f"Í≤ÄÏ∂úÎêú Ïú§Í≥ΩÏÑ† Ïàò: {len(contours)}")
    print(f"Í≥ÑÏ∏µ Íµ¨Ï°∞ shape: {hierarchy.shape if hierarchy is not None else None}")

    # Í≤ÄÏÉâ Î™®Îìú ÏÑ§Î™Ö
    print("\nÍ≤ÄÏÉâ Î™®Îìú (Retrieval Mode):")
    print("  RETR_EXTERNAL: Í∞ÄÏû• Î∞îÍπ• Ïú§Í≥ΩÏÑ†Îßå")
    print("  RETR_LIST: Î™®Îì† Ïú§Í≥ΩÏÑ† (ÌèâÌèâÌïòÍ≤å)")
    print("  RETR_TREE: Ï†ÑÏ≤¥ Í≥ÑÏ∏µ Íµ¨Ï°∞")
    print("  RETR_CCOMP: 2Îã®Í≥Ñ Í≥ÑÏ∏µ")

    return img, contours, hierarchy


def draw_contours_demo():
    """Ïú§Í≥ΩÏÑ† Í∑∏Î¶¨Í∏∞ Îç∞Î™®"""
    print("\n" + "=" * 50)
    print("Ïú§Í≥ΩÏÑ† Í∑∏Î¶¨Í∏∞ (drawContours)")
    print("=" * 50)

    img = create_shapes_image()
    contours, _ = cv2.findContours(img, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)

    # Ïª¨Îü¨ Ïù¥ÎØ∏ÏßÄÎ°ú Î≥ÄÌôò (Í∑∏Î¶¨Í∏∞Ïö©)
    color_img = cv2.cvtColor(img, cv2.COLOR_GRAY2BGR)

    # Î™®Îì† Ïú§Í≥ΩÏÑ† Í∑∏Î¶¨Í∏∞
    all_contours = color_img.copy()
    cv2.drawContours(all_contours, contours, -1, (0, 255, 0), 2)

    # ÌäπÏ†ï Ïú§Í≥ΩÏÑ†Îßå Í∑∏Î¶¨Í∏∞
    specific = color_img.copy()
    cv2.drawContours(specific, contours, 0, (255, 0, 0), 2)  # Ï≤´ Î≤àÏß∏ Ïú§Í≥ΩÏÑ†
    cv2.drawContours(specific, contours, 1, (0, 255, 0), 2)  # Îëê Î≤àÏß∏ Ïú§Í≥ΩÏÑ†

    # Ï±ÑÏö∞Í∏∞
    filled = color_img.copy()
    cv2.drawContours(filled, contours, 0, (0, 0, 255), -1)  # thickness=-1 ‚Üí Ï±ÑÏö∞Í∏∞

    print("drawContours ÌååÎùºÎØ∏ÌÑ∞:")
    print("  contourIdx=-1: Î™®Îì† Ïú§Í≥ΩÏÑ†")
    print("  contourIdx=n: nÎ≤àÏß∏ Ïú§Í≥ΩÏÑ†Îßå")
    print("  thickness=-1: ÎÇ¥Î∂Ä Ï±ÑÏö∞Í∏∞")

    cv2.imwrite('contours_original.jpg', img)
    cv2.imwrite('contours_all.jpg', all_contours)
    cv2.imwrite('contours_specific.jpg', specific)
    cv2.imwrite('contours_filled.jpg', filled)


def contour_hierarchy_demo():
    """Ïú§Í≥ΩÏÑ† Í≥ÑÏ∏µ Íµ¨Ï°∞ Îç∞Î™®"""
    print("\n" + "=" * 50)
    print("Ïú§Í≥ΩÏÑ† Í≥ÑÏ∏µ Íµ¨Ï°∞")
    print("=" * 50)

    img = create_shapes_image()
    contours, hierarchy = cv2.findContours(img, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)

    # hierarchy[i] = [Next, Previous, First_Child, Parent]
    print("Í≥ÑÏ∏µ Íµ¨Ï°∞ ÌòïÏãù: [Next, Previous, First_Child, Parent]")
    print("(-1ÏùÄ Ìï¥Îãπ Í¥ÄÍ≥Ñ ÏóÜÏùå)")

    for i, h in enumerate(hierarchy[0]):
        print(f"Ïú§Í≥ΩÏÑ† {i}: Next={h[0]}, Prev={h[1]}, Child={h[2]}, Parent={h[3]}")

    # Ïô∏Î∂Ä Ïú§Í≥ΩÏÑ†Îßå (Î∂ÄÎ™®Í∞Ä ÏóÜÎäî Í≤É)
    color_img = cv2.cvtColor(img, cv2.COLOR_GRAY2BGR)
    external_only = color_img.copy()

    for i, h in enumerate(hierarchy[0]):
        if h[3] == -1:  # Î∂ÄÎ™®Í∞Ä ÏóÜÏùå = Ïô∏Î∂Ä Ïú§Í≥ΩÏÑ†
            cv2.drawContours(external_only, contours, i, (0, 255, 0), 2)

    cv2.imwrite('contours_external.jpg', external_only)


def contour_approximation_demo():
    """Ïú§Í≥ΩÏÑ† Í∑ºÏÇ¨ Îç∞Î™®"""
    print("\n" + "=" * 50)
    print("Ïú§Í≥ΩÏÑ† Í∑ºÏÇ¨ (approxPolyDP)")
    print("=" * 50)

    # Î≥µÏû°Ìïú Í≥°ÏÑ† Ïù¥ÎØ∏ÏßÄ
    img = np.zeros((300, 400), dtype=np.uint8)
    pts = []
    for angle in range(0, 360, 5):
        r = 80 + 20 * np.sin(5 * np.radians(angle))
        x = int(200 + r * np.cos(np.radians(angle)))
        y = int(150 + r * np.sin(np.radians(angle)))
        pts.append([x, y])
    pts = np.array(pts, np.int32)
    cv2.fillPoly(img, [pts], 255)

    contours, _ = cv2.findContours(img, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

    # ÏõêÎ≥∏ Ïú§Í≥ΩÏÑ†
    color_img = cv2.cvtColor(img, cv2.COLOR_GRAY2BGR)

    # Îã§ÏñëÌïú epsilon Í∞íÏúºÎ°ú Í∑ºÏÇ¨
    epsilons = [0.01, 0.02, 0.05, 0.1]

    for eps in epsilons:
        approx_img = color_img.copy()
        for cnt in contours:
            # epsilon = ÎëòÎ†àÏùò ÎπÑÏú®
            epsilon = eps * cv2.arcLength(cnt, True)
            approx = cv2.approxPolyDP(cnt, epsilon, True)

            print(f"epsilon={eps}: {len(cnt)} Ï†ê ‚Üí {len(approx)} Ï†ê")

            cv2.drawContours(approx_img, [approx], -1, (0, 255, 0), 2)
            # Íº≠ÏßìÏ†ê ÌëúÏãú
            for pt in approx:
                cv2.circle(approx_img, tuple(pt[0]), 3, (0, 0, 255), -1)

        cv2.imwrite(f'approx_eps_{eps}.jpg', approx_img)


def contour_properties_demo():
    """Ïú§Í≥ΩÏÑ† ÏÜçÏÑ± Îç∞Î™®"""
    print("\n" + "=" * 50)
    print("Ïú§Í≥ΩÏÑ† ÏÜçÏÑ±")
    print("=" * 50)

    img = create_shapes_image()
    contours, _ = cv2.findContours(img, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

    for i, cnt in enumerate(contours):
        # Î©¥Ï†Å
        area = cv2.contourArea(cnt)

        # ÎëòÎ†à (Ìò∏ Í∏∏Ïù¥)
        perimeter = cv2.arcLength(cnt, True)

        # Í≤ΩÍ≥Ñ ÏÇ¨Í∞ÅÌòï
        x, y, w, h = cv2.boundingRect(cnt)

        # Ï§ëÏã¨Ï†ê (Î™®Î©òÌä∏)
        M = cv2.moments(cnt)
        if M['m00'] != 0:
            cx = int(M['m10'] / M['m00'])
            cy = int(M['m01'] / M['m00'])
        else:
            cx, cy = 0, 0

        print(f"\nÏú§Í≥ΩÏÑ† {i}:")
        print(f"  Î©¥Ï†Å: {area:.1f}")
        print(f"  ÎëòÎ†à: {perimeter:.1f}")
        print(f"  Í≤ΩÍ≥Ñ ÏÇ¨Í∞ÅÌòï: ({x}, {y}, {w}, {h})")
        print(f"  Ï§ëÏã¨: ({cx}, {cy})")


def detect_shapes():
    """ÎèÑÌòï Ïù∏Ïãù"""
    print("\n" + "=" * 50)
    print("ÎèÑÌòï Ïù∏Ïãù")
    print("=" * 50)

    img = create_shapes_image()
    contours, _ = cv2.findContours(img, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

    color_img = cv2.cvtColor(img, cv2.COLOR_GRAY2BGR)

    for cnt in contours:
        # Ïú§Í≥ΩÏÑ† Í∑ºÏÇ¨
        epsilon = 0.04 * cv2.arcLength(cnt, True)
        approx = cv2.approxPolyDP(cnt, epsilon, True)

        # Ï§ëÏã¨Ï†ê Í≥ÑÏÇ∞
        M = cv2.moments(cnt)
        if M['m00'] != 0:
            cx = int(M['m10'] / M['m00'])
            cy = int(M['m01'] / M['m00'])
        else:
            continue

        # Íº≠ÏßìÏ†ê ÏàòÎ°ú ÎèÑÌòï ÌåêÎ≥Ñ
        vertices = len(approx)

        if vertices == 3:
            shape = "Triangle"
        elif vertices == 4:
            # Ï†ïÏÇ¨Í∞ÅÌòï vs ÏßÅÏÇ¨Í∞ÅÌòï
            x, y, w, h = cv2.boundingRect(approx)
            ar = w / float(h)
            shape = "Square" if 0.95 <= ar <= 1.05 else "Rectangle"
        elif vertices == 5:
            shape = "Pentagon"
        elif vertices > 5:
            # Ïõê ÌåêÎ≥Ñ (Î©¥Ï†Å/ÎëòÎ†à ÎπÑÏú®)
            area = cv2.contourArea(cnt)
            perimeter = cv2.arcLength(cnt, True)
            circularity = 4 * np.pi * area / (perimeter ** 2)
            shape = "Circle" if circularity > 0.8 else f"Polygon({vertices})"
        else:
            shape = f"Polygon({vertices})"

        # Í≤∞Í≥º ÌëúÏãú
        cv2.drawContours(color_img, [approx], -1, (0, 255, 0), 2)
        cv2.putText(color_img, shape, (cx - 30, cy),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 0, 0), 2)

        print(f"  {shape}: Íº≠ÏßìÏ†ê {vertices}Í∞ú")

    cv2.imwrite('shapes_detected.jpg', color_img)

    """Î©îÏù∏ Ìï®Ïàò"""

    contour_approximation_demo()

    contour_properties_demo()

    # ÎèÑÌòï Ïù∏Ïãù
    detect_shapes()

    print("\nÏú§Í≥ΩÏÑ† Í≤ÄÏ∂ú Îç∞Î™® ÏôÑÎ£å!")
