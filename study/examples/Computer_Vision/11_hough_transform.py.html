import cv2
import numpy as np


def create_line_image():
    """ì§ì„ ì´ ìˆëŠ” í…ŒìŠ¤íŠ¸ ì´ë¯¸ì§€"""
    img = np.zeros((400, 500, 3), dtype=np.uint8)
    img[:] = [200, 200, 200]

    # ë‹¤ì–‘í•œ ì§ì„ 
    cv2.line(img, (50, 50), (450, 50), (0, 0, 0), 2)      # ìˆ˜í‰ì„ 
    cv2.line(img, (50, 50), (50, 350), (0, 0, 0), 2)      # ìˆ˜ì§ì„ 
    cv2.line(img, (100, 100), (400, 300), (0, 0, 0), 2)   # ëŒ€ê°ì„ 
    cv2.line(img, (100, 300), (400, 100), (0, 0, 0), 2)   # ëŒ€ê°ì„ 
    cv2.line(img, (250, 150), (250, 350), (0, 0, 0), 2)   # ìˆ˜ì§ì„ 

    return img


def create_circle_image():
    """ì›ì´ ìˆëŠ” í…ŒìŠ¤íŠ¸ ì´ë¯¸ì§€"""
    img = np.zeros((400, 500, 3), dtype=np.uint8)
    img[:] = [200, 200, 200]

    # ë‹¤ì–‘í•œ ì›
    cv2.circle(img, (100, 100), 40, (0, 0, 0), 2)
    cv2.circle(img, (300, 100), 50, (0, 0, 0), 2)
    cv2.circle(img, (150, 250), 60, (0, 0, 0), 2)
    cv2.circle(img, (350, 280), 45, (0, 0, 0), 2)

    # ì±„ì›Œì§„ ì›
    cv2.circle(img, (450, 350), 30, (0, 0, 0), -1)

    return img


def hough_lines_demo():
    """í‘œì¤€ í—ˆí”„ ì§ì„  ë³€í™˜ ë°ëª¨"""
    print("=" * 50)
    print("í‘œì¤€ í—ˆí”„ ì§ì„  ë³€í™˜ (HoughLines)")
    print("=" * 50)

    img = create_line_image()
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # ì—£ì§€ ê²€ì¶œ
    edges = cv2.Canny(gray, 50, 150)

    # í—ˆí”„ ì§ì„  ë³€í™˜
    # rho: ê±°ë¦¬ í•´ìƒë„ (í”½ì…€)
    # theta: ê°ë„ í•´ìƒë„ (ë¼ë””ì•ˆ)
    # threshold: ìµœì†Œ íˆ¬í‘œ ìˆ˜
    lines = cv2.HoughLines(edges, 1, np.pi/180, 100)

    result = img.copy()

    if lines is not None:
        print(f"ê²€ì¶œëœ ì§ì„  ìˆ˜: {len(lines)}")

        for line in lines:
            rho, theta = line[0]

            # ê·¹ì¢Œí‘œ â†’ ì§êµì¢Œí‘œ ë³€í™˜
            a = np.cos(theta)
            b = np.sin(theta)
            x0 = a * rho
            y0 = b * rho

            # ì§ì„  ê·¸ë¦¬ê¸° (ì¶©ë¶„íˆ ê¸´ ì„ ë¶„)
            x1 = int(x0 + 1000 * (-b))
            y1 = int(y0 + 1000 * (a))
            x2 = int(x0 - 1000 * (-b))
            y2 = int(y0 - 1000 * (a))

            cv2.line(result, (x1, y1), (x2, y2), (0, 0, 255), 2)

    print("\nHoughLines íŒŒë¼ë¯¸í„°:")
    print("  rho: ê±°ë¦¬ í•´ìƒë„ (ë³´í†µ 1 í”½ì…€)")
    print("  theta: ê°ë„ í•´ìƒë„ (ë³´í†µ Ï€/180)")
    print("  threshold: ì§ì„ ìœ¼ë¡œ íŒë‹¨í•  ìµœì†Œ íˆ¬í‘œ ìˆ˜")

    cv2.imwrite('hough_lines_input.jpg', img)
    cv2.imwrite('hough_lines_edges.jpg', edges)
    cv2.imwrite('hough_lines_result.jpg', result)

def hough_lines_params_demo():
    """í—ˆí”„ ì§ì„  íŒŒë¼ë¯¸í„° ì˜í–¥"""
    print("\n" + "=" * 50)
    print("í—ˆí”„ ì§ì„  íŒŒë¼ë¯¸í„° ì˜í–¥")
    print("=" * 50)

    img = create_line_image()
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    edges = cv2.Canny(gray, 50, 150)

    # ë‹¤ì–‘í•œ threshold ê°’
    thresholds = [30, 50, 100, 150]

    for thresh in thresholds:
        lines = cv2.HoughLinesP(edges, 1, np.pi/180, thresh,

        if lines is not None:
            for line in lines:
                x1, y1, x2, y2 = line[0]
                cv2.line(result, (x1, y1), (x2, y2), (0, 255, 0), 2)

            print(f"threshold={thresh}: {len(lines)} ì„ ë¶„ ê²€ì¶œ")

        cv2.imwrite(f'hough_thresh_{thresh}.jpg', result)


def hough_circles_demo():
    """í—ˆí”„ ì› ë³€í™˜ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("í—ˆí”„ ì› ë³€í™˜ (HoughCircles)")
    print("=" * 50)

    img = create_circle_image()
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # ë¸”ëŸ¬ ì ìš© (ë…¸ì´ì¦ˆ ê°ì†Œ)
    blurred = cv2.GaussianBlur(gray, (9, 9), 2)

    # í—ˆí”„ ì› ë³€í™˜
    # cv2.HOUGH_GRADIENT: í—ˆí”„ ê·¸ë˜ë””ì–¸íŠ¸ ë°©ë²•
    # dp: ëˆ„ì  ë°°ì—´ í•´ìƒë„ ë¹„ìœ¨ (1 = ì…ë ¥ê³¼ ë™ì¼)
    # minDist: ì› ì¤‘ì‹¬ ê°„ ìµœì†Œ ê±°ë¦¬
    # param1: Canny ì—£ì§€ ê²€ì¶œ ìƒìœ„ ì„ê³„ê°’
    # param2: ì› ê²€ì¶œ ì„ê³„ê°’ (ë‚®ì„ìˆ˜ë¡ ë” ë§ì€ ì› ê²€ì¶œ)
    # minRadius, maxRadius: ì› ë°˜ì§€ë¦„ ë²”ìœ„

    circles = cv2.HoughCircles(
        blurred,
        cv2.HOUGH_GRADIENT,
        dp=1,
        minDist=50,
        param1=100,
        param2=30,
        minRadius=20,
        maxRadius=100
    )

    result = img.copy()

    if circles is not None:
        circles = np.uint16(np.around(circles))
        print(f"ê²€ì¶œëœ ì› ìˆ˜: {len(circles[0])}")

        for circle in circles[0, :]:
            cx, cy, r = circle

            # ì› ê·¸ë¦¬ê¸°
            cv2.circle(result, (cx, cy), r, (0, 255, 0), 2)
            # ì¤‘ì‹¬ì  í‘œì‹œ
            cv2.circle(result, (cx, cy), 3, (0, 0, 255), -1)

            print(f"  ì›: ì¤‘ì‹¬=({cx}, {cy}), ë°˜ì§€ë¦„={r}")

    print("\nHoughCircles íŒŒë¼ë¯¸í„°:")
    print("  dp: ëˆ„ì  ë°°ì—´ í•´ìƒë„ (1 ê¶Œì¥)")
    print("  minDist: ì› ì¤‘ì‹¬ ê°„ ìµœì†Œ ê±°ë¦¬")
    print("  param1: Canny ìƒìœ„ ì„ê³„ê°’")
    print("  param2: ì› ê²€ì¶œ ì„ê³„ê°’ (ë‚®ìœ¼ë©´ ë§ì´ ê²€ì¶œ)")
    print("  minRadius/maxRadius: ë°˜ì§€ë¦„ ë²”ìœ„")

    cv2.imwrite('hough_circles_input.jpg', img)
    cv2.imwrite('hough_circles_result.jpg', result)

    for p2 in param2_values:

        result = img.copy()

        print(f"param2={p2}: {count} ì› ê²€ì¶œ")
        cv2.imwrite(f'hough_circles_p2_{p2}.jpg', result)


def practical_lane_detection():
    """ì‹¤ìš© ì˜ˆì œ: ì°¨ì„  ê²€ì¶œ ì‹œë®¬ë ˆì´ì…˜"""
    print("\n" + "=" * 50)
    print("ì‹¤ìš© ì˜ˆì œ: ì°¨ì„  ê²€ì¶œ")
    print("=" * 50)

    # ë„ë¡œ ì´ë¯¸ì§€ ì‹œë®¬ë ˆì´ì…˜
    img = np.zeros((400, 600, 3), dtype=np.uint8)
    img[:] = [100, 100, 100]  # íšŒìƒ‰ ë„ë¡œ

    # ì°¨ì„  ê·¸ë¦¬ê¸°
    cv2.line(img, (100, 400), (250, 200), (255, 255, 255), 5)  # ì™¼ìª½ ì°¨ì„ 
    cv2.line(img, (500, 400), (350, 200), (255, 255, 255), 5)  # ì˜¤ë¥¸ìª½ ì°¨ì„ 
    cv2.line(img, (300, 400), (300, 250), (255, 255, 0), 3)    # ì¤‘ì•™ì„  (ì ì„  ì—­í• )

    # ê·¸ë ˆì´ìŠ¤ì¼€ì¼ ë° ì—£ì§€
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    edges = cv2.Canny(gray, 50, 150)

    # ROI (Region of Interest) ë§ˆìŠ¤í¬
    mask = np.zeros_like(edges)
    roi_vertices = np.array([[(50, 400), (550, 400), (350, 180), (250, 180)]], np.int32)
    cv2.fillPoly(mask, roi_vertices, 255)
    masked_edges = cv2.bitwise_and(edges, mask)

    # í—ˆí”„ ì§ì„  ê²€ì¶œ
    lines = cv2.HoughLinesP(masked_edges, 1, np.pi/180, 30,
                            minLineLength=50, maxLineGap=100)

    result = img.copy()

    if lines is not None:
        left_lines = []
        right_lines = []

        for line in lines:
            x1, y1, x2, y2 = line[0]

            # ê¸°ìš¸ê¸° ê³„ì‚°
            if x2 - x1 != 0:
                slope = (y2 - y1) / (x2 - x1)

                # ê¸°ìš¸ê¸°ë¡œ ì¢Œ/ìš° ì°¨ì„  ë¶„ë¥˜
                if slope < -0.3:  # ì™¼ìª½ ì°¨ì„ 
                    left_lines.append(line[0])
                elif slope > 0.3:  # ì˜¤ë¥¸ìª½ ì°¨ì„ 
                    right_lines.append(line[0])

        # ì°¨ì„  ê·¸ë¦¬ê¸°
        for line in left_lines:
            cv2.line(result, (line[0], line[1]), (line[2], line[3]), (0, 255, 0), 3)
        for line in right_lines:
            cv2.line(result, (line[0], line[1]), (line[2], line[3]), (0, 0, 255), 3)

        print(f"ì™¼ìª½ ì°¨ì„ : {len(left_lines)}, ì˜¤ë¥¸ìª½ ì°¨ì„ : {len(right_lines)}")

    cv2.imwrite('lane_input.jpg', img)
    cv2.imwrite('lane_edges.jpg', masked_edges)
    cv2.imwrite('lane_result.jpg', result)
    print("ì°¨ì„  ê²€ì¶œ ì´ë¯¸ì§€ ì €ì¥ ì™„ë£Œ")
