import cv2
import numpy as np


def create_low_contrast_image():
    """ì €ëŒ€ë¹„ ì´ë¯¸ì§€ ìƒì„±"""
    img = np.zeros((300, 400), dtype=np.uint8)

    # ì¢ì€ ë²”ìœ„ì˜ ë°ê¸°ê°’ë§Œ ì‚¬ìš© (100-150)
    img[:] = 120

    # ë„í˜• ê·¸ë¦¬ê¸°
    cv2.rectangle(img, (50, 50), (150, 150), 140, -1)
    cv2.circle(img, (300, 150), 60, 130, -1)

def create_color_image():
    """ì»¬ëŸ¬ í…ŒìŠ¤íŠ¸ ì´ë¯¸ì§€"""
    img = np.zeros((300, 400, 3), dtype=np.uint8)

    # ë‹¤ì–‘í•œ ìƒ‰ìƒ ì˜ì—­
    img[0:150, 0:200] = [200, 50, 50]      # íŒŒë‘
    img[0:150, 200:400] = [50, 200, 50]    # ì´ˆë¡
    img[150:300, 0:200] = [50, 50, 200]    # ë¹¨ê°•
    img[150:300, 200:400] = [200, 200, 50] # ì²­ë¡

    return img


def calc_histogram_demo():
    """íˆìŠ¤í† ê·¸ë¨ ê³„ì‚° ë°ëª¨"""
    print("=" * 50)
    print("íˆìŠ¤í† ê·¸ë¨ ê³„ì‚° (calcHist)")
    print("=" * 50)

    img = create_low_contrast_image()

    # íˆìŠ¤í† ê·¸ë¨ ê³„ì‚°
    # images: ì…ë ¥ ì´ë¯¸ì§€ ë¦¬ìŠ¤íŠ¸
    # channels: ì±„ë„ ì¸ë±ìŠ¤ (ê·¸ë ˆì´: [0], BGR: [0], [1], [2])
    # mask: ë§ˆìŠ¤í¬ (None = ì „ì²´)
    # histSize: ë¹ˆ ê°œìˆ˜ (ë³´í†µ 256)
    # ranges: ê°’ ë²”ìœ„

    hist = cv2.calcHist([img], [0], None, [256], [0, 256])

    print(f"íˆìŠ¤í† ê·¸ë¨ shape: {hist.shape}")
    print(f"ì´ í”½ì…€ ìˆ˜: {hist.sum()}")
    print(f"ìµœëŒ€ ë¹ˆë„ ê°’: {hist.max():.0f}")
    print(f"ìµœëŒ€ ë¹ˆë„ ìœ„ì¹˜: {hist.argmax()}")

    # íˆìŠ¤í† ê·¸ë¨ ì‹œê°í™” (í…ìŠ¤íŠ¸)
    print("\níˆìŠ¤í† ê·¸ë¨ ë¶„í¬ (ê°„ëµ):")
    for i in range(0, 256, 32):
        count = hist[i:i+32].sum()
        bar = '#' * int(count / 1000)
        print(f"  {i:3d}-{i+31:3d}: {bar} ({count:.0f})")

    cv2.imwrite('histogram_input.jpg', img)

    return hist


def histogram_color_demo():

def clahe_demo():
    """CLAHE ë°ëª¨"""
    print("\n" + "=" * 50)
    print("CLAHE (Contrast Limited Adaptive Histogram Equalization)")
    print("=" * 50)

    img = create_low_contrast_image()

    # ì¼ë°˜ í‰í™œí™”
    equalized = cv2.equalizeHist(img)

    # CLAHE
    # clipLimit: ëŒ€ë¹„ ì œí•œ (ë†’ì„ìˆ˜ë¡ ëŒ€ë¹„ ê°•í•¨)
    # tileGridSize: íƒ€ì¼ í¬ê¸°

    clahe1 = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
    clahe2 = cv2.createCLAHE(clipLimit=4.0, tileGridSize=(8, 8))
    clahe3 = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(16, 16))

    result1 = clahe1.apply(img)
    result2 = clahe2.apply(img)
    result3 = clahe3.apply(img)

    print("CLAHE vs ì¼ë°˜ í‰í™œí™”:")
    print("  - CLAHE: ì§€ì—­ì ìœ¼ë¡œ ì ìš© (íƒ€ì¼ ë‹¨ìœ„)")
    print("  - ë…¸ì´ì¦ˆ ì¦í­ ë°©ì§€ (clipLimit)")
    print("  - ë¶ˆê· ì¼ ì¡°ëª…ì— íš¨ê³¼ì ")

    print("\nCLAHE íŒŒë¼ë¯¸í„°:")
    print("  clipLimit: ëŒ€ë¹„ ì œí•œ (2.0~4.0 ê¶Œì¥)")
    print("  tileGridSize: íƒ€ì¼ í¬ê¸° (8x8 ê¶Œì¥)")

    cv2.imwrite('clahe_equalized.jpg', equalized)
    cv2.imwrite('clahe_2_8.jpg', result1)
    cv2.imwrite('clahe_4_8.jpg', result2)
    cv2.imwrite('clahe_2_16.jpg', result3)

def histogram_comparison_demo():
    """íˆìŠ¤í† ê·¸ë¨ ë¹„êµ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("íˆìŠ¤í† ê·¸ë¨ ë¹„êµ (compareHist)")
    print("=" * 50)

    # ë¹„êµí•  ì´ë¯¸ì§€ë“¤ ìƒì„±
    img1 = np.zeros((100, 100), dtype=np.uint8)
    img1[:] = 100
    cv2.rectangle(img1, (20, 20), (80, 80), 150, -1)

    img2 = img1.copy()  # ë™ì¼

    img3 = np.zeros((100, 100), dtype=np.uint8)
    img3[:] = 50
    cv2.rectangle(img3, (20, 20), (80, 80), 200, -1)

    # íˆìŠ¤í† ê·¸ë¨ ê³„ì‚°
    hist1 = cv2.calcHist([img1], [0], None, [256], [0, 256])
    hist2 = cv2.calcHist([img2], [0], None, [256], [0, 256])
    hist3 = cv2.calcHist([img3], [0], None, [256], [0, 256])

    # ì •ê·œí™”
    cv2.normalize(hist1, hist1, 0, 1, cv2.NORM_MINMAX)
    cv2.normalize(hist2, hist2, 0, 1, cv2.NORM_MINMAX)
    cv2.normalize(hist3, hist3, 0, 1, cv2.NORM_MINMAX)

    # ë¹„êµ ë°©ë²•
    methods = [
        (cv2.HISTCMP_CORREL, 'Correlation'),
        (cv2.HISTCMP_CHISQR, 'Chi-Square'),
        (cv2.HISTCMP_INTERSECT, 'Intersection'),
        (cv2.HISTCMP_BHATTACHARYYA, 'Bhattacharyya'),
    ]

    print("img1 vs img2 (ë™ì¼), img1 vs img3 (ë‹¤ë¦„):\n")

    for method, name in methods:
        score12 = cv2.compareHist(hist1, hist2, method)
        score13 = cv2.compareHist(hist1, hist3, method)
        print(f"  {name:15}: ë™ì¼={score12:.4f}, ë‹¤ë¦„={score13:.4f}")

    print("\në¹„êµ ë°©ë²• í•´ì„:")
    print("  Correlation: 1ì— ê°€ê¹Œìš¸ìˆ˜ë¡ ìœ ì‚¬")
    print("  Chi-Square: 0ì— ê°€ê¹Œìš¸ìˆ˜ë¡ ìœ ì‚¬")
    print("  Intersection: í´ìˆ˜ë¡ ìœ ì‚¬")
    print("  Bhattacharyya: 0ì— ê°€ê¹Œìš¸ìˆ˜ë¡ ìœ ì‚¬")


def back_projection_demo():
    """íˆìŠ¤í† ê·¸ë¨ ì—­íˆ¬ì˜ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("íˆìŠ¤í† ê·¸ë¨ ì—­íˆ¬ì˜ (Back Projection)")
    print("=" * 50)

    # ëŒ€ìƒ ì´ë¯¸ì§€ (ë‹¤ì–‘í•œ ìƒ‰ìƒ)
    target = np.zeros((300, 400, 3), dtype=np.uint8)
    target[:] = [100, 100, 100]  # íšŒìƒ‰ ë°°ê²½

    # ë¹¨ê°„ ë¬¼ì²´ë“¤
    cv2.circle(target, (100, 100), 40, (50, 50, 200), -1)
    cv2.circle(target, (300, 200), 50, (30, 30, 180), -1)
    cv2.rectangle(target, (150, 200), (220, 280), (40, 40, 210), -1)

    # íŒŒë€ ë¬¼ì²´
    cv2.circle(target, (350, 100), 30, (200, 50, 50), -1)

    # ROI (ë¹¨ê°„ìƒ‰ ìƒ˜í”Œ)
    roi = target[60:140, 60:140]  # ë¹¨ê°„ ì› ì˜ì—­

    # HSV ë³€í™˜
    hsv_target = cv2.cvtColor(target, cv2.COLOR_BGR2HSV)
    hsv_roi = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)

    # ROI íˆìŠ¤í† ê·¸ë¨ ê³„ì‚° (Hue, Saturationë§Œ)
    roi_hist = cv2.calcHist([hsv_roi], [0, 1], None, [180, 256], [0, 180, 0, 256])
    cv2.normalize(roi_hist, roi_hist, 0, 255, cv2.NORM_MINMAX)

    # ì—­íˆ¬ì˜
    back_proj = cv2.calcBackProject([hsv_target], [0, 1], roi_hist, [0, 180, 0, 256], 1)

    # ê²°ê³¼ ê°œì„  (í˜•íƒœí•™ì  ì—°ì‚°)
    kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
    cv2.filter2D(back_proj, -1, kernel, back_proj)
    _, thresh = cv2.threshold(back_proj, 50, 255, cv2.THRESH_BINARY)

    print("ì—­íˆ¬ì˜ ê³¼ì •:")
    print("  1. ROIì˜ íˆìŠ¤í† ê·¸ë¨ ê³„ì‚°")
    print("  2. ëŒ€ìƒ ì´ë¯¸ì§€ì˜ ê° í”½ì…€ì´ ROIì™€ ì–¼ë§ˆë‚˜ ìœ ì‚¬í•œì§€ ê³„ì‚°")
    print("  3. ìœ ì‚¬í•œ ìƒ‰ìƒ ì˜ì—­ì´ ë°ê²Œ í‘œì‹œë¨")
    print("  4. ê°ì²´ ì¶”ì ì— í™œìš© (MeanShift, CamShift)")

    cv2.imwrite('backproj_target.jpg', target)
    cv2.imwrite('backproj_roi.jpg', roi)
    cv2.imwrite('backproj_result.jpg', back_proj)
    cv2.imwrite('backproj_thresh.jpg', thresh)
