def create_test_image():

    img = np.zeros((400, 500, 3), dtype=np.uint8)

def harris_corner_demo():
    """Harris ì½”ë„ˆ ê²€ì¶œ ë°ëª¨"""
    print("=" * 50)
    print("Harris ì½”ë„ˆ ê²€ì¶œ")
    print("=" * 50)

    img = create_test_image()
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    gray = np.float32(gray)

    # Harris ì½”ë„ˆ ê²€ì¶œ
    # blockSize: ì½”ë„ˆ ê²€ì¶œ ìœˆë„ìš° í¬ê¸°
    # ksize: Sobel ì»¤ë„ í¬ê¸°
    # k: Harris íŒŒë¼ë¯¸í„° (0.04~0.06)
    harris = cv2.cornerHarris(gray, blockSize=2, ksize=3, k=0.04)

    # ê²°ê³¼ í™•ì¥ (ì‹œê°í™”ìš©)
    harris_dilated = cv2.dilate(harris, None)

    # ì„ê³„ê°’ ì ìš©
    threshold = 0.01 * harris_dilated.max()
    result = img.copy()
    result[harris_dilated > threshold] = [0, 0, 255]  # ë¹¨ê°„ìƒ‰ìœ¼ë¡œ í‘œì‹œ

    # ì •ë°€ ì½”ë„ˆ ìœ„ì¹˜ (SubPixel)
    _, harris_binary = cv2.threshold(harris_dilated, threshold, 255, cv2.THRESH_BINARY)
    harris_binary = np.uint8(harris_binary)

    # ì—°ê²°ëœ ì»´í¬ë„ŒíŠ¸ë¡œ ì½”ë„ˆ ê°œìˆ˜ ì„¸ê¸°
    num_corners = cv2.connectedComponents(harris_binary)[0] - 1

    print(f"ê²€ì¶œëœ ì½”ë„ˆ ìˆ˜: {num_corners}")
    print("\nHarris ì½”ë„ˆ íŠ¹ì„±:")
    print("  - íšŒì „ ë¶ˆë³€")
    print("  - í¬ê¸° ë³€í™”ì—ëŠ” ë¯¼ê°")
    print("  - ì½”ë„ˆ ì‘ë‹µ í•¨ìˆ˜ R = det(M) - k*trace(M)^2")

    cv2.imwrite('harris_input.jpg', img)
    cv2.imwrite('harris_result.jpg', result)


def shi_tomasi_demo():
    """Shi-Tomasi ì½”ë„ˆ ê²€ì¶œ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("Shi-Tomasi ì½”ë„ˆ ê²€ì¶œ (goodFeaturesToTrack)")
    print("=" * 50)

    img = create_test_image()
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # Shi-Tomasi ì½”ë„ˆ ê²€ì¶œ
    # maxCorners: ìµœëŒ€ ì½”ë„ˆ ìˆ˜
    # qualityLevel: í’ˆì§ˆ ìˆ˜ì¤€ (0~1)
    # minDistance: ì½”ë„ˆ ê°„ ìµœì†Œ ê±°ë¦¬
    corners = cv2.goodFeaturesToTrack(
        gray,
        maxCorners=100,
        qualityLevel=0.01,
        minDistance=10
    )

    result = img.copy()

    if corners is not None:
        corners = np.int32(corners)
        print(f"ê²€ì¶œëœ ì½”ë„ˆ ìˆ˜: {len(corners)}")

        for corner in corners:
            x, y = corner.ravel()
            cv2.circle(result, (x, y), 5, (0, 255, 0), -1)

    print("\nShi-Tomasi vs Harris:")
    print("  - R = min(Î»1, Î»2) ì‚¬ìš©")
    print("  - Harrisë³´ë‹¤ ì•ˆì •ì ")
    print("  - ì¶”ì ì— ì í•©í•œ ì½”ë„ˆ ì„ íƒ")

    cv2.imwrite('shi_tomasi_result.jpg', result)


def fast_demo():

def compare_detectors():
    """íŠ¹ì§•ì  ê²€ì¶œê¸° ë¹„êµ"""
    print("\n" + "=" * 50)
    print("íŠ¹ì§•ì  ê²€ì¶œê¸° ë¹„êµ")
    print("=" * 50)

    img = create_test_image()
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    detectors = []

    # FAST
    fast = cv2.FastFeatureDetector_create()
    kp_fast = fast.detect(gray, None)
    detectors.append(('FAST', len(kp_fast)))

    # ORB
    orb = cv2.ORB_create()
    kp_orb, _ = orb.detectAndCompute(gray, None)
    detectors.append(('ORB', len(kp_orb)))

    # SIFT (ê°€ëŠ¥í•œ ê²½ìš°)
    try:
        sift = cv2.SIFT_create()
        kp_sift, _ = sift.detectAndCompute(gray, None)
        detectors.append(('SIFT', len(kp_sift)))
    except AttributeError:
        detectors.append(('SIFT', 'N/A'))

    # Shi-Tomasi
    corners = cv2.goodFeaturesToTrack(gray, 1000, 0.01, 10)
    detectors.append(('Shi-Tomasi', len(corners) if corners is not None else 0))

    print("\n| ê²€ì¶œê¸° | í‚¤í¬ì¸íŠ¸ ìˆ˜ | íŠ¹ì§• |")
    print("|--------|-----------|------|")
    print(f"| FAST | {detectors[0][1]} | ë¹ ë¦„, ë””ìŠ¤í¬ë¦½í„° ì—†ìŒ |")
    print(f"| ORB | {detectors[1][1]} | ë¹ ë¦„, íŠ¹í—ˆ ë¬´ë£Œ |")
    print(f"| SIFT | {detectors[2][1]} | ì •í™•, ëŠë¦¼ |")
    print(f"| Shi-Tomasi | {detectors[3][1]} | ì¶”ì ìš© |")
