import cv2
import numpy as np


def create_scene_with_objects():
    """Í∞ùÏ≤¥Í∞Ä ÏûàÎäî Ïû•Î©¥ Ïù¥ÎØ∏ÏßÄ ÏÉùÏÑ±"""
    scene = np.zeros((400, 600, 3), dtype=np.uint8)
    scene[:] = [200, 200, 200]

    # Î≥Ñ Î™®Ïñë Í∞ùÏ≤¥ (Ï∞æÏùÑ ÎåÄÏÉÅ)
    def draw_star(img, center, size, color):
        pts = []
        for i in range(5):
            outer = np.radians(i * 72 - 90)
            inner = np.radians(i * 72 + 36 - 90)
            pts.append([int(center[0] + size * np.cos(outer)),
                       int(center[1] + size * np.sin(outer))])
            pts.append([int(center[0] + size * 0.4 * np.cos(inner)),
                       int(center[1] + size * 0.4 * np.sin(inner))])
        cv2.fillPoly(img, [np.array(pts, np.int32)], color)

    # Ïó¨Îü¨ Î≥Ñ Î∞∞Ïπò
    draw_star(scene, (100, 100), 30, (0, 0, 150))
    draw_star(scene, (300, 200), 40, (0, 0, 180))
    draw_star(scene, (500, 300), 35, (0, 0, 160))

    # Î∞©Ìï¥ Í∞ùÏ≤¥
    cv2.circle(scene, (200, 300), 40, (150, 0, 0), -1)
    cv2.rectangle(scene, (400, 50), (480, 130), (0, 150, 0), -1)

    # ÌÖúÌîåÎ¶ø (Î≥Ñ ÌïòÎÇò)
    template = np.zeros((80, 80, 3), dtype=np.uint8)
    template[:] = [200, 200, 200]
    draw_star(template, (40, 40), 30, (0, 0, 150))

    return scene, template


def template_matching_demo():
    """ÌÖúÌîåÎ¶ø Îß§Ïπ≠ Îç∞Î™®"""
    print("=" * 50)
    print("ÌÖúÌîåÎ¶ø Îß§Ïπ≠ (Template Matching)")
    print("=" * 50)

    scene, template = create_scene_with_objects()

    # Í∑∏Î†àÏù¥Ïä§ÏºÄÏùº Î≥ÄÌôò
    scene_gray = cv2.cvtColor(scene, cv2.COLOR_BGR2GRAY)
    template_gray = cv2.cvtColor(template, cv2.COLOR_BGR2GRAY)

    h, w = template_gray.shape

    # ÌÖúÌîåÎ¶ø Îß§Ïπ≠ Î∞©Î≤ïÎì§
    methods = [
        ('TM_CCOEFF', cv2.TM_CCOEFF),
        ('TM_CCOEFF_NORMED', cv2.TM_CCOEFF_NORMED),
        ('TM_CCORR_NORMED', cv2.TM_CCORR_NORMED),
        ('TM_SQDIFF_NORMED', cv2.TM_SQDIFF_NORMED),
    ]

    print("\nÎß§Ïπ≠ Î∞©Î≤ï Í≤∞Í≥º:")

    for name, method in methods:
        result = cv2.matchTemplate(scene_gray, template_gray, method)

        # ÏµúÏÜå/ÏµúÎåÄ ÏúÑÏπò Ï∞æÍ∏∞
        min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(result)

        # SQDIFFÎäî ÏµúÏÜåÍ∞íÏù¥ ÏµúÏ†Å
        if method in [cv2.TM_SQDIFF, cv2.TM_SQDIFF_NORMED]:
            top_left = min_loc
            score = min_val
        else:
            top_left = max_loc
            score = max_val

        print(f"  {name}: score={score:.4f}, loc={top_left}")

        # Í≤∞Í≥º ÏãúÍ∞ÅÌôî
        scene_copy = scene.copy()
        bottom_right = (top_left[0] + w, top_left[1] + h)
        cv2.rectangle(scene_copy, top_left, bottom_right, (0, 255, 0), 2)
        cv2.imwrite(f'template_{name}.jpg', scene_copy)

    print("\nÎß§Ïπ≠ Î∞©Î≤ï ÌäπÏÑ±:")
    print("  TM_SQDIFF: Ï∞®Ïù¥ Ï†úÍ≥±Ìï© (ÏûëÏùÑÏàòÎ°ù Ï¢ãÏùå)")
    print("  TM_CCORR: ÏÉÅÍ¥ÄÍ¥ÄÍ≥Ñ (ÌÅ¥ÏàòÎ°ù Ï¢ãÏùå)")
    print("  TM_CCOEFF: ÏÉÅÍ¥ÄÍ≥ÑÏàò (ÌÅ¥ÏàòÎ°ù Ï¢ãÏùå)")
    print("  _NORMED: Ï†ïÍ∑úÌôî Î≤ÑÏ†Ñ (-1~1 ÎòêÎäî 0~1)")

    cv2.imwrite('template_scene.jpg', scene)
    cv2.imwrite('template_template.jpg', template)


def multi_scale_template_demo():
    """Îã§Ï§ë Ïä§ÏºÄÏùº ÌÖúÌîåÎ¶ø Îß§Ïπ≠"""
    print("\n" + "=" * 50)
    print("Îã§Ï§ë Ïä§ÏºÄÏùº ÌÖúÌîåÎ¶ø Îß§Ïπ≠")
    print("=" * 50)

    scene, template = create_scene_with_objects()
    scene_gray = cv2.cvtColor(scene, cv2.COLOR_BGR2GRAY)
    template_gray = cv2.cvtColor(template, cv2.COLOR_BGR2GRAY)

    best_match = None
    best_val = -1
    best_scale = 1.0
    best_loc = (0, 0)

    # Îã§ÏñëÌïú Ïä§ÏºÄÏùºÎ°ú Îß§Ïπ≠
    scales = [0.5, 0.75, 1.0, 1.25, 1.5]

    for scale in scales:
        # ÌÖúÌîåÎ¶ø ÌÅ¨Í∏∞ Ï°∞Ï†ï
        new_w = int(template_gray.shape[1] * scale)
        new_h = int(template_gray.shape[0] * scale)

        if new_w > scene_gray.shape[1] or new_h > scene_gray.shape[0]:
            continue

        resized = cv2.resize(template_gray, (new_w, new_h))

        # Îß§Ïπ≠
        result = cv2.matchTemplate(scene_gray, resized, cv2.TM_CCOEFF_NORMED)
        _, max_val, _, max_loc = cv2.minMaxLoc(result)

        print(f"  Scale {scale:.2f}: score={max_val:.4f}")

        if max_val > best_val:
            best_val = max_val
            best_scale = scale
            best_loc = max_loc
            best_match = resized.shape

    print(f"\nÏµúÏ†Å Ïä§ÏºÄÏùº: {best_scale}, score={best_val:.4f}")

    # Í≤∞Í≥º Í∑∏Î¶¨Í∏∞
    result_img = scene.copy()
    h, w = best_match
    cv2.rectangle(result_img, best_loc, (best_loc[0]+w, best_loc[1]+h), (0, 255, 0), 2)
    cv2.imwrite('multi_scale_result.jpg', result_img)


def find_all_matches_demo():
    """Î™®Îì† Îß§Ïπ≠ Ï∞æÍ∏∞"""
    print("\n" + "=" * 50)
    print("Î™®Îì† Îß§Ïπ≠ Ï∞æÍ∏∞")
    print("=" * 50)

    scene, template = create_scene_with_objects()
    scene_gray = cv2.cvtColor(scene, cv2.COLOR_BGR2GRAY)
    template_gray = cv2.cvtColor(template, cv2.COLOR_BGR2GRAY)

    h, w = template_gray.shape

    # ÌÖúÌîåÎ¶ø Îß§Ïπ≠
    result = cv2.matchTemplate(scene_gray, template_gray, cv2.TM_CCOEFF_NORMED)

    # ÏûÑÍ≥ÑÍ∞í Ïù¥ÏÉÅÏù∏ Î™®Îì† ÏúÑÏπò Ï∞æÍ∏∞
    threshold = 0.6
    locations = np.where(result >= threshold)

    scene_copy = scene.copy()
    match_count = 0

    # Ï§ëÎ≥µ Ï†úÍ±∞Î•º ÏúÑÌïú NMS Ï†ÅÏö©
    boxes = []
    for pt in zip(*locations[::-1]):
        boxes.append([pt[0], pt[1], pt[0]+w, pt[1]+h])

    # Í∞ÑÎã®Ìïú NMS
    boxes = np.array(boxes)
    if len(boxes) > 0:
        # Ï†êÏàòÏàú Ï†ïÎ†¨
        scores = [result[b[1], b[0]] for b in boxes]
        indices = np.argsort(scores)[::-1]

        keep = []
        while len(indices) > 0:
            i = indices[0]
            keep.append(i)

            # Îã§Î•∏ Î∞ïÏä§ÏôÄÏùò Í≤πÏπ® Í≥ÑÏÇ∞
            remaining = []
            for j in indices[1:]:
                # IoU Í≥ÑÏÇ∞ (Í∞ÑÎã® Î≤ÑÏ†Ñ)
                x_overlap = max(0, min(boxes[i][2], boxes[j][2]) - max(boxes[i][0], boxes[j][0]))
                y_overlap = max(0, min(boxes[i][3], boxes[j][3]) - max(boxes[i][1], boxes[j][1]))
                overlap = x_overlap * y_overlap
                area = w * h
                iou = overlap / area

                if iou < 0.5:  # Í≤πÏπ®Ïù¥ 50% ÎØ∏ÎßåÏù¥Î©¥ Ïú†ÏßÄ
                    remaining.append(j)

            indices = np.array(remaining)

        # Í≤∞Í≥º Í∑∏Î¶¨Í∏∞
        for i in keep:
            pt = (boxes[i][0], boxes[i][1])
            cv2.rectangle(scene_copy, pt, (pt[0]+w, pt[1]+h), (0, 255, 0), 2)
            score = result[pt[1], pt[0]]
            cv2.putText(scene_copy, f'{score:.2f}', (pt[0], pt[1]-5),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 0, 255), 1)
            match_count += 1

    print(f"ÏûÑÍ≥ÑÍ∞í: {threshold}")
    print(f"Í≤ÄÏ∂úÎêú Í∞ùÏ≤¥ Ïàò: {match_count}")

    cv2.imwrite('find_all_matches.jpg', scene_copy)


def haar_cascade_demo():
    """Haar Cascade Îç∞Î™®"""
    print("\n" + "=" * 50)
    print("Haar Cascade Í≤ÄÏ∂úÍ∏∞")
    print("=" * 50)

    # ÌÖåÏä§Ìä∏ Ïù¥ÎØ∏ÏßÄ (ÏñºÍµ¥ ÏãúÎÆ¨Î†àÏù¥ÏÖò)
    img = np.zeros((300, 400, 3), dtype=np.uint8)
    img[:] = [200, 200, 200]

    # ÏñºÍµ¥ ÌòïÌÉú ÏãúÎÆ¨Î†àÏù¥ÏÖò
    cv2.ellipse(img, (200, 150), (50, 60), 0, 0, 360, (180, 150, 130), -1)
    cv2.circle(img, (180, 130), 8, (50, 50, 50), -1)  # Îàà
    cv2.circle(img, (220, 130), 8, (50, 50, 50), -1)
    cv2.ellipse(img, (200, 170), (15, 8), 0, 0, 180, (50, 50, 50), 2)  # ÏûÖ

    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # Haar Cascade Î°úÎìú
    cascade_path = cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
    face_cascade = cv2.CascadeClassifier(cascade_path)

    if face_cascade.empty():
        print("Haar Cascade ÌååÏùºÏùÑ Ï∞æÏùÑ Ïàò ÏóÜÏäµÎãàÎã§.")
    else:
        # Í≤ÄÏ∂ú
        faces = face_cascade.detectMultiScale(
            gray,
            scaleFactor=1.1,
            minNeighbors=5,
            minSize=(30, 30)
        )

        result = img.copy()
        for (x, y, w, h) in faces:
            cv2.rectangle(result, (x, y), (x+w, y+h), (0, 255, 0), 2)

        print(f"Í≤ÄÏ∂úÎêú ÏñºÍµ¥: {len(faces)}")

        cv2.imwrite('haar_input.jpg', img)
        cv2.imwrite('haar_result.jpg', result)

    print("\nHaar Cascade ÌååÎùºÎØ∏ÌÑ∞:")
    print("  scaleFactor: Ïù¥ÎØ∏ÏßÄ ÌîºÎùºÎØ∏Îìú Ïä§ÏºÄÏùº")
    print("  minNeighbors: Í≤ÄÏ∂ú ÌôïÏ†ï ÏµúÏÜå Ïù¥ÏõÉ Ïàò")
    print("  minSize: ÏµúÏÜå Í∞ùÏ≤¥ ÌÅ¨Í∏∞")

    print("\nÏÇ¨Ïö© Í∞ÄÎä•Ìïú Haar Cascade:")
    print(f"  Í≤ΩÎ°ú: {cv2.data.haarcascades}")
    print("  - haarcascade_frontalface_default.xml")
    print("  - haarcascade_eye.xml")
    print("  - haarcascade_smile.xml")
    print("  - haarcascade_fullbody.xml")


def hog_concept_demo():
    """HOG Í∞úÎÖê Îç∞Î™®"""
    print("\n" + "=" * 50)
    print("HOG (Histogram of Oriented Gradients)")
    print("=" * 50)

    # ÌÖåÏä§Ìä∏ Ïù¥ÎØ∏ÏßÄ
    img = np.zeros((128, 64, 3), dtype=np.uint8)
    img[:] = [200, 200, 200]

    # ÏÇ¨Îûå ÌòïÌÉú ÏãúÎÆ¨Î†àÏù¥ÏÖò
    cv2.ellipse(img, (32, 25), (12, 15), 0, 0, 360, (100, 100, 100), -1)  # Î®∏Î¶¨
    cv2.rectangle(img, (20, 40), (44, 90), (100, 100, 100), -1)  # Î™∏ÌÜµ
    cv2.rectangle(img, (18, 90), (30, 125), (100, 100, 100), -1)  # ÏôºÏ™Ω Îã§Î¶¨
    cv2.rectangle(img, (34, 90), (46, 125), (100, 100, 100), -1)  # Ïò§Î•∏Ï™Ω Îã§Î¶¨

    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # HOG ÎîîÏä§ÌÅ¨Î¶ΩÌÑ∞ Í≥ÑÏÇ∞
    # winSize: Í≤ÄÏ∂ú ÏúàÎèÑÏö∞ ÌÅ¨Í∏∞
    # blockSize: Î∏îÎ°ù ÌÅ¨Í∏∞
    # blockStride: Î∏îÎ°ù Ïù¥Îèô Í∞ÑÍ≤©
    # cellSize: ÏÖÄ ÌÅ¨Í∏∞
    # nbins: ÌûàÏä§ÌÜ†Í∑∏Îû® Îπà Ïàò

    hog = cv2.HOGDescriptor(
        _winSize=(64, 128),
        _blockSize=(16, 16),
        _blockStride=(8, 8),
        _cellSize=(8, 8),
        _nbins=9
    )

    # HOG ÌäπÏßï Í≥ÑÏÇ∞
    features = hog.compute(gray)

    print(f"ÏûÖÎ†• Ïù¥ÎØ∏ÏßÄ: {gray.shape}")
    print(f"HOG ÌäπÏßï Î≤°ÌÑ∞ ÌÅ¨Í∏∞: {features.shape}")

    print("\nHOG ÌäπÏÑ±:")
    print("  - Í∑∏ÎûòÎîîÏñ∏Ìä∏ Î∞©Ìñ• ÌûàÏä§ÌÜ†Í∑∏Îû®")
    print("  - Ï°∞Î™Ö Î≥ÄÌôîÏóê Í∞ïÍ±¥")
    print("  - Î≥¥ÌñâÏûê Í≤ÄÏ∂úÏóê Ìö®Í≥ºÏ†Å")
    print("  - SVMÍ≥º Ìï®Íªò ÏÇ¨Ïö©")

    # Í∏∞Î≥∏ HOG Î≥¥ÌñâÏûê Í≤ÄÏ∂úÍ∏∞
    hog_detector = cv2.HOGDescriptor()
    hog_detector.setSVMDetector(cv2.HOGDescriptor_getDefaultPeopleDetector())

    print("\nÏÇ¨Ï†Ñ ÌïôÏäµÎêú HOG+SVM Í≤ÄÏ∂úÍ∏∞:")
    print("  - cv2.HOGDescriptor_getDefaultPeopleDetector()")
    print("  - Î≥¥ÌñâÏûê Í≤ÄÏ∂úÏö© ÌïôÏäµÎêú SVM Í∞ÄÏ§ëÏπò")

    cv2.imwrite('hog_input.jpg', img)
