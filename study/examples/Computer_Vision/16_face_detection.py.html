def create_face_image():
    """ì–¼êµ´ í˜•íƒœ ì‹œë®¬ë ˆì´ì…˜ ì´ë¯¸ì§€"""
    img = np.zeros((400, 500, 3), dtype=np.uint8)

def haar_cascade_face_detection():
    """Haar Cascade ì–¼êµ´ ê²€ì¶œ ë°ëª¨"""
    print("=" * 50)
    print("Haar Cascade ì–¼êµ´ ê²€ì¶œ")
    print("=" * 50)

    img = create_face_image()
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # Haar Cascade ë¶„ë¥˜ê¸° ë¡œë“œ
    face_cascade_path = cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
    face_cascade = cv2.CascadeClassifier(face_cascade_path)

    if face_cascade.empty():
        print("ì–¼êµ´ ê²€ì¶œê¸°ë¥¼ ë¡œë“œí•  ìˆ˜ ì—†ìŠµë‹ˆë‹¤.")
        return

    # ì–¼êµ´ ê²€ì¶œ
    # scaleFactor: ê° ì´ë¯¸ì§€ ìŠ¤ì¼€ì¼ì—ì„œ ì´ë¯¸ì§€ í¬ê¸°ë¥¼ ì¤„ì´ëŠ” ë¹„ìœ¨
    # minNeighbors: ê° í›„ë³´ ì‚¬ê°í˜•ì´ ìœ ì§€ë˜ê¸° ìœ„í•´ í•„ìš”í•œ ì´ì›ƒ ìˆ˜
    # minSize: ìµœì†Œ ê°ì²´ í¬ê¸°
    faces = face_cascade.detectMultiScale(
        gray,
        scaleFactor=1.1,
        minNeighbors=5,
        minSize=(30, 30),
        flags=cv2.CASCADE_SCALE_IMAGE
    )

    result = img.copy()
    print(f"ê²€ì¶œëœ ì–¼êµ´ ìˆ˜: {len(faces)}")

    for i, (x, y, w, h) in enumerate(faces):
        # ì–¼êµ´ ì˜ì—­ í‘œì‹œ
        cv2.rectangle(result, (x, y), (x+w, y+h), (0, 255, 0), 2)
        cv2.putText(result, f'Face {i+1}', (x, y-10),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 1)
        print(f"  Face {i+1}: x={x}, y={y}, w={w}, h={h}")

    print("\nHaar Cascade íŒŒë¼ë¯¸í„°:")
    print("  scaleFactor: 1.1~1.3 (ì‘ì„ìˆ˜ë¡ ì •ë°€, ëŠë¦¼)")
    print("  minNeighbors: 3~6 (í´ìˆ˜ë¡ ì—„ê²©)")
    print("  minSize: ìµœì†Œ ê²€ì¶œ í¬ê¸°")

    cv2.imwrite('face_haar_input.jpg', img)
    cv2.imwrite('face_haar_result.jpg', result)


def cascade_eye_detection():
    """ëˆˆ ê²€ì¶œ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("ëˆˆ ê²€ì¶œ (Haar Cascade)")
    print("=" * 50)

    img = create_face_image()
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # ë¶„ë¥˜ê¸° ë¡œë“œ
    face_cascade = cv2.CascadeClassifier(
        cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
    )
    eye_cascade = cv2.CascadeClassifier(
        cv2.data.haarcascades + 'haarcascade_eye.xml'
    )

    if face_cascade.empty() or eye_cascade.empty():
        print("ë¶„ë¥˜ê¸°ë¥¼ ë¡œë“œí•  ìˆ˜ ì—†ìŠµë‹ˆë‹¤.")
        return

    # ì–¼êµ´ ê²€ì¶œ
    faces = face_cascade.detectMultiScale(gray, 1.1, 5)

    result = img.copy()

    for (x, y, w, h) in faces:
        cv2.rectangle(result, (x, y), (x+w, y+h), (0, 255, 0), 2)

        # ì–¼êµ´ ì˜ì—­ ë‚´ì—ì„œ ëˆˆ ê²€ì¶œ
        roi_gray = gray[y:y+h, x:x+w]
        roi_color = result[y:y+h, x:x+w]

        eyes = eye_cascade.detectMultiScale(
            roi_gray,
            scaleFactor=1.1,
            minNeighbors=10,
            minSize=(15, 15)
        )

        for (ex, ey, ew, eh) in eyes:
            cv2.rectangle(roi_color, (ex, ey), (ex+ew, ey+eh), (255, 0, 0), 2)

        print(f"ì–¼êµ´ ({x}, {y})ì—ì„œ ê²€ì¶œëœ ëˆˆ: {len(eyes)}")

    print("\nëˆˆ ê²€ì¶œ íŒ:")
    print("  - ì–¼êµ´ ì˜ì—­ ë‚´ì—ì„œë§Œ ê²€ì¶œ (ROI)")
    print("  - minNeighborsë¥¼ ë†’ê²Œ ì„¤ì •")
    print("  - ìƒë°˜ë¶€ì—ì„œë§Œ ê²€ì¶œí•˜ë©´ ì •í™•ë„ í–¥ìƒ")

    cv2.imwrite('face_eye_result.jpg', result)

def lbp_face_detection():
    """LBP ê¸°ë°˜ ì–¼êµ´ ê²€ì¶œ ë°ëª¨"""

    print("LBP ì–¼êµ´ ê²€ì¶œ")

    lbp_cascade_path = cv2.data.haarcascades + '../lbpcascades/lbpcascade_frontalface_improved.xml'
