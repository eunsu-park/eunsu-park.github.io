 81def cascade_eye_detection():

 91    face_cascade = cv2.CascadeClassifier(
 92        cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
 93    )

 95        cv2.data.haarcascades + 'haarcascade_eye.xml'
