def create_chessboard_image():
    """ì²´ìŠ¤ë³´ë“œ ì´ë¯¸ì§€ ìƒì„±"""
    # ì²´ìŠ¤ë³´ë“œ íŒ¨í„´
    rows, cols = 7, 9
    square_size = 40

    img = np.zeros((rows * square_size, cols * square_size, 3), dtype=np.uint8)
    img[:] = [255, 255, 255]

    for i in range(rows):
        for j in range(cols):
            if (i + j) % 2 == 1:
                x1, y1 = j * square_size, i * square_size
                x2, y2 = x1 + square_size, y1 + square_size
                cv2.rectangle(img, (x1, y1), (x2, y2), (0, 0, 0), -1)

    return img


def chessboard_detection_demo():
    """ì²´ìŠ¤ë³´ë“œ ì½”ë„ˆ ê²€ì¶œ ë°ëª¨"""
    print("\n" + "=" * 50)
    print("ì²´ìŠ¤ë³´ë“œ ì½”ë„ˆ ê²€ì¶œ")
    print("=" * 50)

    # ì²´ìŠ¤ë³´ë“œ ì´ë¯¸ì§€ ìƒì„±
    chessboard = create_chessboard_image()

    # ì•½ê°„ì˜ ì›ê·¼ ë³€í™˜ ì ìš© (ì‹¤ì œ ì´¬ì˜ ì‹œë®¬ë ˆì´ì…˜)
    h, w = chessboard.shape[:2]
    src_pts = np.float32([[0, 0], [w, 0], [w, h], [0, h]])
    dst_pts = np.float32([[20, 30], [w-30, 20], [w-20, h-10], [10, h-30]])
    M = cv2.getPerspectiveTransform(src_pts, dst_pts)
    warped = cv2.warpPerspective(chessboard, M, (w, h), borderValue=(200, 200, 200))

    # ê·¸ë ˆì´ìŠ¤ì¼€ì¼ ë³€í™˜
    gray = cv2.cvtColor(warped, cv2.COLOR_BGR2GRAY)

    # ì½”ë„ˆ ê²€ì¶œ íŒŒë¼ë¯¸í„°
    # ë‚´ë¶€ ì½”ë„ˆ ìˆ˜ (ê²€ì€ìƒ‰-í°ìƒ‰ êµì°¨ì )
    pattern_size = (8, 6)  # ê°€ë¡œ 8, ì„¸ë¡œ 6 ì½”ë„ˆ

    # ì½”ë„ˆ ê²€ì¶œ
    found, corners = cv2.findChessboardCorners(
        gray, pattern_size,
        flags=cv2.CALIB_CB_ADAPTIVE_THRESH + cv2.CALIB_CB_NORMALIZE_IMAGE
    )

    print(f"íŒ¨í„´ í¬ê¸°: {pattern_size}")
    print(f"ì½”ë„ˆ ê²€ì¶œ ì„±ê³µ: {found}")

    if found:
        # ì„œë¸Œí”½ì…€ ì •ë°€ë„ë¡œ ê°œì„ 
        criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 30, 0.001)
        corners = cv2.cornerSubPix(gray, corners, (11, 11), (-1, -1), criteria)

        print(f"ê²€ì¶œëœ ì½”ë„ˆ ìˆ˜: {len(corners)}")

        # ê²°ê³¼ ì‹œê°í™”
        result = warped.copy()
        cv2.drawChessboardCorners(result, pattern_size, corners, found)

        cv2.imwrite('chessboard_input.jpg', warped)
        cv2.imwrite('chessboard_corners.jpg', result)
        print("ì´ë¯¸ì§€ ì €ì¥ ì™„ë£Œ")

    print("\nê²€ì¶œ í”Œë˜ê·¸:")
    print("  CALIB_CB_ADAPTIVE_THRESH: ì ì‘ì  ì´ì§„í™”")
    print("  CALIB_CB_NORMALIZE_IMAGE: ì´ë¯¸ì§€ ì •ê·œí™”")
    print("  CALIB_CB_FAST_CHECK: ë¹ ë¥¸ ì²´í¬ (ì‹¤íŒ¨ ì‹œ ì¡°ê¸° ì¢…ë£Œ)")

    gray = cv2.cvtColor(chessboard, cv2.COLOR_BGR2GRAY)

    found, corners = cv2.findChessboardCorners(gray, pattern_size)

    if found:
        objpoints.append(objp)
        imgpoints.append(corners)

        print(f"\nì‚¬ìš©ëœ ì´ë¯¸ì§€ ìˆ˜: {len(objpoints)}")

        # ì´ˆê¸° ì¹´ë©”ë¼ í–‰ë ¬ ì¶”ì •
        fx = fy = w  # ëŒ€ëµì  ì´ˆì  ê±°ë¦¬
        cx, cy = w/2, h/2

        camera_matrix = np.array([
            [fx, 0, cx],
            [0, fy, cy],
            [0, 0, 1]
        ], dtype=np.float64)

        dist_coeffs = np.zeros(5)

        print("\nì¶”ì •ëœ ì¹´ë©”ë¼ í–‰ë ¬:")
        print(camera_matrix)

        print("\nìº˜ë¦¬ë¸Œë ˆì´ì…˜ í”„ë¡œì„¸ìŠ¤:")

def undistort_demo():
    """ì™œê³¡ ë³´ì • ë°ëª¨"""
    print("\n" + "=" * 50)
    print("ì™œê³¡ ë³´ì • (Undistortion)")
    print("=" * 50)

    # ì‹œë®¬ë ˆì´ì…˜ìš© ì™œê³¡ ì´ë¯¸ì§€ ìƒì„±
    img = np.zeros((400, 600, 3), dtype=np.uint8)
    img[:] = [200, 200, 200]

    # ê²©ì íŒ¨í„´
    for i in range(0, 600, 50):
        cv2.line(img, (i, 0), (i, 400), (0, 0, 0), 1)
    for j in range(0, 400, 50):
        cv2.line(img, (0, j), (600, j), (0, 0, 0), 1)

    # ê°€ìƒì˜ ì™œê³¡ ì ìš© (barrel distortion ì‹œë®¬ë ˆì´ì…˜)
    h, w = img.shape[:2]
    camera_matrix = np.array([
        [w, 0, w/2],
        [0, w, h/2],
        [0, 0, 1]
    ], dtype=np.float64)

    # ì™œê³¡ ê³„ìˆ˜ (k1ì´ ìŒìˆ˜ë©´ barrel, ì–‘ìˆ˜ë©´ pincushion)
    dist_coeffs = np.array([-0.3, 0.1, 0, 0, 0])

    # ì™œê³¡ ì ìš© (ì—­ìœ¼ë¡œ undistort ì‚¬ìš©)
    distorted = cv2.undistort(img, camera_matrix, -dist_coeffs)

    # ì™œê³¡ ë³´ì •
    undistorted = cv2.undistort(distorted, camera_matrix, dist_coeffs)

    cv2.imwrite('undistort_original.jpg', img)
    cv2.imwrite('undistort_distorted.jpg', distorted)
    cv2.imwrite('undistort_corrected.jpg', undistorted)

    print("ì™œê³¡ ë³´ì • ë°©ë²•:")
    print("  1. cv2.undistort()")
    print("     - ê°„ë‹¨í•˜ê²Œ ì‚¬ìš©")
    print("     - ë§¤ë²ˆ ê³„ì‚°")

    print("\n  2. cv2.initUndistortRectifyMap() + cv2.remap()")
    print("     - ë§µì„ ë¯¸ë¦¬ ê³„ì‚°")
    print("     - ë¹„ë””ì˜¤ì—ì„œ íš¨ìœ¨ì ")

)
