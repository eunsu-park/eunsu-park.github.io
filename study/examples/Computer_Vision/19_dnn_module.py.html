def blob_creation_demo():
    """Blob ìƒì„± ë°ëª¨"""
    print("\n" + "=" * 50)
    print("Blob ìƒì„±")
    print("=" * 50)

    # í…ŒìŠ¤íŠ¸ ì´ë¯¸ì§€
    img = np.zeros((480, 640, 3), dtype=np.uint8)
    img[:] = [150, 150, 150]
    cv2.circle(img, (320, 240), 100, (0, 200, 0), -1)

    # Blob ìƒì„±
    # scalefactor: í”½ì…€ ê°’ ìŠ¤ì¼€ì¼ë§ (ë³´í†µ 1/255)
    # size: ë„¤íŠ¸ì›Œí¬ ì…ë ¥ í¬ê¸°
    # mean: í‰ê·  ê°’ ë¹¼ê¸° (BGR ìˆœì„œ)
    # swapRB: BGR -> RGB ë³€í™˜
    # crop: í¬ê¸° ì¡°ì • ì‹œ í¬ë¡­ ì—¬ë¶€

    blob = cv2.dnn.blobFromImage(
        img,
        scalefactor=1/255.0,
        size=(224, 224),
        mean=(0, 0, 0),
        swapRB=True,
        crop=False
    )

    print(f"ì›ë³¸ ì´ë¯¸ì§€: {img.shape}")
    print(f"Blob shape: {blob.shape}")
    print(f"Blob dtype: {blob.dtype}")

    print("\nblobFromImage íŒŒë¼ë¯¸í„°:")
    print("  scalefactor: ë³´í†µ 1/255.0 (0-1 ì •ê·œí™”)")
    print("  size: ë„¤íŠ¸ì›Œí¬ ì…ë ¥ í¬ê¸° (224x224, 416x416 ë“±)")
    print("  mean: ImageNet í‰ê·  (104.0, 117.0, 123.0)")
    print("  swapRB: OpenCV BGR -> ëª¨ë¸ RGB")
    print("  crop: Trueë©´ í¬ë¡­, Falseë©´ ë¦¬ì‚¬ì´ì¦ˆë§Œ")

    # ì—¬ëŸ¬ ì´ë¯¸ì§€ ì²˜ë¦¬

    blob_batch = cv2.dnn.blobFromImages(
        images,
        scalefactor=1/255.0,
        size=(224, 224),
        mean=(0, 0, 0),
        swapRB=True
    )
    print(f"\nBatch blob shape: {blob_batch.shape}")

    cv2.imwrite('dnn_input.jpg', img)

    print("  - ë²„ì „: YOLOv3, YOLOv4, YOLOv5, YOLOv8")

    code = '''

import numpy as np

# ì¶œë ¥ ë ˆì´ì–´ ì´ë¦„

output_layers = [layer_names[i - 1] for i in net.getUnconnectedOutLayers()]

# ì´ë¯¸ì§€ ì „ì²˜ë¦¬
img = cv2.imread('image.jpg')

    img, 1/255.0, (416, 416), (0, 0, 0), swapRB=True, crop=False

# ì¶”ë¡

# ê²°ê³¼ ì²˜ë¦¬

class_ids = []

    for detection in output:
        scores = detection[5:]
        class_id = np.argmax(scores)
        confidence = scores[class_id]

        if confidence > 0.5:
            # ë°”ìš´ë”© ë°•ìŠ¤ ì¢Œí‘œ
            center_x = int(detection[0] * img.shape[1])
            center_y = int(detection[1] * img.shape[0])
            w = int(detection[2] * img.shape[1])
            h = int(detection[3] * img.shape[0])

            x = int(center_x - w / 2)
            y = int(center_y - h / 2)

            boxes.append([x, y, w, h])
            confidences.append(float(confidence))
            class_ids.append(class_id)

# NMS (Non-Maximum Suppression)

# ê²°ê³¼ ì‹œê°í™”

    x, y, w, h = boxes[i]
    cv2.rectangle(img, (x, y), (x+w, y+h), (0, 255, 0), 2)
    label = f"{classes[class_ids[i]]}: {confidences[i]:.2f}"
    cv2.putText(img, label, (x, y-10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
'''
