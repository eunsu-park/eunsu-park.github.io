def document_scanner():
    """ë¬¸ì„œ ìŠ¤ìºë„ˆ í”„ë¡œì íŠ¸"""
    print("=" * 60)
    print("í”„ë¡œì íŠ¸ 1: ë¬¸ì„œ ìŠ¤ìºë„ˆ")
    print("=" * 60)

    # ì‹œë®¬ë ˆì´ì…˜ìš© ë¬¸ì„œ ì´ë¯¸ì§€ ìƒì„±
    # ì‹¤ì œë¡œëŠ” ì¹´ë©”ë¼ë¡œ ì´¬ì˜í•œ ì´ë¯¸ì§€ ì‚¬ìš©
    img = np.zeros((600, 800, 3), dtype=np.uint8)
    img[:] = [150, 150, 150]  # íšŒìƒ‰ ë°°ê²½

    # ê¸°ìš¸ì–´ì§„ ë¬¸ì„œ (ì‚¬ë‹¤ë¦¬ê¼´)
    doc_pts = np.array([[150, 100], [650, 80], [700, 520], [100, 550]], np.int32)
    cv2.fillPoly(img, [doc_pts], (255, 255, 255))

    # ë¬¸ì„œ ë‚´ìš© ì‹œë®¬ë ˆì´ì…˜
    cv2.putText(img, 'DOCUMENT TITLE', (220, 200),
               cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 0, 0), 2)
    cv2.line(img, (200, 250), (600, 240), (100, 100, 100), 2)
    cv2.line(img, (200, 300), (600, 290), (100, 100, 100), 2)
    cv2.line(img, (200, 350), (550, 340), (100, 100, 100), 2)

    cv2.imwrite('scanner_input.jpg', img)

    # 1. ê·¸ë ˆì´ìŠ¤ì¼€ì¼ ë° ë¸”ëŸ¬
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    blurred = cv2.GaussianBlur(gray, (5, 5), 0)

    # 2. ì—£ì§€ ê²€ì¶œ
    edges = cv2.Canny(blurred, 50, 150)

    # 3. ì»¨íˆ¬ì–´ ê²€ì¶œ
    contours, _ = cv2.findContours(edges, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

    # 4. ê°€ì¥ í° ì‚¬ê°í˜• ì»¨íˆ¬ì–´ ì°¾ê¸°
    doc_contour = None
    max_area = 0

    for cnt in contours:
        area = cv2.contourArea(cnt)
        if area > 10000:  # ìµœì†Œ í¬ê¸°
            peri = cv2.arcLength(cnt, True)
            approx = cv2.approxPolyDP(cnt, 0.02 * peri, True)

            if len(approx) == 4 and area > max_area:
                doc_contour = approx
                max_area = area

    if doc_contour is not None:
        # ì½”ë„ˆ ì •ë ¬
        pts = doc_contour.reshape(4, 2)
        rect = order_points(pts)

        # ê²°ê³¼ ì´ë¯¸ì§€ì— í‘œì‹œ
        result_contour = img.copy()
        cv2.drawContours(result_contour, [doc_contour], -1, (0, 255, 0), 3)
        for pt in rect:
            cv2.circle(result_contour, tuple(pt.astype(int)), 10, (0, 0, 255), -1)

        cv2.imwrite('scanner_contour.jpg', result_contour)

        # 5. ì›ê·¼ ë³€í™˜
        width, height = 500, 700  # ì¶œë ¥ í¬ê¸°
        dst = np.array([
            [0, 0],
            [width - 1, 0],
            [width - 1, height - 1],
            [0, height - 1]
        ], dtype=np.float32)

        M = cv2.getPerspectiveTransform(rect, dst)
        warped = cv2.warpPerspective(img, M, (width, height))

        # 6. ì´ì§„í™” (ìŠ¤ìº” íš¨ê³¼)
        warped_gray = cv2.cvtColor(warped, cv2.COLOR_BGR2GRAY)
        scanned = cv2.adaptiveThreshold(
            warped_gray, 255,
            cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
            cv2.THRESH_BINARY, 11, 2
        )

        cv2.imwrite('scanner_warped.jpg', warped)
        cv2.imwrite('scanner_result.jpg', scanned)

        print("ë¬¸ì„œ ìŠ¤ìºë„ˆ ì™„ë£Œ!")
        print("  - scanner_input.jpg: ì›ë³¸")
        print("  - scanner_contour.jpg: ë¬¸ì„œ ê²€ì¶œ")
        print("  - scanner_warped.jpg: ì›ê·¼ ë³´ì •")
        print("  - scanner_result.jpg: ìµœì¢… ìŠ¤ìº”")
    else:
        print("ë¬¸ì„œë¥¼ ì°¾ì„ ìˆ˜ ì—†ìŠµë‹ˆë‹¤.")

    print("\nì²˜ë¦¬ íŒŒì´í”„ë¼ì¸:")
    print("  1. ê·¸ë ˆì´ìŠ¤ì¼€ì¼ + ë¸”ëŸ¬")
    print("  2. Canny ì—£ì§€ ê²€ì¶œ")
    print("  3. ì»¨íˆ¬ì–´ ê²€ì¶œ ë° ê·¼ì‚¬í™”")
    print("  4. 4ê°í˜• ë¬¸ì„œ ì„ íƒ")
    print("  5. ì›ê·¼ ë³€í™˜")
    print("  6. ì´ì§„í™” (ì„ íƒ)")


def order_points(pts):
    """ì½”ë„ˆ ì ì„ [ì¢Œìƒ, ìš°ìƒ, ìš°í•˜, ì¢Œí•˜] ìˆœì„œë¡œ ì •ë ¬"""
    rect = np.zeros((4, 2), dtype=np.float32)

    # í•©ì´ ê°€ì¥ ì‘ì€ ê²ƒ: ì¢Œìƒ
    # í•©ì´ ê°€ì¥ í° ê²ƒ: ìš°í•˜
    s = pts.sum(axis=1)
    rect[0] = pts[np.argmin(s)]
    rect[2] = pts[np.argmax(s)]

    # ì°¨ì´ê°€ ê°€ì¥ ì‘ì€ ê²ƒ: ìš°ìƒ
    # ì°¨ì´ê°€ ê°€ì¥ í° ê²ƒ: ì¢Œí•˜
    diff = np.diff(pts, axis=1)
    rect[1] = pts[np.argmin(diff)]
    rect[3] = pts[np.argmax(diff)]

    return rect

def panorama_stitching():
    """ì´ë¯¸ì§€ íŒŒë…¸ë¼ë§ˆ í”„ë¡œì íŠ¸"""
    print("\n" + "=" * 60)
    print("í”„ë¡œì íŠ¸ 4: ì´ë¯¸ì§€ íŒŒë…¸ë¼ë§ˆ")
    print("=" * 60)

    # ì‹œë®¬ë ˆì´ì…˜ìš© ê²¹ì¹˜ëŠ” ì´ë¯¸ì§€ ìƒì„±
    # ë°°ê²½
    full_scene = np.zeros((300, 800, 3), dtype=np.uint8)
    full_scene[:] = [200, 200, 200]

    # ì¥ë©´ì— ê°ì²´ ë°°ì¹˜
    cv2.circle(full_scene, (100, 150), 50, (0, 0, 150), -1)
    cv2.rectangle(full_scene, (250, 100), (350, 200), (0, 150, 0), -1)
    cv2.circle(full_scene, (500, 150), 60, (150, 0, 0), -1)
    cv2.rectangle(full_scene, (650, 80), (750, 220), (150, 150, 0), -1)

    # ê²¹ì¹˜ëŠ” ë¶€ë¶„ì´ ìˆëŠ” ë‘ ì´ë¯¸ì§€
    img1 = full_scene[:, :450].copy()
    img2 = full_scene[:, 300:].copy()

    cv2.imwrite('panorama_img1.jpg', img1)
    cv2.imwrite('panorama_img2.jpg', img2)

    print("ìŠ¤í‹°ì¹­í•  ì´ë¯¸ì§€ ìƒì„± ì™„ë£Œ")

    # íŠ¹ì§•ì  ê²€ì¶œ ë° ë§¤ì¹­
    gray1 = cv2.cvtColor(img1, cv2.COLOR_BGR2GRAY)
    gray2 = cv2.cvtColor(img2, cv2.COLOR_BGR2GRAY)

    # ORB íŠ¹ì§•ì 
    orb = cv2.ORB_create(nfeatures=500)
    kp1, des1 = orb.detectAndCompute(gray1, None)
    kp2, des2 = orb.detectAndCompute(gray2, None)

    if des1 is not None and des2 is not None:
        # ë§¤ì¹­
        bf = cv2.BFMatcher(cv2.NORM_HAMMING, crossCheck=False)
        matches = bf.knnMatch(des1, des2, k=2)

        # Ratio test
        good = []
        for m, n in matches:
            if m.distance < 0.75 * n.distance:
                good.append(m)

        print(f"ì¢‹ì€ ë§¤ì¹­: {len(good)}")

        if len(good) >= 4:
            # í˜¸ëª¨ê·¸ë˜í”¼ ê³„ì‚°
            src_pts = np.float32([kp1[m.queryIdx].pt for m in good]).reshape(-1, 1, 2)
            dst_pts = np.float32([kp2[m.trainIdx].pt for m in good]).reshape(-1, 1, 2)
