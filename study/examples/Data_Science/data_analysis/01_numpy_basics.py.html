def linear_algebra():
    """ì„ í˜• ëŒ€ìˆ˜ ì—°ì‚°"""
    print("\n[8] ì„ í˜• ëŒ€ìˆ˜")
    print("=" * 50)

    A = np.array([[1, 2], [3, 4]])
    b = np.array([5, 6])

    print(f"A =\n{A}")
    print(f"b = {b}")

    # í–‰ë ¬ì‹
    det = np.linalg.det(A)
    print(f"\ní–‰ë ¬ì‹ det(A) = {det:.4f}")

    # ì—­í–‰ë ¬
    A_inv = np.linalg.inv(A)
    print(f"\nì—­í–‰ë ¬ A^(-1) =\n{A_inv}")

    # ì„ í˜• ì‹œìŠ¤í…œ í’€ê¸°: Ax = b
    x = np.linalg.solve(A, b)
    print(f"\nAx = bì˜ í•´ x = {x}")
    print(f"ê²€ì¦ A @ x = {A @ x}")

    # ê³ ìœ ê°’/ê³ ìœ ë²¡í„°
    eigenvalues, eigenvectors = np.linalg.eig(A)
    print(f"\nê³ ìœ ê°’: {eigenvalues}")
    print(f"ê³ ìœ ë²¡í„°:\n{eigenvectors}")

    # íŠ¹ì´ê°’ ë¶„í•´ (SVD)
    U, S, Vt = np.linalg.svd(A)
    print(f"\nSVD:")
    print(f"U =\n{U}")
    print(f"S = {S}")
    print(f"Vt =\n{Vt}")
