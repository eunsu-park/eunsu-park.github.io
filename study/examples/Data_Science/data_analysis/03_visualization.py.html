def line_plot():

def bar_plot():

    print("\n[3] ë§‰ëŒ€ ê·¸ë˜í”„ (Bar Plot)")

    ax.set_ylabel('ê°’')
    ax.set_title('ê·¸ë£¹í™”ëœ ë§‰ëŒ€ ê·¸ë˜í”„')

    ax.set_xlabel('ì¹´í…Œê³ ë¦¬')

def pie_chart():

    print("\n[5] íŒŒì´ ì°¨íŠ¸ (Pie Chart)")
    print("=" * 50)

    labels = ['Python', 'JavaScript', 'Java', 'C++', 'Other']
    sizes = [35, 25, 20, 10, 10]
    explode = (0.1, 0, 0, 0, 0)  # Python ê°•ì¡°

    # ë„ë„› ì°¨íŠ¸

                                       autopct='%1.1f%%', startangle=90,
                                       wedgeprops=dict(width=0.5))
    ax.set_title('ë„ë„› ì°¨íŠ¸')

    plt.tight_layout()
    plt.savefig('/opt/projects/01_Personal/03_Study/Data_Analysis/examples/05_pie_chart.png', dpi=150)

    vp = ax.violinplot(data, showmeans=True, showmedians=True)
    ax.set_xticks([1, 2, 3, 4])
    ax.set_xticklabels(['Ïƒ=1', 'Ïƒ=2', 'Ïƒ=3', 'Ïƒ=4'])
    ax.set_xlabel('ê·¸ë£¹')

def heatmap():
    """íˆíŠ¸ë§µ"""
    print("\n[7] íˆíŠ¸ë§µ (Heatmap)")
    print("=" * 50)

    np.random.seed(42)

    # ìƒê´€ í–‰ë ¬ ë°ì´í„°
    data = np.random.randn(5, 5)
    data = (data + data.T) / 2  # ëŒ€ì¹­ í–‰ë ¬
    np.fill_diagonal(data, 1)

    fig, axes = plt.subplots(1, 2, figsize=(14, 5))

    # Matplotlib íˆíŠ¸ë§µ
    ax = axes[0]
    im = ax.imshow(data, cmap='coolwarm', aspect='auto')
    ax.set_xticks(range(5))
    ax.set_yticks(range(5))
    ax.set_xticklabels(['A', 'B', 'C', 'D', 'E'])
    ax.set_yticklabels(['A', 'B', 'C', 'D', 'E'])
    plt.colorbar(im, ax=ax)
    ax.set_title('Matplotlib íˆíŠ¸ë§µ')

    # ê°’ í‘œì‹œ
    for i in range(5):
        for j in range(5):
            ax.text(j, i, f'{data[i, j]:.2f}', ha='center', va='center')

    # Seaborn íˆíŠ¸ë§µ (ìˆëŠ” ê²½ìš°)
    ax = axes[1]
    if HAS_SEABORN:
        sns.heatmap(data, annot=True, fmt='.2f', cmap='coolwarm', ax=ax,
                    xticklabels=['A', 'B', 'C', 'D', 'E'],
                    yticklabels=['A', 'B', 'C', 'D', 'E'])
        ax.set_title('Seaborn íˆíŠ¸ë§µ')
    else:
        ax.text(0.5, 0.5, 'Seaborn not available', ha='center', va='center',
                transform=ax.transAxes)
        ax.set_title('Seaborn íˆíŠ¸ë§µ (ì—†ìŒ)')

    plt.tight_layout()
    plt.savefig('/opt/projects/01_Personal/03_Study/Data_Analysis/examples/07_heatmap.png', dpi=150)
    plt.close()
    print("ì €ì¥: 07_heatmap.png")

def plot_3d():
    """3D í”Œë¡¯"""
    print("\n[9] 3D í”Œë¡¯")
    print("=" * 50)

    from mpl_toolkits.mplot3d import Axes3D

    fig = plt.figure(figsize=(14, 5))

    # 3D í‘œë©´
    ax1 = fig.add_subplot(1, 2, 1, projection='3d')
    x = np.linspace(-5, 5, 50)
    y = np.linspace(-5, 5, 50)
    X, Y = np.meshgrid(x, y)
    Z = np.sin(np.sqrt(X**2 + Y**2))

    ax1.plot_surface(X, Y, Z, cmap='viridis', alpha=0.8)
    ax1.set_xlabel('X')
    ax1.set_ylabel('Y')
    ax1.set_zlabel('Z')
    ax1.set_title('3D í‘œë©´')

    # 3D ì‚°ì ë„
    ax2 = fig.add_subplot(1, 2, 2, projection='3d')
    n = 100
    xs = np.random.randn(n)
    ys = np.random.randn(n)
    zs = np.random.randn(n)
    colors = np.random.rand(n)

    ax2.scatter(xs, ys, zs, c=colors, cmap='plasma')
    ax2.set_xlabel('X')
    ax2.set_ylabel('Y')
    ax2.set_zlabel('Z')
    ax2.set_title('3D ì‚°ì ë„')

    plt.tight_layout()
    plt.savefig('/opt/projects/01_Personal/03_Study/Data_Analysis/examples/09_3d_plot.png', dpi=150)
