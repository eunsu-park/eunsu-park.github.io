def handle_outliers():
    """ì´ìƒì¹˜ íƒì§€ ë° ì²˜ë¦¬"""
    print("\n[2] ì´ìƒì¹˜ íƒì§€ ë° ì²˜ë¦¬")
    print("=" * 50)

    np.random.seed(42)

    # ì´ìƒì¹˜ê°€ í¬í•¨ëœ ë°ì´í„°
    normal_data = np.random.normal(100, 10, 100)
    outliers = np.array([200, -50, 250])
    data = np.concatenate([normal_data, outliers])
    np.random.shuffle(data)

    df = pd.DataFrame({'value': data})

    print(f"ë°ì´í„° í¬ê¸°: {len(df)}")
    print(f"í‰ê· : {df['value'].mean():.2f}")
    print(f"í‘œì¤€í¸ì°¨: {df['value'].std():.2f}")

    # ë°©ë²• 1: IQR ë°©ë²•
    print("\n--- IQR ë°©ë²• ---")
    Q1 = df['value'].quantile(0.25)
    Q3 = df['value'].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR

    print(f"Q1: {Q1:.2f}, Q3: {Q3:.2f}, IQR: {IQR:.2f}")
    print(f"ì •ìƒ ë²”ìœ„: [{lower_bound:.2f}, {upper_bound:.2f}]")

    outliers_iqr = df[(df['value'] < lower_bound) | (df['value'] > upper_bound)]
    print(f"ì´ìƒì¹˜ ê°œìˆ˜: {len(outliers_iqr)}")
    print(f"ì´ìƒì¹˜ ê°’: {outliers_iqr['value'].values}")

    # ë°©ë²• 2: Z-score ë°©ë²•
    print("\n--- Z-score ë°©ë²• ---")
    z_scores = np.abs((df['value'] - df['value'].mean()) / df['value'].std())
    outliers_z = df[z_scores > 3]
    print(f"ì´ìƒì¹˜ ê°œìˆ˜ (|z| > 3): {len(outliers_z)}")

    # ì´ìƒì¹˜ ì²˜ë¦¬
    print("\n--- ì´ìƒì¹˜ ì²˜ë¦¬ ---")

    # 1. ì œê±°
    df_no_outliers = df[(df['value'] >= lower_bound) & (df['value'] <= upper_bound)]
    print(f"1. ì œê±° í›„ í¬ê¸°: {len(df_no_outliers)}")

    # 2. ê²½ê³„ê°’ìœ¼ë¡œ ëŒ€ì²´ (Winsorizing)
    df_winsorized = df.copy()
    df_winsorized['value'] = df_winsorized['value'].clip(lower_bound, upper_bound)
    print(f"2. Winsorizing í›„ ìµœëŒ€ê°’: {df_winsorized['value'].max():.2f}")

    # 3. ì¤‘ì•™ê°’ìœ¼ë¡œ ëŒ€ì²´
    df_median = df.copy()
    median_val = df['value'].median()
    df_median.loc[(df['value'] < lower_bound) | (df['value'] > upper_bound), 'value'] = median_val
    print(f"3. ì¤‘ì•™ê°’ ëŒ€ì²´ í›„ í‰ê· : {df_median['value'].mean():.2f}")

def normalization_standardization():
    """ì •ê·œí™”ì™€ í‘œì¤€í™”"""
    print("\n[5] ì •ê·œí™”ì™€ í‘œì¤€í™”")
    print("=" * 50)

    np.random.seed(42)

    df = pd.DataFrame({
        'feature1': np.random.normal(100, 15, 10),
        'feature2': np.random.normal(50, 5, 10),
        'feature3': np.random.exponential(10, 10)
    })

    print("ì›ë³¸ ë°ì´í„° í†µê³„:")
    print(df.describe().round(2))

    # 1. Min-Max ì •ê·œí™” (0-1 ìŠ¤ì¼€ì¼ë§)
    df_minmax = df.copy()
    for col in df.columns:
        min_val = df[col].min()
        max_val = df[col].max()
        df_minmax[col] = (df[col] - min_val) / (max_val - min_val)

    print("\n1. Min-Max ì •ê·œí™” (0-1):")
    print(df_minmax.describe().round(4))

    # 2. Z-score í‘œì¤€í™”
    df_zscore = df.copy()
    for col in df.columns:
        mean_val = df[col].mean()
        std_val = df[col].std()
        df_zscore[col] = (df[col] - mean_val) / std_val

    print("\n2. Z-score í‘œì¤€í™”:")
    print(df_zscore.describe().round(4))

    # 3. Robust ìŠ¤ì¼€ì¼ë§ (ì´ìƒì¹˜ì— ê°•ê±´)
    df_robust = df.copy()
    for col in df.columns:
        median_val = df[col].median()
        iqr = df[col].quantile(0.75) - df[col].quantile(0.25)
        df_robust[col] = (df[col] - median_val) / iqr

    print("\n3. Robust ìŠ¤ì¼€ì¼ë§ (IQR ê¸°ë°˜):")
    print(df_robust.describe().round(4))

    })

    print("ì›ë³¸ ë°ì´í„°:")

    # 1. ë¼ë²¨ ì¸ì½”ë”©

    # 2. ì›-í•« ì¸ì½”ë”©

    df_freq['color_freq'] = df_freq['color'].map(freq_map)
    print(df_freq)
