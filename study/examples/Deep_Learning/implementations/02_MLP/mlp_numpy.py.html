"""
Multi-Layer Perceptron - NumPy From-Scratch êµ¬í˜„

ì´ íŒŒì¼ì€ MLPë¥¼ ìˆœìˆ˜ NumPyë¡œ êµ¬í˜„í•©ë‹ˆë‹¤.
Backpropagation ì•Œê³ ë¦¬ì¦˜ì„ ì§ì ‘ êµ¬í˜„í•˜ì—¬
ë”¥ëŸ¬ë‹ì˜ í•µì‹¬ ì›ë¦¬ë¥¼ ì´í•´í•©ë‹ˆë‹¤.

í•™ìŠµ ëª©í‘œ:
1. Forward pass: ë‹¤ì¸µ ì‹ ê²½ë§ì˜ ìˆœì „íŒŒ
2. Backward pass: Chain ruleì„ ì´ìš©í•œ ì—­ì „íŒŒ
3. Activation functions: ReLU, Sigmoid, Tanh
4. Weight initialization: Xavier, He ì´ˆê¸°í™”
"""

import numpy as np


class ActivationFunctions:
    """í™œì„±í™” í•¨ìˆ˜ì™€ ê·¸ ë¯¸ë¶„"""

    @staticmethod
    def relu(z):
        return np.maximum(0, z)

    @staticmethod
    def relu_derivative(z):
        return (z > 0).astype(float)

    @staticmethod
    def sigmoid(z):
        # ìˆ˜ì¹˜ ì•ˆì •ì„±ì„ ìœ„í•´ í´ë¦¬í•‘
        z = np.clip(z, -500, 500)
        return 1 / (1 + np.exp(-z))

    @staticmethod
    def sigmoid_derivative(z):
        s = ActivationFunctions.sigmoid(z)
        return s * (1 - s)

    @staticmethod
    def tanh(z):
        return np.tanh(z)

    @staticmethod
    def tanh_derivative(z):
        return 1 - np.tanh(z) ** 2

    @staticmethod
    def softmax(z):
        # ìˆ˜ì¹˜ ì•ˆì •ì„±: ìµœëŒ€ê°’ì„ ë¹¼ì¤Œ
        exp_z = np.exp(z - np.max(z, axis=1, keepdims=True))
        return exp_z / np.sum(exp_z, axis=1, keepdims=True)


class Layer:
    """
    ë‹¨ì¼ Fully Connected Layer

    z = Wx + b (ì„ í˜• ë³€í™˜)
    a = Ïƒ(z)   (í™œì„±í™”)
    """

    def __init__(self, input_dim: int, output_dim: int, activation: str = 'relu'):
        """
        Args:
            input_dim: ì…ë ¥ ì°¨ì›
            output_dim: ì¶œë ¥ ì°¨ì›
            activation: 'relu', 'sigmoid', 'tanh', 'none'
        """
        # He ì´ˆê¸°í™” (ReLUìš©)
        if activation == 'relu':
            self.W = np.random.randn(input_dim, output_dim) * np.sqrt(2.0 / input_dim)
        else:
            # Xavier ì´ˆê¸°í™”
            self.W = np.random.randn(input_dim, output_dim) * np.sqrt(1.0 / input_dim)

        self.b = np.zeros((1, output_dim))

        self.activation = activation
        self._get_activation_fn()

        # Gradients
        self.dW = None
        self.db = None

        # Cache (for backward)
        self.cache = {}

    def _get_activation_fn(self):
        """í™œì„±í™” í•¨ìˆ˜ ì„¤ì •"""
        activations = {
            'relu': (ActivationFunctions.relu, ActivationFunctions.relu_derivative),
            'sigmoid': (ActivationFunctions.sigmoid, ActivationFunctions.sigmoid_derivative),
            'tanh': (ActivationFunctions.tanh, ActivationFunctions.tanh_derivative),
            'none': (lambda x: x, lambda x: np.ones_like(x)),
        }
        self.act_fn, self.act_derivative = activations[self.activation]

    def forward(self, x: np.ndarray) -> np.ndarray:
        """
        Forward pass

        Args:
            x: ì…ë ¥ (batch_size, input_dim)

        Returns:
            a: í™œì„±í™” ì¶œë ¥ (batch_size, output_dim)
        """
        # ìºì‹œ ì €ì¥ (backwardì—ì„œ ì‚¬ìš©)
        self.cache['x'] = x

        # ì„ í˜• ë³€í™˜: z = Wx + b
        z = np.dot(x, self.W) + self.b
        self.cache['z'] = z

        # í™œì„±í™”: a = Ïƒ(z)
        a = self.act_fn(z)
        self.cache['a'] = a

        return a

    def backward(self, da: np.ndarray) -> np.ndarray:
        """
        Backward pass

        Args:
            da: ì¶œë ¥ì˜ gradient (batch_size, output_dim)

        Returns:
            dx: ì…ë ¥ì˜ gradient (batch_size, input_dim)
        """
        x = self.cache['x']
        z = self.cache['z']
        batch_size = x.shape[0]

        # âˆ‚L/âˆ‚z = âˆ‚L/âˆ‚a Ã— âˆ‚a/âˆ‚z = da Ã— Ïƒ'(z)
        dz = da * self.act_derivative(z)

        # âˆ‚L/âˆ‚W = x^T Ã— âˆ‚L/âˆ‚z
        self.dW = np.dot(x.T, dz) / batch_size

        # âˆ‚L/âˆ‚b = sum(âˆ‚L/âˆ‚z)
        self.db = np.sum(dz, axis=0, keepdims=True) / batch_size

        # âˆ‚L/âˆ‚x = âˆ‚L/âˆ‚z Ã— W^T (ë‹¤ìŒ ë ˆì´ì–´ë¡œ ì „íŒŒ)
        dx = np.dot(dz, self.W.T)

        return dx


class MLPNumpy:
    """
    Multi-Layer Perceptron (NumPy êµ¬í˜„)

    ì‚¬ìš© ì˜ˆ:
        model = MLPNumpy([784, 256, 128, 10], activations=['relu', 'relu', 'none'])
        model.fit(X_train, y_train)
        predictions = model.predict(X_test)
    """

    def __init__(self, layer_dims: list, activations: list = None):
        """
        Args:
            layer_dims: ê° ë ˆì´ì–´ì˜ ì°¨ì› [input, hidden1, hidden2, ..., output]
            activations: ê° ë ˆì´ì–´ì˜ í™œì„±í™” í•¨ìˆ˜ (ë§ˆì§€ë§‰ ë ˆì´ì–´ ì œì™¸)
        """
        self.layers = []
        n_layers = len(layer_dims) - 1

        if activations is None:
            activations = ['relu'] * (n_layers - 1) + ['none']

        for i in range(n_layers):
            layer = Layer(layer_dims[i], layer_dims[i + 1], activations[i])
            self.layers.append(layer)

    def forward(self, x: np.ndarray) -> np.ndarray:
        """ì „ì²´ ë„¤íŠ¸ì›Œí¬ forward pass"""
        for layer in self.layers:
            x = layer.forward(x)
        return x

    def backward(self, loss_grad: np.ndarray) -> None:
        """ì „ì²´ ë„¤íŠ¸ì›Œí¬ backward pass"""
        grad = loss_grad
        for layer in reversed(self.layers):
            grad = layer.backward(grad)

    def compute_loss(self, y_true: np.ndarray, y_pred: np.ndarray) -> float:
        """
        Cross-entropy loss (ë¶„ë¥˜ìš©)

        L = -1/n Ã— Î£ y_true Ã— log(y_pred)
        """
        eps = 1e-15  # ìˆ˜ì¹˜ ì•ˆì •ì„±
        y_pred = np.clip(y_pred, eps, 1 - eps)

        if y_true.ndim == 1:
            # Sparse labels â†’ one-hot
            n_classes = y_pred.shape[1]
            y_true_onehot = np.zeros((len(y_true), n_classes))
            y_true_onehot[np.arange(len(y_true)), y_true] = 1
            y_true = y_true_onehot

        loss = -np.mean(np.sum(y_true * np.log(y_pred), axis=1))
        return loss

    def compute_loss_gradient(self, y_true: np.ndarray, y_pred: np.ndarray) -> np.ndarray:
        """
        Cross-entropy gradient (softmax ì¶œë ¥ ê°€ì •)

        âˆ‚L/âˆ‚z = y_pred - y_true (softmax + CEì˜ ê²½ìš° ê°„ë‹¨í•´ì§)
        """
        if y_true.ndim == 1:
            n_classes = y_pred.shape[1]
            y_true_onehot = np.zeros((len(y_true), n_classes))
            y_true_onehot[np.arange(len(y_true)), y_true] = 1
            y_true = y_true_onehot

        return y_pred - y_true

    def update_weights(self, lr: float) -> None:
        """SGD ê°€ì¤‘ì¹˜ ì—…ë°ì´íŠ¸"""
        for layer in self.layers:
            layer.W -= lr * layer.dW
            layer.b -= lr * layer.db

    def fit(
        self,
        X: np.ndarray,
        y: np.ndarray,
        epochs: int = 100,
        lr: float = 0.01,
        batch_size: int = 32,
        verbose: bool = True
    ) -> list:
        """
        ëª¨ë¸ í•™ìŠµ

        Args:
            X: í•™ìŠµ ë°ì´í„° (n_samples, n_features)
            y: ë ˆì´ë¸” (n_samples,) ë˜ëŠ” (n_samples, n_classes)
            epochs: ì—í­ ìˆ˜
            lr: learning rate
            batch_size: ë°°ì¹˜ í¬ê¸°
            verbose: ì§„í–‰ ìƒí™© ì¶œë ¥

        Returns:
            losses: ì—í­ë³„ ì†ì‹¤ ë¦¬ìŠ¤íŠ¸
        """
        n_samples = X.shape[0]
        losses = []

        for epoch in range(epochs):
            # ì…”í”Œ
            indices = np.random.permutation(n_samples)
            X_shuffled = X[indices]
            y_shuffled = y[indices] if y.ndim == 1 else y[indices]

            epoch_loss = 0

            # ë¯¸ë‹ˆë°°ì¹˜ í•™ìŠµ
            for i in range(0, n_samples, batch_size):
                X_batch = X_shuffled[i:i + batch_size]
                y_batch = y_shuffled[i:i + batch_size]

                # Forward
                y_pred = self.forward(X_batch)

                # Softmax (ë§ˆì§€ë§‰ ë ˆì´ì–´ê°€ noneì¼ ê²½ìš°)
                y_pred = ActivationFunctions.softmax(y_pred)

                # Loss
                loss = self.compute_loss(y_batch, y_pred)
                epoch_loss += loss * len(X_batch)

                # Backward
                loss_grad = self.compute_loss_gradient(y_batch, y_pred)
                self.backward(loss_grad)

                # Update
                self.update_weights(lr)

            epoch_loss /= n_samples
            losses.append(epoch_loss)

            if verbose and (epoch + 1) % (epochs // 10) == 0:
                accuracy = self.evaluate(X, y)
                print(f"Epoch {epoch + 1}/{epochs}, Loss: {epoch_loss:.4f}, Accuracy: {accuracy:.4f}")

        return losses

    def predict(self, X: np.ndarray) -> np.ndarray:
        """ì˜ˆì¸¡"""
        logits = self.forward(X)
        probs = ActivationFunctions.softmax(logits)
        return np.argmax(probs, axis=1)

    def evaluate(self, X: np.ndarray, y: np.ndarray) -> float:
        """ì •í™•ë„ í‰ê°€"""
        predictions = self.predict(X)
        if y.ndim > 1:
            y = np.argmax(y, axis=1)
        return np.mean(predictions == y)


def load_mnist_sample(n_samples=1000):
    """MNIST ìƒ˜í”Œ ë°ì´í„° ìƒì„± (í…ŒìŠ¤íŠ¸ìš©)"""
    np.random.seed(42)

    # ê°„ë‹¨í•œ ê°€ìƒ ë°ì´í„° (ì‹¤ì œë¡œëŠ” MNIST ë¡œë“œ)
    n_classes = 10
    n_features = 784  # 28x28

    X = np.random.randn(n_samples, n_features) * 0.5
    y = np.random.randint(0, n_classes, n_samples)

    # í´ë˜ìŠ¤ë³„ë¡œ ì•½ê°„ì˜ íŒ¨í„´ ì¶”ê°€
    for i in range(n_classes):
        mask = y == i
        X[mask, i * 78:(i + 1) * 78] += 1.0

    return X, y


def main():
    """ë©”ì¸ ì‹¤í–‰ í•¨ìˆ˜"""
    print("=" * 60)
    print("Multi-Layer Perceptron - NumPy From-Scratch êµ¬í˜„")
    print("=" * 60)

    # 1. ë°ì´í„° ìƒì„±
    print("\n1. ìƒ˜í”Œ ë°ì´í„° ìƒì„±")
    X_train, y_train = load_mnist_sample(n_samples=1000)
    X_test, y_test = load_mnist_sample(n_samples=200)
    print(f"   Train: {X_train.shape}, Test: {X_test.shape}")

    # 2. ëª¨ë¸ ìƒì„±
    print("\n2. MLP ëª¨ë¸ ì´ˆê¸°í™”")
    model = MLPNumpy(
        layer_dims=[784, 128, 64, 10],
        activations=['relu', 'relu', 'none']
    )
    print(f"   Layers: {[l.W.shape for l in model.layers]}")

    # 3. í•™ìŠµ
    print("\n3. í•™ìŠµ ì‹œì‘")
    losses = model.fit(
        X_train, y_train,
        epochs=50,
        lr=0.1,
        batch_size=32,
        verbose=True
    )

    # 4. í‰ê°€
    print("\n4. í‰ê°€ ê²°ê³¼")
    train_acc = model.evaluate(X_train, y_train)
    test_acc = model.evaluate(X_test, y_test)
    print(f"   Train Accuracy: {train_acc:.4f}")
    print(f"   Test Accuracy: {test_acc:.4f}")

    # 5. ì‹œê°í™”
    try:
        import matplotlib.pyplot as plt

        fig, axes = plt.subplots(1, 2, figsize=(12, 4))

        # Loss ê³¡ì„

        axes[0].set_title('Training Loss')
        axes[0].grid(True)

        # ê°€ì¤‘ì¹˜ ë¶„í¬ (ì²« ë²ˆì§¸ ë ˆì´ì–´)

        axes[1].set_xlabel('Weight Value')
        axes[1].set_ylabel('Frequency')
        axes[1].set_title('First Layer Weight Distribution')
        axes[1].grid(True)

        plt.tight_layout()
        plt.savefig('mlp_result.png', dpi=150)
        plt.show()
        print("\nê²°ê³¼ ì´ë¯¸ì§€ ì €ì¥: mlp_result.png")

    except ImportError:
        print("\n(matplotlib ì—†ìŒ, ì‹œê°í™” ìƒëµ)")
