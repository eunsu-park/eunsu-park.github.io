"""

ëª¨ë“  ê²Œì´íŠ¸ ì—°ì‚°ê³¼ BPTTë¥¼ ì§ì ‘ êµ¬í˜„

import numpy as np
from typing import Tuple, Dict, List


def sigmoid(x: np.ndarray) -> np.ndarray:
    """Sigmoid í™œì„±í™” (ìˆ˜ì¹˜ ì•ˆì •ì„± ê³ ë ¤)"""
    return np.where(x >= 0,
                    1 / (1 + np.exp(-x)),
                    np.exp(x) / (1 + np.exp(x)))


def sigmoid_derivative(s: np.ndarray) -> np.ndarray:
    """Sigmoidì˜ derivative: s * (1 - s)"""
    return s * (1 - s)


def tanh_derivative(t: np.ndarray) -> np.ndarray:
    """Tanhì˜ derivative: 1 - t^2"""
    return 1 - t ** 2


class LSTMCellNumPy:
    """
    ë‹¨ì¼ LSTM Cell (NumPy êµ¬í˜„)

    ìˆ˜ì‹:
        f_t = Ïƒ(W_f Â· [h_{t-1}, x_t] + b_f)
        i_t = Ïƒ(W_i Â· [h_{t-1}, x_t] + b_i)
        cÌƒ_t = tanh(W_c Â· [h_{t-1}, x_t] + b_c)
        c_t = f_t âŠ™ c_{t-1} + i_t âŠ™ cÌƒ_t
        o_t = Ïƒ(W_o Â· [h_{t-1}, x_t] + b_o)
        h_t = o_t âŠ™ tanh(c_t)
    """

    def __init__(self, input_size: int, hidden_size: int):
        self.input_size = input_size
        self.hidden_size = hidden_size

        # Xavier ì´ˆê¸°í™”
        concat_size = input_size + hidden_size
        scale = np.sqrt(2.0 / (concat_size + hidden_size))

        # 4ê°œ ê²Œì´íŠ¸ë¥¼ í•˜ë‚˜ì˜ ê°€ì¤‘ì¹˜ë¡œ ê´€ë¦¬ (íš¨ìœ¨ì„±)
        # ìˆœì„œ: forget, input, candidate, output
        self.W = np.random.randn(4 * hidden_size, concat_size) * scale
        self.b = np.zeros(4 * hidden_size)

        # Gradient ì €ì¥
        self.dW = np.zeros_like(self.W)
        self.db = np.zeros_like(self.b)

        # Forward pass ìºì‹œ
        self.cache = {}

    def forward(
        self,
        x: np.ndarray,
        h_prev: np.ndarray,
        c_prev: np.ndarray
    ) -> Tuple[np.ndarray, np.ndarray]:
        """
        Forward pass

        Args:
            x: (batch_size, input_size) í˜„ì¬ ì…ë ¥
            h_prev: (batch_size, hidden_size) ì´ì „ hidden
            c_prev: (batch_size, hidden_size) ì´ì „ cell

        Returns:
            h_t: (batch_size, hidden_size) í˜„ì¬ hidden
            c_t: (batch_size, hidden_size) í˜„ì¬ cell
        """
        batch_size = x.shape[0]
        H = self.hidden_size

        # Concatenate [h_prev, x]
        concat = np.concatenate([h_prev, x], axis=1)  # (batch, hidden+input)

        # ëª¨ë“  ê²Œì´íŠ¸ í•œë²ˆì— ê³„ì‚°
        gates = concat @ self.W.T + self.b  # (batch, 4*hidden)

        # ë¶„ë¦¬
        f_gate = sigmoid(gates[:, 0:H])           # Forget gate
        i_gate = sigmoid(gates[:, H:2*H])         # Input gate
        c_tilde = np.tanh(gates[:, 2*H:3*H])      # Candidate
        o_gate = sigmoid(gates[:, 3*H:4*H])       # Output gate

        # Cell state ì—…ë°ì´íŠ¸
        c_t = f_gate * c_prev + i_gate * c_tilde

        # Hidden state
        h_t = o_gate * np.tanh(c_t)

        # Backwardë¥¼ ìœ„í•œ ìºì‹œ
        self.cache = {
            'x': x,
            'h_prev': h_prev,
            'c_prev': c_prev,
            'concat': concat,
            'f_gate': f_gate,
            'i_gate': i_gate,
            'c_tilde': c_tilde,
            'o_gate': o_gate,
            'c_t': c_t,
            'h_t': h_t,
            'tanh_c_t': np.tanh(c_t),
        }

        return h_t, c_t

    def backward(
        self,
        dh_next: np.ndarray,
        dc_next: np.ndarray
    ) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
        """
        Backward pass (BPTT í•œ ìŠ¤í…)

        Args:
            dh_next: (batch_size, hidden_size) ë‹¤ìŒ ì‹œì ì—ì„œ ì˜¨ h gradient
            dc_next: (batch_size, hidden_size) ë‹¤ìŒ ì‹œì ì—ì„œ ì˜¨ c gradient

        Returns:
            dx: (batch_size, input_size) ì…ë ¥ì— ëŒ€í•œ gradient
            dh_prev: (batch_size, hidden_size) ì´ì „ hidden gradient
            dc_prev: (batch_size, hidden_size) ì´ì „ cell gradient
        """
        cache = self.cache
        H = self.hidden_size

        # Cell state gradient (ë‘ ê²½ë¡œì—ì„œ ì˜´)
        # 1. dh_next â†’ o_gate â†’ tanh(c_t) â†’ c_t
        # 2. dc_next (ë‹¤ìŒ ì‹œì ì—ì„œ ì§ì ‘)
        do = dh_next * cache['tanh_c_t']
        dc = dh_next * cache['o_gate'] * tanh_derivative(cache['tanh_c_t'])
        dc = dc + dc_next  # ë‘ ê²½ë¡œ í•©ì¹¨

        # ê° ê²Œì´íŠ¸ gradient
        df = dc * cache['c_prev']
        di = dc * cache['c_tilde']
        dc_tilde = dc * cache['i_gate']

        # ì´ì „ cell state gradient (í•µì‹¬: forget gateë¥¼ í†µí•´ ì§ì ‘ ì „íŒŒ)
        dc_prev = dc * cache['f_gate']

        # í™œì„±í™” í•¨ìˆ˜ derivative
        df_gate = df * sigmoid_derivative(cache['f_gate'])
        di_gate = di * sigmoid_derivative(cache['i_gate'])
        dc_tilde_gate = dc_tilde * tanh_derivative(cache['c_tilde'])
        do_gate = do * sigmoid_derivative(cache['o_gate'])

        # ëª¨ë“  ê²Œì´íŠ¸ gradient í•©ì¹˜ê¸°
        dgates = np.concatenate([df_gate, di_gate, dc_tilde_gate, do_gate], axis=1)

        # ê°€ì¤‘ì¹˜ gradient
        self.dW += dgates.T @ cache['concat']
        self.db += dgates.sum(axis=0)

        # Concat gradient â†’ h_prev, x gradient
        dconcat = dgates @ self.W
        dh_prev = dconcat[:, :H]
        dx = dconcat[:, H:]

        return dx, dh_prev, dc_prev

    def zero_grad(self):
        """Gradient ì´ˆê¸°í™”"""
        self.dW.fill(0)
        self.db.fill(0)


class LSTMNumPy:
    """
    ì „ì²´ LSTM (ì—¬ëŸ¬ ì‹œì )
    """

    def __init__(self, input_size: int, hidden_size: int, num_layers: int = 1):
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.num_layers = num_layers

        # ë ˆì´ì–´ë³„ LSTM Cell
        self.cells = []
        for i in range(num_layers):
            in_size = input_size if i == 0 else hidden_size
            self.cells.append(LSTMCellNumPy(in_size, hidden_size))

    def forward(
        self,
        x: np.ndarray,
        h_0: np.ndarray = None,
        c_0: np.ndarray = None
    ) -> Tuple[np.ndarray, Tuple[np.ndarray, np.ndarray]]:
        """
        Forward pass (ì „ì²´ ì‹œí€€ìŠ¤)

        Args:
            x: (seq_len, batch_size, input_size)
            h_0: (num_layers, batch_size, hidden_size) ì´ˆê¸° hidden
            c_0: (num_layers, batch_size, hidden_size) ì´ˆê¸° cell

        Returns:
            output: (seq_len, batch_size, hidden_size) ëª¨ë“  ì‹œì ì˜ hidden
            (h_n, c_n): ë§ˆì§€ë§‰ hidden/cell
        """
        seq_len, batch_size, _ = x.shape

        # ì´ˆê¸° ìƒíƒœ
        if h_0 is None:
            h_0 = np.zeros((self.num_layers, batch_size, self.hidden_size))
        if c_0 is None:
            c_0 = np.zeros((self.num_layers, batch_size, self.hidden_size))

        # ì¶œë ¥ ì €ì¥
        outputs = []
        h_states = [h_0[i] for i in range(self.num_layers)]
        c_states = [c_0[i] for i in range(self.num_layers)]

        # ì‹œê°„ì— ë”°ë¥¸ ìºì‹œ (backwardìš©)
        self.time_cache = []

        for t in range(seq_len):
            layer_input = x[t]

            for layer_idx, cell in enumerate(self.cells):
                h_states[layer_idx], c_states[layer_idx] = cell.forward(
                    layer_input, h_states[layer_idx], c_states[layer_idx]
                )
                layer_input = h_states[layer_idx]

            outputs.append(h_states[-1])
            self.time_cache.append([cell.cache.copy() for cell in self.cells])

        output = np.stack(outputs, axis=0)
        h_n = np.stack(h_states, axis=0)
        c_n = np.stack(c_states, axis=0)

        return output, (h_n, c_n)

    def backward(self, doutput: np.ndarray) -> np.ndarray:
        """
        Backward pass (BPTT)

        Args:
            doutput: (seq_len, batch_size, hidden_size) ì¶œë ¥ì— ëŒ€í•œ gradient

        Returns:
            dx: (seq_len, batch_size, input_size) ì…ë ¥ì— ëŒ€í•œ gradient
        """
        seq_len, batch_size, _ = doutput.shape

        # Gradient ì´ˆê¸°í™”
        for cell in self.cells:
            cell.zero_grad()

        dx = np.zeros((seq_len, batch_size, self.input_size))

        # ë ˆì´ì–´ë³„ gradient ì „íŒŒ
        dh_next = [np.zeros((batch_size, self.hidden_size))
                   for _ in range(self.num_layers)]
        dc_next = [np.zeros((batch_size, self.hidden_size))
                   for _ in range(self.num_layers)]

        # ì‹œê°„ ì—­ìˆœ
        for t in reversed(range(seq_len)):
            # ë§ˆì§€ë§‰ ë ˆì´ì–´ì— ì¶œë ¥ gradient ë”í•¨
            dh_next[-1] += doutput[t]

            # ë ˆì´ì–´ ì—­ìˆœ (ê¹Šì€ ë ˆì´ì–´ â†’ ì–•ì€ ë ˆì´ì–´)
            for layer_idx in reversed(range(self.num_layers)):
                cell = self.cells[layer_idx]
                cell.cache = self.time_cache[t][layer_idx]

                dx_layer, dh_prev, dc_prev = cell.backward(
                    dh_next[layer_idx], dc_next[layer_idx]
                )

                # ë‹¤ìŒ ì‹œì ìœ¼ë¡œ ì „íŒŒ
                dh_next[layer_idx] = dh_prev
                dc_next[layer_idx] = dc_prev

                # ì´ì „ ë ˆì´ì–´ë¡œ ì „íŒŒ
                if layer_idx > 0:
                    dh_next[layer_idx - 1] += dx_layer
                else:
                    dx[t] = dx_layer

        return dx

    def parameters(self) -> List[np.ndarray]:
        """ëª¨ë“  íŒŒë¼ë¯¸í„° ë°˜í™˜"""
        params = []
        for cell in self.cells:
            params.extend([cell.W, cell.b])
        return params

    def gradients(self) -> List[np.ndarray]:
        """ëª¨ë“  gradient ë°˜í™˜"""
        grads = []
        for cell in self.cells:
            grads.extend([cell.dW, cell.db])
        return grads


def sgd_update(params: List[np.ndarray], grads: List[np.ndarray], lr: float):
    """SGD ì—…ë°ì´íŠ¸"""
    for param, grad in zip(params, grads):
        param -= lr * grad


def clip_gradients(grads: List[np.ndarray], max_norm: float = 5.0):
    """Gradient clipping (exploding gradient ë°©ì§€)"""
    total_norm = np.sqrt(sum(np.sum(g ** 2) for g in grads))
    if total_norm > max_norm:
        scale = max_norm / (total_norm + 1e-6)
        for g in grads:
            g *= scale

def train_sequence_classification():
    """ê°„ë‹¨í•œ ì‹œí€€ìŠ¤ ë¶„ë¥˜ ì˜ˆì œ"""
    print("\n=== Sequence Classification ===\n")

    np.random.seed(42)

    # ë°ì´í„°: ì‹œí€€ìŠ¤ì˜ í‰ê· ì´ ì–‘ìˆ˜ë©´ 1, ìŒìˆ˜ë©´ 0
    def generate_data(n_samples, seq_len, input_size):
        X = np.random.randn(n_samples, seq_len, input_size)
        y = (X.mean(axis=(1, 2)) > 0).astype(int)
        return X, y

    X_train, y_train = generate_data(100, 10, 5)
    X_test, y_test = generate_data(20, 10, 5)

    # ëª¨ë¸
    lstm = LSTMNumPy(input_size=5, hidden_size=16, num_layers=1)

    # ì¶œë ¥ ë ˆì´ì–´
    W_out = np.random.randn(2, 16) * 0.1
    b_out = np.zeros(2)

    lr = 0.01
    epochs = 50

    for epoch in range(epochs):
        total_loss = 0
        correct = 0

        for i in range(len(X_train)):
            x = X_train[i:i+1].transpose(1, 0, 2)  # (seq, 1, input)
            target = y_train[i]

            # Forward
            output, _ = lstm.forward(x)
            last_hidden = output[-1]  # (1, hidden)

            # ë¶„ë¥˜
            logits = last_hidden @ W_out.T + b_out
            probs = np.exp(logits - logits.max()) / np.exp(logits - logits.max()).sum()

            # Loss (cross entropy)
            loss = -np.log(probs[0, target] + 1e-7)
            total_loss += loss

            # Accuracy
            pred = logits.argmax()
            correct += (pred == target)

            # Backward
            dlogits = probs.copy()
            dlogits[0, target] -= 1

            dW_out = dlogits.T @ last_hidden
            db_out = dlogits.sum(axis=0)
            dlast_hidden = dlogits @ W_out

            doutput = np.zeros_like(output)
            doutput[-1] = dlast_hidden

            lstm.backward(doutput)

            # Gradient clipping
            clip_gradients(lstm.gradients())

            # Update
            sgd_update(lstm.parameters(), lstm.gradients(), lr)
            W_out -= lr * dW_out
            b_out -= lr * db_out

        if (epoch + 1) % 10 == 0:
            acc = correct / len(X_train)
            print(f"Epoch {epoch+1}: Loss={total_loss/len(X_train):.4f}, Acc={acc:.2f}")

    # í…ŒìŠ¤íŠ¸
    test_correct = 0
    for i in range(len(X_test)):
        x = X_test[i:i+1].transpose(1, 0, 2)
        target = y_test[i]

        output, _ = lstm.forward(x)
        logits = output[-1] @ W_out.T + b_out
        pred = logits.argmax()
        test_correct += (pred == target)

    print(f"\nTest Accuracy: {test_correct/len(X_test):.2f}")


if __name__ == "__main__":
    test_lstm()
    train_sequence_classification()
