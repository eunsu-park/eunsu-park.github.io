"""

NumPyë¡œ ì—­ì „íŒŒë¥¼ ì§ì ‘ êµ¬í˜„í•˜ì—¬ ì›ë¦¬ë¥¼ ì´í•´í•©ë‹ˆë‹¤.

def sigmoid(x):
    return 1 / (1 + np.exp(-np.clip(x, -500, 500)))

def sigmoid_derivative(x):
    """sigmoid'(x) = sigmoid(x) * (1 - sigmoid(x))"""
    s = sigmoid(x)
    return s * (1 - s)

def relu(x):
    return np.maximum(0, x)

def relu_derivative(x):
    """relu'(x) = 1 if x > 0 else 0"""
    return (x > 0).astype(float)

print(f"relu(x): {relu(x)}")
print(f"relu'(x): {relu_derivative(x)}")

class SingleNeuron:
    """
    ë‹¨ì¼ ë‰´ëŸ°: y = sigmoid(w*x + b)
    ì†ì‹¤: L = (y - target)^2
    """
    def __init__(self):
        self.w = np.random.randn()
        self.b = np.random.randn()

    def forward(self, x, target):
        """ìˆœì „íŒŒ"""
        self.x = x
        self.target = target

        # ë‹¨ê³„ë³„ ê³„ì‚° (ìºì‹œì— ì €ì¥)
        self.z = self.w * x + self.b      # ì„ í˜• ë³€í™˜
        self.a = sigmoid(self.z)           # í™œì„±í™”
        self.loss = (self.a - target) ** 2 # MSE

        return self.a, self.loss

    def backward(self):
        """
        ì—­ì „íŒŒ: ì²´ì¸ ë£° ì ìš©

        dL/dw = (dL/da) * (da/dz) * (dz/dw)
        dL/db = (dL/da) * (da/dz) * (dz/db)
        """
        # 1. ì†ì‹¤ â†’ í™œì„±í™”
        dL_da = 2 * (self.a - self.target)

        # 2. í™œì„±í™” â†’ ì„ í˜• (ì‹œê·¸ëª¨ì´ë“œ ë¯¸ë¶„)
        da_dz = sigmoid_derivative(self.z)

        # 3. ì„ í˜• â†’ ê°€ì¤‘ì¹˜/í¸í–¥
        dz_dw = self.x
        dz_db = 1

        # ì²´ì¸ ë£° ì ìš©
        dL_dw = dL_da * da_dz * dz_dw
        dL_db = dL_da * da_dz * dz_db

        return dL_dw, dL_db

# í…ŒìŠ¤íŠ¸

class MLPFromScratch:
    """
    2ì¸µ MLP with ì—­ì „íŒŒ

    êµ¬ì¡°: ì…ë ¥ â†’ [W1, b1] â†’ ReLU â†’ [W2, b2] â†’ Sigmoid â†’ ì¶œë ¥
    """
    def __init__(self, input_dim, hidden_dim, output_dim):
        # Xavier ì´ˆê¸°í™”
        self.W1 = np.random.randn(input_dim, hidden_dim) * np.sqrt(2.0 / input_dim)
        self.b1 = np.zeros(hidden_dim)
        self.W2 = np.random.randn(hidden_dim, output_dim) * np.sqrt(2.0 / hidden_dim)
        self.b2 = np.zeros(output_dim)

        print(f"MLP ìƒì„±: {input_dim} â†’ {hidden_dim} â†’ {output_dim}")

    def forward(self, X):
        """ìˆœì „íŒŒ (ì¤‘ê°„ê°’ ìºì‹œ)"""
        # ì²« ë²ˆì§¸ ì¸µ
        self.z1 = X @ self.W1 + self.b1
        self.a1 = relu(self.z1)

        # ë‘ ë²ˆì§¸ ì¸µ
        self.z2 = self.a1 @ self.W2 + self.b2
        self.a2 = sigmoid(self.z2)

        return self.a2

    def backward(self, X, y_true):
        """
        ì—­ì „íŒŒ: ì²´ì¸ ë£°ë¡œ ëª¨ë“  ê¸°ìš¸ê¸° ê³„ì‚°

        í•µì‹¬ ê³µì‹:
        dL/dW2 = a1.T @ (dL/dz2)
        dL/dW1 = X.T @ (dL/dz1)
        """
        m = X.shape[0]  # ë°°ì¹˜ í¬ê¸°

        # ===== ì¶œë ¥ì¸µ ì—­ì „íŒŒ =====
        # dL/da2 = 2(a2 - y) for MSE
        dL_da2 = 2 * (self.a2 - y_true) / m

        # dL/dz2 = dL/da2 * sigmoid'(z2)
        dL_dz2 = dL_da2 * sigmoid_derivative(self.z2)

        # dL/dW2 = a1.T @ dL/dz2
        dW2 = self.a1.T @ dL_dz2
        db2 = np.sum(dL_dz2, axis=0)

        # ===== ì€ë‹‰ì¸µ ì—­ì „íŒŒ =====
        # dL/da1 = dL/dz2 @ W2.T (ê¸°ìš¸ê¸° ì—­ì „íŒŒ)
        dL_da1 = dL_dz2 @ self.W2.T

        # dL/dz1 = dL/da1 * relu'(z1)
        dL_dz1 = dL_da1 * relu_derivative(self.z1)

        # dL/dW1 = X.T @ dL/dz1
        dW1 = X.T @ dL_dz1
        db1 = np.sum(dL_dz1, axis=0)

        return {'W1': dW1, 'b1': db1, 'W2': dW2, 'b2': db2}

    def update(self, grads, lr):
        """ê²½ì‚¬ í•˜ê°•ë²•ìœ¼ë¡œ ê°€ì¤‘ì¹˜ ì—…ë°ì´íŠ¸"""
        self.W1 -= lr * grads['W1']
        self.b1 -= lr * grads['b1']
        self.W2 -= lr * grads['W2']
        self.b2 -= lr * grads['b2']

    def loss(self, y_pred, y_true):
        """MSE ì†ì‹¤"""
        return np.mean((y_pred - y_true) ** 2)

def numerical_gradient(model, X, y, param_name, h=1e-5):
    """ìˆ˜ì¹˜ ë¯¸ë¶„ìœ¼ë¡œ ê¸°ìš¸ê¸° ê³„ì‚°"""
    param = getattr(model, param_name)
    grad = np.zeros_like(param)

    it = np.nditer(param, flags=['multi_index'])
    while not it.finished:
        idx = it.multi_index
        original = param[idx]

        # f(x + h)
        param[idx] = original + h
        loss_plus = model.loss(model.forward(X), y)

        # f(x - h)
        param[idx] = original - h
        loss_minus = model.loss(model.forward(X), y)

        # ìˆ˜ì¹˜ ë¯¸ë¶„
        grad[idx] = (loss_plus - loss_minus) / (2 * h)

        param[idx] = original
        it.iternext()

    return grad
