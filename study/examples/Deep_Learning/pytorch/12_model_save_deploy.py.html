"""
12. Î™®Îç∏ Ï†ÄÏû• Î∞è Î∞∞Ìè¨

PyTorch Î™®Îç∏ Ï†ÄÏû•, TorchScript, ONNX Î≥ÄÌôòÏùÑ Íµ¨ÌòÑÌï©ÎãàÎã§.

import torch
import torch.nn as nn
import torch.nn.functional as F
import os
import tempfile

print("=" * 60)
print("PyTorch Î™®Îç∏ Ï†ÄÏû• Î∞è Î∞∞Ìè¨")
print("=" * 60)


# ============================================
# 1. ÏÉòÌîå Î™®Îç∏
# ============================================
print("\n[1] ÏÉòÌîå Î™®Îç∏")
print("-" * 40)

class SimpleClassifier(nn.Module):
    def __init__(self, input_size=784, hidden_size=256, num_classes=10):
        super().__init__()
        self.config = {
            'input_size': input_size,
            'hidden_size': hidden_size,
            'num_classes': num_classes
        }
        self.fc1 = nn.Linear(input_size, hidden_size)
        self.bn1 = nn.BatchNorm1d(hidden_size)
        self.fc2 = nn.Linear(hidden_size, num_classes)

    def forward(self, x):
        x = x.view(x.size(0), -1)
        x = F.relu(self.bn1(self.fc1(x)))
        x = self.fc2(x)
        return x

model = SimpleClassifier()
print(f"Î™®Îç∏ Íµ¨Ï°∞:\n{model}")
print(f"ÌååÎùºÎØ∏ÌÑ∞ Ïàò: {sum(p.numel() for p in model.parameters()):,}")


# ============================================
# 2. state_dict Ï†ÄÏû•
# ============================================
print("\n[2] state_dict Ï†ÄÏû•")
print("-" * 40)

# ÏûÑÏãú ÎîîÎ†âÌÜ†Î¶¨ ÏÇ¨Ïö©
save_dir = tempfile.mkdtemp()

# Ï†ÄÏû•
weights_path = os.path.join(save_dir, 'model_weights.pth')
torch.save(model.state_dict(), weights_path)
print(f"Ï†ÄÏû•: {weights_path}")
print(f"ÌååÏùº ÌÅ¨Í∏∞: {os.path.getsize(weights_path) / 1024:.2f} KB")

# Î°úÎìú
loaded_model = SimpleClassifier()
loaded_model.load_state_dict(torch.load(weights_path, weights_only=True))
loaded_model.eval()

# Í≤ÄÏ¶ù
x = torch.randn(2, 1, 28, 28)
model.eval()
with torch.no_grad():
    original_out = model(x)
    loaded_out = loaded_model(x)
    diff = (original_out - loaded_out).abs().max().item()
    print(f"Ï∂úÎ†• Ï∞®Ïù¥: {diff:.10f}")


# ============================================
# 3. Ï≤¥ÌÅ¨Ìè¨Ïù∏Ìä∏ Ï†ÄÏû•
# ============================================
print("\n[3] Ï≤¥ÌÅ¨Ìè¨Ïù∏Ìä∏ Ï†ÄÏû•")
print("-" * 40)

optimizer = torch.optim.Adam(model.parameters(), lr=0.001)

# Í∞ÄÏßú ÌïôÏäµ ÏÉÅÌÉú
epoch = 10
loss = 0.123
best_acc = 0.95

# Ï≤¥ÌÅ¨Ìè¨Ïù∏Ìä∏ Ï†ÄÏû•
checkpoint = {
    'epoch': epoch,
    'model_state_dict': model.state_dict(),
    'optimizer_state_dict': optimizer.state_dict(),
    'loss': loss,
    'best_acc': best_acc,
    'model_config': model.config
}

checkpoint_path = os.path.join(save_dir, 'checkpoint.pth')
torch.save(checkpoint, checkpoint_path)
print(f"Ï≤¥ÌÅ¨Ìè¨Ïù∏Ìä∏ Ï†ÄÏû•: {checkpoint_path}")

# Ï≤¥ÌÅ¨Ìè¨Ïù∏Ìä∏ Î°úÎìú
loaded_checkpoint = torch.load(checkpoint_path, weights_only=False)
print(f"Î°úÎìúÎêú epoch: {loaded_checkpoint['epoch']}")
print(f"Î°úÎìúÎêú best_acc: {loaded_checkpoint['best_acc']}")
print(f"Î™®Îç∏ ÏÑ§Ï†ï: {loaded_checkpoint['model_config']}")


# ============================================
# 4. TorchScript - Tracing
# ============================================

model.eval()
example_input = torch.randn(1, 1, 28, 28)

# Trace

# Ï†ÄÏû•

print(f"TorchScript Ï†ÄÏû•: {traced_path}")
print(f"ÌååÏùº ÌÅ¨Í∏∞: {os.path.getsize(traced_path) / 1024:.2f} KB")

# Î°úÎìú Î∞è Í≤ÄÏ¶ù

with torch.no_grad():
    traced_out = loaded_traced(example_input)
    original_out = model(example_input)
    diff = (traced_out - original_out).abs().max().item()
    print(f"Ï∂úÎ†• Ï∞®Ïù¥: {diff:.10f}")


# ============================================
# 5. TorchScript - Scripting
# ============================================
print("\n[5] TorchScript - Scripting")
print("-" * 40)

class ConditionalModel(nn.Module):
    """Ï°∞Í±¥Î¨∏Ïù¥ ÏûàÎäî Î™®Îç∏"""
    def __init__(self):
        super().__init__()
        self.fc = nn.Linear(10, 5)

    def forward(self, x, use_relu: bool = True):
        x = self.fc(x)
        if use_relu:
            x = F.relu(x)
        return x

cond_model = ConditionalModel()
scripted_model = torch.jit.script(cond_model)

scripted_path = os.path.join(save_dir, 'model_scripted.pt')
scripted_model.save(scripted_path)
print(f"Scripted Î™®Îç∏ Ï†ÄÏû•: {scripted_path}")

# Ï°∞Í±¥Î∂Ä Ïã§Ìñâ ÌÖåÏä§Ìä∏

out_relu = scripted_model(x, True)
out_no_relu = scripted_model(x, False)
print(f"ReLU Ï†ÅÏö©: min={out_relu.min():.4f}")
print(f"ReLU ÎØ∏Ï†ÅÏö©: min={out_no_relu.min():.4f}")

# 6. ONNX Î≥ÄÌôò
# ============================================

try:
    import onnx

    model.eval()
    dummy_input = torch.randn(1, 1, 28, 28)

    onnx_path = os.path.join(save_dir, 'model.onnx')

    torch.onnx.export(
        model,
        dummy_input,
        onnx_path,
        input_names=['input'],
        output_names=['output'],
        dynamic_axes={
            'input': {0: 'batch_size'},
            'output': {0: 'batch_size'}

        opset_version=11

    print(f"ONNX Ï†ÄÏû•: {onnx_path}")
    print(f"ÌååÏùº ÌÅ¨Í∏∞: {os.path.getsize(onnx_path) / 1024:.2f} KB")

    # Í≤ÄÏ¶ù
    onnx_model = onnx.load(onnx_path)
    onnx.checker.check_model(onnx_model)
    print("ONNX Î™®Îç∏ Í≤ÄÏ¶ù ÌÜµÍ≥º")

except ImportError:
    print("onnx ÎØ∏ÏÑ§Ïπò - Ïä§ÌÇµ")

try:
    import onnxruntime as ort
    import numpy as np

    session = ort.InferenceSession(onnx_path)

    input_name = session.get_inputs()[0].name
    output_name = session.get_outputs()[0].name

    # Ï∂îÎ°†
    input_data = np.random.randn(2, 1, 28, 28).astype(np.float32)

    print(f"ONNX Runtime Ï∂úÎ†•: {result[0].shape}")

    # PyTorch Í≤∞Í≥ºÏôÄ ÎπÑÍµê
    model.eval()
    with torch.no_grad():
        torch_out = model(torch.from_numpy(input_data))
        diff = np.abs(result[0] - torch_out.numpy()).max()
        print(f"PyTorch vs ONNX Ï∞®Ïù¥: {diff:.6f}")

    print("onnxruntime ÎØ∏ÏÑ§Ïπò - Ïä§ÌÇµ")
