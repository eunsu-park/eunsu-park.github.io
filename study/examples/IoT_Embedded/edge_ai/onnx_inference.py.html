class ONNXModel:
    """ONNX ëª¨ë¸ ê¸°ë³¸ ë˜í¼"""

    def __init__(self, model_path: str, providers: Optional[List[str]] = None):
        if not HAS_ONNX:
            raise ImportError("onnxruntimeì´ í•„ìš”í•©ë‹ˆë‹¤")

        if not os.path.exists(model_path):
            raise FileNotFoundError(f"ëª¨ë¸ íŒŒì¼ì„ ì°¾ì„ ìˆ˜ ì—†ìŠµë‹ˆë‹¤: {model_path}")

        if providers is None:
            # ì‚¬ìš© ê°€ëŠ¥í•œ í”„ë¡œë°”ì´ë” ìë™ ì„ íƒ
            available = ort.get_available_providers()
            if 'CUDAExecutionProvider' in available:
                providers = ['CUDAExecutionProvider', 'CPUExecutionProvider']
            else:
                providers = ['CPUExecutionProvider']

        # ì„¸ì…˜ ì˜µì…˜ ì„¤ì •
        sess_options = ort.SessionOptions()
        sess_options.graph_optimization_level = ort.GraphOptimizationLevel.ORT_ENABLE_ALL
        sess_options.intra_op_num_threads = 4  # CPU ìŠ¤ë ˆë“œ ìˆ˜

        # ì„¸ì…˜ ìƒì„±
        self.session = ort.InferenceSession(
            model_path,
            sess_options=sess_options,
            providers=providers
        )

        # ì…ì¶œë ¥ ì •ë³´
        self.input_name = self.session.get_inputs()[0].name
        self.input_shape = self.session.get_inputs()[0].shape
        self.input_type = self.session.get_inputs()[0].type
        self.output_name = self.session.get_outputs()[0].name

        print(f"ëª¨ë¸ ë¡œë“œ ì™„ë£Œ: {model_path}")
        print(f"  í”„ë¡œë°”ì´ë”: {self.session.get_providers()}")
        print(f"  ì…ë ¥: {self.input_name} {self.input_shape}")
        print(f"  ì¶œë ¥: {self.output_name}")

    def get_input_shape(self) -> list:
        """ì…ë ¥ í˜•íƒœ ë°˜í™˜"""
        return self.input_shape

    def predict(self, input_data: np.ndarray) -> np.ndarray:
        """ì¶”ë¡  ìˆ˜í–‰"""
        outputs = self.session.run(
            [self.output_name],
            {self.input_name: input_data}
        )
        return outputs[0]

    def benchmark(self, num_iterations: int = 100) -> Dict[str, float]:
        """ì„±ëŠ¥ ë²¤ì¹˜ë§ˆí¬"""
        # ë”ë¯¸ ì…ë ¥ ìƒì„±
        dummy_shape = [1 if x == 'batch' or x == 'N' or x is None else x
                      for x in self.input_shape]
        dummy_input = np.random.randn(*dummy_shape).astype(np.float32)

        # ì›Œë°ì—…
        for _ in range(10):
            self.predict(dummy_input)

        # ì¸¡ì •
        times = []
        for _ in range(num_iterations):
            start = time.perf_counter()
            self.predict(dummy_input)
            elapsed = (time.perf_counter() - start) * 1000  # ms
            times.append(elapsed)

        times = np.array(times)

        results = {
            "mean_ms": float(np.mean(times)),
            "std_ms": float(np.std(times)),
            "min_ms": float(np.min(times)),
            "max_ms": float(np.max(times)),
            "fps": 1000.0 / np.mean(times)
        }

        return results

class ImageClassifier(ONNXModel):
    """ONNX ì´ë¯¸ì§€ ë¶„ë¥˜ ëª¨ë¸"""

    # ImageNet í´ë˜ìŠ¤ (ìƒìœ„ 10ê°œë§Œ ì˜ˆì‹œ)
    IMAGENET_CLASSES = [
        'tench', 'goldfish', 'great_white_shark', 'tiger_shark',
        'hammerhead', 'electric_ray', 'stingray', 'cock', 'hen', 'ostrich'
        # ... ì‹¤ì œë¡œëŠ” 1000ê°œ í´ë˜ìŠ¤
    ]

    def __init__(self, model_path: str, labels_path: Optional[str] = None):
        super().__init__(model_path)

        # ë ˆì´ë¸” ë¡œë“œ (ìˆëŠ” ê²½ìš°)
        if labels_path and os.path.exists(labels_path):
            with open(labels_path, 'r') as f:
                self.labels = [line.strip() for line in f]
        else:
            self.labels = self.IMAGENET_CLASSES

        # ì…ë ¥ í¬ê¸° ì¶”ì¶œ
        self.input_height = self.input_shape[2] if len(self.input_shape) > 2 else 224
        self.input_width = self.input_shape[3] if len(self.input_shape) > 3 else 224

    def preprocess_image(self, image_path: str) -> np.ndarray:
        """ì´ë¯¸ì§€ ì „ì²˜ë¦¬ (PIL ì‚¬ìš©)"""
        if not HAS_PIL:
            raise ImportError("Pillowê°€ í•„ìš”í•©ë‹ˆë‹¤")

        # ì´ë¯¸ì§€ ë¡œë“œ
        image = Image.open(image_path).convert('RGB')

        # ë¦¬ì‚¬ì´ì¦ˆ
        image = image.resize((self.input_width, self.input_height))

        # NumPy ë°°ì—´ë¡œ ë³€í™˜
        img_array = np.array(image).astype(np.float32)

        # ì •ê·œí™” (ImageNet í‘œì¤€)
        mean = np.array([0.485, 0.456, 0.406]) * 255
        std = np.array([0.229, 0.224, 0.225]) * 255
        img_array = (img_array - mean) / std

        # HWC to CHW
        img_array = img_array.transpose(2, 0, 1)

        # ë°°ì¹˜ ì°¨ì› ì¶”ê°€
        img_array = np.expand_dims(img_array, axis=0)

        return img_array

    def classify(self, image_path: str, top_k: int = 5) -> List[Tuple[str, float]]:
        """ì´ë¯¸ì§€ ë¶„ë¥˜"""
        # ì „ì²˜ë¦¬
        input_data = self.preprocess_image(image_path)

        # ì¶”ë¡ 
        start = time.perf_counter()
        output = self.predict(input_data)
        inference_time = (time.perf_counter() - start) * 1000

        # Softmax
        probs = self._softmax(output[0])

        # Top-K ê²°ê³¼
        top_indices = np.argsort(probs)[-top_k:][::-1]

        results = []
        for idx in top_indices:
            label = self.labels[idx] if idx < len(self.labels) else f"class_{idx}"
            results.append((label, float(probs[idx])))

        print(f"ì¶”ë¡  ì‹œê°„: {inference_time:.2f}ms")

        return results

    @staticmethod
    def _softmax(x: np.ndarray) -> np.ndarray:
        """Softmax í•¨ìˆ˜"""
        exp_x = np.exp(x - np.max(x))
        return exp_x / exp_x.sum()

class YOLODetector:
    """YOLO ONNX ê°ì²´ ê²€ì¶œê¸°"""

    # COCO ë°ì´í„°ì…‹ 80ê°œ í´ë˜ìŠ¤
    COCO_CLASSES = [
        'person', 'bicycle', 'car', 'motorcycle', 'airplane', 'bus', 'train',
        'truck', 'boat', 'traffic light', 'fire hydrant', 'stop sign',
        'parking meter', 'bench', 'bird', 'cat', 'dog', 'horse', 'sheep',
        'cow', 'elephant', 'bear', 'zebra', 'giraffe', 'backpack', 'umbrella',
        'handbag', 'tie', 'suitcase', 'frisbee', 'skis', 'snowboard',
        'sports ball', 'kite', 'baseball bat', 'baseball glove', 'skateboard',
        'surfboard', 'tennis racket', 'bottle', 'wine glass', 'cup', 'fork',
        'knife', 'spoon', 'bowl', 'banana', 'apple', 'sandwich', 'orange',
        'broccoli', 'carrot', 'hot dog', 'pizza', 'donut', 'cake', 'chair',
        'couch', 'potted plant', 'bed', 'dining table', 'toilet', 'tv',
        'laptop', 'mouse', 'remote', 'keyboard', 'cell phone', 'microwave',
        'oven', 'toaster', 'sink', 'refrigerator', 'book', 'clock', 'vase',
        'scissors', 'teddy bear', 'hair drier', 'toothbrush'
    ]

    def __init__(self, model_path: str, conf_threshold: float = 0.5,
                 iou_threshold: float = 0.45):
        if not HAS_ONNX:
            raise ImportError("onnxruntimeì´ í•„ìš”í•©ë‹ˆë‹¤")

        if not os.path.exists(model_path):
            # ëª¨ë¸ì´ ì—†ëŠ” ê²½ìš° ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œ
            print(f"ê²½ê³ : ëª¨ë¸ íŒŒì¼ì„ ì°¾ì„ ìˆ˜ ì—†ìŠµë‹ˆë‹¤: {model_path}")
            print("ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œë¡œ ì‹¤í–‰í•©ë‹ˆë‹¤.")
            self.simulation_mode = True
            self.input_height = 640
            self.input_width = 640
            return

        self.simulation_mode = False

        # ONNX ì„¸ì…˜ ìƒì„±
        sess_options = ort.SessionOptions()
        sess_options.graph_optimization_level = ort.GraphOptimizationLevel.ORT_ENABLE_ALL

        self.session = ort.InferenceSession(
            model_path,
            sess_options=sess_options,
            providers=['CPUExecutionProvider']
        )

        self.conf_threshold = conf_threshold
        self.iou_threshold = iou_threshold

        # ì…ë ¥ ì •ë³´
        input_info = self.session.get_inputs()[0]
        self.input_name = input_info.name
        self.input_shape = input_info.shape
        self.input_height = self.input_shape[2]
        self.input_width = self.input_shape[3]

        print(f"YOLO ëª¨ë¸ ë¡œë“œ ì™„ë£Œ")
        print(f"  ì…ë ¥ í¬ê¸°: {self.input_width}x{self.input_height}")

    def preprocess(self, image: np.ndarray) -> Tuple[np.ndarray, Tuple[float, float]]:
        """ì´ë¯¸ì§€ ì „ì²˜ë¦¬"""
        orig_height, orig_width = image.shape[:2]

        # ë¦¬ì‚¬ì´ì¦ˆ
        resized = cv2.resize(image, (self.input_width, self.input_height))

        # BGR to RGB, HWC to CHW
        input_data = resized[:, :, ::-1].transpose(2, 0, 1)

        # ì •ê·œí™” (0-1)
        input_data = input_data.astype(np.float32) / 255.0

        # ë°°ì¹˜ ì°¨ì› ì¶”ê°€
        input_data = np.expand_dims(input_data, axis=0)

        # ìŠ¤ì¼€ì¼ ë¹„ìœ¨ ì €ì¥
        scale = (orig_width / self.input_width, orig_height / self.input_height)

        return input_data, scale

    def detect(self, image: np.ndarray) -> List[Dict]:
        """ê°ì²´ ê²€ì¶œ"""
        if self.simulation_mode:
            # ì‹œë®¬ë ˆì´ì…˜: ëœë¤ ê²€ì¶œ ê²°ê³¼ ë°˜í™˜
            print("ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œ: ëœë¤ ê²€ì¶œ ê²°ê³¼ë¥¼ ìƒì„±í•©ë‹ˆë‹¤.")
            return self._simulate_detection(image)

        if not HAS_CV2:
            raise ImportError("opencv-pythonì´ í•„ìš”í•©ë‹ˆë‹¤")

        # ì „ì²˜ë¦¬
        input_data, scale = self.preprocess(image)

        # ì¶”ë¡ 
        start = time.perf_counter()
        outputs = self.session.run(None, {self.input_name: input_data})
        inference_time = (time.perf_counter() - start) * 1000

        # í›„ì²˜ë¦¬
        detections = self.postprocess(outputs[0], scale)

        print(f"ì¶”ë¡  ì‹œê°„: {inference_time:.2f}ms")
        print(f"ê²€ì¶œëœ ê°ì²´: {len(detections)}ê°œ")

        return detections

    def postprocess(self, output: np.ndarray, scale: Tuple[float, float]) -> List[Dict]:
        """ì¶œë ¥ í›„ì²˜ë¦¬"""
        if not HAS_CV2:
            return []

        predictions = output[0]

        boxes = []
        scores = []
        class_ids = []

        for pred in predictions:
            confidence = pred[4]

            if confidence > self.conf_threshold:
                class_probs = pred[5:]
                class_id = np.argmax(class_probs)
                class_score = class_probs[class_id]

                if class_score > self.conf_threshold:
                    # ë°•ìŠ¤ ì¢Œí‘œ (center_x, center_y, width, height)
                    cx, cy, w, h = pred[:4]

                    # ì›ë³¸ ìŠ¤ì¼€ì¼ë¡œ ë³€í™˜
                    x1 = int((cx - w / 2) * scale[0])
                    y1 = int((cy - h / 2) * scale[1])
                    x2 = int((cx + w / 2) * scale[0])
                    y2 = int((cy + h / 2) * scale[1])

                    boxes.append([x1, y1, x2, y2])
                    scores.append(float(confidence * class_score))
                    class_ids.append(int(class_id))

        # NMS (Non-Maximum Suppression)
        if boxes:
            indices = cv2.dnn.NMSBoxes(
                boxes, scores, self.conf_threshold, self.iou_threshold
            )

            results = []
            for i in indices:
                idx = i[0] if isinstance(i, (list, np.ndarray)) else i
                results.append({
                    'box': boxes[idx],
                    'score': scores[idx],
                    'class_id': class_ids[idx],
                    'class_name': self.COCO_CLASSES[class_ids[idx]]
                })

            return results

        return []

    def _simulate_detection(self, image: np.ndarray) -> List[Dict]:
        """ì‹œë®¬ë ˆì´ì…˜: ëœë¤ ê²€ì¶œ ê²°ê³¼"""
        height, width = image.shape[:2]

        num_detections = np.random.randint(1, 5)
        detections = []

        for _ in range(num_detections):
            x1 = np.random.randint(0, width // 2)
            y1 = np.random.randint(0, height // 2)
            x2 = np.random.randint(x1 + 50, width)
            y2 = np.random.randint(y1 + 50, height)

            class_id = np.random.randint(0, len(self.COCO_CLASSES))

            detections.append({
                'box': [x1, y1, x2, y2],
                'score': np.random.uniform(0.5, 0.95),
                'class_id': class_id,
                'class_name': self.COCO_CLASSES[class_id]
            })

        return detections

    def draw_detections(self, image: np.ndarray, detections: List[Dict]) -> np.ndarray:
        """ê²€ì¶œ ê²°ê³¼ ì‹œê°í™”"""
        if not HAS_CV2:
            print("ê²½ê³ : opencv-pythonì´ ì—†ì–´ ì‹œê°í™”ë¥¼ ê±´ë„ˆëœë‹ˆë‹¤.")
            return image

        result = image.copy()

        for det in detections:
            x1, y1, x2, y2 = det['box']
            label = f"{det['class_name']}: {det['score']:.2f}"

            # ë°•ìŠ¤ ê·¸ë¦¬ê¸°
            cv2.rectangle(result, (x1, y1), (x2, y2), (0, 255, 0), 2)

            # ë¼ë²¨ ë°°ê²½
            (w, h), _ = cv2.getTextSize(label, cv2.FONT_HERSHEY_SIMPLEX, 0.5, 1)
            cv2.rectangle(result, (x1, y1 - 20), (x1 + w, y1), (0, 255, 0), -1)

            # ë¼ë²¨ í…ìŠ¤íŠ¸
            cv2.putText(result, label, (x1, y1 - 5),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 0, 0), 1)

        return result
