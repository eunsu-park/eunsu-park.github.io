def simulate_read_characteristic(char_uuid: str) -> bytes:
    """
    BLE íŠ¹ì„± ì½ê¸° ì‹œë®¬ë ˆì´ì…˜

        # ë°°í„°ë¦¬ ë ˆë²¨ (0-100%)

        # ì˜¨ë„ (0.01ë„ ë‹¨ìœ„, 16-bit ì •ìˆ˜)

async def read_sensor_data(address: str, use_simulation: bool = False) -> Dict:
    """
    BLE ì„¼ì„œ ë°ì´í„° ì½ê¸°

    Args:
        address: BLE ì¥ì¹˜ ì£¼ì†Œ
        use_simulation: ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œ

    Returns:

                pass

            # ì˜¨ë„
            try:
                data = await client.read_gatt_char(BLE_UUID.TEMPERATURE)
                temp_raw = struct.unpack('<h', data[:2])[0]
                result['temperature'] = temp_raw * 0.01
            except Exception:
                pass

            # ìŠµë„
            try:
                data = await client.read_gatt_char(BLE_UUID.HUMIDITY)
                humidity_raw = struct.unpack('<H', data[:2])[0]
                result['humidity'] = humidity_raw * 0.01
            except Exception:

    except Exception as e:
        print(f"ì½ê¸° ì˜¤ë¥˜: {e}")
        print("ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œë¡œ ì¬ì‹œë„...")

def create_notification_handler(sensor_type: str):
    """
    ì•Œë¦¼ í•¸ë“¤ëŸ¬ ìƒì„±

    Args:

    def handler(sender, data):

        print(f"[{timestamp}] ìˆ˜ì‹  ({sensor_type}): {data.hex()}")

        if sensor_type == 'temperature':
            temp_raw = struct.unpack('<h', data[:2])[0]
            temp = temp_raw * 0.01
            print(f"  ì˜¨ë„: {temp:.2f}Â°C")

        elif sensor_type == 'humidity':
            humidity_raw = struct.unpack('<H', data[:2])[0]
            humidity = humidity_raw * 0.01
            print(f"  ìŠµë„: {humidity:.2f}%")

        elif sensor_type == 'battery':
            battery = data[0]
            print(f"  ë°°í„°ë¦¬: {battery}%")

            flags = data[0]
            if flags & 0x01:  # 16-bit heart rate

            else:  # 8-bit heart rate

            print(f"  ì‹¬ë°•ìˆ˜: {hr} bpm")

    return handler

class BLESensorMonitor:
    """BLE í™˜ê²½ ì„¼ì„œ ëª¨ë‹ˆí„°ë§ í´ë˜ìŠ¤"""

    def __init__(self, device_address: Optional[str] = None, use_simulation: bool = False):
        self.device_address = device_address
        self.use_simulation = use_simulation or not BLEAK_AVAILABLE
        self.data_buffer = []

    async def start_monitoring(self, duration: float = 60):
        """
        ëª¨ë‹ˆí„°ë§ ì‹œì‘

        Args:
            duration: ëª¨ë‹ˆí„°ë§ ê¸°ê°„ (ì´ˆ)
        """
        if not self.device_address:
            print("ì˜¤ë¥˜: ì¥ì¹˜ ì£¼ì†Œê°€ ì§€ì •ë˜ì§€ ì•Šì•˜ìŠµë‹ˆë‹¤.")
            return

        print(f"=== BLE ì„¼ì„œ ëª¨ë‹ˆí„°ë§ ì‹œì‘ ===")
        print(f"ì¥ì¹˜: {self.device_address}")
        print(f"ê¸°ê°„: {duration}ì´ˆ")
        print(f"ëª¨ë“œ: {'ì‹œë®¬ë ˆì´ì…˜' if self.use_simulation else 'ì‹¤ì œ'}\n")

        if self.use_simulation:
            # ì‹œë®¬ë ˆì´ì…˜ ëª¨ë‹ˆí„°ë§
            for i in range(int(duration / 5)):
                data = await read_sensor_data(self.device_address, use_simulation=True)
                timestamp = datetime.now()

                print(f"[{timestamp.strftime('%H:%M:%S')}] ìˆ˜ì‹ :")
                for key, value in data.items():
                    print(f"  {key}: {value}")
                    self.data_buffer.append({
                        'type': key,
                        'value': value,
                        'timestamp': timestamp
                    })

                await asyncio.sleep(5)
        else:
            # ì‹¤ì œ BLE ëª¨ë‹ˆí„°ë§
            await subscribe_notifications(
                self.device_address,
                BLE_UUID.TEMPERATURE,
                'temperature',
                duration / 2
            )

        print("\n=== ëª¨ë‹ˆí„°ë§ ì¢…ë£Œ ===")
        print(f"ìˆ˜ì§‘ëœ ë°ì´í„°: {len(self.data_buffer)}ê°œ")

    def get_summary(self) -> Dict:
        """ìˆ˜ì§‘ëœ ë°ì´í„° ìš”ì•½"""
        if not self.data_buffer:
            return {}

        summary = {}
        data_types = set(d['type'] for d in self.data_buffer)

        for dtype in data_types:
            values = [d['value'] for d in self.data_buffer if d['type'] == dtype]
            summary[dtype] = {
                'min': min(values),
                'max': max(values),
                'avg': sum(values) / len(values),
                'count': len(values)
            }

        return summary
