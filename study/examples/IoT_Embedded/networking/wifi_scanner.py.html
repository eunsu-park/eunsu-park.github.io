def is_raspberry_pi() -> bool:
    """ë¼ì¦ˆë² ë¦¬íŒŒì´ í™˜ê²½ì¸ì§€ í™•ì¸"""
    try:
        with open('/proc/cpuinfo', 'r') as f:
            cpuinfo = f.read()
            return 'BCM' in cpuinfo or 'Raspberry' in cpuinfo
    except FileNotFoundError:
        return False


def get_wifi_info() -> dict:
    """
    í˜„ì¬ WiFi ì—°ê²° ì •ë³´ ì¡°íšŒ

    Returns:
        dict: WiFi ì •ë³´ (ssid, ip_addresses, signal_dbm, mac_address)
    """
    info = {}

    if not is_raspberry_pi():
        # ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œ
        print("[ì‹œë®¬ë ˆì´ì…˜] ì‹¤ì œ ë¼ì¦ˆë² ë¦¬íŒŒì´ê°€ ì•„ë‹™ë‹ˆë‹¤. ê°€ìƒ ë°ì´í„° ì‚¬ìš©")
        info = {
            'ssid': 'SimulatedWiFi',
            'ip_addresses': ['192.168.1.100'],
            'signal_dbm': -45,
            'mac_address': 'AA:BB:CC:DD:EE:FF'
        }
        return info

    try:
        # SSID ì¡°íšŒ
        result = subprocess.run(
            ['iwgetid', '-r'],
            capture_output=True,
            text=True,
            timeout=5
        )
        info['ssid'] = result.stdout.strip()

        # IP ì£¼ì†Œ ì¡°íšŒ
        result = subprocess.run(
            ['hostname', '-I'],
            capture_output=True,
            text=True,
            timeout=5
        )
        ips = result.stdout.strip().split()
        info['ip_addresses'] = ips

        # ì‹ í˜¸ ê°•ë„ ì¡°íšŒ
        result = subprocess.run(
            ['iwconfig', 'wlan0'],
            capture_output=True,
            text=True,
            timeout=5
        )
        match = re.search(r'Signal level=(-?\d+)', result.stdout)
        if match:
            info['signal_dbm'] = int(match.group(1))

        # MAC ì£¼ì†Œ
        result = subprocess.run(
            ['cat', '/sys/class/net/wlan0/address'],
            capture_output=True,
            text=True,
            timeout=5

        info['mac_address'] = result.stdout.strip()

    except Exception as e:
        info['error'] = str(e)

    return info


def get_wifi_networks() -> List[dict]:
    """
    ì£¼ë³€ WiFi ë„¤íŠ¸ì›Œí¬ ìŠ¤ìº”

    Returns:
        list: ë„¤íŠ¸ì›Œí¬ ì •ë³´ ë¦¬ìŠ¤íŠ¸ (ssid, quality, signal_dbm)
    """
    networks = []

    if not is_raspberry_pi():
        # ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œ
        print("[ì‹œë®¬ë ˆì´ì…˜] WiFi ìŠ¤ìº” ê²°ê³¼ ìƒì„± ì¤‘...")
        networks = [
            {'ssid': 'HomeNetwork', 'quality': '70/70', 'signal_dbm': -35},
            {'ssid': 'OfficeWiFi', 'quality': '65/70', 'signal_dbm': -42},
            {'ssid': 'GuestNetwork', 'quality': '50/70', 'signal_dbm': -58},
            {'ssid': 'Neighbor_2.4G', 'quality': '40/70', 'signal_dbm': -65},
            {'ssid': 'CafeWiFi', 'quality': '30/70', 'signal_dbm': -75},
        ]
        return networks

    try:
        result = subprocess.run(
            ['sudo', 'iwlist', 'wlan0', 'scan'],
            capture_output=True,
            text=True,
            timeout=10
        )

        current_network = {}
        for line in result.stdout.split('\n'):
            if 'ESSID:' in line:
                ssid = re.search(r'ESSID:"(.+)"', line)
                if ssid and current_network:
                    networks.append(current_network)
                current_network = {'ssid': ssid.group(1) if ssid else ''}

            elif 'Quality=' in line:
                quality = re.search(r'Quality=(\d+)/(\d+)', line)
                if quality:
                    current_network['quality'] = f"{quality.group(1)}/{quality.group(2)}"

                signal = re.search(r'Signal level=(-?\d+)', line)
                if signal:
                    current_network['signal_dbm'] = int(signal.group(1))

        if current_network:
            networks.append(current_network)

    except Exception as e:
        print(f"ìŠ¤ìº” ì‹¤íŒ¨: {e}")
        print("ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œë¡œ ì „í™˜í•©ë‹ˆë‹¤...")
        return get_wifi_networks()  # ì¬ê·€ í˜¸ì¶œë¡œ ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œ ì‚¬ìš©

    return networks


def monitor_wifi_signal(threshold_dbm: int = -70, interval: int = 5, duration: int = 30):
    """
    WiFi ì‹ í˜¸ ê°•ë„ ëª¨ë‹ˆí„°ë§

    Args:

        interval: í™•ì¸ ê°„ê²© (ì´ˆ)
        duration: ëª¨ë‹ˆí„°ë§ ê¸°ê°„ (ì´ˆ)
    """
    print(f"=== WiFi ì‹ í˜¸ ëª¨ë‹ˆí„°ë§ ì‹œì‘ ===")
    print(f"ê²½ê³  ì„ê³„ê°’: {threshold_dbm} dBm")
    print(f"í™•ì¸ ê°„ê²©: {interval}ì´ˆ, ì´ {duration}ì´ˆ")
    print()

    start_time = time.time()

    while time.time() - start_time < duration:

        if 'error' in info:
            print(f"[{datetime.now().strftime('%H:%M:%S')}] ì˜¤ë¥˜: {info['error']}")
        else:
            signal = info.get('signal_dbm', 0)
            ssid = info.get('ssid', 'Unknown')

            status = "ì–‘í˜¸" if signal > threshold_dbm else "âš ï¸  ê²½ê³ "
            print(f"[{datetime.now().strftime('%H:%M:%S')}] {ssid}: {signal} dBm ({status})")

            if signal <= threshold_dbm:
                print(f"  âš ï¸  ì‹ í˜¸ê°€ ì•½í•©ë‹ˆë‹¤! ({signal} dBm <= {threshold_dbm} dBm)")

        time.sleep(interval)

def tcp_server(host: str = '0.0.0.0', port: int = 9999):
    """
    TCP ì„œë²„ - ì„¼ì„œ ë°ì´í„° ìˆ˜ì‹ 

    Args:
        host: ë°”ì¸ë“œí•  IP ì£¼ì†Œ
        port: í¬íŠ¸ ë²ˆí˜¸
    """
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as server:
        # ì£¼ì†Œ ì¬ì‚¬ìš© í—ˆìš©
        server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)

        server.bind((host, port))
        server.listen(5)

        print(f"TCP ì„œë²„ ì‹œì‘: {host}:{port}")
        print("í´ë¼ì´ì–¸íŠ¸ ì—°ê²° ëŒ€ê¸° ì¤‘... (Ctrl+Cë¡œ ì¢…ë£Œ)")

        try:
            while True:
                client, address = server.accept()
                print(f"\ní´ë¼ì´ì–¸íŠ¸ ì—°ê²°: {address}")

                with client:
                    while True:
                        data = client.recv(1024)
                        if not data:
                            break

                        try:
                            # JSON ë°ì´í„° íŒŒì‹±
                            message = json.loads(data.decode('utf-8'))
                            timestamp = datetime.now().strftime("%H:%M:%S")
                            print(f"[{timestamp}] ìˆ˜ì‹ : {message}")

                            # ì‘ë‹µ ì „ì†¡
                            response = {
                                "status": "ok",
                                "received": message.get("sensor_id"),
                                "timestamp": time.time()
                            }
                            client.sendall(json.dumps(response).encode('utf-8'))

                        except json.JSONDecodeError:
                            print(f"ì˜ëª»ëœ JSON: {data}")

                print(f"í´ë¼ì´ì–¸íŠ¸ ì—°ê²° ì¢…ë£Œ: {address}")

        except KeyboardInterrupt:
            print("\nì„œë²„ ì¢…ë£Œ")


def tcp_client(server_host: str, server_port: int = 9999, count: int = 10):

def send_sensor_data_http(api_base: str, sensor_id: str, data: dict) -> bool:
    """
    HTTP POSTë¡œ ì„¼ì„œ ë°ì´í„° ì „ì†¡

    Args:
        api_base: API ë² ì´ìŠ¤ URL
        sensor_id: ì„¼ì„œ ID
        data: ì „ì†¡í•  ë°ì´í„°

    Returns:
        bool: ì„±ê³µ ì—¬ë¶€
    """
    try:
        import requests
    except ImportError:
        print("requests ë¼ì´ë¸ŒëŸ¬ë¦¬ê°€ í•„ìš”í•©ë‹ˆë‹¤: pip install requests")
        return False

    url = f"{api_base}/sensors/{sensor_id}/data"

    try:
        response = requests.post(
            url,
            json=data,
            headers={"Content-Type": "application/json"},
            timeout=5
        )

        if response.status_code in (200, 201):
            print(f"ë°ì´í„° ì „ì†¡ ì„±ê³µ: {data}")
            return True
        else:
            print(f"ì „ì†¡ ì‹¤íŒ¨: {response.status_code} - {response.text}")
            return False

    except requests.exceptions.RequestException as e:
        print(f"ë„¤íŠ¸ì›Œí¬ ì˜¤ë¥˜: {e}")
        return False


def http_client_with_retry(api_base: str, sensor_id: str, retries: int = 3):
    """
    ì¬ì‹œë„ ë¡œì§ì´ ìˆëŠ” HTTP í´ë¼ì´ì–¸íŠ¸

    Args:
        api_base: API ë² ì´ìŠ¤ URL
        sensor_id: ì„¼ì„œ ID
        retries: ì¬ì‹œë„ íšŸìˆ˜
    """
    print(f"=== HTTP í´ë¼ì´ì–¸íŠ¸ (ì¬ì‹œë„: {retries}íšŒ) ===")
    print(f"API: {api_base}")
    print(f"ì„¼ì„œ ID: {sensor_id}\n")

    for i in range(10):
        data = {
            "temperature": round(random.uniform(20, 30), 1),
            "humidity": round(random.uniform(40, 70), 1),
            "timestamp": int(time.time())
        }

        success = False
        for attempt in range(retries):
            if send_sensor_data_http(api_base, sensor_id, data):
                success = True
                break
            else:
                if attempt < retries - 1:
                    wait_time = 2 ** attempt  # Exponential backoff
                    print(f"  ì¬ì‹œë„ ëŒ€ê¸°: {wait_time}ì´ˆ...")
                    time.sleep(wait_time)

        if not success:
            print(f"  âŒ {retries}ë²ˆ ì¬ì‹œë„ í›„ ì‹¤íŒ¨")

        time.sleep(5)
