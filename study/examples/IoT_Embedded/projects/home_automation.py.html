        logging.info(f"EnvironmentMonitor ì´ˆê¸°í™” (ì‹œë®¬ë ˆì´ì…˜={simulate})")

    def read_sensor(self) -> Optional[SensorReading]:
        """
        ì„¼ì„œ ì½ê¸°

        Returns:
            ì„¼ì„œ ë°ì´í„° ë˜ëŠ” None (ì‹¤íŒ¨ ì‹œ)
        """
        if self.simulate:
            # ì‹œë®¬ë ˆì´ì…˜: ëœë¤ ë³€í™” ìƒì„±
            self.sim_temperature += random.uniform(-0.5, 0.5)
            self.sim_temperature = max(10, min(40, self.sim_temperature))

            self.sim_humidity += random.uniform(-2, 2)
            self.sim_humidity = max(30, min(90, self.sim_humidity))

            reading = SensorReading(
                sensor_id=self.sensor_id,
                temperature=round(self.sim_temperature, 1),
                humidity=round(self.sim_humidity, 1),
                timestamp=datetime.now()
            )
            return reading

        else:
            # ì‹¤ì œ ì„¼ì„œ
            try:
                temperature = self.dht.temperature
                humidity = self.dht.humidity

                if temperature is not None and humidity is not None:
                    reading = SensorReading(
                        sensor_id=self.sensor_id,
                        temperature=temperature,
                        humidity=humidity,
                        timestamp=datetime.now()
                    )
                    return reading

            except RuntimeError as e:
                # DHT ì„¼ì„œëŠ” ê°€ë” ì½ê¸° ì‹¤íŒ¨ (ì •ìƒ)
                logging.debug(f"ì„¼ì„œ ì½ê¸° ì‹¤íŒ¨ (ì •ìƒ): {e}")
            except Exception as e:
                logging.error(f"ì„¼ì„œ ì½ê¸° ì˜¤ë¥˜: {e}")

            return None

    def _monitor_loop(self, interval: int):
        """
        ëª¨ë‹ˆí„°ë§ ë£¨í”„ (ë°±ê·¸ë¼ìš´ë“œ ìŠ¤ë ˆë“œ)

        Args:
            interval: ì½ê¸° ê°„ê²© (ì´ˆ)
        """
        while self.running:
            reading = self.read_sensor()

            if reading:
                # ìµœì‹  ë°ì´í„° ì—…ë°ì´íŠ¸
                self.latest_reading = reading

                # íˆìŠ¤í† ë¦¬ ì €ì¥
                self.readings_history.append(reading)
                if len(self.readings_history) > self.max_history:
                    self.readings_history.pop(0)

                # íì— ì¶”ê°€ (ì™¸ë¶€ êµ¬ë…ììš©)
                self.data_queue.put(reading)

            time.sleep(interval)

    def start(self, interval: int = 5):
        """
        ëª¨ë‹ˆí„°ë§ ì‹œì‘

        Args:
            interval: ì½ê¸° ê°„ê²© (ì´ˆ)
        """
        if self.running:
            logging.warning("ì´ë¯¸ ëª¨ë‹ˆí„°ë§ ì¤‘ì…ë‹ˆë‹¤")
            return

        self.running = True
        self.thread = threading.Thread(
            target=self._monitor_loop,
            args=(interval,),
            daemon=True
        )
        self.thread.start()
        logging.info(f"í™˜ê²½ ëª¨ë‹ˆí„°ë§ ì‹œì‘ (ê°„ê²©={interval}ì´ˆ)")

    def stop(self):
        """ëª¨ë‹ˆí„°ë§ ì¤‘ì§€"""
        if not self.running:
            return

        self.running = False
        if self.thread:
            self.thread.join(timeout=5)

        if self.dht and not self.simulate:
            self.dht.exit()

        logging.info("í™˜ê²½ ëª¨ë‹ˆí„°ë§ ì¤‘ì§€")

    def get_latest(self) -> Optional[dict]:
        """ìµœì‹  ì„¼ì„œ ë°ì´í„° ë°˜í™˜"""
        if self.latest_reading:
            return self.latest_reading.to_dict()
        return None

    def get_stats(self) -> dict:
        """
        í†µê³„ ë°ì´í„° ë°˜í™˜

        Returns:
            ìµœì†Œ/ìµœëŒ€/í‰ê·  í†µê³„
        """
        if not self.readings_history:
            return {}

        temps = [r.temperature for r in self.readings_history]
        humids = [r.humidity for r in self.readings_history]

        return {
            "count": len(self.readings_history),
            "temperature": {
                "min": round(min(temps), 1),
                "max": round(max(temps), 1),
                "avg": round(sum(temps) / len(temps), 1)
            },
            "humidity": {
                "min": round(min(humids), 1),
                "max": round(max(humids), 1),
                "avg": round(sum(humids) / len(humids), 1)
            }
        }

class SmartHomeMQTT:
    """
    MQTT ê¸°ë°˜ ìŠ¤ë§ˆíŠ¸í™ˆ ì œì–´ í•¸ë“¤ëŸ¬

    MQTT ë¸Œë¡œì»¤ë¥¼ í†µí•´ ì¥ì¹˜ë¥¼ ì œì–´í•˜ê³  ì„¼ì„œ ë°ì´í„°ë¥¼ ë°œí–‰í•©ë‹ˆë‹¤.
    """

    TOPICS = {
        "light_command": "home/+/light/command",
        "light_status": "home/{}/light/status",
        "sensor_data": "home/sensor/{}",
        "motion": "home/motion/{}",
        "system": "home/system/status",
        "automation": "home/automation/event"
    }

    def __init__(self, light_controller: LightController,
                 env_monitor: EnvironmentMonitor,
                 broker: str = "localhost",
                 port: int = 1883):
        """
        ì´ˆê¸°í™”

        Args:
            light_controller: ì¡°ëª… ì œì–´ê¸°
            env_monitor: í™˜ê²½ ëª¨ë‹ˆí„°
            broker: MQTT ë¸Œë¡œì»¤ ì£¼ì†Œ
            port: MQTT í¬íŠ¸
        """
        self.light_controller = light_controller
        self.env_monitor = env_monitor
        self.broker = broker
        self.port = port

        if not MQTT_AVAILABLE:
            logging.warning("MQTT ë¼ì´ë¸ŒëŸ¬ë¦¬ ì—†ìŒ. MQTT ê¸°ëŠ¥ ë¹„í™œì„±í™”")
            self.client = None
            return

        # MQTT í´ë¼ì´ì–¸íŠ¸ ìƒì„±
        self.client = mqtt.Client(client_id="smart_home_gateway")
        self.client.on_connect = self._on_connect
        self.client.on_message = self._on_message

        # Last Will and Testament (LWT) ì„¤ì •
        self.client.will_set(
            self.TOPICS["system"],
            json.dumps({"status": "offline"}),
            qos=1,
            retain=True
        )

        logging.info(f"MQTT í´ë¼ì´ì–¸íŠ¸ ìƒì„± (ë¸Œë¡œì»¤={broker}:{port})")

    def connect(self):
        """MQTT ë¸Œë¡œì»¤ ì—°ê²°"""
        if not self.client:
            return

        try:
            self.client.connect(self.broker, self.port)
            logging.info("MQTT ë¸Œë¡œì»¤ ì—°ê²° ì‹œë„ ì¤‘...")
        except Exception as e:
            logging.error(f"MQTT ì—°ê²° ì‹¤íŒ¨: {e}")

    def _on_connect(self, client, userdata, flags, rc):
        """MQTT ì—°ê²° ì½œë°±"""
        if rc == 0:
            logging.info("MQTT ë¸Œë¡œì»¤ ì—°ê²° ì„±ê³µ")

            # í† í”½ êµ¬ë…
            client.subscribe(self.TOPICS["light_command"])
            logging.info(f"í† í”½ êµ¬ë…: {self.TOPICS['light_command']}")

            # ì˜¨ë¼ì¸ ìƒíƒœ ë°œí–‰
            client.publish(
                self.TOPICS["system"],
                json.dumps({
                    "status": "online",
                    "timestamp": datetime.now().isoformat()
                }),
                qos=1,
                retain=True
            )
        else:
            logging.error(f"MQTT ì—°ê²° ì‹¤íŒ¨ (ì½”ë“œ={rc})")

    def _on_message(self, client, userdata, msg):
        """MQTT ë©”ì‹œì§€ ìˆ˜ì‹  ì½œë°±"""
        try:
            topic = msg.topic
            payload = json.loads(msg.payload.decode())

            logging.debug(f"MQTT ìˆ˜ì‹ : {topic} = {payload}")

            # ì¡°ëª… ëª…ë ¹ ì²˜ë¦¬
            if "light/command" in topic:
                self._handle_light_command(topic, payload)

        except json.JSONDecodeError:
            logging.error(f"ì˜ëª»ëœ JSON: {msg.payload}")
        except Exception as e:
            logging.error(f"ë©”ì‹œì§€ ì²˜ë¦¬ ì˜¤ë¥˜: {e}")

    def _handle_light_command(self, topic: str, payload: dict):
        """
        ì¡°ëª… ëª…ë ¹ ì²˜ë¦¬

        Args:
            topic: MQTT í† í”½ (home/{room}/light/command)
            payload: ëª…ë ¹ ë°ì´í„° {"command": "on|off|toggle"}
        """
        # í† í”½ì—ì„œ ë°© ID ì¶”ì¶œ
        parts = topic.split('/')
        room = parts[1] if len(parts) >= 2 else None

        if not room:
            logging.warning(f"ë°© ID ì—†ìŒ: {topic}")
            return

        command = payload.get("command")

        # ëª…ë ¹ ì‹¤í–‰
        result = False
        if command == "on":
            result = self.light_controller.turn_on(room)
        elif command == "off":
            result = self.light_controller.turn_off(room)
        elif command == "toggle":
            result = self.light_controller.toggle(room)
        else:
            logging.warning(f"ì•Œ ìˆ˜ ì—†ëŠ” ëª…ë ¹: {command}")

        # ìƒíƒœ ë°œí–‰
        if result:
            status = self.light_controller.get_status(room)
            if status:
                self.publish_light_status(room, status)

    def publish_light_status(self, room: str, status: dict):
        """
        ì¡°ëª… ìƒíƒœ ë°œí–‰

        Args:
            room: ë°© ID
            status: ìƒíƒœ ë°ì´í„°
        """
        if not self.client:
            return

        topic = self.TOPICS["light_status"].format(room)
        self.client.publish(topic, json.dumps(status), qos=1, retain=True)
        logging.debug(f"ì¡°ëª… ìƒíƒœ ë°œí–‰: {topic}")

    def publish_sensor_data(self, sensor_id: str, data: dict):
        """
        ì„¼ì„œ ë°ì´í„° ë°œí–‰

        Args:
            sensor_id: ì„¼ì„œ ID
            data: ì„¼ì„œ ë°ì´í„°
        """
        if not self.client:
            return

        topic = self.TOPICS["sensor_data"].format(sensor_id)
        self.client.publish(topic, json.dumps(data), qos=0)

    def publish_motion(self, sensor_id: str, detected: bool):
        """
        ëª¨ì…˜ ê°ì§€ ë°œí–‰

        Args:
            sensor_id: ì„¼ì„œ ID
            detected: ê°ì§€ ì—¬ë¶€
        """
        if not self.client:
            return

        topic = self.TOPICS["motion"].format(sensor_id)
        data = {
            "detected": detected,
            "timestamp": datetime.now().isoformat()
        }
        self.client.publish(topic, json.dumps(data), qos=1)

    def publish_automation_event(self, event_type: str, details: dict):
        """
        ìë™í™” ì´ë²¤íŠ¸ ë°œí–‰

        Args:
            event_type: ì´ë²¤íŠ¸ íƒ€ì…
            details: ìƒì„¸ ì •ë³´
        """
        if not self.client:
            return

        data = {
            "event_type": event_type,
            "details": details,
            "timestamp": datetime.now().isoformat()
        }
        self.client.publish(self.TOPICS["automation"], json.dumps(data), qos=1)

    def start(self):
        """MQTT ë£¨í”„ ì‹œì‘"""
        if self.client:
            self.client.loop_start()

    def stop(self):
        """MQTT ì¤‘ì§€"""
        if not self.client:
            return

        # ì˜¤í”„ë¼ì¸ ìƒíƒœ ë°œí–‰
        self.client.publish(
            self.TOPICS["system"],
            json.dumps({"status": "offline"}),
            qos=1,
            retain=True
        )

        self.client.loop_stop()
        self.client.disconnect()
        logging.info("MQTT ì—°ê²° ì¢…ë£Œ")

class AutomationEngine:
    """
    ìë™í™” ê·œì¹™ ì—”ì§„

    ì„¼ì„œ ë°ì´í„° ê¸°ë°˜ìœ¼ë¡œ ì¡°ëª…/ê°€ì „ì„ ìë™ ì œì–´í•©ë‹ˆë‹¤.
    """

    def __init__(self, light_controller: LightController, mqtt_handler: SmartHomeMQTT):
        """
        ì´ˆê¸°í™”

        Args:
            light_controller: ì¡°ëª… ì œì–´ê¸°
            mqtt_handler: MQTT í•¸ë“¤ëŸ¬
        """
        self.light_controller = light_controller
        self.mqtt_handler = mqtt_handler
        self.rules: List[dict] = []

        logging.info("AutomationEngine ì´ˆê¸°í™”")

    def add_rule(self, name: str, condition: Callable, action: Callable):
        """
        ê·œì¹™ ì¶”ê°€

        Args:
            name: ê·œì¹™ ì´ë¦„
            condition: ì¡°ê±´ í•¨ìˆ˜ (True/False ë°˜í™˜)
            action: ì‹¤í–‰ í•¨ìˆ˜
        """
        self.rules.append({
            "name": name,
            "condition": condition,
            "action": action,
            "last_triggered": None
        })
        logging.info(f"ìë™í™” ê·œì¹™ ì¶”ê°€: {name}")

    def check_rules(self, sensor_data: dict):
        """
        ê·œì¹™ ê²€ì‚¬ ë° ì‹¤í–‰

        Args:
            sensor_data: ì„¼ì„œ ë°ì´í„°
        """
        for rule in self.rules:
            try:
                if rule["condition"](sensor_data):
                    # ì¡°ê±´ ë§Œì¡± ì‹œ ì•¡ì…˜ ì‹¤í–‰
                    logging.info(f"ìë™í™” ê·œì¹™ íŠ¸ë¦¬ê±°: {rule['name']}")
                    rule["action"](sensor_data)
                    rule["last_triggered"] = datetime.now()

                    # MQTT ì´ë²¤íŠ¸ ë°œí–‰
                    self.mqtt_handler.publish_automation_event(
                        event_type=rule["name"],
                        details=sensor_data
                    )

            except Exception as e:
                logging.error(f"ê·œì¹™ ì‹¤í–‰ ì˜¤ë¥˜ ({rule['name']}): {e}")

    def get_rules_status(self) -> List[dict]:
        """ê·œì¹™ ìƒíƒœ ì¡°íšŒ"""
        return [
            {
                "name": rule["name"],
                "last_triggered": rule["last_triggered"].isoformat() if rule["last_triggered"] else None
            }
            for rule in self.rules
        ]

    def start(self):
        """ê²Œì´íŠ¸ì›¨ì´ ì‹œì‘"""
        if self.running:
            logging.warning("ì´ë¯¸ ì‹¤í–‰ ì¤‘ì…ë‹ˆë‹¤")

        logging.info("=" * 60)
        logging.info("ìŠ¤ë§ˆíŠ¸í™ˆ ê²Œì´íŠ¸ì›¨ì´ ì‹œì‘")
        logging.info(f"ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œ: {self.simulate}")

        self.running = True

        # í™˜ê²½ ëª¨ë‹ˆí„°ë§ ì‹œì‘
        self.env_monitor.start(interval=5)

        # MQTT ì—°ê²° ë° ì‹œì‘
        self.mqtt_handler.connect()
        self.mqtt_handler.start()

        # ì„¼ì„œ ë°ì´í„° ë°œí–‰ ìŠ¤ë ˆë“œ

            target=self._sensor_publish_loop,
            args=(10,),

        self.threads.append(sensor_thread)

        # ìƒíƒœ ë¦¬í¬íŠ¸ ìŠ¤ë ˆë“œ
        status_thread = threading.Thread(
