@dataclass
class BoundingBox:
    """ë°”ìš´ë”© ë°•ìŠ¤"""
    x1: int
    y1: int
    x2: int
    y2: int

    @property
    def width(self) -> int:
        return self.x2 - self.x1

    @property
    def height(self) -> int:
        return self.y2 - self.y1

    @property
    def area(self) -> int:
        return self.width * self.height

class TFLiteObjectDetector:
    """TFLite ê°ì²´ ê²€ì¶œê¸° (ì‹œë®¬ë ˆì´ì…˜)"""

    COCO_LABELS = [
        'person', 'bicycle', 'car', 'motorcycle', 'airplane', 'bus', 'train',
        'truck', 'boat', 'traffic light', 'fire hydrant', 'stop sign',
        'parking meter', 'bench', 'bird', 'cat', 'dog', 'horse', 'sheep',
        'cow', 'elephant', 'bear', 'zebra', 'giraffe'
    ]

    def __init__(self, model_path: str = None, threshold: float = 0.5):
        self.model_path = model_path or "simulated_model.tflite"
        self.threshold = threshold
        self.input_size = (300, 300)
        print(f"[ì‹œë®¬ë ˆì´ì…˜] TFLite ëª¨ë¸ ë¡œë“œ: {self.model_path}")
        print(f"  - ì…ë ¥ í¬ê¸°: {self.input_size}")
        print(f"  - ì„ê³„ê°’: {self.threshold}")

    def preprocess(self, frame):
        """ì „ì²˜ë¦¬"""
        # ë¦¬ì‚¬ì´ì¦ˆ ì‹œë®¬ë ˆì´ì…˜
        target_h, target_w = self.input_size

        if HAS_NUMPY:
            # ì‹¤ì œë¡œëŠ” cv2.resize ì‚¬ìš©
            resized = np.random.randint(0, 256, (target_h, target_w, 3), dtype=np.uint8)
            # ì •ê·œí™”
            normalized = resized.astype(np.float32) / 255.0
            return normalized
        else:
            # numpy ì—†ì„ ë•ŒëŠ” ê°„ë‹¨íˆ ì²˜ë¦¬
            return frame

    def detect(self, frame) -> List[Detection]:
        """ê°ì²´ ê²€ì¶œ (ì‹œë®¬ë ˆì´ì…˜)"""
        # ì „ì²˜ë¦¬
        input_data = self.preprocess(frame)

        # ì¶”ë¡  ì‹œë®¬ë ˆì´ì…˜ (ì¼ë¶€ í™•ë¥ ë¡œ ê°ì²´ ê²€ì¶œ)
        detections = []

        # ëœë¤í•˜ê²Œ 0-3ê°œ ê°ì²´ ê²€ì¶œ
        num_objects = random.choice([0, 0, 0, 1, 1, 2])  # ëŒ€ë¶€ë¶„ 0, ê°€ë” 1-2ê°œ

        if HAS_NUMPY:
            h, w = frame.shape[:2]
        else:
            h, w = len(frame), len(frame[0]) if frame else 0

        for _ in range(num_objects):
            # ëœë¤ í´ë˜ìŠ¤ ì„ íƒ
            class_id = random.randint(0, min(len(self.COCO_LABELS), 10) - 1)
            class_name = self.COCO_LABELS[class_id]

            # ëœë¤ ì‹ ë¢°ë„
            confidence = random.uniform(self.threshold, 1.0)

            # ëœë¤ ë°”ìš´ë”© ë°•ìŠ¤
            x1 = random.randint(0, w // 2)
            y1 = random.randint(0, h // 2)
            x2 = x1 + random.randint(50, w // 3)
            y2 = y1 + random.randint(50, h // 3)

            bbox = BoundingBox(
                x1=min(x1, w-1),
                y1=min(y1, h-1),
                x2=min(x2, w-1),
                y2=min(y2, h-1)
            )

            detection = Detection(
                class_name=class_name,
                confidence=confidence,
                bbox=bbox,
                timestamp=datetime.now()
            )
            detections.append(detection)

        return detections

    def draw_detections(self, frame, detections: List[Detection]):
        """ê²€ì¶œ ê²°ê³¼ ì‹œê°í™” (ì‹œë®¬ë ˆì´ì…˜)"""
        if HAS_NUMPY:
            result = frame.copy()
        else:
            result = frame  # ì‹œë®¬ë ˆì´ì…˜ ëª¨ë“œì—ì„œëŠ” ë³µì‚¬í•˜ì§€ ì•ŠìŒ

        for det in detections:
            # ì‹¤ì œë¡œëŠ” cv2.rectangle, cv2.putText ì‚¬ìš©
            label = f"{det.class_name}: {det.confidence:.2f}"
            print(f"  [ê²€ì¶œ] {label} at ({det.bbox.x1}, {det.bbox.y1})")

        return result

class MotionDetector:
    """ëª¨ì…˜ ê°ì§€ê¸°"""

    def __init__(self, threshold: int = 30, min_area: int = 500):
        self.threshold = threshold
        self.min_area = min_area
        self.prev_frame = None
        self.motion_count = 0
        print(f"[ì‹œë®¬ë ˆì´ì…˜] ëª¨ì…˜ ê°ì§€ê¸° ì´ˆê¸°í™”")
        print(f"  - ì„ê³„ê°’: {threshold}")
        print(f"  - ìµœì†Œ ì˜ì—­: {min_area} pixels")

    def detect_motion(self, frame) -> Tuple[bool, List[BoundingBox]]:
        """ëª¨ì…˜ ê°ì§€"""
        if HAS_NUMPY:
            # ê·¸ë ˆì´ìŠ¤ì¼€ì¼ ë³€í™˜ ì‹œë®¬ë ˆì´ì…˜
            gray = np.mean(frame, axis=2).astype(np.uint8)

            # ê°€ìš°ì‹œì•ˆ ë¸”ëŸ¬ ì‹œë®¬ë ˆì´ì…˜
            # ì‹¤ì œë¡œëŠ” cv2.GaussianBlur ì‚¬ìš©

            if self.prev_frame is None:
                self.prev_frame = gray
                return False, []

            # í”„ë ˆì„ ì°¨ì´ ê³„ì‚°
            frame_delta = np.abs(gray.astype(np.int16) - self.prev_frame.astype(np.int16))

            # ì„ê³„ê°’ ì ìš©
            thresh = (frame_delta > self.threshold).astype(np.uint8) * 255

            # ë³€í™” ì˜ì—­ ë¹„ìœ¨ ê³„ì‚°
            motion_pixels = np.sum(thresh > 0)
            total_pixels = thresh.size
            motion_ratio = motion_pixels / total_pixels

            self.prev_frame = gray

            # ì¼ì • ë¹„ìœ¨ ì´ìƒ ë³€í™” ì‹œ ëª¨ì…˜ ê°ì§€
            motion_detected = motion_ratio > 0.05  # 5% ì´ìƒ ë³€í™”

            h, w = frame.shape[:2]
        else:
            # numpy ì—†ì„ ë•Œ ê°„ë‹¨í•œ ì‹œë®¬ë ˆì´ì…˜
            if self.prev_frame is None:
                self.prev_frame = frame
                return False, []

            # ëœë¤í•˜ê²Œ ëª¨ì…˜ ê°ì§€ (10% í™•ë¥ )
            motion_detected = random.random() < 0.1
            h, w = len(frame), len(frame[0]) if frame else 0

        regions = []
        if motion_detected:
            # ì‹œë®¬ë ˆì´ì…˜: ëœë¤ ëª¨ì…˜ ì˜ì—­ ìƒì„±
            num_regions = random.randint(1, 3)

            for _ in range(num_regions):
                x1 = random.randint(0, w // 2)
                y1 = random.randint(0, h // 2)
                x2 = x1 + random.randint(50, 150)
                y2 = y1 + random.randint(50, 150)

                bbox = BoundingBox(
                    x1=min(x1, w-1),
                    y1=min(y1, h-1),
                    x2=min(x2, w-1),
                    y2=min(y2, h-1)
                )

                if bbox.area >= self.min_area:
                    regions.append(bbox)

            self.motion_count += 1

        return motion_detected, regions

class VideoStreamer:
    """ë¹„ë””ì˜¤ ìŠ¤íŠ¸ë¦¬ë¨¸ (MJPEG over HTTP ê°œë…)"""

    def __init__(self, camera: SimulatedCamera, port: int = 8080):
        self.camera = camera
        self.port = port
        self.is_streaming = False
        self.frame_rate = 30
        self.lock = threading.Lock()
        self.current_frame = None
        print(f"[ì‹œë®¬ë ˆì´ì…˜] ë¹„ë””ì˜¤ ìŠ¤íŠ¸ë¦¬ë¨¸ ì´ˆê¸°í™” (í¬íŠ¸ {port})")

    def _capture_loop(self):
        """ìº¡ì²˜ ë£¨í”„"""
        frame_interval = 1.0 / self.frame_rate

        while self.is_streaming:
            start_time = time.time()

            frame = self.camera.capture_frame()

            # JPEG ì¸ì½”ë”© ì‹œë®¬ë ˆì´ì…˜
            # ì‹¤ì œë¡œëŠ” PIL.Image.fromarray(frame).save(buffer, 'JPEG')

            with self.lock:
                self.current_frame = frame

            # í”„ë ˆì„ ë ˆì´íŠ¸ ìœ ì§€
            elapsed = time.time() - start_time
            sleep_time = max(0, frame_interval - elapsed)
            time.sleep(sleep_time)

    def start_streaming(self):
        """ìŠ¤íŠ¸ë¦¬ë° ì‹œì‘"""
        if self.is_streaming:
            return

        self.is_streaming = True
        self.thread = threading.Thread(target=self._capture_loop, daemon=True)
        self.thread.start()
        print(f"[ì‹œë®¬ë ˆì´ì…˜] ìŠ¤íŠ¸ë¦¬ë° ì‹œì‘: http://0.0.0.0:{self.port}/video_feed")

    def stop_streaming(self):
        """ìŠ¤íŠ¸ë¦¬ë° ì¤‘ì§€"""
        self.is_streaming = False
        if hasattr(self, 'thread'):
            self.thread.join(timeout=2)
        print("[ì‹œë®¬ë ˆì´ì…˜] ìŠ¤íŠ¸ë¦¬ë° ì¤‘ì§€")

    def get_frame(self):
        """í˜„ì¬ í”„ë ˆì„ ë°˜í™˜"""
        with self.lock:
            return self.current_frame

class ResultLogger:
    """ê²€ì¶œ ë° ëª¨ì…˜ ê²°ê³¼ ë¡œê¹…"""

    def __init__(self, log_dir: str = "logs"):
        self.log_dir = log_dir
        os.makedirs(log_dir, exist_ok=True)
        self.log_file = os.path.join(log_dir, f"detection_log_{datetime.now().strftime('%Y%m%d_%H%M%S')}.jsonl")
        print(f"[ì‹œë®¬ë ˆì´ì…˜] ë¡œê±° ì´ˆê¸°í™”: {self.log_file}")

    def log_detection(self, detections: List[Detection], frame_id: int):
        """ê°ì²´ ê²€ì¶œ ë¡œê¹…"""
        log_entry = {
            "type": "detection",
            "timestamp": datetime.now().isoformat(),
            "frame_id": frame_id,
            "count": len(detections),
            "objects": [
                {
                    "class": det.class_name,
                    "confidence": det.confidence,
                    "bbox": [det.bbox.x1, det.bbox.y1, det.bbox.x2, det.bbox.y2]
                }
                for det in detections
            ]
        }

        with open(self.log_file, 'a') as f:
            f.write(json.dumps(log_entry) + '\n')

    def log_motion(self, event: MotionEvent):
        """ëª¨ì…˜ ì´ë²¤íŠ¸ ë¡œê¹…"""
        log_entry = {
            "type": "motion",
            "timestamp": event.timestamp.isoformat(),
            "frame_id": event.frame_id,
            "region_count": len(event.regions),
            "total_area": event.area,
            "regions": [
                [r.x1, r.y1, r.x2, r.y2] for r in event.regions
            ]
        }

        with open(self.log_file, 'a') as f:
            f.write(json.dumps(log_entry) + '\n')

    def get_statistics(self) -> Dict:
        """ë¡œê·¸ í†µê³„"""
        if not os.path.exists(self.log_file):
            return {}

        detection_count = 0
        motion_count = 0
        object_classes = {}

        with open(self.log_file, 'r') as f:
            for line in f:
                entry = json.loads(line)
                if entry['type'] == 'detection':
                    detection_count += 1
                    for obj in entry['objects']:
                        cls = obj['class']
                        object_classes[cls] = object_classes.get(cls, 0) + 1
                elif entry['type'] == 'motion':
                    motion_count += 1

        return {
            "detection_events": detection_count,
            "motion_events": motion_count,
            "object_classes": object_classes
        }

class PerformanceMonitor:
    """ì„±ëŠ¥ ëª¨ë‹ˆí„°"""

    def __init__(self):
        self.metrics = {
            "fps": [],
            "detection_time": [],
            "frame_count": 0,
            "start_time": None
        }

    def start(self):
        """ëª¨ë‹ˆí„°ë§ ì‹œì‘"""
        self.metrics["start_time"] = time.time()

    def record_frame(self, processing_time: float):
        """í”„ë ˆì„ ì²˜ë¦¬ ì‹œê°„ ê¸°ë¡"""
        self.metrics["frame_count"] += 1
        self.metrics["detection_time"].append(processing_time)

        if processing_time > 0:
            fps = 1.0 / processing_time
            self.metrics["fps"].append(fps)

    def get_report(self) -> Dict:
        """ì„±ëŠ¥ ë³´ê³ ì„œ"""
        if self.metrics["start_time"] is None:
            return {}

        elapsed = time.time() - self.metrics["start_time"]

        if HAS_NUMPY:
            avg_fps = np.mean(self.metrics["fps"]) if self.metrics["fps"] else 0
            avg_detection_time = np.mean(self.metrics["detection_time"]) if self.metrics["detection_time"] else 0
        else:
            avg_fps = sum(self.metrics["fps"]) / len(self.metrics["fps"]) if self.metrics["fps"] else 0
            avg_detection_time = sum(self.metrics["detection_time"]) / len(self.metrics["detection_time"]) if self.metrics["detection_time"] else 0

        return {
            "total_frames": self.metrics["frame_count"],
            "elapsed_time": elapsed,
            "average_fps": avg_fps,
            "average_detection_time_ms": avg_detection_time * 1000,
            "frames_per_second_actual": self.metrics["frame_count"] / elapsed if elapsed > 0 else 0
        }

        self.is_running = False

        print("\n" + "="*60)

    def run(self, duration: float = 60, detect_objects: bool = True, detect_motion: bool = True):
        """ì‹œìŠ¤í…œ ì‹¤í–‰"""
        print(f"\nì‹œìŠ¤í…œ ì‹œì‘ (ì‹¤í–‰ ì‹œê°„: {duration}ì´ˆ)")
        print(f"  - ê°ì²´ ê²€ì¶œ: {'ON' if detect_objects else 'OFF'}")

        print(f"  - ìŠ¤íŠ¸ë¦¬ë°: {'ON' if self.enable_streaming else 'OFF'}")

        self.perf_monitor.start()

        if self.enable_streaming:
            self.streamer.start_streaming()

        self.is_running = True
        start_time = time.time()
        frame_id = 0

        try:
            while time.time() - start_time < duration and self.is_running:
                frame_start = time.time()

                # í”„ë ˆì„ ìº¡ì²˜
                frame = self.camera.capture_frame()
                frame_id += 1

                # ê°ì²´ ê²€ì¶œ
                if detect_objects:

                    if detections:
                        print(f"[í”„ë ˆì„ {frame_id}] ê°ì²´ ê²€ì¶œ: {len(detections)}ê°œ")
                        for det in detections:
                            print(f"  - {det.class_name} (ì‹ ë¢°ë„: {det.confidence:.2f})")

                # ëª¨ì…˜ ê°ì§€
                if detect_motion:

                    if motion_detected:
                        total_area = sum(r.area for r in regions)
                        event = MotionEvent(
                            regions=regions,
                            area=total_area,
                            timestamp=datetime.now(),
                            frame_id=frame_id
                        )
                        print(f"[í”„ë ˆì„ {frame_id}] ëª¨ì…˜ ê°ì§€: {len(regions)}ê°œ ì˜ì—­ (ë©´ì : {total_area})")
                        self.logger.log_motion(event)

                # ì„±ëŠ¥ ê¸°ë¡

                # í”„ë ˆì„ ë ˆì´íŠ¸ ì¡°ì ˆ
                sleep_time = max(0, 0.1 - processing_time)  # ~10 FPS

        finally:
            self._cleanup()
