class ReActAgent:
    """ReAct (Reasoning + Acting) ì—ì´ì „íŠ¸"""

    def __init__(self, tools: List[Tool]):
        self.tools = {t.name: t for t in tools}
        self.history = []

    def think(self, question: str, observations: List[str]) -> Dict[str, str]:
        """
        ìƒê° ë‹¨ê³„ (ì‹¤ì œë¡œëŠ” LLM í˜¸ì¶œ)
        ì—¬ê¸°ì„œëŠ” ê·œì¹™ ê¸°ë°˜ìœ¼ë¡œ ì‹œë®¬ë ˆì´ì…˜
        """
        question_lower = question.lower()

        # ê·œì¹™ ê¸°ë°˜ ì˜ì‚¬ê²°ì • (ì‹¤ì œë¡œëŠ” LLMì´ ìˆ˜í–‰)
        if "ë‚ ì”¨" in question_lower:
            # ë„ì‹œ ì¶”ì¶œ
            cities = ["ì„œìš¸", "ë¶€ì‚°", "ì œì£¼"]
            for city in cities:
                if city in question:
                    return {
                        "thought": f"{city}ì˜ ë‚ ì”¨ë¥¼ í™•ì¸í•´ì•¼ í•©ë‹ˆë‹¤.",
                        "action": "get_weather",
                        "action_input": city
                    }

        if "ê³„ì‚°" in question_lower or any(op in question for op in ["+", "-", "*", "/"]):
            # ìˆ˜ì‹ ì¶”ì¶œ
            numbers = re.findall(r'[\d\+\-\*\/\.\(\)\s]+', question)
            if numbers:
                expr = numbers[0].strip()
                return {
                    "thought": f"'{expr}'ì„ ê³„ì‚°í•´ì•¼ í•©ë‹ˆë‹¤.",
                    "action": "calculator",
                    "action_input": expr
                }

        if any(keyword in question_lower for keyword in ["ëˆ„êµ¬", "ë¬´ì—‡", "ì–´ë””", "ê²€ìƒ‰"]):
            return {
                "thought": f"'{question}'ì— ëŒ€í•´ ê²€ìƒ‰í•´ì•¼ í•©ë‹ˆë‹¤.",
                "action": "search",
                "action_input": question
            }

        # ì´ë¯¸ ì¶©ë¶„í•œ ì •ë³´ê°€ ìˆìœ¼ë©´ ìµœì¢… ë‹µë³€
        if observations:
            return {
                "thought": "ì¶©ë¶„í•œ ì •ë³´ë¥¼ ìˆ˜ì§‘í–ˆìŠµë‹ˆë‹¤.",
                "final_answer": " ".join(observations)
            }

        return {
            "thought": "ì§ˆë¬¸ì„ ì´í•´í•˜ì§€ ëª»í–ˆìŠµë‹ˆë‹¤.",
            "final_answer": "ì£„ì†¡í•©ë‹ˆë‹¤, ì§ˆë¬¸ì„ ì²˜ë¦¬í•  ìˆ˜ ì—†ìŠµë‹ˆë‹¤."
        }

    def act(self, action: str, action_input: str) -> str:
        """í–‰ë™ ë‹¨ê³„"""
        if action in self.tools:
            return self.tools[action].run(action_input)
        return f"Error: Unknown tool '{action}'"

    def run(self, question: str, max_steps: int = 5) -> str:
        """ì—ì´ì „íŠ¸ ì‹¤í–‰"""
        observations = []

        print(f"\nì§ˆë¬¸: {question}")
        print("-" * 30)

        for step in range(max_steps):
            # ìƒê°
            result = self.think(question, observations)

            print(f"\n[Step {step + 1}]")
            print(f"Thought: {result.get('thought', '')}")

            # ìµœì¢… ë‹µë³€ í™•ì¸
            if "final_answer" in result:
                print(f"Final Answer: {result['final_answer']}")
                return result["final_answer"]

            # í–‰ë™
            action = result.get("action")
            action_input = result.get("action_input")

            if action:
                print(f"Action: {action}")
                print(f"Action Input: {action_input}")

                observation = self.act(action, action_input)
                observations.append(observation)
                print(f"Observation: {observation}")

        return "ìµœëŒ€ ë‹¨ê³„ ë„ë‹¬"
