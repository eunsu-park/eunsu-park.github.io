% ============================================================================
% hello.tex - Minimal LaTeX Document
% ============================================================================
% This is the simplest possible LaTeX document, demonstrating the basic
% structure required for any LaTeX file.
%
% Compilation:
%   pdflatex hello.tex
%
% This will produce hello.pdf
% ============================================================================

% Document class declaration - required for every LaTeX document
% 'article' is the most common class for short documents
% Optional parameter [12pt] sets the base font size
\documentclass[12pt]{article}

% Preamble section - this is where you load packages and define settings
% For this minimal example, we don't need any additional packages

% The document environment - all content must be inside this
\begin{document}

% This is the actual content of your document
Hello World!

% A blank line creates a new paragraph
This is my first \LaTeX{} document. The \texttt{\textbackslash LaTeX\{\}}
command produces the special LaTeX logo.

% Some basic formatting examples
\textbf{Bold text} and \textit{italic text} are easy to create.

% Lists are straightforward
Here are some reasons to learn LaTeX:
\begin{itemize}
    \item Beautiful mathematical typesetting
    \item Automatic bibliography management
    \item Professional document formatting
    \item Platform independent
\end{itemize}

% End of document
\end{document}

% Notes:
% - Lines starting with % are comments and are ignored by LaTeX
% - Every \begin{...} must have a matching \end{...}
% - LaTeX handles paragraph spacing, line breaking, and hyphenation automatically
% - Special characters like %, $, &, #, _, {, } need to be escaped with \
% ============================================================================
