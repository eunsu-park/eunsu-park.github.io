import mlflow
from mlflow.tracking import MlflowClient
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score
import os

# MLflow ì„¤ì •
TRACKING_URI = os.environ.get("MLFLOW_TRACKING_URI", "http://localhost:5000")
MODEL_NAME = "iris-classifier"


def setup():
    """MLflow ì„¤ì •"""
    mlflow.set_tracking_uri(TRACKING_URI)
    mlflow.set_experiment("model-registry-demo")
    return MlflowClient()


def train_and_register_model(client, version_tag: str):
    """ëª¨ë¸ í•™ìŠµ ë° ë ˆì§€ìŠ¤íŠ¸ë¦¬ ë“±ë¡"""
    # ë°ì´í„° ì¤€ë¹„
    iris = load_iris()
    X_train, X_test, y_train, y_test = train_test_split(
        iris.data, iris.target, test_size=0.2, random_state=42
    )

    with mlflow.start_run(run_name=f"training-{version_tag}") as run:
        # ëª¨ë¸ í•™ìŠµ
        model = RandomForestClassifier(n_estimators=100, random_state=42)
        model.fit(X_train, y_train)

        # í‰ê°€
        accuracy = accuracy_score(y_test, model.predict(X_test))
        mlflow.log_metric("accuracy", accuracy)

        # ëª¨ë¸ ì €ì¥ ë° ë“±ë¡
        mlflow.sklearn.log_model(
            model,
            "model",
            registered_model_name=MODEL_NAME
        )

        print(f"\nëª¨ë¸ ë“±ë¡ ì™„ë£Œ: {MODEL_NAME}")
        print(f"  Run ID: {run.info.run_id}")
        print(f"  Accuracy: {accuracy:.4f}")

        return run.info.run_id


def get_model_versions(client):
    """ë“±ë¡ëœ ëª¨ë¸ ë²„ì „ ì¡°íšŒ"""
    print(f"\n{'='*50}")
    print(f"ëª¨ë¸ '{MODEL_NAME}' ë²„ì „ ëª©ë¡:")
    print("="*50)

    try:
        versions = client.search_model_versions(f"name='{MODEL_NAME}'")
        for v in versions:
            print(f"\në²„ì „ {v.version}:")
            print(f"  ìƒíƒœ: {v.current_stage}")
            print(f"  Run ID: {v.run_id}")
            print(f"  ìƒì„±ì¼: {v.creation_timestamp}")
            if v.description:
                print(f"  ì„¤ëª…: {v.description}")
        return versions
    except Exception as e:
        print(f"ëª¨ë¸ì„ ì°¾ì„ ìˆ˜ ì—†ìŠµë‹ˆë‹¤: {e}")
        return []

    """ëª¨ë¸ì„ Stagingìœ¼ë¡œ ì „í™˜"""

    print(f"\nëª¨ë¸ v{version}ì„ Stagingìœ¼ë¡œ ì „í™˜í–ˆìŠµë‹ˆë‹¤.")

def load_model_by_stage(stage: str):
    """ìŠ¤í…Œì´ì§€ë³„ ëª¨ë¸ ë¡œë“œ"""
    try:

        return model

        print(f"\n{stage} ëª¨ë¸ ë¡œë“œ ì‹¤íŒ¨: {e}")
