import torch
import torch.nn.functional as F
from ts.torch_handler.base_handler import BaseHandler
import json
import logging
import os
import time

logger = logging.getLogger(__name__)


class ChurnPredictionHandler(BaseHandler):
    """
    ê³ ê° ì´íƒˆ ì˜ˆì¸¡ ëª¨ë¸ í•¸ë“¤ëŸ¬

    ì´ í•¸ë“¤ëŸ¬ëŠ” ë‹¤ìŒì„ ìˆ˜í–‰í•©ë‹ˆë‹¤:
    1. ëª¨ë¸ ì´ˆê¸°í™” ë° ë¡œë“œ
    2. ì…ë ¥ ë°ì´í„° ì „ì²˜ë¦¬
    3. ì¶”ë¡  ìˆ˜í–‰
    4. ê²°ê³¼ í›„ì²˜ë¦¬
    """

    def __init__(self):
        super().__init__()
        self.initialized = False
        self.model = None
        self.device = None
        self.class_names = None
        self.feature_names = None

    def initialize(self, context):
        """
        ëª¨ë¸ ì´ˆê¸°í™”

        Args:
            context: TorchServe ì»¨í…ìŠ¤íŠ¸ ê°ì²´
        """
        logger.info("Initializing model...")

        # ì»¨í…ìŠ¤íŠ¸ì—ì„œ ì •ë³´ ì¶”ì¶œ
        self.manifest = context.manifest
        properties = context.system_properties
        model_dir = properties.get("model_dir")

        # ë””ë°”ì´ìŠ¤ ì„¤ì •
        if torch.cuda.is_available() and properties.get("gpu_id") is not None:
            self.device = torch.device(f"cuda:{properties.get('gpu_id')}")
            logger.info(f"Using GPU: {properties.get('gpu_id')}")
        else:
            self.device = torch.device("cpu")
            logger.info("Using CPU")

        # ëª¨ë¸ ë¡œë“œ
        serialized_file = self.manifest["model"]["serializedFile"]
        model_path = os.path.join(model_dir, serialized_file)

        try:
            self.model = torch.jit.load(model_path, map_location=self.device)
            self.model.eval()
            logger.info(f"Model loaded from {model_path}")
        except Exception as e:
            logger.error(f"Failed to load model: {e}")
            raise

        # ì¶”ê°€ ì„¤ì • íŒŒì¼ ë¡œë“œ
        self._load_config(model_dir)

        self.initialized = True
        logger.info("Model initialization complete")

    def _load_config(self, model_dir):
        """ì„¤ì • íŒŒì¼ ë¡œë“œ"""
        # í´ë˜ìŠ¤ ì´ë¦„
        class_file = os.path.join(model_dir, "index_to_name.json")
        if os.path.exists(class_file):
            with open(class_file) as f:
                self.class_names = json.load(f)
            logger.info(f"Loaded class names: {self.class_names}")
        else:
            self.class_names = {"0": "not_churned", "1": "churned"}

        # í”¼ì²˜ ì´ë¦„
        feature_file = os.path.join(model_dir, "feature_names.json")
        if os.path.exists(feature_file):
            with open(feature_file) as f:
                self.feature_names = json.load(f)
            logger.info(f"Loaded feature names: {self.feature_names}")

    def preprocess(self, data):
        """
        ì…ë ¥ ë°ì´í„° ì „ì²˜ë¦¬

        Args:
            data: ìš”ì²­ ë°ì´í„° ë¦¬ìŠ¤íŠ¸

        Returns:
            torch.Tensor: ì „ì²˜ë¦¬ëœ ì…ë ¥ í…ì„œ
        """
        logger.info(f"Preprocessing {len(data)} samples")
        inputs = []

        for row in data:
            # ìš”ì²­ ë°ì´í„° íŒŒì‹±
            if isinstance(row, dict):
                features = row.get("data") or row.get("body")
            else:
                features = row.get("body")

            # ë°”ì´íŠ¸ ë°ì´í„° ì²˜ë¦¬
            if isinstance(features, (bytes, bytearray)):
                features = json.loads(features.decode("utf-8"))

            # JSON ë¬¸ìì—´ ì²˜ë¦¬
            if isinstance(features, str):
                features = json.loads(features)

            # dictì¸ ê²½ìš° ê°’ë§Œ ì¶”ì¶œ
            if isinstance(features, dict):
                if "data" in features:
                    features = features["data"]
                else:
                    features = list(features.values())

            # í…ì„œë¡œ ë³€í™˜
            tensor = torch.tensor(features, dtype=torch.float32)
            inputs.append(tensor)

        # ë°°ì¹˜ë¡œ ë¬¶ê¸°
        batch = torch.stack(inputs).to(self.device)
        logger.info(f"Input batch shape: {batch.shape}")

        return batch

    def inference(self, data):
        """
        ëª¨ë¸ ì¶”ë¡ 

        Args:
            data: ì „ì²˜ë¦¬ëœ ì…ë ¥ í…ì„œ

        Returns:
            torch.Tensor: ëª¨ë¸ ì¶œë ¥
        """
        logger.info("Running inference...")
        start_time = time.time()

        with torch.no_grad():
            outputs = self.model(data)

            # í™•ë¥ ë¡œ ë³€í™˜ (ë¶„ë¥˜ ëª¨ë¸ì¸ ê²½ìš°)
            if outputs.dim() > 1 and outputs.shape[1] > 1:
                probabilities = F.softmax(outputs, dim=1)
            else:
                probabilities = torch.sigmoid(outputs)

        inference_time = time.time() - start_time
        logger.info(f"Inference completed in {inference_time:.4f}s")

        return probabilities

    def postprocess(self, data):
        """
        ì¶œë ¥ í›„ì²˜ë¦¬

        Args:
            data: ëª¨ë¸ ì¶œë ¥ í…ì„œ

        Returns:
            list: JSON ì§ë ¬í™” ê°€ëŠ¥í•œ ê²°ê³¼ ë¦¬ìŠ¤íŠ¸
        """
        logger.info("Postprocessing results...")
        results = []

        for prob in data:
            prob_list = prob.cpu().numpy().tolist()

            # ì´ì§„ ë¶„ë¥˜
            if len(prob_list) == 1:
                prediction = 1 if prob_list[0] > 0.5 else 0
                probabilities = [1 - prob_list[0], prob_list[0]]
            # ë‹¤ì¤‘ í´ë˜ìŠ¤
            else:
                prediction = int(torch.argmax(prob).item())
                probabilities = prob_list

            result = {
                "prediction": prediction,
                "probabilities": probabilities,
                "confidence": max(probabilities)
            }

            # í´ë˜ìŠ¤ ì´ë¦„ ì¶”ê°€
            if self.class_names:
                result["class_name"] = self.class_names.get(
                    str(prediction),
                    f"class_{prediction}"
                )

            results.append(result)

        logger.info(f"Processed {len(results)} results")
        return results

    def handle(self, data, context):
        """
        ì „ì²´ ìš”ì²­ ì²˜ë¦¬ (preprocess -> inference -> postprocess)

        TorchServeê°€ í˜¸ì¶œí•˜ëŠ” ë©”ì¸ ë©”ì„œë“œ
        """
        if not self.initialized:
            self.initialize(context)

        if data is None:
            return None

        # ì „ì²˜ë¦¬
        model_input = self.preprocess(data)

        # ì¶”ë¡ 
        model_output = self.inference(model_input)

        # í›„ì²˜ë¦¬
        return self.postprocess(model_output)
