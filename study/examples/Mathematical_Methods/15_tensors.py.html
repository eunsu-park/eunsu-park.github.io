"""
Tensors - Operations, Einstein Summation, and Coordinate Transformations

This script demonstrates:
- Tensor operations with numpy
- Index notation and Einstein summation (np.einsum)
- Metric tensor
- Christoffel symbols
- Coordinate transformations
- Tensor contraction
- Tensor products
"""

import numpy as np

try:
    import matplotlib.pyplot as plt
    HAS_MATPLOTLIB = True
except ImportError:
    HAS_MATPLOTLIB = False
    print("Warning: matplotlib not available, skipping visualizations")


def kronecker_delta(n):
    """Kronecker delta tensor Œ¥_ij"""
    return np.eye(n)


def levi_civita_3d():
    """
    Levi-Civita (permutation) tensor Œµ_ijk in 3D
    Œµ_ijk = +1 for even permutations, -1 for odd, 0 otherwise
    """
    epsilon = np.zeros((3, 3, 3))
    epsilon[0, 1, 2] = epsilon[1, 2, 0] = epsilon[2, 0, 1] = 1
    epsilon[0, 2, 1] = epsilon[2, 1, 0] = epsilon[1, 0, 2] = -1
    return epsilon


def metric_tensor_euclidean_2d():
    """Euclidean metric in 2D: g_ij = Œ¥_ij"""
    return np.eye(2)


def metric_tensor_polar():
    """
    Metric tensor in polar coordinates (r, Œ∏)
    ds¬≤ = dr¬≤ + r¬≤dŒ∏¬≤
    g_ij = [[1, 0], [0, r¬≤]]
    """
    def g(r):
        return np.array([[1, 0], [0, r**2]])
    return g


def metric_tensor_spherical():
    """
    Metric tensor in spherical coordinates (r, Œ∏, œÜ)
    ds¬≤ = dr¬≤ + r¬≤dŒ∏¬≤ + r¬≤sin¬≤(Œ∏)dœÜ¬≤
    """
    def g(r, theta):
        return np.array([
            [1, 0, 0],
            [0, r**2, 0],
            [0, 0, r**2 * np.sin(theta)**2]
        ])
    return g


def christoffel_symbols_polar(r):
    """
    Compute Christoffel symbols Œì^k_ij for polar coordinates
    Œì^r_Œ∏Œ∏ = -r
    Œì^Œ∏_rŒ∏ = Œì^Œ∏_Œ∏r = 1/r
    """
    Gamma = np.zeros((2, 2, 2))

    # Œì^r_Œ∏Œ∏ = -r (component [0, 1, 1])
    Gamma[0, 1, 1] = -r

    # Œì^Œ∏_rŒ∏ = Œì^Œ∏_Œ∏r = 1/r (components [1, 0, 1] and [1, 1, 0])
    Gamma[1, 0, 1] = 1 / r
    Gamma[1, 1, 0] = 1 / r

    return Gamma


def christoffel_symbols_spherical(r, theta):
    """
    Compute Christoffel symbols Œì^k_ij for spherical coordinates
    """
    Gamma = np.zeros((3, 3, 3))

    # Non-zero components
    # Œì^r_Œ∏Œ∏ = -r
    Gamma[0, 1, 1] = -r

    # Œì^r_œÜœÜ = -r sin¬≤(Œ∏)
    Gamma[0, 2, 2] = -r * np.sin(theta)**2

    # Œì^Œ∏_rŒ∏ = Œì^Œ∏_Œ∏r = 1/r
    Gamma[1, 0, 1] = 1 / r
    Gamma[1, 1, 0] = 1 / r

    # Œì^Œ∏_œÜœÜ = -sin(Œ∏)cos(Œ∏)
    Gamma[1, 2, 2] = -np.sin(theta) * np.cos(theta)

    # Œì^œÜ_rœÜ = Œì^œÜ_œÜr = 1/r
    Gamma[2, 0, 2] = 1 / r
    Gamma[2, 2, 0] = 1 / r

    # Œì^œÜ_Œ∏œÜ = Œì^œÜ_œÜŒ∏ = cot(Œ∏)
    if np.sin(theta) != 0:
        Gamma[2, 1, 2] = np.cos(theta) / np.sin(theta)
        Gamma[2, 2, 1] = np.cos(theta) / np.sin(theta)

    return Gamma


def coordinate_transform_cartesian_to_polar(x, y):
    """Transform from Cartesian to polar coordinates"""
    r = np.sqrt(x**2 + y**2)
    theta = np.arctan2(y, x)
    return r, theta


def coordinate_transform_polar_to_cartesian(r, theta):
    """Transform from polar to Cartesian coordinates"""
    x = r * np.cos(theta)
    y = r * np.sin(theta)
    return x, y


def jacobian_cartesian_to_polar(x, y):
    """
    Jacobian matrix for Cartesian to polar transformation
    J_ij = ‚àÇx^i_new / ‚àÇx^j_old
    """
    r = np.sqrt(x**2 + y**2)
    if r == 0:
        return np.eye(2)

    # dr/dx, dr/dy
    # dŒ∏/dx, dŒ∏/dy
    J = np.array([
        [x/r, y/r],
        [-y/(r**2), x/(r**2)]
    ])
    return J


def transform_vector_contravariant(v, J):
    """
    Transform contravariant vector: v'^i = J^i_j v^j
    """
    return J @ v


def transform_covector_covariant(w, J):
    """
    Transform covariant vector (covector): w'_i = (J^T)^(-1)_ij w_j
    """
    return np.linalg.inv(J.T) @ w

# Matrix multiplication: C_ij = A_ik B_kj
A = np.array([[1, 2], [3, 4]])
B = np.array([[5, 6], [7, 8]])

C_matmul = np.einsum('ik,kj->ij', A, B)
C_direct = A @ B

print("Matrix multiplication: C = A @ B")
print("Using einsum:")
print(C_matmul)
print("Direct computation:")
print(C_direct)
print(f"Match: {np.allclose(C_matmul, C_direct)}")

# Trace: tr(A) = A_ii

print(f"Trace using np.trace: {np.trace(A)}")

print("Using einsum:")
print(outer_einsum)
print("Using np.outer:")
print(outer_direct)
