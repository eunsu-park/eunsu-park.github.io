 61def newton_raphson(
 62    f: Callable[[float], float],
 63    df: Callable[[float], float],
 64    x0: float,
 65    tol: float = 1e-10,
 66    max_iter: int = 100
 67) -> Tuple[float, int, list]:
 68    """
 69    ë‰´í„´-ë©ìŠ¨ ë°©ë²•ìœ¼ë¡œ f(x) = 0ì˜ ê·¼ ì°¾ê¸°
 70
 71    x_{n+1} = x_n - f(x_n) / f'(x_n)

 73    ìˆ˜ë ´ ì†ë„: 2ì°¨ (ì œê³± ìˆ˜ë ´)
 74    ë‹¨ì : ë„í•¨ìˆ˜ í•„ìš”, ì´ˆê¸°ê°’ì— ë¯¼ê°

 76    Args:
 77        f: ëª©í‘œ í•¨ìˆ˜
 78        df: ë„í•¨ìˆ˜

 80        tol: í—ˆìš© ì˜¤ì°¨
 81        max_iter: ìµœëŒ€ ë°˜ë³µ íšŸìˆ˜
 82
 83    Returns:
 84        (ê·¼, ë°˜ë³µ íšŸìˆ˜, íˆìŠ¤í† ë¦¬)
 85    """
 86    x = x0
 87    history = [x]
 88
 89    for i in range(max_iter):
 90        fx = f(x)
 91        dfx = df(x)
 92
 93        if abs(dfx) < 1e-15:
 94            raise ValueError("ë„í•¨ìˆ˜ê°€ 0ì— ê°€ê¹Œì›€: ë°œì‚° ìœ„í—˜")

 96        x_new = x - fx / dfx
 97        history.append(x_new)
 98
 99        if abs(x_new - x) < tol:
100            return x_new, i + 1, history
101
102        x = x_new
103
104    return x, max_iter, history

110def secant(
111    f: Callable[[float], float],
112    x0: float,
113    x1: float,
114    tol: float = 1e-10,
115    max_iter: int = 100
116) -> Tuple[float, int, list]:
117    """
118    í• ì„ ë²•ìœ¼ë¡œ f(x) = 0ì˜ ê·¼ ì°¾ê¸°
119
120    ë‰´í„´ë²•ì˜ ë„í•¨ìˆ˜ë¥¼ ì°¨ë¶„ìœ¼ë¡œ ê·¼ì‚¬:
121    x_{n+1} = x_n - f(x_n) * (x_n - x_{n-1}) / (f(x_n) - f(x_{n-1}))

123    ìˆ˜ë ´ ì†ë„: ì•½ 1.618ì°¨ (í™©ê¸ˆë¹„)
124    ì¥ì : ë„í•¨ìˆ˜ ë¶ˆí•„ìš”

126    Args:
127        f: ëª©í‘œ í•¨ìˆ˜
128        x0, x1: ë‘ ì´ˆê¸°ê°’
129        tol: í—ˆìš© ì˜¤ì°¨
130        max_iter: ìµœëŒ€ ë°˜ë³µ íšŸìˆ˜
131
132    Returns:
133        (ê·¼, ë°˜ë³µ íšŸìˆ˜, íˆìŠ¤í† ë¦¬)
134    """
135    history = [x0, x1]
136
137    for i in range(max_iter):
138        f0, f1 = f(x0), f(x1)
139
140        if abs(f1 - f0) < 1e-15:
141            raise ValueError("ë¶„ëª¨ê°€ 0ì— ê°€ê¹Œì›€")
142
143        x2 = x1 - f1 * (x1 - x0) / (f1 - f0)
144        history.append(x2)
145
146        if abs(x2 - x1) < tol:
147            return x2, i + 1, history
148
149        x0, x1 = x1, x2
150
151    return x1, max_iter, history

198def brents_method(
199    f: Callable[[float], float],
200    a: float,
201    b: float,
202    tol: float = 1e-10,
203    max_iter: int = 100
204) -> Tuple[float, int]:
205    """
206    Brent's Method (ê°„ì†Œí™” ë²„ì „)
207    ì´ë¶„ë²•, í• ì„ ë²•, ì—­2ì°¨ë³´ê°„ì„ ì¡°í•©í•œ ë°©ë²•
208
209    ì‹¤ë¬´ì—ì„œëŠ” scipy.optimize.brentq ì‚¬ìš© ê¶Œì¥
210    """
211    fa, fb = f(a), f(b)
212    if fa * fb >= 0:
213        raise ValueError("f(a)ì™€ f(b)ì˜ ë¶€í˜¸ê°€ ë‹¬ë¼ì•¼ í•©ë‹ˆë‹¤")
214
215    if abs(fa) < abs(fb):
216        a, b = b, a
217        fa, fb = fb, fa
218
219    c, fc = a, fa
220    d = c  # d ì´ˆê¸°í™” (ì²« iterationì—ì„œ ì‚¬ìš©ë¨)
221    mflag = True
222
223    for i in range(max_iter):
224        if fa != fc and fb != fc:
225            # ì—­2ì°¨ ë³´ê°„
226            s = (a * fb * fc / ((fa - fb) * (fa - fc)) +
227                 b * fa * fc / ((fb - fa) * (fb - fc)) +
228                 c * fa * fb / ((fc - fa) * (fc - fb)))
229        else:
230            # í• ì„ ë²•
231            s = b - fb * (b - a) / (fb - fa)
232
233        # ì¡°ê±´ ì²´í¬ í›„ ì´ë¶„ë²•ìœ¼ë¡œ ëŒ€ì²´
234        conditions = [
235            not ((3 * a + b) / 4 <= s <= b or b <= s <= (3 * a + b) / 4),
236            mflag and abs(s - b) >= abs(b - c) / 2,
237            not mflag and abs(s - b) >= abs(c - d) / 2,
238        ]
239
240        if any(conditions):
241            s = (a + b) / 2
242            mflag = True
243        else:
244            mflag = False
245
246        fs = f(s)
247        d, c, fc = c, b, fb
248
249        if fa * fs < 0:
250            b, fb = s, fs
251        else:
252            a, fa = s, fs
253
254        if abs(fa) < abs(fb):
255            a, b = b, a
256            fa, fb = fb, fa
257
258        if abs(b - a) < tol or abs(fb) < tol:
259            return b, i + 1
260
261    return b, max_iter
