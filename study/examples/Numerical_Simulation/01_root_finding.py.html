# 1. ì´ë¶„ë²• (Bisection Method)
# =============================================================================
def bisection(
    f: Callable[[float], float],
    a: float,
    b: float,
    tol: float = 1e-10,
    max_iter: int = 100
) -> Tuple[float, int, list]:
    """
    ì´ë¶„ë²•ìœ¼ë¡œ f(x) = 0ì˜ ê·¼ ì°¾ê¸°

    ì¡°ê±´: f(a)ì™€ f(b)ì˜ ë¶€í˜¸ê°€ ë‹¬ë¼ì•¼ í•¨ (ì¤‘ê°„ê°’ ì •ë¦¬)
    ìˆ˜ë ´ ì†ë„: ì„ í˜• (ë§¤ ë°˜ë³µë§ˆë‹¤ êµ¬ê°„ ì ˆë°˜)

    Args:
        f: ëª©í‘œ í•¨ìˆ˜
        a, b: ì´ˆê¸° êµ¬ê°„
        tol: í—ˆìš© ì˜¤ì°¨
        max_iter: ìµœëŒ€ ë°˜ë³µ íšŸìˆ˜

    Returns:
        (ê·¼, ë°˜ë³µ íšŸìˆ˜, ì¤‘ê°„ê°’ íˆìŠ¤í† ë¦¬)
    """
    if f(a) * f(b) >= 0:
        raise ValueError("f(a)ì™€ f(b)ì˜ ë¶€í˜¸ê°€ ë‹¬ë¼ì•¼ í•©ë‹ˆë‹¤")

    history = []

    for i in range(max_iter):
        c = (a + b) / 2
        history.append(c)

        if abs(f(c)) < tol or (b - a) / 2 < tol:
            return c, i + 1, history

        if f(a) * f(c) < 0:
            b = c
        else:
            a = c

    return (a + b) / 2, max_iter, history


# =============================================================================
# 2. ë‰´í„´-ë©ìŠ¨ ë°©ë²• (Newton-Raphson Method)
# =============================================================================
def newton_raphson(
    f: Callable[[float], float],
    df: Callable[[float], float],
    x0: float,
    tol: float = 1e-10,
    max_iter: int = 100
) -> Tuple[float, int, list]:
    """
    ë‰´í„´-ë©ìŠ¨ ë°©ë²•ìœ¼ë¡œ f(x) = 0ì˜ ê·¼ ì°¾ê¸°

    x_{n+1} = x_n - f(x_n) / f'(x_n)

    ìˆ˜ë ´ ì†ë„: 2ì°¨ (ì œê³± ìˆ˜ë ´)
    ë‹¨ì : ë„í•¨ìˆ˜ í•„ìš”, ì´ˆê¸°ê°’ì— ë¯¼ê°

    Args:
        f: ëª©í‘œ í•¨ìˆ˜
        df: ë„í•¨ìˆ˜
        x0: ì´ˆê¸°ê°’
        tol: í—ˆìš© ì˜¤ì°¨
        max_iter: ìµœëŒ€ ë°˜ë³µ íšŸìˆ˜

    Returns:
        (ê·¼, ë°˜ë³µ íšŸìˆ˜, íˆìŠ¤í† ë¦¬)
    """
    x = x0
    history = [x]

    for i in range(max_iter):
        fx = f(x)
        dfx = df(x)

        if abs(dfx) < 1e-15:
            raise ValueError("ë„í•¨ìˆ˜ê°€ 0ì— ê°€ê¹Œì›€: ë°œì‚° ìœ„í—˜")

        x_new = x - fx / dfx
        history.append(x_new)

        if abs(x_new - x) < tol:
            return x_new, i + 1, history

        x = x_new

    return x, max_iter, history

# 3. í• ì„ ë²• (Secant Method)
# =============================================================================
def secant(
    f: Callable[[float], float],
    x0: float,
    x1: float,
    tol: float = 1e-10,
    max_iter: int = 100
) -> Tuple[float, int, list]:
    """
    í• ì„ ë²•ìœ¼ë¡œ f(x) = 0ì˜ ê·¼ ì°¾ê¸°

    ë‰´í„´ë²•ì˜ ë„í•¨ìˆ˜ë¥¼ ì°¨ë¶„ìœ¼ë¡œ ê·¼ì‚¬:
    x_{n+1} = x_n - f(x_n) * (x_n - x_{n-1}) / (f(x_n) - f(x_{n-1}))

    ìˆ˜ë ´ ì†ë„: ì•½ 1.618ì°¨ (í™©ê¸ˆë¹„)
    ì¥ì : ë„í•¨ìˆ˜ ë¶ˆí•„ìš”

    Args:
        f: ëª©í‘œ í•¨ìˆ˜
        x0, x1: ë‘ ì´ˆê¸°ê°’
        tol: í—ˆìš© ì˜¤ì°¨
        max_iter: ìµœëŒ€ ë°˜ë³µ íšŸìˆ˜

    Returns:
        (ê·¼, ë°˜ë³µ íšŸìˆ˜, íˆìŠ¤í† ë¦¬)
    """
    history = [x0, x1]

    for i in range(max_iter):
        f0, f1 = f(x0), f(x1)

        if abs(f1 - f0) < 1e-15:
            raise ValueError("ë¶„ëª¨ê°€ 0ì— ê°€ê¹Œì›€")

        x2 = x1 - f1 * (x1 - x0) / (f1 - f0)
        history.append(x2)

        if abs(x2 - x1) < tol:
            return x2, i + 1, history

        x0, x1 = x1, x2

    return x1, max_iter, history

# 4. ê³ ì •ì  ë°˜ë³µë²• (Fixed-Point Iteration)
# =============================================================================
def fixed_point(
    g: Callable[[float], float],
    x0: float,
    tol: float = 1e-10,
    max_iter: int = 100
) -> Tuple[float, int, list]:
    """
    ê³ ì •ì  ë°˜ë³µ: x = g(x)ë¥¼ ë§Œì¡±í•˜ëŠ” x ì°¾ê¸°

    f(x) = 0 ì„ x = g(x) í˜•íƒœë¡œ ë³€í™˜
    ì˜ˆ: xÂ² - 2 = 0  â†’  x = 2/x ë˜ëŠ” x = (x + 2/x)/2

    ìˆ˜ë ´ ì¡°ê±´: |g'(x*)| < 1

    Args:
        g: ë°˜ë³µ í•¨ìˆ˜
        x0: ì´ˆê¸°ê°’
        tol: í—ˆìš© ì˜¤ì°¨
        max_iter: ìµœëŒ€ ë°˜ë³µ íšŸìˆ˜

    Returns:
        (ê³ ì •ì , ë°˜ë³µ íšŸìˆ˜, íˆìŠ¤í† ë¦¬)
    """
    x = x0
    history = [x]

    for i in range(max_iter):
        x_new = g(x)
        history.append(x_new)

        if abs(x_new - x) < tol:
            return x_new, i + 1, history

        x = x_new

    return x, max_iter, history

# 5. Brent's Method (scipyì™€ ë¹„êµ)
# =============================================================================
def brents_method(
    f: Callable[[float], float],
    a: float,
    b: float,
    tol: float = 1e-10,
    max_iter: int = 100
) -> Tuple[float, int]:
    """
    Brent's Method (ê°„ì†Œí™” ë²„ì „)
    ì´ë¶„ë²•, í• ì„ ë²•, ì—­2ì°¨ë³´ê°„ì„ ì¡°í•©í•œ ë°©ë²•

    ì‹¤ë¬´ì—ì„œëŠ” scipy.optimize.brentq ì‚¬ìš© ê¶Œì¥
    """
    fa, fb = f(a), f(b)
    if fa * fb >= 0:
        raise ValueError("f(a)ì™€ f(b)ì˜ ë¶€í˜¸ê°€ ë‹¬ë¼ì•¼ í•©ë‹ˆë‹¤")

    if abs(fa) < abs(fb):
        a, b = b, a
        fa, fb = fb, fa

    c, fc = a, fa
    d = c  # d ì´ˆê¸°í™” (ì²« iterationì—ì„œ ì‚¬ìš©ë¨)
    mflag = True

    for i in range(max_iter):
        if fa != fc and fb != fc:
            # ì—­2ì°¨ ë³´ê°„
            s = (a * fb * fc / ((fa - fb) * (fa - fc)) +
                 b * fa * fc / ((fb - fa) * (fb - fc)) +
                 c * fa * fb / ((fc - fa) * (fc - fb)))
        else:
            # í• ì„ ë²•
            s = b - fb * (b - a) / (fb - fa)

        # ì¡°ê±´ ì²´í¬ í›„ ì´ë¶„ë²•ìœ¼ë¡œ ëŒ€ì²´
        conditions = [
            not ((3 * a + b) / 4 <= s <= b or b <= s <= (3 * a + b) / 4),
            mflag and abs(s - b) >= abs(b - c) / 2,
            not mflag and abs(s - b) >= abs(c - d) / 2,
        ]

        if any(conditions):
            s = (a + b) / 2
            mflag = True
        else:
            mflag = False

        fs = f(s)
        d, c, fc = c, b, fb

        if fa * fs < 0:
            b, fb = s, fs
        else:
            a, fa = s, fs

        if abs(fa) < abs(fb):
            a, b = b, a
            fa, fb = fb, fa

        if abs(b - a) < tol or abs(fb) < tol:
            return b, i + 1

    return b, max_iter

def plot_convergence(f, methods_data, x_range, title="ê·¼ ì°¾ê¸° ìˆ˜ë ´ ë¹„êµ"):
    """ìˆ˜ë ´ ê³¼ì • ì‹œê°í™”"""
    fig, axes = plt.subplots(1, 2, figsize=(14, 5))

    # ì™¼ìª½: í•¨ìˆ˜ì™€ ê·¼

    y = [f(xi) for xi in x]
    axes[0].plot(x, y, 'b-', label='f(x)')
    axes[0].axhline(y=0, color='k', linestyle='-', linewidth=0.5)

    colors = plt.cm.tab10.colors
    for i, (name, root, _, history) in enumerate(methods_data):
        axes[0].scatter([root], [0], s=100, color=colors[i], zorder=5, label=f'{name}: x={root:.6f}')

    axes[0].set_xlabel('x')
    axes[0].set_ylabel('f(x)')
    axes[0].set_title('í•¨ìˆ˜ì™€ ê·¼')
