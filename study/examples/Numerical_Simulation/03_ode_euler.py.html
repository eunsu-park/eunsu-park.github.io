 14# 1. ì „ì§„ ì˜¤ì¼ëŸ¬ ë°©ë²• (Forward Euler Method)
 15# =============================================================================
 16def euler_forward(
 17    f: Callable[[float, float], float],
 18    y0: float,
 19    t_span: Tuple[float, float],
 20    h: float
 21) -> Tuple[np.ndarray, np.ndarray]:
 22    """
 23    ì „ì§„ ì˜¤ì¼ëŸ¬ ë°©ë²• (ëª…ì‹œì  ì˜¤ì¼ëŸ¬)
 24
 25    y_{n+1} = y_n + h * f(t_n, y_n)
 26
 27    ì˜¤ì°¨: O(h) - 1ì°¨ ì •í™•ë„
 28    ì•ˆì •ì„±: ì¡°ê±´ë¶€ ì•ˆì •
 29
 30    Args:
 31        f: dy/dt = f(t, y)
 32        y0: ì´ˆê¸°ê°’ y(t0)
 33        t_span: (t0, tf) ì‹œê°„ êµ¬ê°„
 34        h: ì‹œê°„ ê°„ê²©
 35
 36    Returns:
 37        (t ë°°ì—´, y ë°°ì—´)
 38    """
 39    t0, tf = t_span
 40    n_steps = int((tf - t0) / h)
 41
 42    t = np.linspace(t0, tf, n_steps + 1)
 43    y = np.zeros(n_steps + 1)
 44    y[0] = y0
 45
 46    for i in range(n_steps):
 47        y[i + 1] = y[i] + h * f(t[i], y[i])
 48
 49    return t, y

 53# 2. í›„ì§„ ì˜¤ì¼ëŸ¬ ë°©ë²• (Backward Euler Method)
 54# =============================================================================
 55def euler_backward(
 56    f: Callable[[float, float], float],
 57    df_dy: Callable[[float, float], float],
 58    y0: float,
 59    t_span: Tuple[float, float],
 60    h: float,
 61    newton_tol: float = 1e-10,
 62    max_iter: int = 50
 63) -> Tuple[np.ndarray, np.ndarray]:
 64    """
 65    í›„ì§„ ì˜¤ì¼ëŸ¬ ë°©ë²• (ì•”ì‹œì  ì˜¤ì¼ëŸ¬)
 66
 67    y_{n+1} = y_n + h * f(t_{n+1}, y_{n+1})
 68
 69    ì•”ì‹œì  ë°©ì •ì‹ì„ ë‰´í„´ë²•ìœ¼ë¡œ í•´ê²°
 70    ì˜¤ì°¨: O(h) - 1ì°¨ ì •í™•ë„
 71    ì•ˆì •ì„±: ë¬´ì¡°ê±´ ì•ˆì • (stiff ë¬¸ì œì— ì í•©)
 72
 73    Args:
 74        f: dy/dt = f(t, y)
 75        df_dy: âˆ‚f/âˆ‚y (ì•¼ì½”ë¹„ì•ˆ)
 76        y0: ì´ˆê¸°ê°’
 77        t_span: ì‹œê°„ êµ¬ê°„
 78        h: ì‹œê°„ ê°„ê²©
 79    """
 80    t0, tf = t_span
 81    n_steps = int((tf - t0) / h)
 82
 83    t = np.linspace(t0, tf, n_steps + 1)
 84    y = np.zeros(n_steps + 1)
 85    y[0] = y0
 86
 87    for i in range(n_steps):
 88        # ë‰´í„´ë²•ìœ¼ë¡œ y_{n+1} êµ¬í•˜ê¸°
 89        # g(y) = y - y_n - h*f(t_{n+1}, y) = 0
 90        y_new = y[i]  # ì´ˆê¸° ì¶”ì •ê°’ (ì „ì§„ ì˜¤ì¼ëŸ¬)
 91        t_new = t[i + 1]
 92
 93        for _ in range(max_iter):
 94            g = y_new - y[i] - h * f(t_new, y_new)
 95            dg = 1 - h * df_dy(t_new, y_new)
 96
 97            if abs(dg) < 1e-15:
 98                break
 99
100            delta = g / dg
101            y_new = y_new - delta
102
103            if abs(delta) < newton_tol:
104                break
105
106        y[i + 1] = y_new
107
108    return t, y

112# 3. ìˆ˜ì • ì˜¤ì¼ëŸ¬ ë°©ë²• (Modified Euler / Heun's Method)
113# =============================================================================
114def euler_modified(
115    f: Callable[[float, float], float],
116    y0: float,
117    t_span: Tuple[float, float],
118    h: float
119) -> Tuple[np.ndarray, np.ndarray]:
120    """
121    ìˆ˜ì • ì˜¤ì¼ëŸ¬ ë°©ë²• (Heun's Method)
122
123    ì˜ˆì¸¡: y*_{n+1} = y_n + h * f(t_n, y_n)
124    ìˆ˜ì •: y_{n+1} = y_n + h/2 * [f(t_n, y_n) + f(t_{n+1}, y*_{n+1})]
125
126    ì˜¤ì°¨: O(hÂ²) - 2ì°¨ ì •í™•ë„
127    RK2ì˜ ì¼ì¢…
128    """
129    t0, tf = t_span
130    n_steps = int((tf - t0) / h)
131
132    t = np.linspace(t0, tf, n_steps + 1)
133    y = np.zeros(n_steps + 1)
134    y[0] = y0
135
136    for i in range(n_steps):
137        k1 = f(t[i], y[i])
138        y_pred = y[i] + h * k1
139        k2 = f(t[i + 1], y_pred)
140        y[i + 1] = y[i] + h * (k1 + k2) / 2
141
142    return t, y

146# 4. ì—°ë¦½ ODE í’€ê¸°
147# =============================================================================
148def euler_system(
149    f: Callable[[float, np.ndarray], np.ndarray],
150    y0: np.ndarray,
151    t_span: Tuple[float, float],
152    h: float
153) -> Tuple[np.ndarray, np.ndarray]:
154    """
155    ì—°ë¦½ ODEë¥¼ ì „ì§„ ì˜¤ì¼ëŸ¬ë¡œ í’€ê¸°
156
157    dy/dt = f(t, y), y = [y1, y2, ..., yn]
158
159    Args:
160        f: ë²¡í„° í•¨ìˆ˜ f(t, y) -> dy/dt
161        y0: ì´ˆê¸°ê°’ ë²¡í„°
162        t_span: ì‹œê°„ êµ¬ê°„
163        h: ì‹œê°„ ê°„ê²©
164
165    Returns:
166        (t ë°°ì—´, y ë°°ì—´) - y.shape = (n_steps+1, n_vars)
167    """
168    t0, tf = t_span
169    n_steps = int((tf - t0) / h)
170    n_vars = len(y0)
171
172    t = np.linspace(t0, tf, n_steps + 1)
173    y = np.zeros((n_steps + 1, n_vars))
174    y[0] = y0
175
176    for i in range(n_steps):
177        y[i + 1] = y[i] + h * np.array(f(t[i], y[i]))
178
179    return t, y

183# 5. 2ì°¨ ODEë¥¼ 1ì°¨ ì—°ë¦½ìœ¼ë¡œ ë³€í™˜
184# =============================================================================
185def solve_second_order(
186    f: Callable[[float, float, float], float],
187    y0: float,
188    v0: float,
189    t_span: Tuple[float, float],
190    h: float
191) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
192    """
193    2ì°¨ ODE: y'' = f(t, y, y')
194    ë³€í™˜: yâ‚ = y, yâ‚‚ = y'
195          yâ‚' = yâ‚‚
196          yâ‚‚' = f(t, yâ‚, yâ‚‚)
197
198    Args:
199        f: y'' = f(t, y, y')
200        y0: ì´ˆê¸° ìœ„ì¹˜
201        v0: ì´ˆê¸° ì†ë„
202        t_span, h: ì‹œê°„ êµ¬ê°„ê³¼ ê°„ê²©
203
204    Returns:
205        (t, y, v) - ìœ„ì¹˜ì™€ ì†ë„
206    """
207    def system(t, state):
208        y, v = state
209        return np.array([v, f(t, y, v)])
210
211    t, solution = euler_system(system, np.array([y0, v0]), t_span, h)
212    return t, solution[:, 0], solution[:, 1]
