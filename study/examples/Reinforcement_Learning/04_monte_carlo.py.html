ëª¬í…Œì¹´ë¥¼ë¡œ ë°©ë²• (Monte Carlo Methods) êµ¬í˜„

- Every-visit MC Prediction

import numpy as np
import matplotlib.pyplot as plt
from collections import defaultdict
import gymnasium as gym


def calculate_returns(episode, gamma=0.99):
    """
    ì—í”¼ì†Œë“œì—ì„œ ê° ì‹œì ì˜ ë¦¬í„´ ê³„ì‚°

    Args:
        episode: [(state, action, reward), ...] í˜•íƒœì˜ ë¦¬ìŠ¤íŠ¸
        gamma: í• ì¸ìœ¨

    Returns:
        returns: [(state, G), ...] ê° ì‹œì ì˜ ë¦¬í„´
    """
    G = 0  # ë¦¬í„´ ì´ˆê¸°í™”
    returns = []

    # ì—­ìˆœìœ¼ë¡œ ê³„ì‚° (íš¨ìœ¨ì ì¸ ê³„ì‚°)
    for t in range(len(episode) - 1, -1, -1):
        state, action, reward = episode[t]
        G = reward + gamma * G  # í• ì¸ëœ ë¦¬í„´
        returns.insert(0, (state, G))

    return returns


def first_visit_mc_prediction(env, policy, n_episodes=10000, gamma=0.99):
    """
    First-visit MC ì •ì±… í‰ê°€

    Args:
        env: Gymnasium í™˜ê²½
        policy: ì •ì±… í•¨ìˆ˜ policy(state) -> action
        n_episodes: ì—í”¼ì†Œë“œ ìˆ˜
        gamma: í• ì¸ìœ¨

    Returns:
        V: ìƒíƒœ ê°€ì¹˜ í•¨ìˆ˜
    """
    # ê° ìƒíƒœì˜ ë¦¬í„´ í•©ê³¼ ë°©ë¬¸ íšŸìˆ˜
    returns_sum = defaultdict(float)
    returns_count = defaultdict(int)
    V = defaultdict(float)

    for episode_num in range(n_episodes):
        # ì—í”¼ì†Œë“œ ìƒì„±
        episode = []
        state, _ = env.reset()
        done = False

        while not done:
            action = policy(state)
            next_state, reward, terminated, truncated, _ = env.step(action)
            episode.append((state, action, reward))
            state = next_state
            done = terminated or truncated

        # First-visit: ê° ìƒíƒœì˜ ì²« ë°©ë¬¸ ì¸ë±ìŠ¤ ì°¾ê¸°
        visited = set()
        G = 0

        # ì—­ìˆœìœ¼ë¡œ ë¦¬í„´ ê³„ì‚°
        for t in range(len(episode) - 1, -1, -1):
            state_t, action_t, reward_t = episode[t]

            G = gamma * G + reward_t

            # First-visit ì²´í¬
            if state_t not in visited:
                visited.add(state_t)
                returns_sum[state_t] += G
                returns_count[state_t] += 1
                V[state_t] = returns_sum[state_t] / returns_count[state_t]

        if (episode_num + 1) % 2000 == 0:
            print(f"Episode {episode_num + 1}/{n_episodes}")

    return dict(V)


def every_visit_mc_prediction(env, policy, n_episodes=10000, gamma=0.99):
    """
    Every-visit MC ì •ì±… í‰ê°€

    ëª¨ë“  ë°©ë¬¸ì„ ì¹´ìš´íŠ¸
    """
    returns_sum = defaultdict(float)
    returns_count = defaultdict(int)
    V = defaultdict(float)

        episode = []

            action = policy(state)
            next_state, reward, terminated, truncated, _ = env.step(action)
            episode.append((state, action, reward))
            state = next_state
            done = terminated or truncated

        G = 0

        # Every-visit: ëª¨ë“  ë°©ë¬¸ì—ì„œ ì—…ë°ì´íŠ¸
        for t in range(len(episode) - 1, -1, -1):
            state_t, action_t, reward_t = episode[t]

            G = gamma * G + reward_t

            # ëª¨ë“  ë°©ë¬¸ ì¹´ìš´íŠ¸
            returns_sum[state_t] += G
            returns_count[state_t] += 1
            V[state_t] = returns_sum[state_t] / returns_count[state_t]

        if (episode_num + 1) % 2000 == 0:
            print(f"Episode {episode_num + 1}/{n_episodes}")

    return dict(V)


def epsilon_greedy_policy(Q, state, n_actions, epsilon=0.1):
    """
    Îµ-íƒìš• í–‰ë™ ì„ íƒ

    Args:
        Q: í–‰ë™ ê°€ì¹˜ í•¨ìˆ˜
        state: í˜„ì¬ ìƒíƒœ
        n_actions: í–‰ë™ ìˆ˜
        epsilon: íƒí—˜ í™•ë¥ 

    Returns:
        action: ì„ íƒëœ í–‰ë™
    """
    if np.random.random() < epsilon:
        # íƒí—˜: ëœë¤ í–‰ë™
        return np.random.randint(n_actions)
    else:
        # í™œìš©: ìµœì„ ì˜ í–‰ë™
        return np.argmax(Q[state])


def mc_on_policy_control(env, n_episodes=100000, gamma=0.99,
                         epsilon=0.1, epsilon_decay=0.9999):
    """
    On-policy MC ì œì–´ (Îµ-greedy)

    Args:
        env: Gymnasium í™˜ê²½
        n_episodes: ì—í”¼ì†Œë“œ ìˆ˜
        gamma: í• ì¸ìœ¨
        epsilon: íƒí—˜ìœ¨
        epsilon_decay: epsilon ê°ì†Œìœ¨

    Returns:
        Q: í–‰ë™ ê°€ì¹˜ í•¨ìˆ˜
        policy: í•™ìŠµëœ ì •ì±…
        episode_rewards: ì—í”¼ì†Œë“œë³„ ë³´ìƒ
    """
    n_actions = env.action_space.n

    Q = defaultdict(lambda: np.zeros(n_actions))
    returns_sum = defaultdict(lambda: np.zeros(n_actions))
    returns_count = defaultdict(lambda: np.zeros(n_actions))

    episode_rewards = []

    print("MC On-Policy Control í•™ìŠµ ì‹œì‘...")
    for episode_num in range(n_episodes):
        episode = []
        state, _ = env.reset()
        done = False
        total_reward = 0

        # Îµ-greedy ì •ì±…ìœ¼ë¡œ ì—í”¼ì†Œë“œ ìƒì„±
        while not done:
            action = epsilon_greedy_policy(Q, state, n_actions, epsilon)
            next_state, reward, terminated, truncated, _ = env.step(action)

            episode.append((state, action, reward))
            total_reward += reward

            state = next_state
            done = terminated or truncated

        episode_rewards.append(total_reward)

        # Q ì—…ë°ì´íŠ¸ (First-visit)
        G = 0
        visited = set()

        for t in range(len(episode) - 1, -1, -1):
            state_t, action_t, reward_t = episode[t]
            G = gamma * G + reward_t

            if (state_t, action_t) not in visited:
                visited.add((state_t, action_t))
                returns_sum[state_t][action_t] += G
                returns_count[state_t][action_t] += 1
                Q[state_t][action_t] = (returns_sum[state_t][action_t] /
                                        returns_count[state_t][action_t])

        # epsilon ê°ì†Œ
        epsilon = max(0.01, epsilon * epsilon_decay)

        if (episode_num + 1) % 10000 == 0:
            avg_reward = np.mean(episode_rewards[-1000:])
            print(f"Episode {episode_num + 1}: avg_reward = {avg_reward:.3f}, "
                  f"epsilon = {epsilon:.4f}")

    # ìµœì¢… íƒìš•ì  ì •ì±…
    policy = {}
    for state in Q:
        policy[state] = np.argmax(Q[state])

    return dict(Q), policy, episode_rewards


def mc_off_policy_control(env, n_episodes=100000, gamma=0.99):
    """
    Off-policy MC ì œì–´ (Weighted Importance Sampling)

    í–‰ë™ ì •ì±…: Îµ-greedy (íƒí—˜)
    ëª©í‘œ ì •ì±…: greedy (í™œìš©)

    Returns:
        Q: í–‰ë™ ê°€ì¹˜ í•¨ìˆ˜
        target_policy: ëª©í‘œ ì •ì±…
        episode_rewards: ì—í”¼ì†Œë“œë³„ ë³´ìƒ
    """
    n_actions = env.action_space.n

    Q = defaultdict(lambda: np.zeros(n_actions))
    C = defaultdict(lambda: np.zeros(n_actions))  # ê°€ì¤‘ì¹˜ í•©

    episode_rewards = []
    epsilon = 0.1  # í–‰ë™ ì •ì±…ì˜ epsilon

    print("MC Off-Policy Control í•™ìŠµ ì‹œì‘...")
    for episode_num in range(n_episodes):
        # í–‰ë™ ì •ì±… (Îµ-greedy)ìœ¼ë¡œ ì—í”¼ì†Œë“œ ìƒì„±
        episode = []
        state, _ = env.reset()
        done = False
        total_reward = 0

        while not done:
            action = epsilon_greedy_policy(Q, state, n_actions, epsilon)
            next_state, reward, terminated, truncated, _ = env.step(action)
            episode.append((state, action, reward))
            total_reward += reward
            state = next_state
            done = terminated or truncated

        episode_rewards.append(total_reward)

        G = 0
        W = 1.0  # ì¤‘ìš”ë„ ìƒ˜í”Œë§ ê°€ì¤‘ì¹˜

        # ì—­ìˆœ ì²˜ë¦¬
        for t in range(len(episode) - 1, -1, -1):
            state_t, action_t, reward_t = episode[t]
            G = gamma * G + reward_t

            # ê°€ì¤‘ ì¤‘ìš”ë„ ìƒ˜í”Œë§ ì—…ë°ì´íŠ¸
            C[state_t][action_t] += W
            Q[state_t][action_t] += (W / C[state_t][action_t] *
                                     (G - Q[state_t][action_t]))

            # ëª©í‘œ ì •ì±…ì—ì„œì˜ í–‰ë™ (greedy)
            target_action = np.argmax(Q[state_t])

            # í–‰ë™ì´ ëª©í‘œ ì •ì±…ê³¼ ë‹¤ë¥´ë©´ ì¤‘ë‹¨
            if action_t != target_action:
                break

            # ì¤‘ìš”ë„ ë¹„ìœ¨ ì—…ë°ì´íŠ¸
            # Ï€(a|s) = 1 (ê²°ì •ì ), b(a|s) = (1-Îµ) + Îµ/|A| or Îµ/|A|
            if action_t == target_action:
                b_prob = (1 - epsilon) + epsilon / n_actions
            else:
                b_prob = epsilon / n_actions

            W = W * 1.0 / b_prob

        if (episode_num + 1) % 10000 == 0:
            avg_reward = np.mean(episode_rewards[-1000:])
            print(f"Episode {episode_num + 1}: avg_reward = {avg_reward:.3f}")

    # ìµœì¢… íƒìš•ì  ì •ì±…
    target_policy = {}
    for state in Q:
        target_policy[state] = np.argmax(Q[state])

    return dict(Q), target_policy, episode_rewards


def blackjack_example():
    """ë¸”ë™ì­ í™˜ê²½ì—ì„œ MC í•™ìŠµ"""
    print("\n" + "=" * 60)
    print("ë¸”ë™ì­ ì˜ˆì œ - MC On-Policy Control")

    env = gym.make('Blackjack-v1', sab=True)

    n_actions = env.action_space.n
    Q = defaultdict(lambda: np.zeros(n_actions))
    returns_sum = defaultdict(lambda: np.zeros(n_actions))
    returns_count = defaultdict(lambda: np.zeros(n_actions))

    n_episodes = 500000
    gamma = 1.0
    epsilon = 0.1

    wins = 0
    losses = 0
    draws = 0

    print(f"\n{n_episodes} ì—í”¼ì†Œë“œ í•™ìŠµ ì¤‘...")
    for ep in range(n_episodes):
        episode = []
        state, _ = env.reset()
        done = False

        while not done:
            action = epsilon_greedy_policy(Q, state, n_actions, epsilon)
            next_state, reward, terminated, truncated, _ = env.step(action)
            episode.append((state, action, reward))
            state = next_state
            done = terminated or truncated

        # ê²°ê³¼ ê¸°ë¡
        final_reward = episode[-1][2]
        if final_reward == 1:
            wins += 1
        elif final_reward == -1:
            losses += 1
        else:
            draws += 1

        # Q ì—…ë°ì´íŠ¸
        G = 0
        visited = set()

        for t in range(len(episode) - 1, -1, -1):
            state_t, action_t, reward_t = episode[t]
            G = gamma * G + reward_t

            if (state_t, action_t) not in visited:
                visited.add((state_t, action_t))
                returns_sum[state_t][action_t] += G
                returns_count[state_t][action_t] += 1
                Q[state_t][action_t] = (returns_sum[state_t][action_t] /
                                        returns_count[state_t][action_t])

        if (ep + 1) % 100000 == 0:
            win_rate = wins / (ep + 1)
            print(f"Episode {ep + 1}: ìŠ¹ë¥  = {win_rate:.3f}")

    env.close()

    # ìµœì¢… í†µê³„
    print("\ní•™ìŠµ ì™„ë£Œ!")

    print(f"íŒ¨ë°°: {losses} ({losses/n_episodes*100:.1f}%)")
    print(f"ë¬´ìŠ¹ë¶€: {draws} ({draws/n_episodes*100:.1f}%)")
    print(f"í•™ìŠµëœ ìƒíƒœ-í–‰ë™ ìŒ ìˆ˜: {len(Q)}")

def plot_learning_curve(episode_rewards, window=1000):
    """í•™ìŠµ ê³¡ì„  ì‹œê°í™”"""
    # ì´ë™ í‰ê·  ê³„ì‚°
    moving_avg = []
    for i in range(len(episode_rewards)):
        start = max(0, i - window + 1)
        moving_avg.append(np.mean(episode_rewards[start:i+1]))

    plt.figure(figsize=(12, 6))
    plt.plot(moving_avg, label=f'Moving Average (window={window})')
    plt.xlabel('Episode')
    plt.ylabel('Average Reward')
    plt.title('Monte Carlo Learning Curve')
    plt.legend()
    plt.grid(True)
    plt.tight_layout()
    plt.savefig('mc_learning_curve.png', dpi=150)
    print("í•™ìŠµ ê³¡ì„  ì €ì¥: mc_learning_curve.png")
