"""
Actor-Critic (A2C) êµ¬í˜„
Actor-Critic ì•„í‚¤í…ì²˜, Advantage ì¶”ì •, GAE í¬í•¨
"""
import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
import gymnasium as gym
import matplotlib.pyplot as plt


class ActorCriticNetwork(nn.Module):
    """Actor-Critic ê³µìœ  ë„¤íŠ¸ì›Œí¬"""

    def __init__(self, state_dim, action_dim, hidden_dim=128):
        super().__init__()

        # ê³µìœ  íŠ¹ì§• ì¶”ì¶œ ë ˆì´ì–´
        self.shared = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU()
        )

        # Actor (ì •ì±… ë„¤íŠ¸ì›Œí¬)
        self.actor = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, action_dim)
        )

        # Critic (ê°€ì¹˜ ë„¤íŠ¸ì›Œí¬)
        self.critic = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, 1)
        )

    def forward(self, state):
        """ìˆœì „íŒŒ: ì •ì±…ê³¼ ê°€ì¹˜ë¥¼ ë™ì‹œì— ì¶œë ¥"""
        features = self.shared(state)
        policy_logits = self.actor(features)
        value = self.critic(features)
        return policy_logits, value

    def get_action(self, state):
        """í–‰ë™ ìƒ˜í”Œë§"""
        policy_logits, value = self.forward(state)
        policy = F.softmax(policy_logits, dim=-1)
        dist = torch.distributions.Categorical(policy)
        action = dist.sample()
        log_prob = dist.log_prob(action)
        entropy = dist.entropy()
        return action.item(), log_prob, value, entropy


class A2CAgent:
    """A2C (Advantage Actor-Critic) ì—ì´ì „íŠ¸"""

    def __init__(self, state_dim, action_dim, lr=3e-4, gamma=0.99,
                 value_coef=0.5, entropy_coef=0.01):
        self.network = ActorCriticNetwork(state_dim, action_dim)
        self.optimizer = torch.optim.Adam(self.network.parameters(), lr=lr)

        self.gamma = gamma
        self.value_coef = value_coef  # Critic ì†ì‹¤ ê°€ì¤‘ì¹˜
        self.entropy_coef = entropy_coef  # ì—”íŠ¸ë¡œí”¼ ë³´ë„ˆìŠ¤ ê°€ì¤‘ì¹˜

        # ì—í”¼ì†Œë“œ ë²„í¼
        self.reset_buffers()

    def reset_buffers(self):
        """ë²„í¼ ì´ˆê¸°í™”"""
        self.log_probs = []
        self.values = []
        self.rewards = []
        self.dones = []
        self.entropies = []

    def choose_action(self, state):
        """í–‰ë™ ì„ íƒ"""
        state_tensor = torch.FloatTensor(state).unsqueeze(0)
        action, log_prob, value, entropy = self.network.get_action(state_tensor)

        # ë²„í¼ì— ì €ì¥
        self.log_probs.append(log_prob)
        self.values.append(value)
        self.entropies.append(entropy)

        return action

    def store_transition(self, reward, done):
        """ì „ì´ ì €ì¥"""
        self.rewards.append(reward)
        self.dones.append(done)

    def compute_returns(self, next_value):
        """n-step returns ê³„ì‚° (ë¶€íŠ¸ìŠ¤íŠ¸ë˜í•‘)"""
        returns = []
        R = next_value

        for reward, done in zip(reversed(self.rewards), reversed(self.dones)):
            if done:
                R = 0
            R = reward + self.gamma * R
            returns.insert(0, R)

        return torch.tensor(returns, dtype=torch.float32)

    def update(self, next_state):
        """A2C ì—…ë°ì´íŠ¸"""
        if len(self.rewards) == 0:
            return 0, 0

        # ë‹¤ìŒ ìƒíƒœì˜ ê°€ì¹˜ (ë¶€íŠ¸ìŠ¤íŠ¸ë˜í•‘ìš©)
        with torch.no_grad():
            state_tensor = torch.FloatTensor(next_state).unsqueeze(0)
            _, next_value = self.network(state_tensor)
            next_value = next_value.item()

        # Returns ê³„ì‚°
        returns = self.compute_returns(next_value)
        values = torch.cat(self.values).squeeze()
        log_probs = torch.stack(self.log_probs)
        entropies = torch.stack(self.entropies)

        # Advantage ê³„ì‚°: A(s,a) = Q(s,a) - V(s) â‰ˆ R - V(s)
        advantages = returns - values.detach()

        # ì†ì‹¤ ê³„ì‚°
        actor_loss = -(log_probs * advantages).mean()  # Policy gradient
        critic_loss = F.mse_loss(values, returns)  # Value function loss
        entropy_loss = -entropies.mean()  # íƒìƒ‰ ì¥ë ¤

        total_loss = (actor_loss +
                      self.value_coef * critic_loss +
                      self.entropy_coef * entropy_loss)

        # ê·¸ë˜ë””ì–¸íŠ¸ ì—…ë°ì´íŠ¸
        self.optimizer.zero_grad()
        total_loss.backward()
        torch.nn.utils.clip_grad_norm_(self.network.parameters(), max_norm=0.5)
        self.optimizer.step()

        # ë²„í¼ ì´ˆê¸°í™”
        self.reset_buffers()

        return actor_loss.item(), critic_loss.item()


class A2CWithGAE(A2CAgent):
    """GAE (Generalized Advantage Estimation)ë¥¼ ì‚¬ìš©í•˜ëŠ” A2C"""

    def __init__(self, *args, gae_lambda=0.95, **kwargs):
        super().__init__(*args, **kwargs)
        self.gae_lambda = gae_lambda

    def compute_gae(self, next_value):
        """GAEë¥¼ ì‚¬ìš©í•œ Advantage ê³„ì‚°"""
        values = torch.cat(self.values).squeeze().tolist()
        values.append(next_value)  # ë§ˆì§€ë§‰ì— ë¶€íŠ¸ìŠ¤íŠ¸ë© ê°€ì¹˜ ì¶”ê°€

        advantages = []
        gae = 0

        # ì—­ë°©í–¥ìœ¼ë¡œ GAE ê³„ì‚°
        for t in reversed(range(len(self.rewards))):
            if self.dones[t]:
                delta = self.rewards[t] - values[t]
                gae = delta
            else:
                delta = self.rewards[t] + self.gamma * values[t + 1] - values[t]
                gae = delta + self.gamma * self.gae_lambda * gae

            advantages.insert(0, gae)

        advantages = torch.tensor(advantages, dtype=torch.float32)
        returns = advantages + torch.tensor(values[:-1], dtype=torch.float32)

        return advantages, returns

    def update(self, next_state):
        """GAEë¥¼ ì‚¬ìš©í•œ ì—…ë°ì´íŠ¸"""
        if len(self.rewards) == 0:
            return 0, 0

        # ë‹¤ìŒ ìƒíƒœì˜ ê°€ì¹˜
        with torch.no_grad():
            state_tensor = torch.FloatTensor(next_state).unsqueeze(0)
            _, next_value = self.network(state_tensor)
            next_value = next_value.item()

        # GAEë¡œ advantageì™€ returns ê³„ì‚°
        advantages, returns = self.compute_gae(next_value)

        values = torch.cat(self.values).squeeze()
        log_probs = torch.stack(self.log_probs)
        entropies = torch.stack(self.entropies)

        # ì†ì‹¤ ê³„ì‚°
        actor_loss = -(log_probs * advantages.detach()).mean()
        critic_loss = F.mse_loss(values, returns)
        entropy_loss = -entropies.mean()

        total_loss = (actor_loss +
                      self.value_coef * critic_loss +
                      self.entropy_coef * entropy_loss)

        # ì—…ë°ì´íŠ¸
        self.optimizer.zero_grad()
        total_loss.backward()
        torch.nn.utils.clip_grad_norm_(self.network.parameters(), max_norm=0.5)
        self.optimizer.step()

        self.reset_buffers()

        return actor_loss.item(), critic_loss.item()


def train_a2c(env_name='CartPole-v1', n_episodes=1000, n_steps=5, use_gae=False):
    """A2C í•™ìŠµ"""
    env = gym.make(env_name)
    state_dim = env.observation_space.shape[0]
    action_dim = env.action_space.n

    # ì—ì´ì „íŠ¸ ìƒì„±
    if use_gae:
        agent = A2CWithGAE(state_dim, action_dim, lr=3e-4, gamma=0.99,
                          value_coef=0.5, entropy_coef=0.01, gae_lambda=0.95)
        method_name = "A2C with GAE"
    else:
        agent = A2CAgent(state_dim, action_dim, lr=3e-4, gamma=0.99,
                        value_coef=0.5, entropy_coef=0.01)
        method_name = "A2C"

    print(f"=== {method_name} í•™ìŠµ ì‹œì‘ ({env_name}) ===\n")

    scores = []
    actor_losses = []
    critic_losses = []

    for episode in range(n_episodes):
        state, _ = env.reset()
        total_reward = 0
        step_count = 0
        done = False

        while not done:
            action = agent.choose_action(state)
            next_state, reward, terminated, truncated, _ = env.step(action)
            done = terminated or truncated

            agent.store_transition(reward, done)
            state = next_state
            total_reward += reward
            step_count += 1

            # n-step ì—…ë°ì´íŠ¸ ë˜ëŠ” ì—í”¼ì†Œë“œ ì¢…ë£Œ ì‹œ ì—…ë°ì´íŠ¸
            if step_count % n_steps == 0 or done:
                actor_loss, critic_loss = agent.update(next_state)
                actor_losses.append(actor_loss)
                critic_losses.append(critic_loss)

        scores.append(total_reward)

        if (episode + 1) % 50 == 0:
            avg_score = np.mean(scores[-50:])
            avg_actor_loss = np.mean(actor_losses[-50:]) if actor_losses else 0
            avg_critic_loss = np.mean(critic_losses[-50:]) if critic_losses else 0
            print(f"Episode {episode + 1:4d} | "
                  f"Avg Score: {avg_score:7.2f} | "
                  f"Actor Loss: {avg_actor_loss:.4f} | "
                  f"Critic Loss: {avg_critic_loss:.4f}")

        # CartPole í•´ê²° ì¡°ê±´: ì—°ì† 100 ì—í”¼ì†Œë“œ í‰ê·  475 ì´ìƒ
        if len(scores) >= 100 and np.mean(scores[-100:]) >= 475:
            print(f"\ní™˜ê²½ í•´ê²°! ({episode + 1} ì—í”¼ì†Œë“œ)")
            break

    env.close()
    return agent, scores, actor_losses, critic_losses


def compare_a2c_with_reinforce():
    """A2Cì™€ REINFORCE ë¹„êµ"""
    print("=== A2C vs REINFORCE ë¹„êµ ===\n")

    # A2C í•™ìŠµ
    _, a2c_scores, _, _ = train_a2c('CartPole-v1', n_episodes=500, use_gae=False)

    # A2C with GAE í•™ìŠµ
    print("\n" + "="*60 + "\n")
    _, a2c_gae_scores, _, _ = train_a2c('CartPole-v1', n_episodes=500, use_gae=True)

    # í•™ìŠµ ê³¡ì„  ë¹„êµ ì‹œê°í™”
    plot_comparison(a2c_scores, a2c_gae_scores)

    return a2c_scores, a2c_gae_scores


def plot_comparison(a2c_scores, a2c_gae_scores):
    """í•™ìŠµ ê³¡ì„  ë¹„êµ ì‹œê°í™”"""
    window = 10

    def smooth(data, window):
        if len(data) < window:
            return data
        return np.convolve(data, np.ones(window)/window, mode='valid')

    plt.figure(figsize=(14, 5))

    # ì›ë³¸ ë°ì´í„°
    plt.subplot(1, 2, 1)
    plt.plot(a2c_scores, alpha=0.3, label='A2C (raw)', color='blue')
    plt.plot(a2c_gae_scores, alpha=0.3, label='A2C+GAE (raw)', color='green')
    plt.axhline(y=475, color='red', linestyle='--', linewidth=1, label='Solved threshold')
    plt.xlabel('Episode')
    plt.ylabel('Episode Reward')
    plt.title('A2C vs A2C+GAE - Raw Data')
    plt.legend()
    plt.grid(True, alpha=0.3)

    # í‰í™œí™”ëœ ë°ì´í„°
    plt.subplot(1, 2, 2)
    plt.plot(smooth(a2c_scores, window), label='A2C (smoothed)', linewidth=2, color='blue')
    plt.plot(smooth(a2c_gae_scores, window), label='A2C+GAE (smoothed)', linewidth=2, color='green')
    plt.axhline(y=475, color='red', linestyle='--', linewidth=1, label='Solved threshold')
    plt.xlabel('Episode')
    plt.ylabel('Episode Reward (smoothed)')
    plt.title(f'A2C vs A2C+GAE - Smoothed (window={window})')
    plt.legend()
    plt.grid(True, alpha=0.3)

    plt.tight_layout()
    plt.savefig('a2c_comparison.png', dpi=150)
    print("\ní•™ìŠµ ê³¡ì„ ì´ 'a2c_comparison.png'ë¡œ ì €ì¥ë˜ì—ˆìŠµë‹ˆë‹¤.")


def train_lunarlander():
    """LunarLander í™˜ê²½ì—ì„œ A2C í•™ìŠµ"""
    try:
        env = gym.make('LunarLander-v2')
    except:
        print("LunarLander-v2 í™˜ê²½ì„ ì°¾ì„ ìˆ˜ ì—†ìŠµë‹ˆë‹¤.")
        print("ì„¤ì¹˜: pip install gymnasium[box2d]")
        return None, None

    state_dim = env.observation_space.shape[0]
    action_dim = env.action_space.n

    # GAEë¥¼ ì‚¬ìš©í•˜ëŠ” A2C
    agent = A2CWithGAE(
        state_dim, action_dim,
        lr=7e-4, gamma=0.99,
        value_coef=0.5, entropy_coef=0.01,
        gae_lambda=0.95
    )

    print("=== LunarLander A2C í•™ìŠµ ì‹œì‘ ===\n")

    scores = []
    n_steps = 5
    n_episodes = 2000

    for episode in range(n_episodes):
        state, _ = env.reset()
        total_reward = 0
        steps = 0

        while True:
            action = agent.choose_action(state)
            next_state, reward, terminated, truncated, _ = env.step(action)
            done = terminated or truncated

            agent.store_transition(reward, done)
            state = next_state
            total_reward += reward
            steps += 1

            # n-step ì—…ë°ì´íŠ¸ ë˜ëŠ” ì¢…ë£Œ ì‹œ
            if steps % n_steps == 0 or done:
                agent.update(next_state)

            if done:
                break

        scores.append(total_reward)

        if (episode + 1) % 100 == 0:
            avg = np.mean(scores[-100:])
            print(f"Episode {episode + 1:4d} | Avg Score: {avg:.2f}")

            # LunarLander í•´ê²° ì¡°ê±´: í‰ê·  200 ì´ìƒ
            if avg >= 200:
                print(f"\ní™˜ê²½ í•´ê²°! ({episode + 1} ì—í”¼ì†Œë“œ)")
                break

    env.close()
    return agent, scores
