262@test "integration: gcd and lcm relationship" {
263    # For any two numbers a and b: a * b = gcd(a, b) * lcm(a, b)
264    a=24
265    b=36
266
267    gcd_val=$(gcd $a $b)
268    lcm_val=$(lcm $a $b)
269
270    product=$((a * b))
271    gcd_lcm_product=$((gcd_val * lcm_val))
272
273    [ "$product" -eq "$gcd_lcm_product" ]
274}
