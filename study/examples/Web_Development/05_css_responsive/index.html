8. 미디어 쿼리 예제 코드

/* Mobile First 접근법 */

/* 기본 (모바일) 스타일 */
.container {
    padding: 10px;
}

/* 태블릿 (576px 이상) */
@media (min-width: 576px) {
    .container {
        padding: 20px;
    }
}

/* 작은 데스크탑 (768px 이상) */
@media (min-width: 768px) {
    .container {
        max-width: 720px;
        margin: 0 auto;
    }
}

/* 데스크탑 (992px 이상) */
@media (min-width: 992px) {
    .container {
        max-width: 960px;
    }
}

/* 큰 화면 (1200px 이상) */
@media (min-width: 1200px) {
    .container {
        max-width: 1140px;
    }
}
