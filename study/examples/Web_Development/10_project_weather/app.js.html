async function fetchWeather(city) {
    showLoading();
    hideError();
    hideWeather();

    try {
        let data;

        if (DEMO_MODE) {
            // ë°ëª¨ ëª¨ë“œ: ìƒ˜í”Œ ë°ì´í„° ì‚¬ìš©
            await simulateDelay(800);
            data = getDemoData(city);
        } else {
            // ì‹¤ì œ API í˜¸ì¶œ
            const url = `${API_BASE}?q=${encodeURIComponent(city)}&appid=${API_KEY}&units=metric&lang=kr`;
            const response = await fetch(url);

            if (!response.ok) {
                if (response.status === 404) {
                    throw new Error(`'${city}'ë¥¼ ì°¾ì„ ìˆ˜ ì—†ìŠµë‹ˆë‹¤.`);
                } else if (response.status === 401) {
                    throw new Error('API í‚¤ê°€ ìœ íš¨í•˜ì§€ ì•ŠìŠµë‹ˆë‹¤.');
                } else {
                    throw new Error('ë‚ ì”¨ ì •ë³´ë¥¼ ê°€ì ¸ì˜¤ëŠ”ë° ì‹¤íŒ¨í–ˆìŠµë‹ˆë‹¤.');

    } finally {
        hideLoading();
    }
}

function updateBackground(weatherMain) {
    const app = document.querySelector('.app');

    // ê¸°ì¡´ í´ë˜ìŠ¤ ì œê±°
    app.classList.remove('sunny', 'cloudy', 'rainy', 'snowy');

    // ë‚ ì”¨ì— ë”°ë¥¸ í´ë˜ìŠ¤ ì¶”ê°€

            break;
        case 'snow':
            app.classList.add('snowy');
            break;
    }
}
