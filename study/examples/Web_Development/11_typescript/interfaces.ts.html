233class Stack<T> {
234    private items: T[] = [];
235
236    push(item: T): void {
237        this.items.push(item);
238    }
239
240    pop(): T | undefined {
241        return this.items.pop();
242    }
243
244    peek(): T | undefined {
245        return this.items[this.items.length - 1];
246    }
247
248    isEmpty(): boolean {
249        return this.items.length === 0;
250    }
251
252    size(): number {
253        return this.items.length;
254    }
255}
