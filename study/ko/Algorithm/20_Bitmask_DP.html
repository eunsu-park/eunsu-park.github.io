def bitmask_dp_2d(n, start, data):
    """
    dp[mask][i] = mask ìƒíƒœì—ì„œ í˜„ì¬ ìœ„ì¹˜ê°€ iì¼ ë•Œì˜ ìµœì ê°’
    TSP ë“±ì— ì‚¬ìš©
    """
    INF = float('inf')
    dp = [[INF] * n for _ in range(1 << n)]
    dp[1 << start][start] = 0

    for mask in range(1 << n):
        for last in range(n):
            if dp[mask][last] == INF:
                continue
            if not (mask & (1 << last)):
                continue

            for next_node in range(n):
                if mask & (1 << next_node):
                    continue

                new_mask = mask | (1 << next_node)
                cost = data[last][next_node]
                dp[new_mask][next_node] = min(
                    dp[new_mask][next_node],
                    dp[mask][last] + cost
                )

    return dp

def tsp(dist):
    """
    ì™¸íŒì› ë¬¸ì œ - ë¹„íŠ¸ë§ˆìŠ¤í¬ DP
    dist[i][j] = ë„ì‹œ iì—ì„œ jë¡œ ê°€ëŠ” ë¹„ìš©
    ì‹œê°„: O(nÂ² Ã— 2^n)
    ê³µê°„: O(n Ã— 2^n)

    INF = float('inf')

    # dp[mask][i] = mask ìƒíƒœì—ì„œ iì— ìˆì„ ë•Œ ìµœì†Œ ë¹„ìš©
    dp = [[INF] * n for _ in range(1 << n)]
    dp[1][0] = 0  # ì‹œì‘ì  0ì—ì„œ ì¶œë°œ

    for mask in range(1 << n):
        for last in range(n):
            if dp[mask][last] == INF:
                continue
            if not (mask & (1 << last)):
                continue

                if mask & (1 << next_city):
                    continue
                if dist[last][next_city] == INF:
                    continue

                new_mask = mask | (1 << next_city)

            result = min(result, dp[full_mask][last] + dist[last][0])

    return result if result != INF else -1

def tsp_with_path(dist):
    """TSP + ê²½ë¡œ ë³µì›"""
    n = len(dist)
    INF = float('inf')

    dp = [[INF] * n for _ in range(1 << n)]
    parent = [[-1] * n for _ in range(1 << n)]
    dp[1][0] = 0

    for mask in range(1 << n):
        for last in range(n):
            if dp[mask][last] == INF:
                continue

            for next_city in range(n):
                if mask & (1 << next_city):
                    continue
                if dist[last][next_city] == INF:
                    continue

                new_mask = mask | (1 << next_city)
                new_cost = dp[mask][last] + dist[last][next_city]

                if new_cost < dp[new_mask][next_city]:
                    dp[new_mask][next_city] = new_cost
                    parent[new_mask][next_city] = last

    # ìµœì†Œ ë¹„ìš© ì°¾ê¸°
    full_mask = (1 << n) - 1
    min_cost = INF
    last_city = -1

    for i in range(n):
        if dp[full_mask][i] != INF and dist[i][0] != INF:
            total = dp[full_mask][i] + dist[i][0]
            if total < min_cost:
                min_cost = total
                last_city = i

    if last_city == -1:
        return -1, []

    # ê²½ë¡œ ë³µì›
    path = [0]  # ë§ˆì§€ë§‰ì— 0ìœ¼ë¡œ ëŒì•„ê°
    mask = full_mask
    curr = last_city

    while curr != -1:
        path.append(curr)
        prev = parent[mask][curr]
        mask ^= (1 << curr)
        curr = prev

    path.reverse()
    return min_cost, path

# ì˜ˆì‹œ

print(f"ìµœì†Œ ë¹„ìš©: {cost}")  # 80
print(f"ê²½ë¡œ: {path}")  # [0, 1, 3, 2, 0]

#include <vector>

using namespace std;

const int INF = 1e9;

int tsp(vector<vector<int>>& dist) {
    int n = dist.size();
    vector<vector<int>> dp(1 << n, vector<int>(n, INF));
    dp[1][0] = 0;

    for (int mask = 1; mask < (1 << n); mask++) {
        for (int last = 0; last < n; last++) {
            if (dp[mask][last] == INF) continue;
            if (!(mask & (1 << last))) continue;

            for (int next = 0; next < n; next++) {
                if (mask & (1 << next)) continue;
                if (dist[last][next] == INF) continue;

                int newMask = mask | (1 << next);
                dp[newMask][next] = min(dp[newMask][next],
                                        dp[mask][last] + dist[last][next]);
            }
        }
    }

    int fullMask = (1 << n) - 1;
    int result = INF;

    for (int last = 0; last < n; last++) {
        if (dp[fullMask][last] != INF && dist[last][0] != INF) {
            result = min(result, dp[fullMask][last] + dist[last][0]);
        }
    }

    return result == INF ? -1 : result;
}

def min_set_cover(n, sets):
    """
    ìµœì†Œ ì§‘í•© ì»¤ë²„: ëª¨ë“  ì›ì†Œë¥¼ ì»¤ë²„í•˜ëŠ” ìµœì†Œ ë¶€ë¶„ì§‘í•© ìˆ˜
    sets[i] = ië²ˆì§¸ ì§‘í•©ì˜ ì›ì†Œë“¤ (ë¹„íŠ¸ë§ˆìŠ¤í¬)
    """

    full = (1 << n) - 1

    # dp[mask] = maskë¥¼ ì»¤ë²„í•˜ëŠ” ìµœì†Œ ì§‘í•© ìˆ˜
    INF = float('inf')

    for mask in range(1 << n):

            new_mask = mask | s

    return dp[full] if dp[full] != INF else -1

def min_assignment(cost):
    """
    nëª…ì˜ ì‚¬ëŒì—ê²Œ nê°œì˜ ì‘ì—… í• ë‹¹
    cost[i][j] = ì‚¬ëŒ iê°€ ì‘ì—… jë¥¼ í•  ë•Œ ë¹„ìš©
    ê° ì‚¬ëŒì€ í•˜ë‚˜ì˜ ì‘ì—…ë§Œ ìˆ˜í–‰
    """
    n = len(cost)
    INF = float('inf')

    # dp[mask] = maskì— í•´ë‹¹í•˜ëŠ” ì‘ì—…ë“¤ì´ í• ë‹¹ëœ ìƒíƒœì—ì„œ ìµœì†Œ ë¹„ìš©
    # ì‚¬ëŒ iëŠ” popcount(mask) ì´ì „ ì‚¬ëŒë“¤ì´ ì²˜ë¦¬
    dp = [INF] * (1 << n)
    dp[0] = 0

        if dp[mask] == INF:
            continue

        person = bin(mask).count('1')  # í˜„ì¬ í• ë‹¹í•  ì‚¬ëŒ
        if person >= n:
            continue

                continue

            new_mask = mask | (1 << job)
            dp[new_mask] = min(dp[new_mask], dp[mask] + cost[person][job])

    return dp[(1 << n) - 1]

def count_hamiltonian_paths(n, adj):
    """
    í•´ë°€í„´ ê²½ë¡œ ê°œìˆ˜: ëª¨ë“  ì •ì ì„ ì •í™•íˆ í•œ ë²ˆ ë°©ë¬¸í•˜ëŠ” ê²½ë¡œ
    adj[i][j] = True if edge (i, j) exists
    """
    # dp[mask][i] = mask ë°©ë¬¸, iì—ì„œ ëë‚˜ëŠ” ê²½ë¡œ ìˆ˜
    dp = [[0] * n for _ in range(1 << n)]

    # ì‹œì‘ì  ì„¤ì •
    for i in range(n):
        dp[1 << i][i] = 1

    for mask in range(1, 1 << n):
        for last in range(n):
            if not (mask & (1 << last)):
                continue
            if dp[mask][last] == 0:
                continue

            for next_node in range(n):
                if mask & (1 << next_node):
                    continue
                if not adj[last][next_node]:
                    continue

                new_mask = mask | (1 << next_node)
                dp[new_mask][next_node] += dp[mask][last]

    # ëª¨ë“  ì •ì  ë°©ë¬¸í•œ ê²½ë¡œ ìˆ˜
    full = (1 << n) - 1
    return sum(dp[full][i] for i in range(n))

    ë¶€ë¶„ì§‘í•© í•©ì´ targetì¸ ê²½ìš°ì˜ ìˆ˜
    """
    n = len(arr)
    count = 0

    for mask in range(1 << n):
        total = sum(arr[i] for i in range(n) if mask & (1 << i))
        if total == target:
            count += 1

    return count

def subset_sum_mitm(arr, target):
    """
    Meet in the Middle: O(2^(n/2))
    """
    n = len(arr)
    mid = n // 2

    # ì™¼ìª½ ì ˆë°˜ì˜ ëª¨ë“  ë¶€ë¶„ì§‘í•© í•©
    left_sums = []
    for mask in range(1 << mid):
        total = sum(arr[i] for i in range(mid) if mask & (1 << i))
        left_sums.append(total)

    # ì •ë ¬
    left_sums.sort()

    # ì˜¤ë¥¸ìª½ ì ˆë°˜ + ì´ë¶„íƒìƒ‰
    from bisect import bisect_left, bisect_right
    count = 0

    for mask in range(1 << (n - mid)):
        total = sum(arr[mid + i] for i in range(n - mid) if mask & (1 << i))
        need = target - total
        # left_sumsì—ì„œ needì˜ ê°œìˆ˜
        count += bisect_right(left_sums, need) - bisect_left(left_sums, need)

    return count

def sos_dp(arr):
    """
    SOS DP: ê° maskì— ëŒ€í•´ ë¶€ë¶„ì§‘í•© í•© ê³„ì‚°
    F[mask] = sum(A[i]) for all i that is subset of mask
    ì‹œê°„: O(n Ã— 2^n)
    """
    n = len(arr).bit_length()
    N = 1 << n

    # arrì„ N í¬ê¸°ë¡œ í™•ì¥
    F = arr + [0] * (N - len(arr))

    for i in range(n):
        for mask in range(N):
            if mask & (1 << i):
                F[mask] += F[mask ^ (1 << i)]

    return F
