def extended_gcd(a, b):
    """
    ax + by = gcd(a, b)ë¥¼ ë§Œì¡±í•˜ëŠ” x, yì™€ gcd ë°˜í™˜
    """

    x = y1
    y = x1 - (a // b) * y1

    return (x % mod + mod) % mod

def sieve_of_eratosthenes(n):
    """
    ì—ë¼í† ìŠ¤í…Œë„¤ìŠ¤ì˜ ì²´
    ì‹œê°„: O(n log log n)
    ê³µê°„: O(n)
    """
    if n < 2:
        return []

    is_prime = [True] * (n + 1)
    is_prime[0] = is_prime[1] = False

    for i in range(2, int(n**0.5) + 1):
        if is_prime[i]:
            for j in range(i * i, n + 1, i):
                is_prime[j] = False

    return [i for i in range(n + 1) if is_prime[i]]

# ì˜ˆì‹œ

print(primes)  # [2, 3, 5, 7, 11, 13, ...]
print(len(primes))  # 25

// C++
vector<int> sieve(int n) {
    vector<bool> is_prime(n + 1, true);
    is_prime[0] = is_prime[1] = false;

    for (int i = 2; i * i <= n; i++) {
        if (is_prime[i]) {
            for (int j = i * i; j <= n; j += i) {
                is_prime[j] = false;
            }
        }
    }

    vector<int> primes;
    for (int i = 2; i <= n; i++) {
        if (is_prime[i]) primes.push_back(i);
    }
    return primes;
}

def factorize(n):
    """
    ì†Œì¸ìˆ˜ë¶„í•´
    ì‹œê°„: O(âˆšn)
    """
    factors = []
    d = 2

    while d * d <= n:
        while n % d == 0:
            factors.append(d)
            n //= d
        d += 1

    if n > 1:
        factors.append(n)

    return factors

def factorize_with_count(n):
    """ì†Œì¸ìˆ˜ì™€ ì§€ìˆ˜"""
    factors = {}
    d = 2

    while d * d <= n:
        while n % d == 0:
            factors[d] = factors.get(d, 0) + 1
            n //= d
        d += 1

    if n > 1:
        factors[n] = 1

    return factors

# ì˜ˆì‹œ

print(factorize_with_count(60))  # {2: 2, 3: 1, 5: 1}

def get_divisors(n):

    nì˜ ëª¨ë“  ì•½ìˆ˜
    ì‹œê°„: O(âˆšn)

def pascal_triangle(n):
    """
    íŒŒìŠ¤ì¹¼ ì‚¼ê°í˜• ìƒì„±
    ì‹œê°„: O(nÂ²)

def matrix_mult(A, B, mod):
    """
    í–‰ë ¬ ê³±ì…ˆ A * B
    ì‹œê°„: O(nÂ³)
    """
    n = len(A)
    m = len(B[0])
    k = len(B)

    C = [[0] * m for _ in range(n)]

    for i in range(n):
        for j in range(m):
            for l in range(k):
                C[i][j] = (C[i][j] + A[i][l] * B[l][j]) % mod

    return C

    í–‰ë ¬ Mì˜ nì œê³±
    ì‹œê°„: O(kÂ³ log n) (k = í–‰ë ¬ í¬ê¸°)

def fibonacci_matrix(n, mod=10**9 + 7):
    """
    í”¼ë³´ë‚˜ì¹˜ ìˆ˜ F(n) - O(log n)
    """
    if n <= 1:
        return n

    M = [[1, 1], [1, 0]]
    result = matrix_pow(M, n, mod)

    return result[1][0]

5.4 ì¼ë°˜ ì„ í˜• ì í™”ì‹¶

ì í™”ì‹: f(n) = a*f(n-1) + b*f(n-2) + c*f(n-3)

[f(n)  ]   [a b c]^(n-2)   [f(2)]
[f(n-1)] = [1 0 0]       * [f(1)]
[f(n-2)]   [0 1 0]         [f(0)]

def solve_linear_recurrence(coeffs, initial, n, mod):
    """
    ì„ í˜• ì í™”ì‹ í•´ê²°
    coeffs: [a, b, c, ...] (f(n) = a*f(n-1) + b*f(n-2) + ...)
    initial: [f(0), f(1), f(2), ...] (kê°œ)
    """
    k = len(coeffs)

    if n < k:
        return initial[n]

    # ì „ì´ í–‰ë ¬ êµ¬ì„±
    M = [[0] * k for _ in range(k)]
    for j in range(k):
        M[0][j] = coeffs[j]
    for i in range(1, k):
        M[i][i - 1] = 1

    result = matrix_pow(M, n - k + 1, mod)

    ans = 0
    for j in range(k):
        ans = (ans + result[0][j] * initial[k - 1 - j]) % mod

    return ans

# ì˜ˆì‹œ: f(n) = f(n-1) + f(n-2), f(0)=0, f(1)=1
print(solve_linear_recurrence([1, 1], [0, 1], 10, 10**9 + 7))  # 55

6.1 ì˜¤ì¼ëŸ¬ í”¼ í•¨ìˆ˜¶

Ï†(n) = n ì´í•˜ì˜ nê³¼ ì„œë¡œì†Œì¸ ìˆ˜ì˜ ê°œìˆ˜

Ï†(p) = p - 1  (pê°€ ì†Œìˆ˜)
Ï†(p^k) = p^k - p^(k-1)
Ï†(mn) = Ï†(m) * Ï†(n)  (m, nì´ ì„œë¡œì†Œ)

def euler_phi(n):
    """
    ì˜¤ì¼ëŸ¬ í”¼ í•¨ìˆ˜ Ï†(n)
    ì‹œê°„: O(âˆšn)
    """
    result = n
    p = 2

    while p * p <= n:
        if n % p == 0:
            while n % p == 0:
                n //= p
            result -= result // p
        p += 1

    if n > 1:
        result -= result // n

    return result

# ì˜ˆì‹œ
print(euler_phi(12))  # 4 (1, 5, 7, 11)
print(euler_phi(7))   # 6 (1, 2, 3, 4, 5, 6)

6.2 ë¤¼ì¹´ ì •ë¦¬ (Lucas' Theorem)¶

pê°€ ì†Œìˆ˜ì¼ ë•Œ:
C(m, n) mod p = Î  C(m_i, n_i) mod p

mê³¼ nì„ pì§„ë²•ìœ¼ë¡œ í‘œí˜„í–ˆì„ ë•Œ ê° ìë¦¿ìˆ˜ì˜ ì´í•­ê³„ìˆ˜ ê³±

def lucas(m, n, p):
    """
    ë¤¼ì¹´ ì •ë¦¬: C(m, n) mod p
    pê°€ ì†Œìˆ˜ì¼ ë•Œ ì‚¬ìš©
    """
    def nCr_small(a, b, p):
        if b > a:
            return 0
        if b == 0 or a == b:
            return 1

        num = den = 1
        for i in range(b):
            num = num * (a - i) % p
            den = den * (i + 1) % p

        return num * mod_pow(den, p - 2, p) % p

    result = 1
    while m > 0 or n > 0:
        mi, ni = m % p, n % p
        result = result * nCr_small(mi, ni, p) % p
        m //= p
        n //= p

    return result

# ì˜ˆì‹œ
print(lucas(1000, 500, 13))  # C(1000, 500) mod 13

6.3 ì¤‘êµ­ì¸ì˜ ë‚˜ë¨¸ì§€ ì •ë¦¬ (CRT)¶

def chinese_remainder_theorem(remainders, moduli):
    """
    ì¤‘êµ­ì¸ì˜ ë‚˜ë¨¸ì§€ ì •ë¦¬
    remainders: [a1, a2, ...]
    moduli: [m1, m2, ...]  (ìŒë§ˆë‹¤ ì„œë¡œì†Œ)
    """
    M = 1
    for m in moduli:
        M *= m

    result = 0
    for a, m in zip(remainders, moduli):
        Mi = M // m
        _, y, _ = extended_gcd(Mi, m)
        result = (result + a * Mi * y) % M

    return (result + M) % M
