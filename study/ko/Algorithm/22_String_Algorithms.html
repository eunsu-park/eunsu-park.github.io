def brute_force(text, pattern):
    """
    ë¸Œë£¨íŠ¸í¬ìŠ¤ íŒ¨í„´ ë§¤ì¹­
    ì‹œê°„: O(n * m), ê³µê°„: O(1)
    n = len(text), m = len(pattern)
    """
    n, m = len(text), len(pattern)
    result = []

    for i in range(n - m + 1):
        match = True
        for j in range(m):
            if text[i + j] != pattern[j]:
                match = False
                break
        if match:
            result.append(i)

    return result

    ì‹œê°„: O(n + m), ê³µê°„: O(m)

vector<int> computeFailure(const string& pattern) {
    int m = pattern.length();
    vector<int> pi(m, 0);
    int j = 0;

    for (int i = 1; i < m; i++) {
        while (j > 0 && pattern[i] != pattern[j]) {
            j = pi[j - 1];
        }
        if (pattern[i] == pattern[j]) {
            pi[i] = ++j;
        }
    }

    return pi;
}

vector<int> kmpSearch(const string& text, const string& pattern) {
    int n = text.length(), m = pattern.length();
    vector<int> pi = computeFailure(pattern);
    vector<int> result;
    int j = 0;

    for (int i = 0; i < n; i++) {
        while (j > 0 && text[i] != pattern[j]) {
            j = pi[j - 1];
        }
        if (text[i] == pattern[j]) {
            if (j == m - 1) {
                result.push_back(i - m + 1);
                j = pi[j];
            } else {
                j++;
            }
        }
    }

    return result;
}

def rabin_karp(text, pattern, d=256, q=101):
    """
    Rabin-Karp íŒ¨í„´ ë§¤ì¹­
    d: ê¸°ìˆ˜ (ë¬¸ì ì¢…ë¥˜ ìˆ˜)
    q: ëª¨ë“ˆëŸ¬ (í° ì†Œìˆ˜)
    ì‹œê°„: í‰ê·  O(n + m), ìµœì•… O(nm)
    """
    n, m = len(text), len(pattern)
    if m > n:
        return []

    result = []
    h = pow(d, m - 1, q)  # d^(m-1) mod q

    # ì´ˆê¸° í•´ì‹œê°’ ê³„ì‚°
    p_hash = 0  # íŒ¨í„´ í•´ì‹œ
    t_hash = 0  # í…ìŠ¤íŠ¸ ìœˆë„ìš° í•´ì‹œ

    for i in range(m):
        p_hash = (d * p_hash + ord(pattern[i])) % q
        t_hash = (d * t_hash + ord(text[i])) % q

    for i in range(n - m + 1):
        # í•´ì‹œê°€ ê°™ìœ¼ë©´ ì‹¤ì œ ë¹„êµ

                result.append(i)

        # ë‹¤ìŒ ìœˆë„ìš° í•´ì‹œ ê³„ì‚° (ë¡¤ë§)
        if i < n - m:
            t_hash = (d * (t_hash - ord(text[i]) * h) + ord(text[i + m])) % q
            if t_hash < 0:
                t_hash += q

    return result

def rabin_karp_multiple(text, patterns, d=256, q=101):
    """ì—¬ëŸ¬ íŒ¨í„´ ë™ì‹œ ê²€ìƒ‰"""
    n = len(text)
    result = {p: [] for p in patterns}

    # íŒ¨í„´ì„ ê¸¸ì´ë³„ë¡œ ê·¸ë£¹í™”
    by_length = {}
    for p in patterns:
        m = len(p)
        if m not in by_length:
            by_length[m] = {}
        # íŒ¨í„´ í•´ì‹œ ê³„ì‚°
        p_hash = 0
        for c in p:
            p_hash = (d * p_hash + ord(c)) % q
        if p_hash not in by_length[m]:
            by_length[m][p_hash] = []
        by_length[m][p_hash].append(p)

    # ê° ê¸¸ì´ì— ëŒ€í•´ ê²€ìƒ‰
    for m, hash_to_patterns in by_length.items():
        if m > n:
            continue

        h = pow(d, m - 1, q)
        t_hash = 0

        for i in range(m):
            t_hash = (d * t_hash + ord(text[i])) % q

        for i in range(n - m + 1):
            if t_hash in hash_to_patterns:
                for p in hash_to_patterns[t_hash]:
                    if text[i:i + m] == p:
                        result[p].append(i)

            if i < n - m:
                t_hash = (d * (t_hash - ord(text[i]) * h) + ord(text[i + m])) % q
                if t_hash < 0:
                    t_hash += q

    return result

def z_function(s):
    """
    Z ë°°ì—´ ê³„ì‚°
    ì‹œê°„: O(n)
    """
    n = len(s)
    z = [0] * n
    z[0] = n  # ì •ì˜ìƒ ì „ì²´ ë¬¸ìì—´

    l, r = 0, 0  # Z-boxì˜ ì™¼ìª½, ì˜¤ë¥¸ìª½ ê²½ê³„

    for i in range(1, n):
        if i < r:
            # Z-box ë‚´ë¶€: ì´ì „ ì •ë³´ í™œìš©
            z[i] = min(r - i, z[i - l])

        # í™•ì¥ ì‹œë„

            z[i] += 1

        # Z-box ì—…ë°ì´íŠ¸
        if i + z[i] > r:
            l, r = i, i + z[i]

    return z

def z_search(text, pattern):
    """
    Z-ì•Œê³ ë¦¬ì¦˜ì„ ì´ìš©í•œ íŒ¨í„´ ë§¤ì¹­
    ì‹œê°„: O(n + m)

#include <vector>
#include <string>
using namespace std;

vector<int> zFunction(const string& s) {
    int n = s.length();
    vector<int> z(n, 0);
    z[0] = n;
    int l = 0, r = 0;

    for (int i = 1; i < n; i++) {
        if (i < r) {
            z[i] = min(r - i, z[i - l]);
        }
        while (i + z[i] < n && s[z[i]] == s[i + z[i]]) {
            z[i]++;
        }
        if (i + z[i] > r) {
            l = i;
            r = i + z[i];
        }
    }

    return z;
}

vector<int> zSearch(const string& text, const string& pattern) {
    string concat = pattern + "$" + text;
    vector<int> z = zFunction(concat);
    int m = pattern.length();
    vector<int> result;

    for (int i = m + 1; i < concat.length(); i++) {
        if (z[i] == m) {
            result.push_back(i - m - 1);
        }
    }

    return result;
}

def polynomial_hash(s, base=31, mod=10**9 + 9):
    """
    ë‹¤í•­ì‹ ë¡¤ë§ í•´ì‹œ
    hash(s) = s[0]*base^(n-1) + s[1]*base^(n-2) + ... + s[n-1]
    """
    h = 0
    for c in s:
        h = (h * base + ord(c) - ord('a') + 1) % mod
    return h

class StringHash:
    """
    ë¬¸ìì—´ í•´ì‹œ (êµ¬ê°„ í•´ì‹œ ì¿¼ë¦¬ O(1))
    """
    def __init__(self, s, base=31, mod=10**9 + 9):
        self.base = base
        self.mod = mod
        self.n = len(s)

        # í”„ë¦¬í”½ìŠ¤ í•´ì‹œ
        self.prefix = [0] * (self.n + 1)
        # baseì˜ ê±°ë“­ì œê³±
        self.power = [1] * (self.n + 1)

        for i in range(self.n):
            self.prefix[i + 1] = (self.prefix[i] * base + ord(s[i]) - ord('a') + 1) % mod
            self.power[i + 1] = (self.power[i] * base) % mod

    def get_hash(self, l, r):
        """s[l:r+1]ì˜ í•´ì‹œê°’ (0-indexed)"""
        h = (self.prefix[r + 1] - self.prefix[l] * self.power[r - l + 1]) % self.mod
        return (h + self.mod) % self.mod

def longest_palindrome_substring(s):
    """
    ì¤‘ì‹¬ í™•ì¥ë²•
    ì‹œê°„: O(nÂ²)
    """

def repeated_substring_pattern(s):
    """
    ë¬¸ìì—´ì´ ë°˜ë³µ íŒ¨í„´ìœ¼ë¡œ êµ¬ì„±ë˜ì–´ ìˆëŠ”ì§€ í™•ì¸
    ì˜ˆ: "abab" â†’ True (abê°€ 2ë²ˆ ë°˜ë³µ)

def longest_common_substring(s1, s2):
    """
    ì´ë¶„íƒìƒ‰ + ë¡¤ë§ í•´ì‹œ
    ì‹œê°„: O((n+m) log(min(n,m)))
    """
    def get_hashes(s, length, base=31, mod=10**9 + 9):
        """ê¸¸ì´ lengthì¸ ëª¨ë“  ë¶€ë¶„ ë¬¸ìì—´ í•´ì‹œ"""
        if length > len(s):
            return set()

        hashes = set()
        h = 0
        power = pow(base, length - 1, mod)

        for i in range(length):
            h = (h * base + ord(s[i])) % mod

        hashes.add(h)

        for i in range(length, len(s)):
            h = (h - ord(s[i - length]) * power) % mod
            h = (h * base + ord(s[i])) % mod
            hashes.add(h)

        return hashes

    def check(length):
        """ê¸¸ì´ lengthì˜ ê³µí†µ ë¶€ë¶„ ë¬¸ìì—´ ì¡´ì¬ ì—¬ë¶€"""
        h1 = get_hashes(s1, length)
        h2 = get_hashes(s2, length)
        return len(h1 & h2) > 0

    left, right = 0, min(len(s1), len(s2))
    result = 0

    while left <= right:
        mid = (left + right) // 2
        if check(mid):
            result = mid
            left = mid + 1
        else:
            right = mid - 1

    return result

def find_anagrams(s, p):
    """
    sì—ì„œ pì˜ ì•„ë‚˜ê·¸ë¨ì¸ ë¶€ë¶„ ë¬¸ìì—´ ì‹œì‘ ì¸ë±ìŠ¤
    ìŠ¬ë¼ì´ë”© ìœˆë„ìš° + í•´ì‹œë§µ
    """
    from collections import Counter

    result = []
    p_count = Counter(p)
    s_count = Counter()
    m = len(p)

    for i, c in enumerate(s):
        s_count[c] += 1

        # ìœˆë„ìš° í¬ê¸° ìœ ì§€
        if i >= m:
            left = s[i - m]
            s_count[left] -= 1
            if s_count[left] == 0:
                del s_count[left]

        if s_count == p_count:
            result.append(i - m + 1)

    return result


# ì˜ˆì‹œ
print(find_anagrams("cbaebabacd", "abc"))  # [0, 6]

def compress_string(s):
    """
    ì—°ì†ëœ ê°™ì€ ë¬¸ìë¥¼ ì••ì¶•
    "aabcccccaaa" â†’ "a2b1c5a3"
    """
    if not s:
        return s

    result = []
    count = 1

    for i in range(1, len(s)):
        if s[i] == s[i - 1]:
            count += 1
        else:
            result.append(s[i - 1] + str(count))
            count = 1

    result.append(s[-1] + str(count))

    compressed = "".join(result)
    return compressed if len(compressed) < len(s) else s
