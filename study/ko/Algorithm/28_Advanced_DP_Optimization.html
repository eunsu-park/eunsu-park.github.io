class ConvexHullTrickMin:
    """
    ìµœì†Ÿê°’ ì¿¼ë¦¬ìš© CHT
    ì¡°ê±´: ì§ì„ ì˜ ê¸°ìš¸ê¸°ê°€ ë‹¨ì¡° ê°ì†Œ, ì¿¼ë¦¬ xê°€ ë‹¨ì¡° ì¦ê°€
    """
    def __init__(self):
        self.lines = []  # (ê¸°ìš¸ê¸°, yì ˆí¸)
        self.ptr = 0

    def bad(self, l1, l2, l3):
        """l2ê°€ ë¶ˆí•„ìš”í•œì§€ í™•ì¸"""
        # êµì (l1, l2).x >= êµì (l2, l3).x ë©´ l2 ë¶ˆí•„ìš”
        return (l3[1] - l1[1]) * (l1[0] - l2[0]) <= (l2[1] - l1[1]) * (l1[0] - l3[0])

    def add_line(self, m, b):
        """ì§ì„  y = mx + b ì¶”ê°€"""
        line = (m, b)
        while len(self.lines) >= 2 and self.bad(self.lines[-2], self.lines[-1], line):
            self.lines.pop()
        self.lines.append(line)

    def query(self, x):
        """xì—ì„œì˜ ìµœì†Ÿê°’"""
        if not self.lines:
            return float('inf')

        # í¬ì¸í„° ì´ë™ (xê°€ ë‹¨ì¡° ì¦ê°€í•  ë•Œ)
        while self.ptr < len(self.lines) - 1:
            m1, b1 = self.lines[self.ptr]
            m2, b2 = self.lines[self.ptr + 1]
            if m1 * x + b1 > m2 * x + b2:
                self.ptr += 1
            else:
                break

        m, b = self.lines[self.ptr]
        return m * x + b

# ì‚¬ìš© ì˜ˆì‹œ: dp[i] = min(dp[j] + a[j] * b[i])
def solve_with_cht(a, b):
    n = len(a)
    dp = [0] * n
    cht = ConvexHullTrickMin()

    # ì²« ë²ˆì§¸ ì§ì„  ì¶”ê°€ (j=0)
    cht.add_line(a[0], dp[0])

    for i in range(1, n):
        dp[i] = cht.query(b[i])
        cht.add_line(a[i], dp[i])

    return dp[n-1]

class LiChaoTree:
    def __init__(self, lo, hi):
        self.lo = lo
        self.hi = hi
        self.tree = {}  # ë…¸ë“œë³„ ì§ì„  ì €ì¥

    def add_line(self, m, b, node=1, lo=None, hi=None):
        """ì§ì„  y = mx + b ì¶”ê°€"""
        if lo is None:
            lo, hi = self.lo, self.hi

        if lo > hi:
            return

        mid = (lo + hi) // 2

        if node not in self.tree:
            self.tree[node] = (m, b)
            return

        old_m, old_b = self.tree[node]

        # midì—ì„œ ì–´ëŠ ì§ì„ ì´ ë” ì¢‹ì€ì§€ ë¹„êµ
        left_better = m * lo + b < old_m * lo + old_b
        mid_better = m * mid + b < old_m * mid + old_b

        if mid_better:
            self.tree[node] = (m, b)
            m, b = old_m, old_b

        if lo == hi:
            return

        # êµì°¨ê°€ ì–´ëŠ ìª½ì—ì„œ ì¼ì–´ë‚˜ëŠ”ì§€ì— ë”°ë¼ ì¬ê·€
        if left_better != mid_better:
            self.add_line(m, b, 2 * node, lo, mid)
        else:
            self.add_line(m, b, 2 * node + 1, mid + 1, hi)

    def query(self, x, node=1, lo=None, hi=None):
        """xì—ì„œì˜ ìµœì†Ÿê°’"""
        if lo is None:
            lo, hi = self.lo, self.hi

        if node not in self.tree:
            return float('inf')

        m, b = self.tree[node]
        result = m * x + b

        if lo == hi:
            return result

        mid = (lo + hi) // 2
        if x <= mid:
            result = min(result, self.query(x, 2 * node, lo, mid))
        else:
            result = min(result, self.query(x, 2 * node + 1, mid + 1, hi))

        return result

#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

struct Line {
    ll m, b;
    ll eval(ll x) { return m * x + b; }
};

class CHT {
private:
    deque<Line> lines;

    bool bad(Line l1, Line l2, Line l3) {
        // l2ê°€ ë¶ˆí•„ìš”í•œì§€ í™•ì¸
        return (__int128)(l3.b - l1.b) * (l1.m - l2.m) <=
               (__int128)(l2.b - l1.b) * (l1.m - l3.m);
    }

public:
    void addLine(ll m, ll b) {
        Line line = {m, b};
        while (lines.size() >= 2 && bad(lines[lines.size()-2], lines[lines.size()-1], line))
            lines.pop_back();
        lines.push_back(line);
    }

    ll query(ll x) {
        // ì´ì§„ íƒìƒ‰ ë˜ëŠ” í¬ì¸í„° ì‚¬ìš©
        int lo = 0, hi = lines.size() - 1;
        while (lo < hi) {
            int mid = (lo + hi) / 2;
            if (lines[mid].eval(x) > lines[mid + 1].eval(x))
                lo = mid + 1;
            else
                hi = mid;
        }
        return lines[lo].eval(x);
    }
};

def dnc_optimization(n, k, cost):
    """
    dp[k][n] ê³„ì‚° (kê°œì˜ ê·¸ë£¹ìœ¼ë¡œ nê°œ ì›ì†Œ ë¶„í• )
    cost(i, j): êµ¬ê°„ [i, j)ì˜ ë¹„ìš©
    """
    INF = float('inf')
    dp = [[INF] * (n + 1) for _ in range(k + 1)]
    dp[0][0] = 0

    def compute(row, dp_lo, dp_hi, opt_lo, opt_hi):
        if dp_lo > dp_hi:
            return

        dp_mid = (dp_lo + dp_hi) // 2
        best_cost = INF
        best_opt = opt_lo

        for opt in range(opt_lo, min(opt_hi, dp_mid) + 1):
            current_cost = dp[row - 1][opt] + cost(opt, dp_mid)
            if current_cost < best_cost:
                best_cost = current_cost
                best_opt = opt

        dp[row][dp_mid] = best_cost

        # ë¶„í•  ì •ë³µ
        compute(row, dp_lo, dp_mid - 1, opt_lo, best_opt)
        compute(row, dp_mid + 1, dp_hi, best_opt, opt_hi)

    for row in range(1, k + 1):
        compute(row, 1, n, 0, n - 1)

    return dp[k][n]

def solve():

#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int n, k;
vector<ll> arr, prefix;
vector<vector<ll>> dp;

ll cost(int i, int j) {
    ll sum = prefix[j] - prefix[i];
    return sum * sum;
}

void compute(int row, int lo, int hi, int opt_lo, int opt_hi) {
    if (lo > hi) return;

    int mid = (lo + hi) / 2;
    ll best = LLONG_MAX;
    int best_opt = opt_lo;

    for (int opt = opt_lo; opt <= min(opt_hi, mid - 1); opt++) {
        ll val = dp[row - 1][opt] + cost(opt, mid);
        if (val < best) {
            best = val;
            best_opt = opt;
        }
    }

    dp[row][mid] = best;

    compute(row, lo, mid - 1, opt_lo, best_opt);
    compute(row, mid + 1, hi, best_opt, opt_hi);
}

ll solve() {
    dp.assign(k + 1, vector<ll>(n + 1, LLONG_MAX));
    dp[0][0] = 0;

    for (int row = 1; row <= k; row++) {
        compute(row, 1, n, 0, n - 1);
    }

    return dp[k][n];
}

def optimal_bst(freq):
    """
    ìµœì  ì´ì§„ íƒìƒ‰ íŠ¸ë¦¬ êµ¬ì„± ë¹„ìš©
    freq[i]: ië²ˆì§¸ í‚¤ì˜ íƒìƒ‰ ë¹ˆë„
    """
    n = len(freq)
    INF = float('inf')

    # prefix sum for range cost
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + freq[i]

    def cost(i, j):
        return prefix[j] - prefix[i]

    # dp[i][j]: êµ¬ê°„ [i, j)ì˜ ìµœì†Œ ë¹„ìš©
    dp = [[0] * (n + 1) for _ in range(n + 1)]
    opt = [[0] * (n + 1) for _ in range(n + 1)]

    # ê¸¸ì´ 1
    for i in range(n):
        dp[i][i + 1] = freq[i]
        opt[i][i + 1] = i

    # ê¸¸ì´ 2 ì´ìƒ
    for length in range(2, n + 1):
        for i in range(n - length + 1):
            j = i + length
            dp[i][j] = INF

            # Knuth ìµœì í™”: opt[i][j-1] ~ opt[i+1][j]
            lo = opt[i][j - 1]
            hi = opt[i + 1][j] if i + 1 <= n and j <= n else j - 1

            for k in range(lo, min(hi, j - 1) + 1):
                val = dp[i][k] + dp[k + 1][j] + cost(i, j)
                if val < dp[i][j]:
                    dp[i][j] = val
                    opt[i][j] = k

    return dp[0][n]

def matrix_chain_multiplication(dims):
    """
    í–‰ë ¬ ì²´ì¸ ê³±ì…ˆì˜ ìµœì†Œ ì—°ì‚° íšŸìˆ˜
    dims[i]: ië²ˆì§¸ í–‰ë ¬ì˜ í–‰ ìˆ˜ (ë§ˆì§€ë§‰ì€ ì—´ ìˆ˜)
    """
    n = len(dims) - 1
    INF = float('inf')

    dp = [[0] * n for _ in range(n)]
    opt = [[0] * n for _ in range(n)]

    for i in range(n):
        opt[i][i] = i

    for length in range(2, n + 1):
        for i in range(n - length + 1):
            j = i + length - 1
            dp[i][j] = INF

            lo = opt[i][j - 1] if j > i else i
            hi = opt[i + 1][j] if i + 1 < n else j

            for k in range(lo, hi + 1):
                cost = dp[i][k] + dp[k + 1][j] + dims[i] * dims[k + 1] * dims[j + 1]
                if cost < dp[i][j]:
                    dp[i][j] = cost
                    opt[i][j] = k

    return dp[0][n - 1]

def check_quadrangle_inequality(cost, n):
    """
    cost(a, c) + cost(b, d) â‰¤ cost(a, d) + cost(b, c)
    for all a â‰¤ b â‰¤ c â‰¤ d
    """
    for a in range(n):
        for b in range(a, n):
            for c in range(b, n):
                for d in range(c, n):
                    lhs = cost(a, c) + cost(b, d)
                    rhs = cost(a, d) + cost(b, c)
                    if lhs > rhs:
                        return False
    return True

def special_post_office(villages, costs):
    """
    ìš°ì²´êµ­ ì„¤ì¹˜ ë¹„ìš© ìµœì†Œí™”

def merge_files(sizes):
    """
    ì—°ì†ëœ íŒŒì¼ì„ í•©ì¹  ë•Œ ìµœì†Œ ë¹„ìš©
    """
    n = len(sizes)
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + sizes[i]

    dp = [[0] * (n + 1) for _ in range(n + 1)]
    opt = [[0] * (n + 1) for _ in range(n + 1)]

    for i in range(n):
        opt[i][i + 1] = i

    for length in range(2, n + 1):
        for i in range(n - length + 1):
            j = i + length
            dp[i][j] = float('inf')

            lo = opt[i][j - 1]
            hi = opt[i + 1][j] if i + 1 < n else j - 1

            for k in range(lo, hi + 1):
                val = dp[i][k] + dp[k][j] + prefix[j] - prefix[i]
                if val < dp[i][j]:
                    dp[i][j] = val
                    opt[i][j] = k

    return dp[0][n]
