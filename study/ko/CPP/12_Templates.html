    while (*format) {
        if (*format == '%' && *(format + 1) != '%') {
            throw std::runtime_error("ì¸ì ë¶€ì¡±");
        }

template<typename T, typename... Args>
void safePrint(std::ostream& os, const char* format, T value, Args... args) {
    while (*format) {
        if (*format == '%') {
            if (*(format + 1) == '%') {
                os << '%';
                format += 2;
                continue;
            }
            os << value;
            safePrint(os, format + 1, args...);

}

    std::ostringstream oss;
    safePrint(oss, fmt, args...);

}
