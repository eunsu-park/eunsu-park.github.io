import cv2

def extract_face_region(img, x, y, w, h):
    """ì–¼êµ´ ì˜ì—­ ì¶”ì¶œ (ê²½ê³„ ì²´í¬ í¬í•¨)"""
    h_img, w_img = img.shape[:2]

    # ê²½ê³„ ì²´í¬
    x1 = max(0, x)
    y1 = max(0, y)
    x2 = min(w_img, x + w)
    y2 = min(h_img, y + h)

    return img[y1:y2, x1:x2].copy()


def apply_mosaic(img, x, y, w, h, ratio=0.1):
    """íŠ¹ì • ì˜ì—­ì— ëª¨ìì´í¬ ì ìš©"""
    roi = img[y:y+h, x:x+w]

    # ì¶•ì†Œ í›„ í™•ëŒ€ (ëª¨ìì´í¬ íš¨ê³¼)
    small = cv2.resize(roi, None, fx=ratio, fy=ratio,
                       interpolation=cv2.INTER_NEAREST)
    mosaic = cv2.resize(small, (w, h),
                        interpolation=cv2.INTER_NEAREST)

    img[y:y+h, x:x+w] = mosaic
    return img

# íŒíŠ¸: PSNR ê³„ì‚°
def calculate_psnr(original, compressed):
    mse = np.mean((original.astype(float) - compressed.astype(float)) ** 2)
    if mse == 0:
        return float('inf')
    max_pixel = 255.0
    psnr = 20 * np.log10(max_pixel / np.sqrt(mse))
    return psnr
