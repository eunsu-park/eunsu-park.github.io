    if M['m00'] == 0:
        return None, None

import cv2
import numpy as np

class ShapeClassifier:
    """ë„í˜• ë¶„ë¥˜ê¸°"""

    def __init__(self):
        self.shape_names = {
            3: 'Triangle',
            4: 'Rectangle',
            5: 'Pentagon',
            6: 'Hexagon'
        }

    def classify(self, contour):
        """ìœ¤ê³½ì„ ìœ¼ë¡œ ë„í˜• ë¶„ë¥˜"""
        # ê¸°ë³¸ ì†ì„±
        area = cv2.contourArea(contour)
        perimeter = cv2.arcLength(contour, True)

        if area < 100 or perimeter == 0:
            return None, {}

        # ë‹¤ê°í˜• ê·¼ì‚¬í™”
        approx = cv2.approxPolyDP(contour, 0.04 * perimeter, True)
        vertices = len(approx)

        # ê²½ê³„ ì‚¬ê°í˜•
        x, y, w, h = cv2.boundingRect(contour)
        aspect_ratio = w / h

        # ì›í˜•ë„
        circularity = 4 * np.pi * area / (perimeter ** 2)

        # ì†”ë¦¬ë””í‹°
        hull = cv2.convexHull(contour)
        hull_area = cv2.contourArea(hull)
        solidity = area / hull_area if hull_area > 0 else 0

        # ë¶„ë¥˜
        properties = {
            'vertices': vertices,
            'area': area,
            'perimeter': perimeter,
            'aspect_ratio': aspect_ratio,
            'circularity': circularity,
            'solidity': solidity
        }

        # ë„í˜• íŒë³„
        if vertices == 3:
            shape = 'Triangle'
        elif vertices == 4:
            if 0.95 <= aspect_ratio <= 1.05:
                shape = 'Square'
            else:
                shape = 'Rectangle'
        elif vertices == 5:
            shape = 'Pentagon'
        elif vertices == 6:
            shape = 'Hexagon'
        elif circularity > 0.85:
            shape = 'Circle'
        elif 0.6 < circularity < 0.85 and solidity > 0.9:
            shape = 'Ellipse'
        elif solidity < 0.7:
            shape = 'Star' if vertices > 6 else 'Irregular'
        else:
            shape = f'Polygon-{vertices}'

        return shape, properties

    def process_image(self, image_path):
        """ì´ë¯¸ì§€ì˜ ëª¨ë“  ë„í˜• ë¶„ë¥˜"""
        img = cv2.imread(image_path)

        _, binary = cv2.threshold(gray, 127, 255, cv2.THRESH_BINARY_INV)

        contours, _ = cv2.findContours(
            binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE
        )

        result = img.copy()
        classifications = []

        for contour in contours:
            shape, props = self.classify(contour)
            if shape is None:
                continue

            classifications.append((shape, props))

            # ë¬´ê²Œì¤‘ì‹¬
            M = cv2.moments(contour)
            if M['m00'] != 0:
                cx = int(M['m10'] / M['m00'])
                cy = int(M['m01'] / M['m00'])

                # ìœ¤ê³½ì„  ê·¸ë¦¬ê¸°
                color = self._get_shape_color(shape)
                cv2.drawContours(result, [contour], 0, color, 2)

                # ë¼ë²¨
                cv2.putText(result, shape, (cx - 30, cy),
                            cv2.FONT_HERSHEY_SIMPLEX, 0.5, color, 2)

        cv2.imshow('Classified Shapes', result)
        cv2.waitKey(0)

        return classifications

            'Star': (0, 255, 255),      # ë…¸ë‘
        }
        return colors.get(shape, (128, 128, 128))

import cv2

def rank_shapes_by_area(image_path):
    """ë„í˜•ì„ ë©´ì ìˆœìœ¼ë¡œ ì •ë ¬"""
    img = cv2.imread(image_path)
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    _, binary = cv2.threshold(gray, 127, 255, cv2.THRESH_BINARY)

    contours, _ = cv2.findContours(
        binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE
    )

    # ë©´ì ê³¼ í•¨ê»˜ ì €ì¥
    contour_areas = []
    for contour in contours:
        area = cv2.contourArea(contour)
        if area > 100:

    # ë©´ì  ê¸°ì¤€ ì •ë ¬ (ë‚´ë¦¼ì°¨ìˆœ)
    contour_areas.sort(key=lambda x: x[1], reverse=True)

    result = img.copy()

        # ë¬´ê²Œì¤‘ì‹¬
        M = cv2.moments(contour)
        cx = int(M['m10'] / M['m00'])
        cy = int(M['m01'] / M['m00'])

        cv2.drawContours(result, [contour], 0, (0, 255, 0), 2)
        cv2.putText(result, f'#{rank}', (cx - 15, cy),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 0, 255), 2)
        cv2.putText(result, f'{area:.0f}', (cx - 25, cy + 20),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.4, (255, 0, 0), 1)

    cv2.imshow('Ranked Shapes', result)
    cv2.waitKey(0)

def find_2to1_rectangles(image_path, tolerance=0.2):
    """2:1 ë¹„ìœ¨ ì‚¬ê°í˜• ì°¾ê¸°"""
    img = cv2.imread(image_path)
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    _, binary = cv2.threshold(gray, 127, 255, cv2.THRESH_BINARY)

    contours, _ = cv2.findContours(
        binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE
    )

    result = img.copy()
    found = []

    target_ratio = 2.0

    for contour in contours:
        if cv2.contourArea(contour) < 100:
            continue

        # ë‹¤ê°í˜• ê·¼ì‚¬í™”
        peri = cv2.arcLength(contour, True)
        approx = cv2.approxPolyDP(contour, 0.04 * peri, True)

        if len(approx) != 4:
            continue

        # ê²½ê³„ ì‚¬ê°í˜•ìœ¼ë¡œ ë¹„ìœ¨ í™•ì¸
        x, y, w, h = cv2.boundingRect(contour)
        aspect_ratio = max(w, h) / min(w, h)

        # 2:1 ë¹„ìœ¨ í™•ì¸ (í—ˆìš© ì˜¤ì°¨ í¬í•¨)
        if abs(aspect_ratio - target_ratio) < tolerance:

            # ë¹„ìœ¨ í‘œì‹œ
            cv2.putText(result, f'{aspect_ratio:.2f}:1', (x, y - 10),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 0, 0), 1)

    print(f"2:1 ì‚¬ê°í˜• ìˆ˜: {len(found)}")
    cv2.imshow('2:1 Rectangles', result)
    cv2.waitKey(0)

import cv2
import numpy as np

def find_most_circular(image_path):
    """ê°€ì¥ ì›í˜•ì— ê°€ê¹Œìš´ ë„í˜• ì°¾ê¸°"""
    img = cv2.imread(image_path)
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    _, binary = cv2.threshold(gray, 127, 255, cv2.THRESH_BINARY)

    contours, _ = cv2.findContours(
        binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE
    )

    best_circularity = 0
    best_contour = None

    for contour in contours:
        area = cv2.contourArea(contour)
        perimeter = cv2.arcLength(contour, True)

        if area < 100 or perimeter == 0:
            continue

        circularity = 4 * np.pi * area / (perimeter ** 2)

        if circularity > best_circularity:
            best_circularity = circularity
            best_contour = contour

    result = img.copy()

    if best_contour is not None:
        # ëª¨ë“  ìœ¤ê³½ì„  (íšŒìƒ‰)
        cv2.drawContours(result, contours, -1, (128, 128, 128), 1)

        # ê°€ì¥ ì›í˜•ì¸ ê²ƒ (ë…¹ìƒ‰)
        cv2.drawContours(result, [best_contour], 0, (0, 255, 0), 3)

        # ì •ë³´ í‘œì‹œ
        M = cv2.moments(best_contour)
        cx = int(M['m10'] / M['m00'])
        cy = int(M['m01'] / M['m00'])
        cv2.putText(result, f'Circularity: {best_circularity:.3f}',
                    (cx - 50, cy), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 0, 255), 2)

    print(f"ìµœëŒ€ ì›í˜•ë„: {best_circularity:.4f}")
    cv2.imshow('Most Circular', result)
    cv2.waitKey(0)
