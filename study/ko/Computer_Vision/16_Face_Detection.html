import cv2

class HaarFaceEyeDetector:
    """Haar Cascade ê¸°ë°˜ ì–¼êµ´/ëˆˆ ê²€ì¶œê¸°"""

    def __init__(self):
        self.face_cascade = cv2.CascadeClassifier(
            cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
        )
        self.eye_cascade = cv2.CascadeClassifier(
            cv2.data.haarcascades + 'haarcascade_eye.xml'
        )
        self.eye_glasses_cascade = cv2.CascadeClassifier(
            cv2.data.haarcascades + 'haarcascade_eye_tree_eyeglasses.xml'
        )

    def detect(self, img, detect_eyes=True):
        """ì–¼êµ´ê³¼ ëˆˆ ê²€ì¶œ"""
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        gray = cv2.equalizeHist(gray)

        results = []

        # ì–¼êµ´ ê²€ì¶œ
        faces = self.face_cascade.detectMultiScale(
            gray, 1.1, 5, minSize=(60, 60)
        )

        for (x, y, w, h) in faces:
            face_data = {
                'face_rect': (x, y, w, h),
                'eyes': []
            }

            if detect_eyes:
                # ì–¼êµ´ ìƒë‹¨ 50%ì—ì„œ ëˆˆ ê²€ì¶œ
                roi_gray = gray[y:y+h//2, x:x+w]

                # ì¼ë°˜ ëˆˆ ê²€ì¶œ ì‹œë„
                eyes = self.eye_cascade.detectMultiScale(
                    roi_gray, 1.1, 3, minSize=(20, 20)
                )

                # ì•ˆê²½ ì°©ìš©ììš© ê²€ì¶œ ì‹œë„ (ì¼ë°˜ ê²€ì¶œ ì‹¤íŒ¨ ì‹œ)
                if len(eyes) < 2:
                    eyes = self.eye_glasses_cascade.detectMultiScale(
                        roi_gray, 1.1, 3, minSize=(20, 20)
                    )

                # ê°€ì¥ ê·¸ëŸ´ë“¯í•œ ë‘ ëˆˆ ì„ íƒ
                eyes = self._select_best_eyes(eyes, w)

                for (ex, ey, ew, eh) in eyes:
                    face_data['eyes'].append((x + ex, y + ey, ew, eh))

            results.append(face_data)

        return results

    def _select_best_eyes(self, eyes, face_width):
        """ê°€ì¥ ì¢‹ì€ ë‘ ëˆˆ ì„ íƒ"""
        if len(eyes) <= 2:
            return eyes

        # ëˆˆ í¬ê¸°ì™€ y ì¢Œí‘œë¡œ í•„í„°ë§
        eyes = sorted(eyes, key=lambda e: e[1])  # y ì¢Œí‘œë¡œ ì •ë ¬

        # ìƒìœ„ 4ê°œ í›„ë³´ì—ì„œ ì„ íƒ
        candidates = eyes[:4]

        # x ì¢Œí‘œ ê¸°ì¤€ ì™¼ìª½/ì˜¤ë¥¸ìª½ ë¶„ë¦¬
        mid_x = face_width // 2
        left_eyes = [e for e in candidates if e[0] + e[2]//2 < mid_x]
        right_eyes = [e for e in candidates if e[0] + e[2]//2 >= mid_x]

        result = []
        if left_eyes:
            result.append(left_eyes[0])
        if right_eyes:
            result.append(right_eyes[0])

        return result

    def draw_results(self, img, results):
        """ê²°ê³¼ ì‹œê°í™”"""
        output = img.copy()

        for face_data in results:
            x, y, w, h = face_data['face_rect']

            # ì–¼êµ´ ì‚¬ê°í˜•
            cv2.rectangle(output, (x, y), (x+w, y+h), (0, 255, 0), 2)

            # ëˆˆ
            for (ex, ey, ew, eh) in face_data['eyes']:
                center = (ex + ew//2, ey + eh//2)
                radius = (ew + eh) // 4
                cv2.circle(output, center, radius, (255, 0, 0), 2)

        return output

class FaceLandmarkAnalyzer:
    """ì–¼êµ´ ëœë“œë§ˆí¬ ë¶„ì„ê¸°"""

    # ëœë“œë§ˆí¬ ì¸ë±ìŠ¤ ì •ì˜
    JAWLINE = list(range(0, 17))
    LEFT_EYEBROW = list(range(17, 22))
    RIGHT_EYEBROW = list(range(22, 27))
    NOSE_BRIDGE = list(range(27, 31))
    NOSE_TIP = list(range(31, 36))
    LEFT_EYE = list(range(36, 42))
    RIGHT_EYE = list(range(42, 48))
    OUTER_LIP = list(range(48, 60))
    INNER_LIP = list(range(60, 68))

    def __init__(self, predictor_path):
        self.detector = dlib.get_frontal_face_detector()
        self.predictor = dlib.shape_predictor(predictor_path)

    def get_landmarks(self, img):
        """ëœë“œë§ˆí¬ ì¶”ì¶œ"""
        rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        faces = self.detector(rgb, 0)

        if len(faces) == 0:
            return None

        shape = self.predictor(rgb, faces[0])
        landmarks = np.array([[shape.part(i).x, shape.part(i).y]
                             for i in range(68)])
        return landmarks

    def eye_aspect_ratio(self, eye_points):
        """ëˆˆ ì¢…íš¡ë¹„ (EAR) ê³„ì‚° - ì¡¸ìŒ ê°ì§€ì— ì‚¬ìš©"""
        # ìˆ˜ì§ ê±°ë¦¬
        A = dist.euclidean(eye_points[1], eye_points[5])
        B = dist.euclidean(eye_points[2], eye_points[4])
        # ìˆ˜í‰ ê±°ë¦¬
        C = dist.euclidean(eye_points[0], eye_points[3])

        ear = (A + B) / (2.0 * C)
        return ear

    def mouth_aspect_ratio(self, mouth_points):
        """ì… ì¢…íš¡ë¹„ (MAR) ê³„ì‚° - í•˜í’ˆ ê°ì§€ì— ì‚¬ìš©"""
        # ìˆ˜ì§ ê±°ë¦¬
        A = dist.euclidean(mouth_points[2], mouth_points[10])  # 51, 59
        B = dist.euclidean(mouth_points[4], mouth_points[8])   # 53, 57
        # ìˆ˜í‰ ê±°ë¦¬
        C = dist.euclidean(mouth_points[0], mouth_points[6])   # 49, 55

        mar = (A + B) / (2.0 * C)
        return mar

    def get_face_angle(self, landmarks):
        """ì–¼êµ´ ê¸°ìš¸ê¸° ê°ë„ ê³„ì‚°"""
        # ì–‘ìª½ ëˆˆ ì¤‘ì‹¬ì  ì‚¬ìš©
        left_eye_center = landmarks[self.LEFT_EYE].mean(axis=0)
        right_eye_center = landmarks[self.RIGHT_EYE].mean(axis=0)

        # ê°ë„ ê³„ì‚°
        dY = right_eye_center[1] - left_eye_center[1]
        dX = right_eye_center[0] - left_eye_center[0]
        angle = np.degrees(np.arctan2(dY, dX))

        return angle

    def analyze_face(self, img):
        """ì–¼êµ´ ì¢…í•© ë¶„ì„"""
        landmarks = self.get_landmarks(img)
        if landmarks is None:
            return None

        # ëˆˆ ë¶„ì„
        left_eye = landmarks[self.LEFT_EYE]
        right_eye = landmarks[self.RIGHT_EYE]
        left_ear = self.eye_aspect_ratio(left_eye)
        right_ear = self.eye_aspect_ratio(right_eye)
        avg_ear = (left_ear + right_ear) / 2.0

        # ì… ë¶„ì„
        outer_lip = landmarks[self.OUTER_LIP]
        mar = self.mouth_aspect_ratio(outer_lip)

        # ì–¼êµ´ ê°ë„
        angle = self.get_face_angle(landmarks)

        return {
            'landmarks': landmarks,
            'eye_aspect_ratio': avg_ear,
            'mouth_aspect_ratio': mar,
            'face_angle': angle,
            'eyes_closed': avg_ear < 0.2,  # ì„ê³„ê°’ ê¸°ë°˜
            'mouth_open': mar > 0.5
        }

class LBPHFaceRecognizer:
    """LBPH ê¸°ë°˜ ì–¼êµ´ ì¸ì‹ê¸°"""

    def __init__(self):
        self.face_cascade = cv2.CascadeClassifier(
            cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
        )
        self.recognizer = cv2.face.LBPHFaceRecognizer_create(
            radius=1,        # LBP ë°˜ê²½
            neighbors=8,     # ì´ì›ƒ ìˆ˜
            grid_x=8,        # x ë°©í–¥ ì…€ ìˆ˜
            grid_y=8         # y ë°©í–¥ ì…€ ìˆ˜
        )
        self.label_names = {}

    def prepare_training_data(self, data_dir):
        """í•™ìŠµ ë°ì´í„° ì¤€ë¹„"""
        faces = []
        labels = []

        for label_id, person_name in enumerate(os.listdir(data_dir)):
            person_dir = os.path.join(data_dir, person_name)
            if not os.path.isdir(person_dir):
                continue

            self.label_names[label_id] = person_name

            for img_name in os.listdir(person_dir):
                img_path = os.path.join(person_dir, img_name)
                img = cv2.imread(img_path, cv2.IMREAD_GRAYSCALE)

                if img is None:
                    continue

                # ì–¼êµ´ ê²€ì¶œ
                detected_faces = self.face_cascade.detectMultiScale(
                    img, 1.1, 5, minSize=(50, 50)
                )

                for (x, y, w, h) in detected_faces:
                    face_roi = img[y:y+h, x:x+w]
                    # í¬ê¸° ì •ê·œí™”
                    face_roi = cv2.resize(face_roi, (100, 100))
                    faces.append(face_roi)
                    labels.append(label_id)

        return faces, labels

    def train(self, faces, labels):
        """ëª¨ë¸ í•™ìŠµ"""
        self.recognizer.train(faces, np.array(labels))
        print(f"í•™ìŠµ ì™„ë£Œ: {len(set(labels))}ëª…, {len(faces)}ê°œ ì´ë¯¸ì§€")

    def save_model(self, path):
        """ëª¨ë¸ ì €ì¥"""
        self.recognizer.save(path)
        # ë¼ë²¨ ì´ë¦„ë„ ì €ì¥
        np.save(path + '_labels.npy', self.label_names)

    def load_model(self, path):
        """ëª¨ë¸ ë¡œë“œ"""
        self.recognizer.read(path)
        self.label_names = np.load(path + '_labels.npy',
                                   allow_pickle=True).item()

    def predict(self, img):
        """ì–¼êµ´ ì¸ì‹"""
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY) if len(img.shape) == 3 else img

        faces = self.face_cascade.detectMultiScale(gray, 1.1, 5, minSize=(50, 50))

        results = []
        for (x, y, w, h) in faces:
            face_roi = gray[y:y+h, x:x+w]
            face_roi = cv2.resize(face_roi, (100, 100))

            label, confidence = self.recognizer.predict(face_roi)

            # confidenceê°€ ë‚®ì„ìˆ˜ë¡ ì¢‹ìŒ (LBPH)
            # ì¼ë°˜ì ìœ¼ë¡œ 50 ì´í•˜ë©´ ë§¤ìš° ì¢‹ì€ ë§¤ì¹­
            name = self.label_names.get(label, "Unknown")
            if confidence > 100:
                name = "Unknown"

            results.append({
                'rect': (x, y, w, h),
                'name': name,
                'confidence': confidence
            })

        return results

    def draw_results(self, img, results):
        """ê²°ê³¼ ì‹œê°í™”"""
        output = img.copy()

        for result in results:
            x, y, w, h = result['rect']

            label = f"{name} ({conf:.1f})"
            cv2.putText(output, label, (x, y-10),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.6, color, 2)

        return output

import cv2
import time

def realtime_face_detection():
    """ì‹¤ì‹œê°„ ì–¼êµ´ ê²€ì¶œ (Haar Cascade)"""

    face_cascade = cv2.CascadeClassifier(
        cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
    )

    cap = cv2.VideoCapture(0)
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

    fps_counter = 0
    fps = 0
    start_time = time.time()

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        # í”„ë ˆì„ ì¶•ì†Œ (ì†ë„ í–¥ìƒ)
        small_frame = cv2.resize(frame, None, fx=0.5, fy=0.5)
        gray = cv2.cvtColor(small_frame, cv2.COLOR_BGR2GRAY)

        faces = face_cascade.detectMultiScale(
            gray,
            scaleFactor=1.2,  # ë¹ ë¥¸ ê²€ì¶œì„ ìœ„í•´ ì¦ê°€
            minNeighbors=4,
            minSize=(30, 30)
        )

        # ì¢Œí‘œ ìŠ¤ì¼€ì¼ ë³µì›

            x, y, w, h = x*2, y*2, w*2, h*2

        # FPS ê³„ì‚°
        fps_counter += 1
        elapsed = time.time() - start_time
        if elapsed >= 1.0:
            fps = fps_counter / elapsed
            fps_counter = 0
            start_time = time.time()

        # FPS í‘œì‹œ
        cv2.putText(frame, f'FPS: {fps:.1f}', (10, 30),
                   cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2)
        cv2.putText(frame, f'Faces: {len(faces)}', (10, 60),
                   cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2)

        cv2.imshow('Face Detection', frame)

        if cv2.waitKey(1) & 0xFF == ord('q'):

        """ì‹¤ì‹œê°„ ì¸ì‹ ì‹¤í–‰"""
        cap = cv2.VideoCapture(camera_id)
        cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)

        face_names = []

                face_encodings = face_recognition.face_encodings(
                    rgb_small, face_locations

                face_names = []
                for encoding in face_encodings:
                    name = "Unknown"

                        matches = face_recognition.compare_faces(
                            self.known_encodings, encoding, tolerance=0.6
                        )

                            self.known_encodings, encoding
                        )

                        if len(distances) > 0:
                            best_idx = np.argmin(distances)
                            if matches[best_idx]:

                    face_names.append(name)

            # ê²°ê³¼ í‘œì‹œ (ì¢Œí‘œ ìŠ¤ì¼€ì¼ ë³µì›)

                right *= 4
                bottom *= 4
                left *= 4

                cv2.rectangle(frame, (left, top), (right, bottom), color, 2)

                # ì´ë¦„ ë°°ê²½

            frame_count += 1
            if frame_count % 30 == 0:
