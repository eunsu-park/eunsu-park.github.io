2.2 í‘œì¤€ì˜¤ì°¨ (Standard Error)¶

# í‘œì¤€ì˜¤ì°¨ = í‘œë³¸í†µê³„ëŸ‰ì˜ í‘œì¤€í¸ì°¨
# í‘œë³¸í‰ê· ì˜ í‘œì¤€ì˜¤ì°¨: SE(XÌ„) = Ïƒ/âˆšn (ë˜ëŠ” s/âˆšn)

def standard_error(sample):
    """í‘œë³¸í‰ê· ì˜ í‘œì¤€ì˜¤ì°¨ ê³„ì‚°"""
    n = len(sample)
    s = np.std(sample, ddof=1)  # í‘œë³¸í‘œì¤€í¸ì°¨
    se = s / np.sqrt(n)
    return se

# MSE = BiasÂ² + Variance

def mse_analysis(estimates, true_value):
    """MSE ë¶„í•´"""
    estimates = np.array(estimates)
    bias = np.mean(estimates) - true_value
    variance = np.var(estimates)
    mse = np.mean((estimates - true_value)**2)
    return bias, variance, mse

# ê°€ëŠ¥ë„ í•¨ìˆ˜: L(Î¸|x) = P(X=x|Î¸)
# MLE: L(Î¸)ë¥¼ ìµœëŒ€í™”í•˜ëŠ” Î¸ ì°¾ê¸°

# ì˜ˆì‹œ: ë² ë¥´ëˆ„ì´ ë¶„í¬ì˜ MLE
def bernoulli_likelihood(p, data):
    """ë² ë¥´ëˆ„ì´ ë¶„í¬ì˜ ê°€ëŠ¥ë„ í•¨ìˆ˜"""
    n = len(data)
    k = sum(data)  # ì„±ê³µ íšŸìˆ˜
    return (p ** k) * ((1 - p) ** (n - k))

def bernoulli_log_likelihood(p, data):
    """ë² ë¥´ëˆ„ì´ ë¶„í¬ì˜ ë¡œê·¸ ê°€ëŠ¥ë„ í•¨ìˆ˜"""
    n = len(data)
    k = sum(data)
    if p <= 0 or p >= 1:
        return -np.inf
    return k * np.log(p) + (n - k) * np.log(1 - p)

# ì •ê·œë¶„í¬ N(Î¼, ÏƒÂ²)ì˜ MLE
# MLE: Î¼Ì‚ = xÌ„, ÏƒÌ‚Â² = (1/n)Î£(x_i - xÌ„)Â²

def normal_neg_log_likelihood(params, data):
    """ì •ê·œë¶„í¬ì˜ ìŒì˜ ë¡œê·¸ ê°€ëŠ¥ë„ í•¨ìˆ˜"""
    mu, sigma = params
    if sigma <= 0:
        return np.inf
    n = len(data)
    ll = -n/2 * np.log(2*np.pi) - n*np.log(sigma) - np.sum((data - mu)**2) / (2*sigma**2)
    return -ll

# kì°¨ í‘œë³¸ì ë¥ : m_k = (1/n)Î£x_i^k
# kì°¨ ì¤‘ì‹¬ì ë¥ : m'_k = (1/n)Î£(x_i - xÌ„)^k

def sample_moments(data, k):
    """kì°¨ í‘œë³¸ì ë¥  ê³„ì‚°"""

def sample_central_moments(data, k):
    """kì°¨ í‘œë³¸ì¤‘ì‹¬ì ë¥  ê³„ì‚°"""
    return np.mean((data - data.mean()) ** k)

# ê°ë§ˆë¶„í¬ Gamma(Î±, Î²): E[X] = Î±/Î², Var(X) = Î±/Î²Â²
# ì ë¥ ì¶”ì •: Î±Ì‚ = xÌ„Â² / sÂ², Î²Ì‚ = xÌ„ / sÂ²

def gamma_mom_estimator(data):
    """ê°ë§ˆë¶„í¬ì˜ ì ë¥ ì¶”ì •"""
    x_bar = data.mean()
    s_squared = data.var(ddof=1)

    alpha_hat = x_bar ** 2 / s_squared
    beta_hat = x_bar / s_squared

    return alpha_hat, beta_hat

# ë² íƒ€ë¶„í¬ Beta(Î±, Î²): E[X] = Î±/(Î±+Î²), Var(X) = Î±Î²/((Î±+Î²)Â²(Î±+Î²+1))
# ì ë¥ ì¶”ì • ìœ ë„

def beta_mom_estimator(data):
    """ë² íƒ€ë¶„í¬ì˜ ì ë¥ ì¶”ì •"""
    x_bar = data.mean()
    s_squared = data.var(ddof=1)

    alpha_hat = x_bar * temp
    beta_hat = (1 - x_bar) * temp

    return alpha_hat, beta_hat

# ì •ê·œë¶„í¬ ì í•©ë„ ê²€ì •

def goodness_of_fit_tests(data, dist_name='norm'):
    """ì í•©ë„ ê²€ì • ìˆ˜í–‰"""
    # ë¶„í¬ í”¼íŒ…
    if dist_name == 'norm':
        params = stats.norm.fit(data)
        dist = stats.norm(*params)
    elif dist_name == 'expon':
        params = stats.expon.fit(data)
        dist = stats.expon(*params)

    # KS ê²€ì •
    ks_stat, ks_pvalue = stats.kstest(data, dist.cdf)

    # Shapiro-Wilk ê²€ì • (ì •ê·œì„±)
    if dist_name == 'norm':
        sw_stat, sw_pvalue = stats.shapiro(data)
    else:
