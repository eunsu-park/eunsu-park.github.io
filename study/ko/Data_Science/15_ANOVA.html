1.2 ANOVA ê³„ì‚°¶

def one_way_anova_manual(groups):
    """ì¼ì›ë°°ì¹˜ ANOVA ìˆ˜ë™ ê³„ì‚°"""
    # ì „ì²´ ë°ì´í„°
    all_data = np.concatenate(groups)
    n_total = len(all_data)
    k = len(groups)  # ê·¸ë£¹ ìˆ˜
    grand_mean = all_data.mean()

    # ê·¸ë£¹ ê°„ ë³€ë™ (SSB: Sum of Squares Between)
    SSB = sum(len(g) * (g.mean() - grand_mean)**2 for g in groups)

    # ê·¸ë£¹ ë‚´ ë³€ë™ (SSW: Sum of Squares Within)
    SSW = sum(sum((x - g.mean())**2 for x in g) for g in groups)

    # ì´ ë³€ë™ (SST: Sum of Squares Total)
    SST = sum((x - grand_mean)**2 for x in all_data)

    # ììœ ë„
    df_between = k - 1
    df_within = n_total - k
    df_total = n_total - 1

    # í‰ê· ì œê³± (Mean Squares)
    MSB = SSB / df_between
    MSW = SSW / df_within

    # F-í†µê³„ëŸ‰
    F_stat = MSB / MSW

    # p-value
    p_value = 1 - stats.f.cdf(F_stat, df_between, df_within)

    return {
        'SSB': SSB, 'SSW': SSW, 'SST': SST,
        'df_between': df_between, 'df_within': df_within,
        'MSB': MSB, 'MSW': MSW,
        'F_stat': F_stat, 'p_value': p_value
    }

# ê³„ì‚°
groups = [method_A, method_B, method_C]
result = one_way_anova_manual(groups)

print("ì¼ì›ë°°ì¹˜ ANOVA ê²°ê³¼ (ìˆ˜ë™ ê³„ì‚°):")
print("=" * 60)
print("ë¶„ì‚°ë¶„ì„í‘œ:")
print("-" * 60)
print(f"{'ë³€ë™ì›':<12} {'SS':<12} {'df':<8} {'MS':<12} {'F':<10} {'p-value':<10}")
print("-" * 60)
print(f"{'ì²˜ë¦¬(ê·¸ë£¹ê°„)':<12} {result['SSB']:<12.2f} {result['df_between']:<8} "
