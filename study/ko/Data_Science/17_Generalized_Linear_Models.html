def identity_link(mu):
    """í•­ë“± ë§í¬: Î· = Î¼"""
    return mu

def logit_link(mu):
    """ë¡œì§“ ë§í¬: Î· = log(Î¼/(1-Î¼))"""
    return np.log(mu / (1 - mu))

    """ë¡œê·¸ ë§í¬: Î· = log(Î¼)"""
    return np.log(mu)

    """ì—­ìˆ˜ ë§í¬: Î· = 1/Î¼"""

def check_overdispersion(model):
    """ê³¼ëŒ€ì‚°í¬ ê²€ì •"""
    pearson_chi2 = model.pearson_chi2
    df_resid = model.df_resid

    print(f"â†’ {dispersion:.2f} > 1 ì´ë©´ ê³¼ëŒ€ì‚°í¬ ì˜ì‹¬")

    return dispersion
