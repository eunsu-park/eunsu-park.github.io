def simulate_ar1(phi, n=500, c=0, sigma=1):
    """AR(1) ì‹œë®¬ë ˆì´ì…˜: Y_t = c + Ï†*Y_{t-1} + Îµ_t"""
    y = np.zeros(n)
    y[0] = np.random.normal(0, sigma / np.sqrt(1 - phi**2))  # ì •ìƒ ì´ˆê¸°ê°’

    for t in range(1, n):
        y[t] = c + phi * y[t-1] + np.random.normal(0, sigma)

    return y

def simulate_ar2(phi1, phi2, n=500, c=0, sigma=1):

def check_ar2_stationarity(phi1, phi2):
    """AR(2) ì •ìƒì„± ì¡°ê±´ í™•ì¸"""
    # ì¡°ê±´: |Ï†â‚‚| < 1, Ï†â‚‚ + Ï†â‚ < 1, Ï†â‚‚ - Ï†â‚ < 1
    cond1 = abs(phi2) < 1
    cond2 = phi2 + phi1 < 1
    cond3 = phi2 - phi1 < 1

    print(f"AR(2) Ï†â‚={phi1}, Ï†â‚‚={phi2}")
    print(f"  |Ï†â‚‚| < 1: {cond1} ({abs(phi2):.2f})")
    print(f"  Ï†â‚‚ + Ï†â‚ < 1: {cond2} ({phi2 + phi1:.2f})")
    print(f"  Ï†â‚‚ - Ï†â‚ < 1: {cond3} ({phi2 - phi1:.2f})")
    print(f"  ì •ìƒì„±: {cond1 and cond2 and cond3}")
    return cond1 and cond2 and cond3

def simulate_ma(theta_list, n=500, c=0, sigma=1):
    """MA(q) ì‹œë®¬ë ˆì´ì…˜"""
    q = len(theta_list)
    epsilon = np.random.normal(0, sigma, n + q)
    y = np.zeros(n)

    for t in range(n):
        y[t] = c + epsilon[t + q]
        for i, theta in enumerate(theta_list):
            y[t] += theta * epsilon[t + q - i - 1]

    return y

# MA(1) ì‹œë®¬ë ˆì´ì…˜
theta_values = [0.8, 0.3, -0.3, -0.8]
fig, axes = plt.subplots(len(theta_values), 3, figsize=(15, 12))

for i, theta in enumerate(theta_values):
    y = simulate_ma([theta], n=500)

    axes[i, 0].plot(y[:200])
    axes[i, 0].set_title(f'MA(1), Î¸ = {theta}')
    axes[i, 0].grid(True, alpha=0.3)

    plot_acf(y, lags=20, ax=axes[i, 1], alpha=0.05)
    axes[i, 1].set_title(f'ACF (Î¸ = {theta})')

    plot_pacf(y, lags=20, ax=axes[i, 2], alpha=0.05, method='ywm')
    axes[i, 2].set_title(f'PACF (Î¸ = {theta})')

plt.tight_layout()
plt.show()

# MA(1)ì˜ ì´ë¡ ì  ACF

def compare_arima_models(y, max_p=3, max_d=2, max_q=3):
    """
    ê·¸ë¦¬ë“œ ì„œì¹˜ë¡œ ìµœì  ARIMA ëª¨ë¸ ì„ íƒ
    """
    results = []

    for p in range(max_p + 1):
        for d in range(max_d + 1):
            for q in range(max_q + 1):
                if p == 0 and q == 0:
                    continue  # ì˜ë¯¸ ì—†ëŠ” ëª¨ë¸ ì œì™¸
                try:
                    model = ARIMA(y, order=(p, d, q))
                    fit = model.fit()
                    results.append({
                        'order': (p, d, q),
                        'aic': fit.aic,
                        'bic': fit.bic,
                        'llf': fit.llf
                    })
                except:
                    pass

    # ê²°ê³¼ ì •ë ¬
    results_df = pd.DataFrame(results)
    results_df = results_df.sort_values('aic')

    return results_df

def diagnose_arima_model(result, title=''):
    """
    ARIMA ëª¨ë¸ ì§„ë‹¨
    """
    fig, axes = plt.subplots(2, 2, figsize=(12, 10))

    residuals = result.resid

    # ì”ì°¨ ì‹œê³„ì—´
    axes[0, 0].plot(residuals)
    axes[0, 0].axhline(0, color='r', linestyle='--')
    axes[0, 0].set_title('ì”ì°¨ ì‹œê³„ì—´')
    axes[0, 0].grid(True, alpha=0.3)

    # ì”ì°¨ íˆìŠ¤í† ê·¸ë¨
    axes[0, 1].hist(residuals, bins=30, density=True, alpha=0.7)
    x = np.linspace(residuals.min(), residuals.max(), 100)
    from scipy.stats import norm
    axes[0, 1].plot(x, norm.pdf(x, residuals.mean(), residuals.std()), 'r-', lw=2)
    axes[0, 1].set_title('ì”ì°¨ ë¶„í¬')
    axes[0, 1].grid(True, alpha=0.3)

    # ì”ì°¨ ACF
    plot_acf(residuals, lags=20, ax=axes[1, 0], alpha=0.05)
    axes[1, 0].set_title('ì”ì°¨ ACF')

    # Q-Q Plot

    stats.probplot(residuals, dist="norm", plot=axes[1, 1])
    axes[1, 1].set_title('Q-Q Plot')

    plt.tight_layout()
    plt.show()

    lb_result = acorr_ljungbox(residuals, lags=[10, 20, 30], return_df=True)

from sklearn.metrics import mean_squared_error, mean_absolute_error

def evaluate_forecast(y_train, y_test, order):
    """
    ì˜ˆì¸¡ ì„±ëŠ¥ í‰ê°€

    Parameters:
    -----------
    y_train : array
        í›ˆë ¨ ë°ì´í„°
    y_test : array
        í…ŒìŠ¤íŠ¸ ë°ì´í„°
    order : tuple
        ARIMA ì°¨ìˆ˜ (p, d, q)
    """
    # ëª¨ë¸ ì í•©
    model = ARIMA(y_train, order=order)
    result = model.fit()

    # ì˜ˆì¸¡
    forecast = result.get_forecast(steps=len(y_test))
    y_pred = forecast.predicted_mean

    # í‰ê°€ ì§€í‘œ
    mse = mean_squared_error(y_test, y_pred)
    rmse = np.sqrt(mse)
    mae = mean_absolute_error(y_test, y_pred)
    mape = np.mean(np.abs((y_test - y_pred) / y_test)) * 100

    print(f"=== ì˜ˆì¸¡ ì„±ëŠ¥ ({order}) ===")
    print(f"MSE:  {mse:.4f}")
    print(f"RMSE: {rmse:.4f}")
    print(f"MAE:  {mae:.4f}")
    print(f"MAPE: {mape:.2f}%")

    # ì‹œê°í™”
    fig, ax = plt.subplots(figsize=(12, 5))
    ax.plot(range(len(y_train)-50, len(y_train)), y_train[-50:], 'b-', label='í›ˆë ¨')
    ax.plot(range(len(y_train), len(y_train)+len(y_test)), y_test, 'g-', label='ì‹¤ì œ')
    ax.plot(range(len(y_train), len(y_train)+len(y_test)), y_pred, 'r--', label='ì˜ˆì¸¡')
    ax.axvline(len(y_train), color='k', linestyle=':', alpha=0.5)
    ax.legend()
    ax.set_title(f'ì˜ˆì¸¡ vs ì‹¤ì œ (ARIMA{order})')
    ax.grid(True, alpha=0.3)
    plt.show()

    return {'mse': mse, 'rmse': rmse, 'mae': mae, 'mape': mape}

def rolling_forecast(y, order, initial_train_size, horizon=1):
    """
    ë¡¤ë§ ìœˆë„ìš° ì˜ˆì¸¡

    Parameters:
    -----------
    y : array
        ì „ì²´ ì‹œê³„ì—´
    order : tuple
        ARIMA ì°¨ìˆ˜
    initial_train_size : int
        ì´ˆê¸° í›ˆë ¨ ë°ì´í„° í¬ê¸°
    horizon : int
        ì˜ˆì¸¡ í˜¸ë¼ì´ì¦Œ
    """
    predictions = []
    actuals = []

    for i in range(initial_train_size, len(y) - horizon + 1):
        train = y[:i]
        actual = y[i:i+horizon]

        model = ARIMA(train, order=order)
        result = model.fit()

        pred = result.get_forecast(steps=horizon).predicted_mean.values
        predictions.append(pred[0])
        actuals.append(actual[0])

    predictions = np.array(predictions)
    actuals = np.array(actuals)

    # í‰ê°€
    rmse = np.sqrt(mean_squared_error(actuals, predictions))
    mae = mean_absolute_error(actuals, predictions)

    print(f"=== ë¡¤ë§ ì˜ˆì¸¡ (ARIMA{order}) ===")
    print(f"RMSE: {rmse:.4f}")
    print(f"MAE:  {mae:.4f}")

    # ì‹œê°í™”
    fig, ax = plt.subplots(figsize=(14, 5))
    ax.plot(actuals, 'b-', label='ì‹¤ì œ', alpha=0.7)
    ax.plot(predictions, 'r--', label='ì˜ˆì¸¡', alpha=0.7)
    ax.set_xlabel('ì‹œê°„')
    ax.set_ylabel('ê°’')
    ax.set_title('ë¡¤ë§ 1-step ì˜ˆì¸¡')
    ax.legend()
    ax.grid(True, alpha=0.3)
    plt.show()

    return predictions, actuals

predictions, actuals = rolling_forecast(y, (1, 1, 1), initial_train_size=200)

def complete_arima_analysis(y, title=''):
    """
    ARIMA ëª¨ë¸ë§ ì „ì²´ íŒŒì´í”„ë¼ì¸
    """
    print(f"{'='*60}")
    print(f"ì‹œê³„ì—´ ë¶„ì„: {title}")
    print(f"{'='*60}")

    # 1. ì‹œê°í™” ë° ì •ìƒì„± ê²€ì •
    print("\n[1ë‹¨ê³„] ì •ìƒì„± ë¶„ì„")
    from statsmodels.tsa.stattools import adfuller, kpss

    adf_p = adfuller(y)[1]
    kpss_p = kpss(y, regression='c')[1]

    print(f"ADF p-value: {adf_p:.4f}")
    print(f"KPSS p-value: {kpss_p:.4f}")

    # ì°¨ë¶„ ê²°ì •
    d = 0
    y_diff = y.copy()
    while adf_p > 0.05 and d < 2:
        d += 1
        y_diff = np.diff(y_diff)
        adf_p = adfuller(y_diff)[1]
        print(f"{d}ì°¨ ì°¨ë¶„ í›„ ADF p-value: {adf_p:.4f}")

    print(f"ê²°ì •ëœ d: {d}")

    # 2. ACF/PACF ë¶„ì„
    print("\n[2ë‹¨ê³„] ACF/PACF ë¶„ì„")
    fig, axes = plt.subplots(1, 2, figsize=(12, 4))
    plot_acf(y_diff if d > 0 else y, lags=20, ax=axes[0], alpha=0.05)
    plot_pacf(y_diff if d > 0 else y, lags=20, ax=axes[1], alpha=0.05, method='ywm')
    plt.suptitle(f'ACF/PACF (d={d} ì°¨ë¶„ í›„)')
    plt.tight_layout()
    plt.show()

    # 3. ëª¨ë¸ ë¹„êµ
    print("\n[3ë‹¨ê³„] ëª¨ë¸ ë¹„êµ (AIC)")
    results = []
    for p in range(4):
        for q in range(4):
            if p == 0 and q == 0:
                continue
            try:
                model = ARIMA(y, order=(p, d, q))
                fit = model.fit()
                results.append({'p': p, 'd': d, 'q': q, 'aic': fit.aic, 'bic': fit.bic})
            except:
                pass

    results_df = pd.DataFrame(results).sort_values('aic')
    print(results_df.head(5))

    # ìµœì  ëª¨ë¸
    best = results_df.iloc[0]
    best_order = (int(best['p']), int(best['d']), int(best['q']))
    print(f"\nìµœì  ëª¨ë¸: ARIMA{best_order}")

    # 4. ìµœì¢… ëª¨ë¸ ì í•© ë° ì§„ë‹¨
    print(f"\n[4ë‹¨ê³„] ìµœì¢… ëª¨ë¸ ì í•©: ARIMA{best_order}")
    final_model = ARIMA(y, order=best_order)
    final_result = final_model.fit()
    print(final_result.summary())

    # ì§„ë‹¨
    diagnose_arima_model(final_result, f'ARIMA{best_order}')

    # 5. ì˜ˆì¸¡
    print("\n[5ë‹¨ê³„] ì˜ˆì¸¡")
    forecast = final_result.get_forecast(steps=20)

    fig, ax = plt.subplots(figsize=(14, 5))
    ax.plot(y[-100:], label='ê´€ì¸¡ê°’')
    forecast_index = range(len(y), len(y) + 20)
    ax.plot(forecast_index, forecast.predicted_mean, 'r-', label='ì˜ˆì¸¡')
    ci = forecast.conf_int()
    ax.fill_between(forecast_index, ci.iloc[:, 0], ci.iloc[:, 1],
                    color='r', alpha=0.2, label='95% CI')
    ax.legend()
    ax.set_title(f'ARIMA{best_order} ì˜ˆì¸¡')
    ax.grid(True, alpha=0.3)
    plt.show()

    return final_result
