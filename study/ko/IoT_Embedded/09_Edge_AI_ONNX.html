#!/usr/bin/env python3
"""PyTorch ëª¨ë¸ì„ ONNXë¡œ ë³€í™˜"""

import torch

# ì˜ˆì‹œ ëª¨ë¸
class SimpleNet(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc1 = nn.Linear(10, 50)
        self.fc2 = nn.Linear(50, 3)

    def forward(self, x):
        x = torch.relu(self.fc1(x))
        return self.fc2(x)

def export_to_onnx(model, output_path: str, input_shape: tuple):
    """PyTorch ëª¨ë¸ì„ ONNXë¡œ ë‚´ë³´ë‚´ê¸°"""
    model.eval()

    # ë”ë¯¸ ì…ë ¥
    dummy_input = torch.randn(*input_shape)

    # ONNX ë‚´ë³´ë‚´ê¸°
    torch.onnx.export(
        model,
        dummy_input,
        output_path,
        input_names=['input'],
        output_names=['output'],
        dynamic_axes={
            'input': {0: 'batch_size'},
            'output': {0: 'batch_size'}
        },
        opset_version=13
    )

    print(f"ONNX ëª¨ë¸ ì €ì¥: {output_path}")

#!/usr/bin/env python3
"""ONNX Runtime ê¸°ë³¸ ì¶”ë¡ """

import onnxruntime as ort
import numpy as np

class ONNXModel:
    """ONNX ëª¨ë¸ ë˜í¼"""

    def __init__(self, model_path: str, providers: list = None):
        if providers is None:
            providers = ['CPUExecutionProvider']

        # ì„¸ì…˜ ì˜µì…˜
        sess_options = ort.SessionOptions()
        sess_options.graph_optimization_level = ort.GraphOptimizationLevel.ORT_ENABLE_ALL
        sess_options.intra_op_num_threads = 4

        # ì„¸ì…˜ ìƒì„±
        self.session = ort.InferenceSession(
            model_path,
            sess_options=sess_options,
            providers=providers
        )

        # ì…ì¶œë ¥ ì •ë³´
        self.input_name = self.session.get_inputs()[0].name
        self.input_shape = self.session.get_inputs()[0].shape
        self.output_name = self.session.get_outputs()[0].name

    def get_input_shape(self):
        return self.input_shape

    def predict(self, input_data: np.ndarray) -> np.ndarray:
        """ì¶”ë¡  ìˆ˜í–‰"""
        outputs = self.session.run(
            [self.output_name],
            {self.input_name: input_data}
        )
        return outputs[0]

#!/usr/bin/env python3
"""ONNX Runtime ë°°ì¹˜ ì¶”ë¡ """

import onnxruntime as ort
import numpy as np
import time

def batch_inference(model_path: str, data: np.ndarray,
                    batch_size: int = 32) -> np.ndarray:
    """ë°°ì¹˜ ì¶”ë¡  ìˆ˜í–‰"""
    session = ort.InferenceSession(model_path, providers=['CPUExecutionProvider'])

    input_name = session.get_inputs()[0].name
    output_name = session.get_outputs()[0].name

    results = []
    num_samples = len(data)

    for i in range(0, num_samples, batch_size):
        batch = data[i:i + batch_size]
        output = session.run([output_name], {input_name: batch})[0]
        results.append(output)

    return np.concatenate(results, axis=0)

# ì„±ëŠ¥ ì¸¡ì •
def benchmark_batch_sizes(model_path: str, input_shape: tuple):
    """ë°°ì¹˜ í¬ê¸°ë³„ ì„±ëŠ¥ ë¹„êµ"""
    session = ort.InferenceSession(model_path, providers=['CPUExecutionProvider'])

    input_name = session.get_inputs()[0].name

    total_samples = 1000

    for batch_size in [1, 8, 16, 32, 64]:
        data = np.random.randn(total_samples, *input_shape[1:]).astype(np.float32)

        start = time.perf_counter()

        for i in range(0, total_samples, batch_size):
            batch = data[i:i + batch_size]
            _ = session.run(None, {input_name: batch})

        elapsed = time.perf_counter() - start
        throughput = total_samples / elapsed

        print(f"ë°°ì¹˜ í¬ê¸° {batch_size:2d}: {throughput:.1f} samples/sec")

        'handbag', 'tie', 'suitcase', 'frisbee', 'skis', 'snowboard',

        'surfboard', 'tennis racket', 'bottle', 'wine glass', 'cup', 'fork',

        'oven', 'toaster', 'sink', 'refrigerator', 'book', 'clock', 'vase',
        'scissors', 'teddy bear', 'hair drier', 'toothbrush'
    ]

    def __init__(self, model_path: str, conf_threshold: float = 0.5,
                 iou_threshold: float = 0.45):
        self.session = ort.InferenceSession(
            model_path,
            providers=['CPUExecutionProvider']
        )

        self.conf_threshold = conf_threshold
        self.iou_threshold = iou_threshold

        # ì…ë ¥ ì •ë³´
        input_info = self.session.get_inputs()[0]
        self.input_name = input_info.name
        self.input_shape = input_info.shape
        self.input_height = self.input_shape[2]
        self.input_width = self.input_shape[3]

    def preprocess(self, image: np.ndarray) -> tuple:
        """ì´ë¯¸ì§€ ì „ì²˜ë¦¬"""
        orig_height, orig_width = image.shape[:2]

        # ë¦¬ì‚¬ì´ì¦ˆ
        resized = cv2.resize(image, (self.input_width, self.input_height))

        # BGR to RGB, HWC to CHW
        input_data = resized[:, :, ::-1].transpose(2, 0, 1)

        # ì •ê·œí™” (0-1)
        input_data = input_data.astype(np.float32) / 255.0

        # ë°°ì¹˜ ì°¨ì› ì¶”ê°€
        input_data = np.expand_dims(input_data, axis=0)

        # ìŠ¤ì¼€ì¼ ë¹„ìœ¨ ì €ì¥
        scale = (orig_width / self.input_width, orig_height / self.input_height)

        return input_data, scale

    def postprocess(self, output: np.ndarray, scale: tuple) -> list:
        """ì¶œë ¥ í›„ì²˜ë¦¬"""
        predictions = output[0]

        boxes = []
        scores = []
        class_ids = []

        for pred in predictions:
            confidence = pred[4]

            if confidence > self.conf_threshold:
                class_probs = pred[5:]
                class_id = np.argmax(class_probs)
                class_score = class_probs[class_id]

                if class_score > self.conf_threshold:
                    # ë°•ìŠ¤ ì¢Œí‘œ (center_x, center_y, width, height)
                    cx, cy, w, h = pred[:4]

                    # ì›ë³¸ ìŠ¤ì¼€ì¼ë¡œ ë³€í™˜
                    x1 = int((cx - w / 2) * scale[0])
                    y1 = int((cy - h / 2) * scale[1])
                    x2 = int((cx + w / 2) * scale[0])
                    y2 = int((cy + h / 2) * scale[1])

                    boxes.append([x1, y1, x2, y2])
                    scores.append(float(confidence * class_score))
                    class_ids.append(int(class_id))

        # NMS (Non-Maximum Suppression)
        if boxes:
            indices = cv2.dnn.NMSBoxes(
                boxes, scores, self.conf_threshold, self.iou_threshold
            )

            results = []
            for i in indices:
                idx = i[0] if isinstance(i, (list, np.ndarray)) else i
                results.append({
                    'box': boxes[idx],
                    'score': scores[idx],
                    'class_id': class_ids[idx],
                    'class_name': self.CLASSES[class_ids[idx]]
                })

            return results

        return []

    def detect(self, image: np.ndarray) -> list:
        """ê°ì²´ ê²€ì¶œ"""
        input_data, scale = self.preprocess(image)

        outputs = self.session.run(None, {self.input_name: input_data})

        detections = self.postprocess(outputs[0], scale)

        return detections

    def draw_detections(self, image: np.ndarray, detections: list) -> np.ndarray:
        """ê²€ì¶œ ê²°ê³¼ ì‹œê°í™”"""
        result = image.copy()

        for det in detections:
            x1, y1, x2, y2 = det['box']
            label = f"{det['class_name']}: {det['score']:.2f}"

            # ë°•ìŠ¤
            cv2.rectangle(result, (x1, y1), (x2, y2), (0, 255, 0), 2)

            # ë¼ë²¨
            (w, h), _ = cv2.getTextSize(label, cv2.FONT_HERSHEY_SIMPLEX, 0.5, 1)
            cv2.rectangle(result, (x1, y1 - 20), (x1 + w, y1), (0, 255, 0), -1)
            cv2.putText(result, label, (x1, y1 - 5),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 0, 0), 1)

        return result

class RealtimeDetector:
    """ì‹¤ì‹œê°„ ê°ì²´ ê²€ì¶œê¸°"""

    def __init__(self, model_path: str):
        self.detector = YOLODetector(model_path)

        if HAS_CAMERA:
            self.camera = Picamera2()
            config = self.camera.create_preview_configuration(
                main={"size": (640, 480), "format": "RGB888"}
            )
            self.camera.configure(config)

    def run(self, duration: float = 60, display: bool = False):
        """ì‹¤ì‹œê°„ ê²€ì¶œ ì‹¤í–‰"""
        if not HAS_CAMERA:
            print("ì¹´ë©”ë¼ ì—†ìŒ")
            return

        self.camera.start()
        print(f"ì‹¤ì‹œê°„ ê²€ì¶œ ì‹œì‘ ({duration}ì´ˆ)")

        start_time = time.time()
        frame_count = 0
        fps_time = time.time()

        try:
            while time.time() - start_time < duration:
                # í”„ë ˆì„ ìº¡ì²˜
                frame = self.camera.capture_array()

                # BGR ë³€í™˜ (OpenCV í˜•ì‹)
                frame_bgr = cv2.cvtColor(frame, cv2.COLOR_RGB2BGR)

                # ê²€ì¶œ
                detections = self.detector.detect(frame_bgr)

                frame_count += 1

                # FPS ê³„ì‚°
                if frame_count % 10 == 0:
                    elapsed = time.time() - fps_time
                    fps = 10 / elapsed
                    fps_time = time.time()

                    print(f"\rFPS: {fps:.1f}, ê²€ì¶œ: {len(detections)}ê°œ", end="")

                    for det in detections:
                        print(f" | {det['class_name']}", end="")

                # ë””ìŠ¤í”Œë ˆì´ (ì„ íƒ)
                if display:
                    result = self.detector.draw_detections(frame_bgr, detections)
                    cv2.imshow("Detection", result)
                    if cv2.waitKey(1) & 0xFF == ord('q'):
                        break

        except KeyboardInterrupt:
            pass
        finally:
            self.camera.stop()
            if display:
                cv2.destroyAllWindows()

            total_time = time.time() - start_time
            avg_fps = frame_count / total_time
            print(f"\n\ní‰ê·  FPS: {avg_fps:.1f}")

if __name__ == "__main__":
    detector = RealtimeDetector("yolov5s.onnx")
    detector.run(duration=30, display=False)

#!/usr/bin/env python3
"""ê°ì²´ ê²€ì¶œ ê²°ê³¼ MQTT ë°œí–‰"""

import paho.mqtt.client as mqtt
import json
import time

class DetectionPublisher:
    """ê²€ì¶œ ê²°ê³¼ MQTT ë°œí–‰ê¸°"""

    def __init__(self, model_path: str, mqtt_broker: str = "localhost"):
        self.detector = YOLODetector(model_path)

        self.mqtt_client = mqtt.Client()
        self.mqtt_client.connect(mqtt_broker, 1883)
        self.mqtt_client.loop_start()

        self.node_id = "detector_01"

    def process_and_publish(self, image_path: str):
        """ì´ë¯¸ì§€ ì²˜ë¦¬ ë° ê²°ê³¼ ë°œí–‰"""
        import cv2

        image = cv2.imread(image_path)
        if image is None:
            print(f"ì´ë¯¸ì§€ ë¡œë“œ ì‹¤íŒ¨: {image_path}")
            return

        # ê²€ì¶œ
        start = time.perf_counter()
        detections = self.detector.detect(image)
        inference_time = (time.perf_counter() - start) * 1000

        # ê²°ê³¼ ìƒì„±
        result = {
            "node_id": self.node_id,
            "image": image_path,
            "detections": [
                {
                    "class": det['class_name'],
                    "score": round(det['score'], 3),
                    "box": det['box']
                }
                for det in detections
            ],
            "count": len(detections),
            "inference_time_ms": round(inference_time, 2),
            "timestamp": time.time()
        }

        # MQTT ë°œí–‰
        topic = f"edge/{self.node_id}/detection"
        self.mqtt_client.publish(topic, json.dumps(result))

        print(f"ë°œí–‰: {topic}")
        print(f"  ê²€ì¶œ: {len(detections)}ê°œ, ì‹œê°„: {inference_time:.1f}ms")

        return result

    def shutdown(self):
        self.mqtt_client.loop_stop()
        self.mqtt_client.disconnect()

if __name__ == "__main__":
    publisher = DetectionPublisher("yolov5s.onnx")

    try:
        publisher.process_and_publish("test_image.jpg")
    finally:
        publisher.shutdown()
