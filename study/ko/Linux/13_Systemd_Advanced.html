def get_systemd_socket():
    """systemdì—ì„œ ì „ë‹¬ëœ ì†Œì¼“ ê°€ì ¸ì˜¤ê¸°"""
    # LISTEN_FDS: ì „ë‹¬ëœ fd ê°œìˆ˜
    # LISTEN_PID: ëŒ€ìƒ í”„ë¡œì„¸ìŠ¤ PID
    listen_fds = int(os.environ.get('LISTEN_FDS', 0))
    listen_pid = int(os.environ.get('LISTEN_PID', 0))

        return None

    if listen_fds >= 1:
        # fd 3ë¶€í„° ì‹œì‘ (0=stdin, 1=stdout, 2=stderr)
        return socket.fromfd(3, socket.AF_INET, socket.SOCK_STREAM)

def main():
    # systemd ì†Œì¼“ ë˜ëŠ” ìƒˆ ì†Œì¼“ ìƒì„±
    sock = get_systemd_socket()

        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        sock.bind(('0.0.0.0', 8080))
        sock.listen(128)
        print("Listening on port 8080")
    else:
        print("Using systemd socket")

    while True:
        conn, addr = sock.accept()
        with conn:
            print(f"Connection from {addr}")
            conn.sendall(b"Hello from socket-activated service!\n")
