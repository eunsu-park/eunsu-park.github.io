"""
custom_handler.py - TorchServe ì»¤ìŠ¤í…€ í•¸ë“¤ëŸ¬
"""

import torch
import torch.nn.functional as F
from ts.torch_handler.base_handler import BaseHandler
import json
import logging

logger = logging.getLogger(__name__)

class CustomHandler(BaseHandler):
    """ì»¤ìŠ¤í…€ TorchServe í•¸ë“¤ëŸ¬"""

    def __init__(self):
        super().__init__()
        self.initialized = False

    def initialize(self, context):
        """ëª¨ë¸ ì´ˆê¸°í™”"""
        self.manifest = context.manifest
        properties = context.system_properties
        model_dir = properties.get("model_dir")

        # ë””ë°”ì´ìŠ¤ ì„¤ì •
        self.device = torch.device(
            "cuda" if torch.cuda.is_available() else "cpu"
        )

        # ëª¨ë¸ ë¡œë“œ
        serialized_file = self.manifest["model"]["serializedFile"]
        model_pt_path = f"{model_dir}/{serialized_file}"

        self.model = torch.jit.load(model_pt_path, map_location=self.device)
        self.model.eval()

        # ì¶”ê°€ ì„¤ì • ë¡œë“œ (ìˆëŠ” ê²½ìš°)
        self.class_names = self._load_class_names(model_dir)

        self.initialized = True
        logger.info("Model initialized successfully")

    def _load_class_names(self, model_dir):
        """í´ë˜ìŠ¤ ì´ë¦„ ë¡œë“œ"""
        try:
            with open(f"{model_dir}/index_to_name.json") as f:
                return json.load(f)
        except FileNotFoundError:
            return None

    def preprocess(self, data):
        """ì…ë ¥ ì „ì²˜ë¦¬"""
        inputs = []

        for row in data:
            # JSON ì…ë ¥ ì²˜ë¦¬
            if isinstance(row, dict):
                features = row.get("data") or row.get("body")
            else:
                features = row.get("body")

            if isinstance(features, (bytes, bytearray)):
                features = json.loads(features.decode("utf-8"))

            tensor = torch.tensor(features, dtype=torch.float32)
            inputs.append(tensor)

        # ë°°ì¹˜ë¡œ ë¬¶ê¸°
        return torch.stack(inputs).to(self.device)

    def inference(self, data):
        """ì¶”ë¡  ìˆ˜í–‰"""
        with torch.no_grad():
            outputs = self.model(data)
            probabilities = F.softmax(outputs, dim=1)
        return probabilities

    def postprocess(self, data):
        """ì¶œë ¥ í›„ì²˜ë¦¬"""
        results = []

        for prob in data:
            prob_list = prob.cpu().numpy().tolist()
            prediction = int(torch.argmax(prob).item())

            result = {
                "prediction": prediction,
                "probabilities": prob_list
            }

            # í´ë˜ìŠ¤ ì´ë¦„ ì¶”ê°€
            if self.class_names:
                result["class_name"] = self.class_names.get(str(prediction))

            results.append(result)

        return results

"""
image_classifier_handler.py - ì´ë¯¸ì§€ ë¶„ë¥˜ í•¸ë“¤ëŸ¬
"""

import torch
import torch.nn.functional as F
from torchvision import transforms
from ts.torch_handler.vision_handler import VisionHandler
from PIL import Image
import io
import base64

class ImageClassifierHandler(VisionHandler):
    """ì´ë¯¸ì§€ ë¶„ë¥˜ í•¸ë“¤ëŸ¬"""

    def __init__(self):
        super().__init__()
        self.transform = transforms.Compose([
            transforms.Resize(256),
            transforms.CenterCrop(224),
            transforms.ToTensor(),
            transforms.Normalize(
                mean=[0.485, 0.456, 0.406],
                std=[0.229, 0.224, 0.225]
            )
        ])

    def preprocess(self, data):
        """ì´ë¯¸ì§€ ì „ì²˜ë¦¬"""
        images = []

        for row in data:
            image_data = row.get("data") or row.get("body")

            # Base64 ë””ì½”ë”©
            if isinstance(image_data, str):
                image_data = base64.b64decode(image_data)

            # ì´ë¯¸ì§€ ë¡œë“œ
            image = Image.open(io.BytesIO(image_data)).convert("RGB")
            image = self.transform(image)
            images.append(image)

        return torch.stack(images).to(self.device)

    def postprocess(self, data):
        """ê²°ê³¼ í›„ì²˜ë¦¬"""
        probabilities = F.softmax(data, dim=1)
        top_k = torch.topk(probabilities, 5)

        results = []
        for probs, indices in zip(top_k.values, top_k.indices):
            result = {
                "predictions": [
                    {
                        "class_id": int(idx),
                        "class_name": self.mapping.get(str(int(idx)), "unknown"),
                        "probability": float(prob)
                    }
                    for prob, idx in zip(probs, indices)
                ]
            }
            results.append(result)

        return results

# HTTP í´ë¼ì´ì–¸íŠ¸
def triton_http_inference(model_name: str, input_data: np.ndarray):
    """HTTPë¥¼ í†µí•œ ì¶”ë¡ """
    client = httpclient.InferenceServerClient(url="localhost:8000")

    # ì…ë ¥ ì„¤ì •
    inputs = [
        httpclient.InferInput("input", input_data.shape, "FP32")
    ]
    inputs[0].set_data_from_numpy(input_data)

    # ì¶œë ¥ ì„¤ì •
    outputs = [
        httpclient.InferRequestedOutput("output")
    ]

    # ì¶”ë¡ 
    response = client.infer(
        model_name=model_name,
        inputs=inputs,
        outputs=outputs
    )

    return response.as_numpy("output")

# gRPC í´ë¼ì´ì–¸íŠ¸ (ë” ë¹ ë¦„)

    inputs = [
        grpcclient.InferInput("input", input_data.shape, "FP32")

        model_name=model_name,

    )

    return response.as_numpy("output")

def build_engine(onnx_path: str, engine_path: str):
    """ONNXì—ì„œ TensorRT ì—”ì§„ ë¹Œë“œ"""
    logger = trt.Logger(trt.Logger.WARNING)
    builder = trt.Builder(logger)
    network = builder.create_network(
        1 << int(trt.NetworkDefinitionCreationFlag.EXPLICIT_BATCH)
    )
    parser = trt.OnnxParser(network, logger)

    # ONNX íŒŒì‹±
    with open(onnx_path, "rb") as f:
        if not parser.parse(f.read()):
            for error in range(parser.num_errors):
                print(parser.get_error(error))
            return None

    # ë¹Œë” ì„¤ì •
    config = builder.create_builder_config()
    config.set_memory_pool_limit(trt.MemoryPoolType.WORKSPACE, 1 << 30)  # 1GB

    # FP16 ìµœì í™” (ì§€ì›ë˜ëŠ” ê²½ìš°)
    if builder.platform_has_fast_fp16:
        config.set_flag(trt.BuilderFlag.FP16)

    # ë™ì  ë°°ì¹˜ ì„¤ì •
    profile = builder.create_optimization_profile()
    profile.set_shape("input", (1, 4), (16, 4), (64, 4))  # min, opt, max
    config.add_optimization_profile(profile)

    # ì—”ì§„ ë¹Œë“œ
    engine = builder.build_serialized_network(network, config)

    # ì €ì¥
    with open(engine_path, "wb") as f:
        f.write(engine)

    return engine

import random
import requests

class ABTestRouter:
    """A/B í…ŒìŠ¤íŠ¸ ë¼ìš°í„°"""

    def __init__(self, model_a: str, model_b: str, traffic_split: float = 0.5):
        self.model_a = model_a
        self.model_b = model_b
        self.traffic_split = traffic_split
        self.base_url = "http://localhost:8080/predictions"

    def predict(self, data: dict) -> dict:
        """A/B ë¶„ë°°ëœ ì˜ˆì¸¡"""
        # íŠ¸ë˜í”½ ë¶„ë°°
        if random.random() < self.traffic_split:
            model = self.model_a
            variant = "A"
        else:
            model = self.model_b
            variant = "B"

        # ì˜ˆì¸¡
        response = requests.post(
            f"{self.base_url}/{model}",
            json=data
        )

        result = response.json()
        result["variant"] = variant
        result["model"] = model

        return result
