5. ê³„ì‚° ë¹„ìš© ì¦ê°€

import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import load_digits, load_iris, fetch_olivetti_faces
from sklearn.preprocessing import StandardScaler

# ì°¨ì›ì˜ ì €ì£¼ ë°ëª¨: ê³ ì°¨ì›ì—ì„œ ê±°ë¦¬ ë¶„í¬
np.random.seed(42)

def distance_distribution(n_dims, n_points=1000):
    """ê³ ì°¨ì›ì—ì„œ ê±°ë¦¬ ë¶„í¬ í™•ì¸"""
    points = np.random.rand(n_points, n_dims)
    # ëœë¤ í¬ì¸íŠ¸ ìŒ ê°„ ê±°ë¦¬
    idx = np.random.choice(n_points, size=(500, 2), replace=False)
    distances = [np.linalg.norm(points[i] - points[j]) for i, j in idx]
    return distances

# ë‹¤ì–‘í•œ ì°¨ì›ì—ì„œ ê±°ë¦¬ ë¶„í¬
dims = [2, 10, 100, 1000]

fig, axes = plt.subplots(1, 4, figsize=(16, 4))

    distances = distance_distribution(d)
    ax.hist(distances, bins=30, edgecolor='black')
    ax.set_title(f'Dim={d}\nMean={np.mean(distances):.2f}, Std={np.std(distances):.2f}')
    ax.set_xlabel('Distance')

plt.tight_layout()
plt.show()

print("ì°¨ì›ì´ ì¦ê°€í• ìˆ˜ë¡ ê±°ë¦¬ ë¶„í¬ê°€ ì¢ì•„ì§ â†’ í¬ì¸íŠ¸ë“¤ì´ ë¹„ìŠ·í•œ ê±°ë¦¬ì— ìœ„ì¹˜")

"""
PCA (Principal Component Analysis):
- ë°ì´í„°ì˜ ë¶„ì‚°ì„ ìµœëŒ€í™”í•˜ëŠ” ì¶•(ì£¼ì„±ë¶„)ì„ ì°¾ìŒ
- ê³ ì°¨ì› â†’ ì €ì°¨ì› íˆ¬ì˜
- ì„ í˜• ë³€í™˜

ìˆ˜í•™ì  ì›ë¦¬:
1. ë°ì´í„° ì¤‘ì‹¬í™” (í‰ê·  0)
2. ê³µë¶„ì‚° í–‰ë ¬ ê³„ì‚°
3. ê³ ìœ ê°’ ë¶„í•´ (eigendecomposition)
4. ê³ ìœ ê°’ì´ í° ìˆœì„œë¡œ ê³ ìœ ë²¡í„°(ì£¼ì„±ë¶„) ì„ íƒ
5. ì„ íƒëœ ì£¼ì„±ë¶„ìœ¼ë¡œ ë°ì´í„° íˆ¬ì˜

- ì²« ë²ˆì§¸ ì£¼ì„±ë¶„: ë¶„ì‚°ì´ ê°€ì¥ í° ë°©í–¥
- ë‘ ë²ˆì§¸ ì£¼ì„±ë¶„: ì²« ë²ˆì§¸ì™€ ì§êµí•˜ë©´ì„œ ë¶„ì‚°ì´ í° ë°©í–¥
- në²ˆì§¸ ì£¼ì„±ë¶„: ì´ì „ ì£¼ì„±ë¶„ë“¤ê³¼ ì§êµ
"""

from sklearn.decomposition import PCA

# 2D ì˜ˆì‹œë¡œ PCA ì‹œê°í™”
np.random.seed(42)
X_2d = np.dot(np.random.randn(200, 2), [[2, 1], [1, 2]])

# PCA ì ìš©
pca = PCA(n_components=2)
pca.fit(X_2d)

# ì‹œê°í™”
plt.figure(figsize=(10, 8))
plt.scatter(X_2d[:, 0], X_2d[:, 1], alpha=0.5)

# ì£¼ì„±ë¶„ ë°©í–¥ (í™”ì‚´í‘œ)
mean = pca.mean_
for i, (comp, var) in enumerate(zip(pca.components_, pca.explained_variance_)):
    end = mean + comp * np.sqrt(var) * 3
    plt.arrow(mean[0], mean[1], end[0]-mean[0], end[1]-mean[1],
              head_width=0.3, head_length=0.2, fc=f'C{i}', ec=f'C{i}',
              linewidth=2, label=f'PC{i+1} (Var: {var:.2f})')

plt.xlabel('X1')
plt.ylabel('X2')
plt.title('PCA: Principal Components')
plt.legend()
plt.axis('equal')
plt.grid(True, alpha=0.3)
plt.show()

print(f"ì£¼ì„±ë¶„:\n{pca.components_}")
print(f"ì„¤ëª…ëœ ë¶„ì‚°: {pca.explained_variance_}")
print(f"ì„¤ëª…ëœ ë¶„ì‚° ë¹„ìœ¨: {pca.explained_variance_ratio_}")

def remove_highly_correlated(df, threshold=0.9):
    corr_matrix = df.corr().abs()
    upper = corr_matrix.where(np.triu(np.ones(corr_matrix.shape), k=1).astype(bool))
    to_drop = [column for column in upper.columns if any(upper[column] > threshold)]
    return df.drop(columns=to_drop), to_drop

df_cleaned, dropped = remove_highly_correlated(df, threshold=0.9)
print(f"ì œê±°ëœ íŠ¹ì„±: {dropped}")
