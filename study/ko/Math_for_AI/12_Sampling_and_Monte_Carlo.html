# ì—­ë³€í™˜ ìƒ˜í”Œë§ ì˜ˆì œ: ì§€ìˆ˜ë¶„í¬

2.2 ê¸°ê° ìƒ˜í”Œë§ (Rejection Sampling)¶

ëª©í‘œ ë¶„í¬ $p(x)$ì—ì„œ ì§ì ‘ ìƒ˜í”Œë§í•˜ê¸° ì–´ë ¤ìš¸ ë•Œ, ì œì•ˆ ë¶„í¬(proposal) $q(x)$ë¥¼ ì‚¬ìš©í•©ë‹ˆë‹¤.

ì•Œê³ ë¦¬ì¦˜: 1. ìƒìˆ˜ $M$ì„ ì„ íƒí•˜ì—¬ $M q(x) \geq p(x)$ (ëª¨ë“  $x$ì— ëŒ€í•´) 2. $x \sim q(x)$ ìƒì„± 3. $u \sim \text{Uniform}(0, 1)$ ìƒì„± 4. $u \leq \frac{p(x)}{M q(x)}$ì´ë©´ $x$ ìˆ˜ë½, ì•„ë‹ˆë©´ ê±°ë¶€

ìˆ˜ë½ë¥ : $\frac{1}{M}$

# ê¸°ê° ìƒ˜í”Œë§ ì˜ˆì œ: ë² íƒ€ë¶„í¬ Beta(2, 5)
def target_pdf(x):
    """ëª©í‘œ ë¶„í¬: Beta(2, 5)"""
    if 0 <= x <= 1:
        return 30 * x * (1-x)**4  # ì •ê·œí™”ëœ Beta(2,5)
    return 0

def proposal_pdf(x):
    """ì œì•ˆ ë¶„í¬: Uniform(0, 1)"""
    return 1.0 if 0 <= x <= 1 else 0

# M ì°¾ê¸° (ìµœëŒ€ê°’)

def rejection_sampling(n_samples):
    """ê¸°ê° ìƒ˜í”Œë§ êµ¬í˜„"""
    samples = []
    n_rejected = 0

    while len(samples) < n_samples:
        x = np.random.uniform(0, 1)  # ì œì•ˆ ë¶„í¬ì—ì„œ ìƒ˜í”Œ
        u = np.random.uniform(0, 1)

            samples.append(x)
        else:
            n_rejected += 1

    acceptance_rate = n_samples / (n_samples + n_rejected)
    print(f"ìˆ˜ë½ë¥ : {acceptance_rate:.2%} (ì´ë¡ ê°’: {1/M:.2%})")
    return np.array(samples)

3. ì¤‘ìš”ë„ ìƒ˜í”Œë§ (Importance Sampling)¶

3.1 ê¸°ë³¸ ì›ë¦¬¶

ëª©í‘œ: $\mathbb{E}_{p(x)}[f(x)]$ ê³„ì‚°

ì¤‘ìš”ë„ ìƒ˜í”Œë§ í•­ë“±ì‹:

$$ \mathbb{E}_{p(x)}[f(x)] = \int f(x) p(x) dx = \int f(x) \frac{p(x)}{q(x)} q(x) dx = \mathbb{E}_{q(x)}\left[f(x) \frac{p(x)}{q(x)}\right] $$

$q(x)$ë¥¼ ì œì•ˆ ë¶„í¬, $w(x) = \frac{p(x)}{q(x)}$ë¥¼ ì¤‘ìš”ë„ ê°€ì¤‘ì¹˜ë¼ê³  í•©ë‹ˆë‹¤.

ëª¬í…Œì¹´ë¥¼ë¡œ ì¶”ì •: $$ \mathbb{E}_{p(x)}[f(x)] \approx \frac{1}{N} \sum_{i=1}^{N} f(x_i) w(x_i), \quad x_i \sim q(x) $$

3.2 ì œì•ˆ ë¶„í¬ ì„ íƒ¶

ì¢‹ì€ ì œì•ˆ ë¶„í¬: - $q(x)$ê°€ $|f(x)| p(x)$ì™€ ë¹„ë¡€í•˜ë©´ ë¶„ì‚° ìµœì†Œ - ì‹¤ì œë¡œëŠ” $p(x)$ì˜ ê¼¬ë¦¬ê°€ ë‘êº¼ìš´ ê³³ì„ ì»¤ë²„í•´ì•¼ í•¨

ë‚˜ìœ ì„ íƒ: $q(x)$ê°€ $p(x)$ë³´ë‹¤ ì–‡ì€ ê¼¬ë¦¬ â†’ ê°€ì¤‘ì¹˜ê°€ í­ë°œí•˜ì—¬ ë†’ì€ ë¶„ì‚°

3.3 ìê¸°ì •ê·œí™” ì¤‘ìš”ë„ ìƒ˜í”Œë§¶

$p(x)$ë¥¼ ì •ê·œí™” ìƒìˆ˜ê¹Œì§€ë§Œ ì•Œ ë•Œ ($p(x) = \tilde{p}(x)/Z$):

$$ \mathbb{E}_{p(x)}[f(x)] \approx \frac{\sum_{i=1}^{N} f(x_i) \tilde{w}(x_i)}{\sum_{i=1}^{N} \tilde{w}(x_i)}, \quad \tilde{w}(x_i) = \frac{\tilde{p}(x_i)}{q(x_i)} $$

# ì¤‘ìš”ë„ ìƒ˜í”Œë§ ì˜ˆì œ
def target_unnormalized(x):
    """ì •ê·œí™”ë˜ì§€ ì•Šì€ ëª©í‘œ ë¶„í¬: í˜¼í•© ê°€ìš°ìŠ¤"""
    return 0.3 * np.exp(-0.5 * ((x-2)/0.5)**2) + \
           0.7 * np.exp(-0.5 * ((x+2)/1.0)**2)

def proposal(x):
    """ì œì•ˆ ë¶„í¬: N(0, 3)"""
    return np.exp(-0.5 * (x/3)**2) / (3 * np.sqrt(2*np.pi))

def f(x):
    """ê³„ì‚°í•˜ë ¤ëŠ” í•¨ìˆ˜: x^2"""
    return x**2

# ì¼ë°˜ ëª¬í…Œì¹´ë¥¼ë¡œ (ëª©í‘œ ë¶„í¬ì—ì„œ ì§ì ‘ ìƒ˜í”Œë§ì´ ê°€ëŠ¥í•˜ë‹¤ê³  ê°€ì •)
def naive_monte_carlo(n_samples):
    # ê¸°ê° ìƒ˜í”Œë§ìœ¼ë¡œ ëª©í‘œ ë¶„í¬ì—ì„œ ìƒ˜í”Œ
    samples = []
    while len(samples) < n_samples:
        x = np.random.normal(0, 3)
        u = np.random.uniform(0, target_unnormalized(x))
        if u <= target_unnormalized(x):
            samples.append(x)
    samples = np.array(samples)
    return np.mean(f(samples))

# ì¤‘ìš”ë„ ìƒ˜í”Œë§
def importance_sampling(n_samples):
    # ì œì•ˆ ë¶„í¬ì—ì„œ ìƒ˜í”Œ
    samples = np.random.normal(0, 3, n_samples)
    weights = target_unnormalized(samples) / proposal(samples)
    weights = weights / np.sum(weights)  # ìê¸°ì •ê·œí™”
    return np.sum(f(samples) * weights * n_samples)

# ë¹„êµ
n_trials = 100
n_samples = 1000

naive_results = [naive_monte_carlo(n_samples) for _ in range(n_trials)]
is_results = [importance_sampling(n_samples) for _ in range(n_trials)]

print(f"ì¼ë°˜ MC: í‰ê·  = {np.mean(naive_results):.4f}, í‘œì¤€í¸ì°¨ = {np.std(naive_results):.4f}")
print(f"ì¤‘ìš”ë„ ìƒ˜í”Œë§: í‰ê·  = {np.mean(is_results):.4f}, í‘œì¤€í¸ì°¨ = {np.std(is_results):.4f}")
print(f"ë¶„ì‚° ê°ì†Œìœ¨: {np.var(naive_results) / np.var(is_results):.2f}x")

ì •ì±… ê²½ì‚¬ë²•(Policy Gradient)ì—ì„œ ì¤‘ìš”ë„ ìƒ˜í”Œë§ì´ í•µì‹¬ì…ë‹ˆë‹¤:

4. ë§ˆë¥´ì½”í”„ ì²´ì¸ ëª¬í…Œì¹´ë¥¼ë¡œ (MCMC)¶

4.1 ë§ˆë¥´ì½”í”„ ì²´ì¸ ê¸°ì´ˆ¶

ë§ˆë¥´ì½”í”„ ì²´ì¸: í™•ë¥ ì  ìƒíƒœ ì „ì´ $P(x_{t+1} | x_t)$

ì„¸ë¶€ ê· í˜• ì¡°ê±´(Detailed Balance): $$ \pi(x) P(x'|x) = \pi(x') P(x|x') $$

ì„¸ë¶€ ê· í˜•ì„ ë§Œì¡±í•˜ë©´ $\pi$ê°€ ì •ìƒ ë¶„í¬ì…ë‹ˆë‹¤.

# ë©”íŠ¸ë¡œí´ë¦¬ìŠ¤-í—¤ì´ìŠ¤íŒ…ìŠ¤ êµ¬í˜„
def target_distribution(x):
    """ëª©í‘œ ë¶„í¬: ì´ë´‰ ë¶„í¬ (bimodal)"""
    return np.exp(-0.5 * ((x-3)/0.8)**2) + np.exp(-0.5 * ((x+3)/0.8)**2)

def metropolis_hastings(n_samples, proposal_std=1.0):
    """MH ì•Œê³ ë¦¬ì¦˜ (ëŒ€ì¹­ ì œì•ˆ ë¶„í¬)"""
    samples = np.zeros(n_samples)
    x = 0.0  # ì´ˆê¸° ìƒíƒœ
    n_accepted = 0

    for i in range(n_samples):
        # ì œì•ˆ ìƒì„± (ëŒ€ì¹­ ê°€ìš°ìŠ¤)
        x_proposed = x + np.random.normal(0, proposal_std)

        # ìˆ˜ë½ í™•ë¥  ê³„ì‚° (ëŒ€ì¹­ ì œì•ˆì´ë¯€ë¡œ q í•­ ì†Œê±°)
        acceptance_prob = min(1, target_distribution(x_proposed) / target_distribution(x))

        # ìˆ˜ë½/ê±°ë¶€ ê²°ì •
        if np.random.uniform() < acceptance_prob:
            x = x_proposed
            n_accepted += 1

        samples[i] = x

    acceptance_rate = n_accepted / n_samples
    print(f"ìˆ˜ë½ë¥ : {acceptance_rate:.2%}")
    return samples

4.3 ê¹ìŠ¤ ìƒ˜í”Œë§ (Gibbs Sampling)¶

ë‹¤ë³€ëŸ‰ ë¶„í¬ $p(x_1, \ldots, x_d)$ì—ì„œ ì¡°ê±´ë¶€ ë¶„í¬ $p(x_i | x_{-i})$ë¥¼ ì•Œ ë•Œ ì‚¬ìš©í•©ë‹ˆë‹¤.

# ê¹ìŠ¤ ìƒ˜í”Œë§: ì´ë³€ëŸ‰ ê°€ìš°ìŠ¤
def gibbs_sampling_bivariate_gaussian(n_samples, rho=0.9):
    """ì´ë³€ëŸ‰ ê°€ìš°ìŠ¤ N([0,0], [[1,Ï],[Ï,1]])ì—ì„œ ìƒ˜í”Œë§"""
    samples = np.zeros((n_samples, 2))
    x, y = 0.0, 0.0  # ì´ˆê¸°ê°’

    for i in range(n_samples):
        # x | y ~ N(Ïy, 1-ÏÂ²)
        x = np.random.normal(rho * y, np.sqrt(1 - rho**2))
        # y | x ~ N(Ïx, 1-ÏÂ²)
        y = np.random.normal(rho * x, np.sqrt(1 - rho**2))
        samples[i] = [x, y]

    return samples

5.1 ë¬¸ì œ: í™•ë¥ ì  ë…¸ë“œì˜ ì—­ì „íŒŒ¶

VAEì—ì„œ ì¸ì½”ë”ëŠ” $q_\phi(z|x)$ë¥¼ ì¶œë ¥í•˜ê³ , $z \sim q_\phi(z|x)$ë¥¼ ìƒ˜í”Œë§í•©ë‹ˆë‹¤. ì†ì‹¤ í•¨ìˆ˜:

$$ \mathcal{L}(\phi) = \mathbb{E}_{z \sim q_\phi(z|x)}[f(z)] $$

ë¬¸ì œ: $\frac{\partial}{\partial \phi} \mathbb{E}_{z \sim q_\phi}[f(z)]$ë¥¼ ì–´ë–»ê²Œ ê³„ì‚°í• ê¹Œ?

5.2 ì¬ë§¤ê°œë³€ìˆ˜í™” í•´ë²•¶

ì•„ì´ë””ì–´: í™•ë¥ ì„±ì„ $\phi$ì™€ ë¬´ê´€í•œ ë…¸ì´ì¦ˆë¡œ ë¶„ë¦¬í•©ë‹ˆë‹¤.

ê°€ìš°ìŠ¤ ë¶„í¬: $z \sim \mathcal{N}(\mu_\phi, \sigma_\phi^2)$

ì¬ë§¤ê°œë³€ìˆ˜í™”: $z = \mu_\phi + \sigma_\phi \cdot \epsilon$, ì—¬ê¸°ì„œ $\epsilon \sim \mathcal{N}(0, 1)$

ì´ì œ: $$ \frac{\partial}{\partial \phi} \mathbb{E}_{\epsilon \sim \mathcal{N}(0,1)}[f(\mu_\phi + \sigma_\phi \epsilon)] = \mathbb{E}_{\epsilon}\left[\frac{\partial f(z)}{\partial z} \frac{\partial z}{\partial \phi}\right] $$

ê¸°ìš¸ê¸°ê°€ ìƒ˜í”Œë§ ë‚´ë¶€ë¡œ ë“¤ì–´ê°ˆ ìˆ˜ ìˆìŠµë‹ˆë‹¤!

import torch
import torch.nn as nn

# ì¬ë§¤ê°œë³€ìˆ˜í™” íŠ¸ë¦­ êµ¬í˜„
class VAEEncoder(nn.Module):
    def __init__(self, input_dim, latent_dim):
        super().__init__()
        self.fc = nn.Linear(input_dim, 128)
        self.fc_mu = nn.Linear(128, latent_dim)
        self.fc_logvar = nn.Linear(128, latent_dim)

    def forward(self, x):
        h = torch.relu(self.fc(x))
        mu = self.fc_mu(h)
        logvar = self.fc_logvar(h)

        eps = torch.randn_like(std)  # Îµ ~ N(0, 1)
        z = mu + std * eps           # z = Î¼ + ÏƒÎµ
        return z

5.4 VAEì˜ ELBO¶

VAE ì†ì‹¤ í•¨ìˆ˜ (ELBO):

$$ \mathcal{L} = \mathbb{E}_{q_\phi(z|x)}[\log p_\theta(x|z)] - D_{KL}(q_\phi(z|x) \| p(z)) $$

ì¬ë§¤ê°œë³€ìˆ˜í™” ì ìš©: - ì²« ë²ˆì§¸ í•­: ì¬ë§¤ê°œë³€ìˆ˜í™” íŠ¸ë¦­ìœ¼ë¡œ ê¸°ìš¸ê¸° ê³„ì‚° - ë‘ ë²ˆì§¸ í•­: ê°€ìš°ìŠ¤ ì‚¬ì „ë¶„í¬ ê°€ì • ì‹œ í•´ì„ì ìœ¼ë¡œ ê³„ì‚° ê°€ëŠ¥

def vae_loss(x, x_recon, mu, logvar):
    """VAE ì†ì‹¤ í•¨ìˆ˜ (ELBO)"""
    # ì¬êµ¬ì„± ì†ì‹¤ (log p(x|z))
    recon_loss = nn.functional.binary_cross_entropy(x_recon, x, reduction='sum')

    # KL ë°œì‚° (ê°€ìš°ìŠ¤ ê°€ì • í•˜ì— í•´ì„ì  ê³„ì‚°)
    # KL(N(Î¼,ÏƒÂ²) || N(0,1)) = -0.5 * Î£(1 + log(ÏƒÂ²) - Î¼Â² - ÏƒÂ²)
    kl_loss = -0.5 * torch.sum(1 + logvar - mu.pow(2) - logvar.exp())

    return recon_loss + kl_loss
