2.4 ì–´í…ì…˜ ë©”ì»¤ë‹ˆì¦˜¶

íŠ¸ëœìŠ¤í¬ë¨¸ì˜ í•µì‹¬ ì—°ì‚°ì„ einsumìœ¼ë¡œ í‘œí˜„í•˜ë©´ ë§¤ìš° ê°„ê²°í•©ë‹ˆë‹¤.

$$ \text{Attention}(Q, K, V) = \text{softmax}\left(\frac{QK^T}{\sqrt{d_k}}\right) V $$

def attention_einsum(Q, K, V):
    """
    Q: (batch, query_len, d_k)
    K: (batch, key_len, d_k)
    V: (batch, key_len, d_v)
    """
    d_k = Q.shape[-1]

    # QK^T: (batch, query_len, key_len)
    scores = np.einsum('bqd,bkd->bqk', Q, K) / np.sqrt(d_k)

    # Softmax
    attention_weights = np.exp(scores - scores.max(axis=-1, keepdims=True))
    attention_weights /= attention_weights.sum(axis=-1, keepdims=True)

    # Attention weights Ã— V: (batch, query_len, d_v)
    output = np.einsum('bqk,bkv->bqv', attention_weights, V)

    return output, attention_weights

print(f"Attention output shape: {output.shape}")

def clip_gradients(gradients, max_norm=1.0):
    """ê·¸ë˜ë””ì–¸íŠ¸ ë…¸ë¦„ í´ë¦¬í•‘"""
    total_norm = np.sqrt(sum(np.sum(g**2) for g in gradients))
    clip_coef = max_norm / (total_norm + 1e-6)
    if clip_coef < 1:
        for g in gradients:
            g *= clip_coef
    return gradients, total_norm

5.2 Xavier/Glorot ì´ˆê¸°í™”¶

ì„¤ì •: ì„ í˜• ë ˆì´ì–´ $y = Wx + b$, í™œì„±í™” í•¨ìˆ˜ ì—†ìŒ

ê°€ì •: - $x_i$ëŠ” í‰ê·  0, ë¶„ì‚° $\text{Var}(x)$ - $W_{ij}$ëŠ” ë…ë¦½, í‰ê·  0

ë¶„ì‚° ì „íŒŒ: $$ \text{Var}(y_i) = n_{\text{in}} \cdot \text{Var}(W) \cdot \text{Var}(x) $$

ìˆœë°©í–¥: $\text{Var}(y) = \text{Var}(x)$ ì›í•  ë•Œ $$ \text{Var}(W) = \frac{1}{n_{\text{in}}} $$

ì—­ë°©í–¥: ê·¸ë˜ë””ì–¸íŠ¸ ë¶„ì‚° ìœ ì§€ ì›í•  ë•Œ $$ \text{Var}(W) = \frac{1}{n_{\text{out}}} $$

Xavier ì´ˆê¸°í™”: ì ˆì¶©ì•ˆ $$ \text{Var}(W) = \frac{2}{n_{\text{in}} + n_{\text{out}}} $$

def xavier_uniform(n_in, n_out):
    """Xavier ê· ë“± ì´ˆê¸°í™”"""
    limit = np.sqrt(6 / (n_in + n_out))
    return np.random.uniform(-limit, limit, (n_in, n_out))

def xavier_normal(n_in, n_out):

    std = np.sqrt(2 / (n_in + n_out))

6.1 ë°°ì¹˜ ì •ê·œí™” (Batch Normalization)¶

ìˆ˜ì‹: $$ \hat{x} = \frac{x - \mathbb{E}[x]}{\sqrt{\text{Var}[x] + \epsilon}} $$

$$ y = \gamma \hat{x} + \beta $$

ì—¬ê¸°ì„œ $\gamma, \beta$ëŠ” í•™ìŠµ ê°€ëŠ¥í•œ íŒŒë¼ë¯¸í„°ì…ë‹ˆë‹¤.

class BatchNorm1d:
    def __init__(self, num_features, eps=1e-5, momentum=0.1):
        self.eps = eps
        self.momentum = momentum
        self.gamma = np.ones(num_features)
        self.beta = np.zeros(num_features)
        self.running_mean = np.zeros(num_features)
        self.running_var = np.ones(num_features)

    def forward(self, x, training=True):
        if training:
            # ë°°ì¹˜ í†µê³„ ê³„ì‚°
            batch_mean = x.mean(axis=0)
            batch_var = x.var(axis=0)

            # ëŸ¬ë‹ í†µê³„ ì—…ë°ì´íŠ¸ (ì§€ìˆ˜ ì´ë™ í‰ê· )
            self.running_mean = (1 - self.momentum) * self.running_mean + \
                                self.momentum * batch_mean
            self.running_var = (1 - self.momentum) * self.running_var + \
                               self.momentum * batch_var

            # ì •ê·œí™”
            x_norm = (x - batch_mean) / np.sqrt(batch_var + self.eps)
        else:
            # ì¶”ë¡  ì‹œ: ëŸ¬ë‹ í†µê³„ ì‚¬ìš©
            x_norm = (x - self.running_mean) / np.sqrt(self.running_var + self.eps)

        # ìŠ¤ì¼€ì¼/ì‹œí”„íŠ¸
        return self.gamma * x_norm + self.beta

def layer_norm(x, gamma, beta, eps=1e-5):
    """ë ˆì´ì–´ ì •ê·œí™”"""
    # x: (batch, features)
    mean = x.mean(axis=1, keepdims=True)
    var = x.var(axis=1, keepdims=True)
    x_norm = (x - mean) / np.sqrt(var + eps)
    return gamma * x_norm + beta
