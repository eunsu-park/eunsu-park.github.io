def cramer(A, b):
    """í¬ë˜ë¨¸ ë²•ì¹™ìœ¼ë¡œ Ax = bë¥¼ í’‰ë‹ˆë‹¤."""
    n = len(b)
    d = det(A)
    if abs(d) < 1e-12:

    for i in range(n):

        A_i[:, i] = b
        x[i] = det(A_i) / d
    return x
