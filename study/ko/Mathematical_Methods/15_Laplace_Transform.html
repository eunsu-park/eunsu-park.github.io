4.1 ë¶€ë¶„ë¶„ìˆ˜ ë¶„í•´ (Partial Fraction Decomposition)¶

ì—­ ë¼í”Œë¼ìŠ¤ ë³€í™˜ì˜ í•µì‹¬ ê¸°ë²•ì€ $F(s)$ë¥¼ ë¶€ë¶„ë¶„ìˆ˜ë¡œ ë¶„í•´í•œ í›„ ë³€í™˜í‘œë¥¼ ì—­ìœ¼ë¡œ ì ìš©í•˜ëŠ” ê²ƒì´ë‹¤.

$$\frac{P(s)}{(s-a_1)(s-a_2)\cdots(s-a_n)} = \frac{A_1}{s-a_1} + \frac{A_2}{s-a_2} + \cdots + \frac{A_n}{s-a_n}$$

$$\frac{P(s)}{(s-a)^n} = \frac{A_1}{s-a} + \frac{A_2}{(s-a)^2} + \cdots + \frac{A_n}{(s-a)^n}$$

ì´í›„ ì™„ì „ì œê³±ì‹ìœ¼ë¡œ ë³€í™˜í•˜ì—¬ $\sin$, $\cos$ ë³€í™˜ ìŒì„ ì ìš©í•œë‹¤.

import numpy as np
import matplotlib.pyplot as plt
from math import factorial

def stehfest_weights(N):
    """ìŠ¤í…Œí˜ìŠ¤íŠ¸ ì•Œê³ ë¦¬ì¦˜ì˜ ê°€ì¤‘ì¹˜ ê³„ì‚°"""
    V = np.zeros(N)
    for k in range(1, N + 1):
        s = 0
        for j in range(int((k + 1) / 2), min(k, N // 2) + 1):
            numer = j**(N // 2) * factorial(2 * j)
            denom = (factorial(N // 2 - j) *
                     factorial(j) *
                     factorial(j - 1) *
                     factorial(k - j) *
                     factorial(2 * j - k))
            s += numer / denom
        V[k - 1] = (-1)**(k + N // 2) * s
    return V

def numerical_inverse_laplace(F_func, t_values, N=12):
    """ìŠ¤í…Œí˜ìŠ¤íŠ¸ ì•Œê³ ë¦¬ì¦˜ì„ ì´ìš©í•œ ìˆ˜ì¹˜ì  ì—­ ë¼í”Œë¼ìŠ¤ ë³€í™˜

        t_values: ì—­ë³€í™˜ì„ êµ¬í•  ì‹œê°„ê°’ ë°°ì—´
        N: ìŠ¤í…Œí˜ìŠ¤íŠ¸ ì°¨ìˆ˜ (ì§ìˆ˜, ê¸°ë³¸ê°’ 12)
    """

    ln2 = np.log(2)
    result = np.zeros_like(t_values, dtype=float)

        if t <= 0:

            continue
        s_vals = np.arange(1, N + 1) * ln2 / t
        F_vals = np.array([F_func(sv) for sv in s_vals])
        result[i] = (ln2 / t) * np.sum(V * F_vals)

    return result
