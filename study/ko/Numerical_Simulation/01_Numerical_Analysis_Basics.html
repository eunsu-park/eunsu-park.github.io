def numerical_derivatives(f, x, h=1e-5):
    """ë‹¤ì–‘í•œ ìœ í•œì°¨ë¶„ ê³µì‹"""

    forward = (f(x + h) - f(x)) / h

    # ì¤‘ì‹¬ì°¨ë¶„ (central difference) - ë” ì •í™•í•¨

def trapezoidal(f, a, b, n):
    """ì‚¬ë‹¤ë¦¬ê¼´ ê³µì‹ìœ¼ë¡œ ì ë¶„"""
    h = (b - a) / n
    x = np.linspace(a, b, n + 1)
    y = f(x)

    # ì‚¬ë‹¤ë¦¬ê¼´ ê³µì‹
    integral = h * (0.5 * y[0] + np.sum(y[1:-1]) + 0.5 * y[-1])
    return integral

def simpson(f, a, b, n):
    """ì‹¬ìŠ¨ 1/3 ê³µì‹ (nì€ ì§ìˆ˜ì—¬ì•¼ í•¨)"""
    if n % 2 != 0:
        n += 1
