def convert_to_system():
    # 2ì°¨ ODE: y'' + 4y' + 3y = 0
    # ì´ˆê¸°ì¡°ê±´: y(0) = 1, y'(0) = 0

    # 1ì°¨ ì‹œìŠ¤í…œìœ¼ë¡œ í•´ì„
    # íŠ¹ì„± ë°©ì •ì‹: rÂ² + 4r + 3 = 0 â†’ r = -1, -3
    # í•´: y = Ae^(-t) + Be^(-3t)
    # ì´ˆê¸°ì¡°ê±´ ì ìš©: A + B = 1, -A - 3B = 0
    # A = 3/2, B = -1/2

    t = np.linspace(0, 5, 100)
    y = 1.5 * np.exp(-t) - 0.5 * np.exp(-3*t)
    y_prime = -1.5 * np.exp(-t) + 1.5 * np.exp(-3*t)

    fig, axes = plt.subplots(1, 2, figsize=(12, 4))

    axes[0].plot(t, y, label='y(t)')
    axes[0].plot(t, y_prime, label="y'(t)")
    axes[0].set_xlabel('t')
    axes[0].set_title('ì‹œê°„ ë„ë©”ì¸')
    axes[0].legend()
    axes[0].grid(True)

    # ìœ„ìƒ í‰ë©´
    axes[1].plot(y, y_prime)
    axes[1].set_xlabel('y')
    axes[1].set_ylabel("y'")
    axes[1].set_title('ìœ„ìƒ í‰ë©´')
    axes[1].grid(True)

    plt.tight_layout()
    plt.show()

def equilibrium_analysis():
    # dy/dt = y(1 - y) (ë¡œì§€ìŠ¤í‹± ì„±ì¥)
    # í‰í˜•ì : y* = 0 ë˜ëŠ” y* = 1

    y = np.linspace(-0.5, 1.5, 100)
    f = y * (1 - y)

    plt.figure(figsize=(10, 5))

    # dy/dt vs y
    plt.subplot(1, 2, 1)
    plt.plot(y, f)
    plt.axhline(y=0, color='k', linewidth=0.5)
    plt.scatter([0, 1], [0, 0], color='red', s=100, zorder=5)
    plt.xlabel('y')
    plt.ylabel('dy/dt')
    plt.title('dy/dt = y(1-y)')
    plt.grid(True)

    # f'(y) = 1 - 2y
    # f'(0) = 1 > 0: ë¶ˆì•ˆì •
    # f'(1) = -1 < 0: ì•ˆì •

    # ì‹œê°„ ì§„í™”

    t_span = (0, 10)
    t_eval = np.linspace(0, 10, 100)

    for y0 in [-0.1, 0.1, 0.5, 0.9, 1.1, 1.5]:
        sol = solve_ivp(lambda t, y: y*(1-y), t_span, [y0], t_eval=t_eval)
        plt.plot(sol.t, sol.y[0], label=f'yâ‚€={y0}')

    plt.axhline(y=1, color='r', linestyle='--', label='ì•ˆì • í‰í˜•')
    plt.axhline(y=0, color='b', linestyle='--', label='ë¶ˆì•ˆì • í‰í˜•')
    plt.xlabel('t')
    plt.ylabel('y')
    plt.legend(loc='right')
    plt.title('ë‹¤ì–‘í•œ ì´ˆê¸°ì¡°ê±´ì—ì„œì˜ í•´')

def phase_plane_2d():

    # ì‹œìŠ¤í…œ: x' = v, v' = -x

    axes[0].set_ylabel('v')

def free_fall():
    """ì¤‘ë ¥ í•˜ì˜ ììœ  ë‚™í•˜ (ê³µê¸° ì €í•­ í¬í•¨)"""
    from scipy.integrate import solve_ivp

    # íŒŒë¼ë¯¸í„°
    g = 9.8  # ì¤‘ë ¥ ê°€ì†ë„
    k = 0.1  # ê³µê¸° ì €í•­ ê³„ìˆ˜
    m = 1.0  # ì§ˆëŸ‰

    # ìš´ë™ ë°©ì •ì‹: m*dv/dt = mg - kvÂ²
    # dv/dt = g - (k/m)vÂ²

    def fall_with_drag(t, state):
        v = state[0]
        return [g - (k/m) * v * abs(v)]

    # ì¢…ë‹¨ ì†ë„: v_terminal = sqrt(mg/k)
    v_terminal = np.sqrt(m * g / k)

    t_span = (0, 20)
    t_eval = np.linspace(0, 20, 200)

    sol = solve_ivp(fall_with_drag, t_span, [0], t_eval=t_eval)

    plt.figure(figsize=(10, 5))
    plt.plot(sol.t, sol.y[0])
    plt.axhline(y=v_terminal, color='r', linestyle='--',
                label=f'ì¢…ë‹¨ ì†ë„: {v_terminal:.2f} m/s')
    plt.xlabel('ì‹œê°„ (s)')
    plt.ylabel('ì†ë„ (m/s)')
    plt.title('ê³µê¸° ì €í•­ì´ ìˆëŠ” ììœ  ë‚™í•˜')
    plt.legend()

def rc_circuit():
    """RC íšŒë¡œì˜ ê³¼ë„ ì‘ë‹µ"""
    from scipy.integrate import solve_ivp

    # íŒŒë¼ë¯¸í„°
    R = 1000  # ì €í•­ (Î©)
    C = 1e-6  # ì»¤íŒ¨ì‹œí„´ìŠ¤ (F)
    V_source = 5  # ì „ì› ì „ì•• (V)
    tau = R * C  # ì‹œì •ìˆ˜

    # ì¶©ì „: V_C' = (V_source - V_C) / (RC)
    def charging(t, V_C):
        return [(V_source - V_C[0]) / (R * C)]

    t_span = (0, 5 * tau)
    t_eval = np.linspace(0, 5*tau, 200)

    sol = solve_ivp(charging, t_span, [0], t_eval=t_eval)

    # í•´ì„ì  í•´
    t_analytical = np.linspace(0, 5*tau, 200)
    V_analytical = V_source * (1 - np.exp(-t_analytical / tau))

    plt.figure(figsize=(10, 5))
    plt.plot(sol.t * 1000, sol.y[0], 'b-', label='ìˆ˜ì¹˜ í•´')
    plt.plot(t_analytical * 1000, V_analytical, 'r--', label='í•´ì„ì  í•´')
    plt.axhline(y=V_source * 0.632, color='g', linestyle=':',
                label=f'Ï„ = {tau*1000:.3f} ms')
    plt.xlabel('ì‹œê°„ (ms)')
    plt.ylabel('ì»¤íŒ¨ì‹œí„° ì „ì•• (V)')
    plt.title('RC íšŒë¡œ ì¶©ì „')
    plt.legend()
    plt.grid(True)
    plt.show()
