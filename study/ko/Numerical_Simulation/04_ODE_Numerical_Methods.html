import numpy as np

def forward_euler(f, y0, t_span, n_steps):
    """
    ì „ì§„ ì˜¤ì¼ëŸ¬ ë°©ë²•

    dy/dt = f(t, y)
    y_{n+1} = y_n + h * f(t_n, y_n)
    """
    t = np.linspace(t_span[0], t_span[1], n_steps + 1)
    h = t[1] - t[0]

    y = np.zeros(n_steps + 1)
    y[0] = y0

    for i in range(n_steps):
        y[i+1] = y[i] + h * f(t[i], y[i])

    return t, y

def backward_euler(f, df_dy, y0, t_span, n_steps, tol=1e-10):
    """
    í›„ì§„ ì˜¤ì¼ëŸ¬ ë°©ë²• (ì•”ì‹œì )

    y_{n+1} = y_n + h * f(t_{n+1}, y_{n+1})

    ë‰´í„´-ë©ìŠ¨ìœ¼ë¡œ ì•”ì‹œì  ë°©ì •ì‹ í’€ì´
    """
    t = np.linspace(t_span[0], t_span[1], n_steps + 1)
    h = t[1] - t[0]

    y = np.zeros(n_steps + 1)
    y[0] = y0

    for i in range(n_steps):
        # ë‰´í„´-ë©ìŠ¨: g(y_{n+1}) = y_{n+1} - y_n - h*f(t_{n+1}, y_{n+1}) = 0
        y_guess = y[i]  # ì´ˆê¸° ì¶”ì¸¡

        for _ in range(100):  # ìµœëŒ€ ë°˜ë³µ íšŸìˆ˜
            g = y_guess - y[i] - h * f(t[i+1], y_guess)
            dg = 1 - h * df_dy(t[i+1], y_guess)

            y_new = y_guess - g / dg

            if abs(y_new - y_guess) < tol:
                break
            y_guess = y_new

        y[i+1] = y_new

    return t, y

def rk2_midpoint(f, y0, t_span, n_steps):
    """RK2 ì¤‘ì ë²•"""
    t = np.linspace(t_span[0], t_span[1], n_steps + 1)
    h = t[1] - t[0]

    y = np.zeros(n_steps + 1)
    y[0] = y0

    for i in range(n_steps):
        k1 = f(t[i], y[i])
        k2 = f(t[i] + h/2, y[i] + h/2 * k1)
        y[i+1] = y[i] + h * k2

    return t, y

def rk2_heun(f, y0, t_span, n_steps):
    """RK2 Heun ë°©ë²• (ìˆ˜ì • ì˜¤ì¼ëŸ¬)"""
    t = np.linspace(t_span[0], t_span[1], n_steps + 1)
    h = t[1] - t[0]

    y = np.zeros(n_steps + 1)
    y[0] = y0

    for i in range(n_steps):
        k1 = f(t[i], y[i])
        k2 = f(t[i] + h, y[i] + h * k1)
        y[i+1] = y[i] + h/2 * (k1 + k2)

    return t, y

def rk4(f, y0, t_span, n_steps):
    """
    ê³ ì „ì  4ì°¨ ë£½ê²Œ-ì¿ íƒ€ ë°©ë²•

    k1 = f(t_n, y_n)
    k2 = f(t_n + h/2, y_n + h*k1/2)
    k3 = f(t_n + h/2, y_n + h*k2/2)
    k4 = f(t_n + h, y_n + h*k3)

    y_{n+1} = y_n + h/6 * (k1 + 2*k2 + 2*k3 + k4)
    """
    t = np.linspace(t_span[0], t_span[1], n_steps + 1)
    h = t[1] - t[0]

    y = np.zeros(n_steps + 1)
    y[0] = y0

    for i in range(n_steps):
        k1 = f(t[i], y[i])
        k2 = f(t[i] + h/2, y[i] + h/2 * k1)
        k3 = f(t[i] + h/2, y[i] + h/2 * k2)
        k4 = f(t[i] + h, y[i] + h * k3)

        y[i+1] = y[i] + h/6 * (k1 + 2*k2 + 2*k3 + k4)

    return t, y

def rk4_system(f, y0, t_span, n_steps):
    """
    ì—°ë¦½ ODE ì‹œìŠ¤í…œì„ ìœ„í•œ RK4

    y: ë²¡í„°
    f: ë²¡í„° í•¨ìˆ˜
    """
    t = np.linspace(t_span[0], t_span[1], n_steps + 1)
    h = t[1] - t[0]

    y = np.zeros((n_steps + 1, len(y0)))
    y[0] = y0

    for i in range(n_steps):
        k1 = np.array(f(t[i], y[i]))
        k2 = np.array(f(t[i] + h/2, y[i] + h/2 * k1))
        k3 = np.array(f(t[i] + h/2, y[i] + h/2 * k2))
        k4 = np.array(f(t[i] + h, y[i] + h * k3))

        y[i+1] = y[i] + h/6 * (k1 + 2*k2 + 2*k3 + k4)

    return t, y

def rk45_step(f, t, y, h):
    """
    RK4-5 (Dormand-Prince) í•œ ìŠ¤í…

    4ì°¨ì™€ 5ì°¨ ê·¼ì‚¬ë¥¼ ë™ì‹œì— ê³„ì‚°í•˜ì—¬ ì˜¤ì°¨ ì¶”ì •
    """
    # Dormand-Prince ê³„ìˆ˜
    c = [0, 1/5, 3/10, 4/5, 8/9, 1, 1]
    a = [
        [],
        [1/5],
        [3/40, 9/40],
        [44/45, -56/15, 32/9],
        [19372/6561, -25360/2187, 64448/6561, -212/729],
        [9017/3168, -355/33, 46732/5247, 49/176, -5103/18656],
        [35/384, 0, 500/1113, 125/192, -2187/6784, 11/84]
    ]
    b5 = [35/384, 0, 500/1113, 125/192, -2187/6784, 11/84, 0]
    b4 = [5179/57600, 0, 7571/16695, 393/640, -92097/339200, 187/2100, 1/40]

    # k ê³„ì‚°
    k = [np.array(f(t, y))]
    for i in range(1, 7):
        yi = y.copy()
        for j in range(i):
            yi = yi + h * a[i][j] * k[j]
        k.append(np.array(f(t + c[i]*h, yi)))

    # 5ì°¨ ê·¼ì‚¬
    y5 = y.copy()
    for i in range(7):
        y5 = y5 + h * b5[i] * k[i]

    # 4ì°¨ ê·¼ì‚¬ (ì˜¤ì°¨ ì¶”ì •ìš©)
    y4 = y.copy()
    for i in range(7):
        y4 = y4 + h * b4[i] * k[i]

    # ì˜¤ì°¨ ì¶”ì •
    error = np.max(np.abs(y5 - y4))

    return y5, error

def adaptive_rk45(f, y0, t_span, tol=1e-6, h_init=0.1):
    """ì ì‘í˜• RK4-5 ì†”ë²„"""
    t = [t_span[0]]
    y = [np.array(y0)]
    h = h_init

    while t[-1] < t_span[1]:
        # ìŠ¤í… í¬ê¸°ê°€ ë²”ìœ„ë¥¼ ë²—ì–´ë‚˜ì§€ ì•Šë„ë¡
        if t[-1] + h > t_span[1]:
            h = t_span[1] - t[-1]

        # RK4-5 ìŠ¤í…
        y_new, error = rk45_step(f, t[-1], y[-1], h)

        # ì˜¤ì°¨ì— ë”°ë¥¸ ìŠ¤í… ì¡°ì ˆ
        if error < tol:
            t.append(t[-1] + h)
            y.append(y_new)

            # ìŠ¤í… í¬ê¸° ì¦ê°€ (ì•ˆì „ ê³„ìˆ˜ 0.9)
            if error > 0:
                h = min(h * 0.9 * (tol / error) ** 0.2, 2 * h)
        else:
            # ìŠ¤í… í¬ê¸° ê°ì†Œ
            h = max(h * 0.9 * (tol / error) ** 0.25, h / 4)

    return np.array(t), np.array(y)

def lorenz(t, state, sigma=10, rho=28, beta=8/3):

def projectile_motion():
    """í¬ë¬¼ì„  ìš´ë™ - ì§€ë©´ ì¶©ëŒ ê²€ì¶œ"""

    g = 9.8

    def projectile(t, state):
