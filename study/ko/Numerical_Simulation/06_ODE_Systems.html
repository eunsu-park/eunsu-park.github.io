import numpy as np
import matplotlib.pyplot as plt

"""
Lotka-Volterra ë°©ì •ì‹:

dx/dt = Î±x - Î²xy    (í”¼ì‹ì: í† ë¼)
dy/dt = Î´xy - Î³y    (í¬ì‹ì: ì—¬ìš°)

Î³: í¬ì‹ì ì‚¬ë§ë¥

def lotka_volterra():
    alpha = 1.0  # í† ë¼ ì„±ì¥ë¥ 
    beta = 0.1   # í¬ì‹ë¥ 
    gamma = 1.5  # ì—¬ìš° ì‚¬ë§ë¥ 
    delta = 0.075  # ì—¬ìš° ì„±ì¥ íš¨ìœ¨

    def lv(t, y):
        x, y_pred = y
        dx = alpha*x - beta*x*y_pred
        dy = delta*x*y_pred - gamma*y_pred
        return [dx, dy]

    # ì´ˆê¸° ì¡°ê±´
    y0 = [40, 9]  # í† ë¼ 40ë§ˆë¦¬, ì—¬ìš° 9ë§ˆë¦¬
    t_span = (0, 50)
    t_eval = np.linspace(0, 50, 1000)

    sol = solve_ivp(lv, t_span, y0, t_eval=t_eval, method='RK45')

    fig, axes = plt.subplots(1, 2, figsize=(14, 5))

    # ì‹œê°„ì— ë”°ë¥¸ ê°œì²´ìˆ˜
    axes[0].plot(sol.t, sol.y[0], 'b-', label='í† ë¼ (í”¼ì‹ì)')
    axes[0].plot(sol.t, sol.y[1], 'r-', label='ì—¬ìš° (í¬ì‹ì)')
    axes[0].set_xlabel('ì‹œê°„')
    axes[0].set_ylabel('ê°œì²´ìˆ˜')
    axes[0].set_title('Lotka-Volterra ê°œì²´ ë™ì—­í•™')
    axes[0].legend()
    axes[0].grid(True)

    # ìœ„ìƒ í‰ë©´
    axes[1].plot(sol.y[0], sol.y[1], 'g-')
    axes[1].scatter([y0[0]], [y0[1]], color='red', s=100, zorder=5, label='ì‹œì‘ì ')
    axes[1].scatter([gamma/delta], [alpha/beta], color='black', s=100,
                    marker='x', zorder=5, label='í‰í˜•ì ')
    axes[1].set_xlabel('í† ë¼')
    axes[1].set_ylabel('ì—¬ìš°')
    axes[1].set_title('ìœ„ìƒ ê³µê°„')
    axes[1].legend()

    plt.tight_layout()
    plt.show()

def lv_phase_portrait():
    """Lotka-Volterra ë²¡í„°ì¥ê³¼ ì—¬ëŸ¬ ì´ˆê¸°ì¡°ê±´"""
    alpha, beta, gamma, delta = 1.0, 0.1, 1.5, 0.075

    def lv(t, y):
        return [alpha*y[0] - beta*y[0]*y[1],
                delta*y[0]*y[1] - gamma*y[1]]

    # ë²¡í„°ì¥
    x_range = np.linspace(0.1, 80, 20)
    y_range = np.linspace(0.1, 30, 20)
    X, Y = np.meshgrid(x_range, y_range)

    U = alpha*X - beta*X*Y
    V = delta*X*Y - gamma*Y

    # ì •ê·œí™”
    N = np.sqrt(U**2 + V**2)
    U, V = U/N, V/N

    plt.figure(figsize=(12, 8))
    plt.quiver(X, Y, U, V, alpha=0.5, color='gray')

    # ë‹¤ì–‘í•œ ì´ˆê¸°ì¡°ê±´
    colors = plt.cm.viridis(np.linspace(0, 1, 5))
    for i, (x0, y0) in enumerate([(10, 5), (30, 5), (50, 10), (70, 15), (20, 20)]):
        sol = solve_ivp(lv, (0, 50), [x0, y0], max_step=0.1)
        plt.plot(sol.y[0], sol.y[1], color=colors[i], linewidth=1.5)
        plt.scatter([x0], [y0], color=colors[i], s=50)

    # í‰í˜•ì 
    plt.scatter([gamma/delta], [alpha/beta], color='red', s=150,
                marker='*', zorder=10, label='í‰í˜•ì ')

    plt.xlabel('í”¼ì‹ì (x)')
    plt.ylabel('í¬ì‹ì (y)')
    plt.title('Lotka-Volterra ìœ„ìƒ ì´ˆìƒí™”')
    plt.legend()
    plt.xlim(0, 80)
    plt.ylim(0, 30)
    plt.grid(True)
    plt.show()

def simple_pendulum():
    """
    ë‹¨ìˆœ ì§„ì: Î¸'' + (g/L)sin(Î¸) = 0

    ì‘ì€ ê°ë„ ê·¼ì‚¬: Î¸'' + (g/L)Î¸ = 0
    í•´: Î¸ = Î¸â‚€cos(Ï‰t), Ï‰ = âˆš(g/L)
    """
    g = 9.8  # ì¤‘ë ¥ ê°€ì†ë„
    L = 1.0  # ì§„ì ê¸¸ì´

    def pendulum(t, y):
        theta, omega = y
        return [omega, -(g/L) * np.sin(theta)]

    # ë‹¤ì–‘í•œ ì´ˆê¸° ê°ë„
    fig, axes = plt.subplots(2, 2, figsize=(14, 10))

    for ax, theta0 in zip(axes.flat, [0.1, 0.5, np.pi/2, np.pi - 0.1]):
        sol = solve_ivp(pendulum, (0, 10), [theta0, 0], max_step=0.01)

        ax.plot(sol.y[0], sol.y[1])
        ax.scatter([theta0], [0], color='red', s=100, label='ì‹œì‘')
        ax.set_xlabel('Î¸ (rad)')
        ax.set_ylabel('Ï‰ (rad/s)')
        ax.set_title(f'Î¸â‚€ = {theta0:.2f} rad ({np.degrees(theta0):.1f}Â°)')
        ax.legend()
        ax.grid(True)
        ax.set_aspect('equal', adjustable='box')

    plt.tight_layout()
    plt.show()

def driven_pendulum():
    """
    ê°ì‡  ê°•ì œ ì§„ì:
    Î¸'' + Î³Î¸' + (g/L)sin(Î¸) = A*cos(Ï‰t)

    ë¹„ì„ í˜• â†’ í˜¼ëˆ ê°€ëŠ¥ì„±
    """
    g, L = 9.8, 1.0
    gamma = 0.5  # ê°ì‡  ê³„ìˆ˜
    A = 1.5      # êµ¬ë™ë ¥ ì§„í­
    omega_d = 2/3 * np.sqrt(g/L)  # êµ¬ë™ ì£¼íŒŒìˆ˜

    def driven(t, y):
        theta, w = y
        return [w, -gamma*w - (g/L)*np.sin(theta) + A*np.cos(omega_d*t)]

    t_span = (0, 200)
    sol = solve_ivp(driven, t_span, [0.1, 0], max_step=0.01)

    fig, axes = plt.subplots(2, 2, figsize=(14, 10))

    # ì‹œê°„ ì‘ë‹µ
    axes[0, 0].plot(sol.t, sol.y[0])
    axes[0, 0].set_xlabel('ì‹œê°„')
    axes[0, 0].set_ylabel('Î¸')
    axes[0, 0].set_title('ì‹œê°„ ì‘ë‹µ')
    axes[0, 0].grid(True)

    # ìœ„ìƒ ê³µê°„
    axes[0, 1].plot(sol.y[0], sol.y[1], linewidth=0.5)
    axes[0, 1].set_xlabel('Î¸')
    axes[0, 1].set_ylabel('Ï‰')
    axes[0, 1].set_title('ìœ„ìƒ ê³µê°„')
    axes[0, 1].grid(True)

    # í¬ì•™ì¹´ë ˆ ë‹¨ë©´ (êµ¬ë™ë ¥ ì£¼ê¸°ë§ˆë‹¤ ìƒ˜í”Œë§)
    T_drive = 2 * np.pi / omega_d
    t_poincare = np.arange(0, t_span[1], T_drive)

    from scipy.interpolate import interp1d
    theta_interp = interp1d(sol.t, sol.y[0])
    omega_interp = interp1d(sol.t, sol.y[1])

    valid_t = t_poincare[(t_poincare >= sol.t[0]) & (t_poincare <= sol.t[-1])]
    theta_p = theta_interp(valid_t)
    omega_p = omega_interp(valid_t)

    axes[1, 0].scatter(theta_p[100:], omega_p[100:], s=1)  # ê³¼ë„ ìƒíƒœ ì œì™¸
    axes[1, 0].set_xlabel('Î¸')
    axes[1, 0].set_ylabel('Ï‰')
    axes[1, 0].set_title('í¬ì•™ì¹´ë ˆ ë‹¨ë©´')
    axes[1, 0].grid(True)

    # ì—ë„ˆì§€
    E = 0.5 * sol.y[1]**2 + (g/L) * (1 - np.cos(sol.y[0]))
    axes[1, 1].plot(sol.t, E)
    axes[1, 1].set_xlabel('ì‹œê°„')
    axes[1, 1].set_ylabel('ì—ë„ˆì§€')
    axes[1, 1].set_title('ì—­í•™ì  ì—ë„ˆì§€')
    axes[1, 1].grid(True)

    plt.tight_layout()
    plt.show()

def double_pendulum():
    """
    ì´ì¤‘ ì§„ì: í˜¼ëˆì  ì‹œìŠ¤í…œ

    Î¸â‚, Î¸â‚‚: ê° ì§„ìì˜ ê°ë„
    Ï‰â‚, Ï‰â‚‚: ê°ì†ë„
    """
    g = 9.8
    L1, L2 = 1.0, 1.0
    m1, m2 = 1.0, 1.0

    def double_pend(t, y):
        t1, t2, w1, w2 = y

        delta = t2 - t1
        denom1 = (m1 + m2) * L1 - m2 * L1 * np.cos(delta)**2
        denom2 = (L2 / L1) * denom1

        dw1 = (m2 * L1 * w1**2 * np.sin(delta) * np.cos(delta) +
               m2 * g * np.sin(t2) * np.cos(delta) +
               m2 * L2 * w2**2 * np.sin(delta) -
               (m1 + m2) * g * np.sin(t1)) / denom1

        dw2 = (-m2 * L2 * w2**2 * np.sin(delta) * np.cos(delta) +
               (m1 + m2) * g * np.sin(t1) * np.cos(delta) -
               (m1 + m2) * L1 * w1**2 * np.sin(delta) -
               (m1 + m2) * g * np.sin(t2)) / denom2

        return [w1, w2, dw1, dw2]

    # ì´ˆê¸° ì¡°ê±´ì— ë¯¼ê°
    fig, axes = plt.subplots(2, 2, figsize=(14, 10))

    initial_conditions = [
        [np.pi/2, np.pi/2, 0, 0],
        [np.pi/2 + 0.001, np.pi/2, 0, 0],  # ë¯¸ì„¸í•œ ì°¨ì´
    ]

    for ic, color in zip(initial_conditions, ['blue', 'red']):
        sol = solve_ivp(double_pend, (0, 20), ic, max_step=0.01)

        # ìœ„ì¹˜ ê³„ì‚°
        x1 = L1 * np.sin(sol.y[0])
        y1 = -L1 * np.cos(sol.y[0])
        x2 = x1 + L2 * np.sin(sol.y[1])
        y2 = y1 - L2 * np.cos(sol.y[1])

        axes[0, 0].plot(sol.t, sol.y[0], color=color, alpha=0.7)
        axes[0, 1].plot(sol.t, sol.y[1], color=color, alpha=0.7)
        axes[1, 0].plot(x2, y2, color=color, linewidth=0.5, alpha=0.7)

    axes[0, 0].set_xlabel('ì‹œê°„')
    axes[0, 0].set_ylabel('Î¸â‚')
    axes[0, 0].set_title('ì²« ë²ˆì§¸ ì§„ì')
    axes[0, 0].grid(True)

    axes[0, 1].set_xlabel('ì‹œê°„')
    axes[0, 1].set_ylabel('Î¸â‚‚')
    axes[0, 1].set_title('ë‘ ë²ˆì§¸ ì§„ì')
    axes[0, 1].grid(True)

    axes[1, 0].set_xlabel('x')
    axes[1, 0].set_ylabel('y')
    axes[1, 0].set_title('ëì  ê¶¤ì  (ì´ˆê¸°ì¡°ê±´ ë¯¼ê°ë„)')
    axes[1, 0].set_aspect('equal')
    axes[1, 0].grid(True)

    # ì´ˆê¸°ì¡°ê±´ ì°¨ì´ ì‹œê°í™”
    sol1 = solve_ivp(double_pend, (0, 20), initial_conditions[0], max_step=0.01)
    sol2 = solve_ivp(double_pend, (0, 20), initial_conditions[1], max_step=0.01)

    from scipy.interpolate import interp1d
    t_common = np.linspace(0, 20, 1000)
    theta1_1 = interp1d(sol1.t, sol1.y[0])(t_common)
    theta1_2 = interp1d(sol2.t, sol2.y[0])(t_common)

    axes[1, 1].semilogy(t_common, np.abs(theta1_1 - theta1_2))
    axes[1, 1].set_xlabel('ì‹œê°„')
    axes[1, 1].set_ylabel('|Î”Î¸â‚|')
    axes[1, 1].set_title('ê¶¤ì  ë¶„ë¦¬ (ì§€ìˆ˜ì  ì¦ê°€)')
    axes[1, 1].grid(True)

    plt.tight_layout()
    plt.show()

def lorenz_system():
    """
    ë¡œë Œì¸  ì‹œìŠ¤í…œ (1963):

    dx/dt = Ïƒ(y - x)
    dy/dt = x(Ï - z) - y
    dz/dt = xy - Î²z

    í‘œì¤€ íŒŒë¼ë¯¸í„°: Ïƒ=10, Ï=28, Î²=8/3
    â†’ ì´ìƒí•œ ëŒê°œ (strange attractor)
    """
    sigma, rho, beta = 10, 28, 8/3

    def lorenz(t, state):
        x, y, z = state
        return [
            sigma * (y - x),
            x * (rho - z) - y,
            x * y - beta * z
        ]

    sol = solve_ivp(lorenz, (0, 50), [1, 1, 1], max_step=0.01)

    fig = plt.figure(figsize=(15, 5))

    # 3D ê¶¤ì 
    ax1 = fig.add_subplot(131, projection='3d')
    ax1.plot(sol.y[0], sol.y[1], sol.y[2], linewidth=0.5)
    ax1.set_xlabel('x')
    ax1.set_ylabel('y')
    ax1.set_zlabel('z')
    ax1.set_title('ë¡œë Œì¸  ëŒê°œ')

    # x-z íˆ¬ì˜
    ax2 = fig.add_subplot(132)
    ax2.plot(sol.y[0], sol.y[2], linewidth=0.3)
    ax2.set_xlabel('x')
    ax2.set_ylabel('z')
    ax2.set_title('x-z íˆ¬ì˜')
    ax2.grid(True)

    # ì‹œê°„ ì‘ë‹µ
    ax3 = fig.add_subplot(133)
    ax3.plot(sol.t[:500], sol.y[0][:500])
    ax3.set_xlabel('ì‹œê°„')
    ax3.set_ylabel('x')
    ax3.set_title('x(t)')
    ax3.grid(True)

    plt.tight_layout()
    plt.show()

def lorenz_sensitivity():
    """ë¡œë Œì¸  ì‹œìŠ¤í…œì˜ ì´ˆê¸°ì¡°ê±´ ë¯¼ê°ë„"""
    sigma, rho, beta = 10, 28, 8/3

    def lorenz(t, state):
        x, y, z = state
        return [sigma*(y-x), x*(rho-z)-y, x*y-beta*z]

    # ë¯¸ì„¸í•˜ê²Œ ë‹¤ë¥¸ ì´ˆê¸°ì¡°ê±´
    eps = 1e-10
    sol1 = solve_ivp(lorenz, (0, 50), [1, 1, 1], max_step=0.01)
    sol2 = solve_ivp(lorenz, (0, 50), [1+eps, 1, 1], max_step=0.01)

    fig, axes = plt.subplots(2, 2, figsize=(14, 10))

    # ì‹œê°„ì— ë”°ë¥¸ x
    axes[0, 0].plot(sol1.t, sol1.y[0], 'b-', alpha=0.7, label='IC 1')
    axes[0, 0].plot(sol2.t, sol2.y[0], 'r-', alpha=0.7, label='IC 2')
    axes[0, 0].set_xlabel('ì‹œê°„')
    axes[0, 0].set_ylabel('x')
    axes[0, 0].set_title(f'x(t), ì´ˆê¸°ì°¨ì´ = {eps}')
    axes[0, 0].legend()
    axes[0, 0].grid(True)

    # ê¶¤ì  ë¶„ë¦¬
    from scipy.interpolate import interp1d
    t_common = np.linspace(0, 50, 5000)
    x1 = interp1d(sol1.t, sol1.y[0])(t_common)
    x2 = interp1d(sol2.t, sol2.y[0])(t_common)
    y1 = interp1d(sol1.t, sol1.y[1])(t_common)
    y2 = interp1d(sol2.t, sol2.y[1])(t_common)
    z1 = interp1d(sol1.t, sol1.y[2])(t_common)
    z2 = interp1d(sol2.t, sol2.y[2])(t_common)

    dist = np.sqrt((x1-x2)**2 + (y1-y2)**2 + (z1-z2)**2)

    axes[0, 1].semilogy(t_common, dist)
    axes[0, 1].set_xlabel('ì‹œê°„')
    axes[0, 1].set_ylabel('ê±°ë¦¬')
    axes[0, 1].set_title('ê¶¤ì  ê°„ ê±°ë¦¬ (ë¡œê·¸ ìŠ¤ì¼€ì¼)')
    axes[0, 1].grid(True)

    # ë¦¬ì•„í‘¸ë…¸í”„ ì§€ìˆ˜ ì¶”ì •
    # ì´ˆê¸° ì§€ìˆ˜ì  ì„±ì¥ êµ¬ê°„ì—ì„œ
    early = (t_common > 5) & (t_common < 20)
    from scipy.stats import linregress
    slope, _, _, _, _ = linregress(t_common[early], np.log(dist[early] + 1e-20))
    print(f"ì¶”ì • ë¦¬ì•„í‘¸ë…¸í”„ ì§€ìˆ˜: {slope:.3f}")

    axes[1, 0].plot(t_common[early], np.log(dist[early]))
    axes[1, 0].set_xlabel('ì‹œê°„')
    axes[1, 0].set_ylabel('log(ê±°ë¦¬)')
    axes[1, 0].set_title(f'ë¦¬ì•„í‘¸ë…¸í”„ ì§€ìˆ˜ ì¶”ì •: Î» â‰ˆ {slope:.3f}')
    axes[1, 0].grid(True)

    # 3D ë¹„êµ
    ax = fig.add_subplot(224, projection='3d')
    ax.plot(sol1.y[0][::10], sol1.y[1][::10], sol1.y[2][::10],
            'b-', alpha=0.5, linewidth=0.5)
    ax.plot(sol2.y[0][::10], sol2.y[1][::10], sol2.y[2][::10],
            'r-', alpha=0.5, linewidth=0.5)
    ax.set_title('ë‘ ê¶¤ì  ë¹„êµ')

    plt.tight_layout()
    plt.show()

def lorenz_bifurcation():
    """Ïì— ë”°ë¥¸ ë¡œë Œì¸  ì‹œìŠ¤í…œ ë¶„ê¸°"""
    sigma, beta = 10, 8/3

    rho_values = np.linspace(0, 50, 100)
    bifurcation_points = []

    for rho in rho_values:
        def lorenz(t, state):
            x, y, z = state
            return [sigma*(y-x), x*(rho-z)-y, x*y-beta*z]

def sir_model():
    """
    SIR ëª¨ë¸:
    dS/dt = -Î²SI
    dI/dt = Î²SI - Î³I
    dR/dt = Î³I

    S: ê°ìˆ˜ì„±(Susceptible)
    I: ê°ì—¼(Infected)
    R: íšŒë³µ(Recovered)
    """
    beta = 0.3   # ê°ì—¼ë¥ 
    gamma = 0.1  # íšŒë³µë¥ 

    def sir(t, y):
        S, I, R = y
        dS = -beta * S * I
        dI = beta * S * I - gamma * I
        dR = gamma * I
        return [dS, dI, dR]

    # ì´ˆê¸° ì¡°ê±´ (ì¸êµ¬ì˜ ë¹„ìœ¨)
    S0, I0, R0 = 0.99, 0.01, 0.0
    t_span = (0, 100)

    sol = solve_ivp(sir, t_span, [S0, I0, R0],
                    t_eval=np.linspace(0, 100, 1000))

    plt.figure(figsize=(10, 6))
    plt.plot(sol.t, sol.y[0], 'b-', label='ê°ìˆ˜ì„± (S)')
    plt.plot(sol.t, sol.y[1], 'r-', label='ê°ì—¼ (I)')
    plt.plot(sol.t, sol.y[2], 'g-', label='íšŒë³µ (R)')
    plt.xlabel('ì‹œê°„')
    plt.ylabel('ì¸êµ¬ ë¹„ìœ¨')
    plt.title(f'SIR ëª¨ë¸ (Î²={beta}, Î³={gamma}, Râ‚€={beta/gamma:.1f})')
    plt.legend()
    plt.grid(True)
    plt.show()

    # ê¸°ë³¸ ì¬ìƒì‚° ìˆ˜ Râ‚€ = Î²/Î³
    R0 = beta / gamma
    print(f"ê¸°ë³¸ ì¬ìƒì‚° ìˆ˜ Râ‚€ = {R0:.2f}")
    print(f"ì„ê³„ ë©´ì—­ë¥  = 1 - 1/Râ‚€ = {1 - 1/R0:.2%}")

def rossler_attractor():
    """
    ë¢°ìŠ¬ëŸ¬ ëŒê°œ:
    dx/dt = -y - z
    dy/dt = x + ay
    dz/dt = b + z(x - c)
    """
    a, b, c = 0.2, 0.2, 5.7

    def rossler(t, state):
        x, y, z = state
        return [-y - z, x + a*y, b + z*(x - c)]

    sol = solve_ivp(rossler, (0, 200), [0, 1, 0], max_step=0.01)

    fig = plt.figure(figsize=(15, 5))

    ax1 = fig.add_subplot(131, projection='3d')

def three_body_simplified():
    """ë‹¨ìˆœí™”ëœ 3ì²´ ë¬¸ì œ (í‰ë©´, ë™ì¼ ì§ˆëŸ‰)"""
    G = 1  # ì¤‘ë ¥ ìƒìˆ˜ (ë‹¨ìœ„ê³„ ì¡°ì •)
    m = 1  # ëª¨ë“  ì§ˆëŸ‰ ë™ì¼

    def three_body(t, y):
        # y = [x1, y1, x2, y2, x3, y3, vx1, vy1, vx2, vy2, vx3, vy3]
        x1, y1, x2, y2, x3, y3, vx1, vy1, vx2, vy2, vx3, vy3 = y

        # ê±°ë¦¬
        r12 = np.sqrt((x2-x1)**2 + (y2-y1)**2)
        r13 = np.sqrt((x3-x1)**2 + (y3-y1)**2)
        r23 = np.sqrt((x3-x2)**2 + (y3-y2)**2)

        # ê°€ì†ë„
        ax1 = G*m*((x2-x1)/r12**3 + (x3-x1)/r13**3)
        ay1 = G*m*((y2-y1)/r12**3 + (y3-y1)/r13**3)
        ax2 = G*m*((x1-x2)/r12**3 + (x3-x2)/r23**3)
        ay2 = G*m*((y1-y2)/r12**3 + (y3-y2)/r23**3)
        ax3 = G*m*((x1-x3)/r13**3 + (x2-x3)/r23**3)
        ay3 = G*m*((y1-y3)/r13**3 + (y2-y3)/r23**3)

        return [vx1, vy1, vx2, vy2, vx3, vy3,
                ax1, ay1, ax2, ay2, ax3, ay3]

    # 8ìí˜• í•´ ì´ˆê¸°ì¡°ê±´ (Chenciner & Montgomery, 2000)
    # íŠ¹ë³„í•œ ì£¼ê¸°í•´
    x0 = 0.97000436
    y0 = -0.24308753
    vx0 = 0.4662036850
    vy0 = 0.4323657300

    y0_state = [-x0, -y0, x0, y0, 0, 0,
                vx0, vy0, vx0, vy0, -2*vx0, -2*vy0]

    sol = solve_ivp(three_body, (0, 6.3), y0_state,
                    method='DOP853', max_step=0.01)

    plt.figure(figsize=(10, 8))
    plt.plot(sol.y[0], sol.y[1], 'b-', linewidth=0.8, label='ë¬¼ì²´ 1')
    plt.plot(sol.y[2], sol.y[3], 'r-', linewidth=0.8, label='ë¬¼ì²´ 2')
    plt.plot(sol.y[4], sol.y[5], 'g-', linewidth=0.8, label='ë¬¼ì²´ 3')

    # ì´ˆê¸° ìœ„ì¹˜
    plt.scatter([sol.y[0][0], sol.y[2][0], sol.y[4][0]],
                [sol.y[1][0], sol.y[3][0], sol.y[5][0]],
                s=100, c=['blue', 'red', 'green'])

    plt.xlabel('x')
    plt.ylabel('y')
    plt.title('3ì²´ ë¬¸ì œ: 8ìí˜• ì£¼ê¸°í•´')
    plt.legend()
    plt.axis('equal')
    plt.grid(True)
    plt.show()
