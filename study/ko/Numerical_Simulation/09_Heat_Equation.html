import numpy as np
import matplotlib.pyplot as plt
from matplotlib.animation import FuncAnimation

class HeatEquation1D_FTCS:
    """
    1D ì—´ë°©ì •ì‹ FTCS ì–‘í•´ë²•

    âˆ‚u/âˆ‚t = Î± Â· âˆ‚Â²u/âˆ‚xÂ²
    """

    def __init__(self, L=1.0, alpha=0.01, nx=51, T=1.0, safety=0.4):
        """
        Parameters:
        -----------
        L : float - ì˜ì—­ ê¸¸ì´
        alpha : float - ì—´í™•ì‚°ê³„ìˆ˜
        nx : int - ê³µê°„ ê²©ìì  ìˆ˜
        T : float - ìµœì¢… ì‹œê°„
        safety : float - CFL ì•ˆì „ê³„ìˆ˜ (0 < safety â‰¤ 0.5)
        """

        self.alpha = alpha
        self.nx = nx
        self.T = T

        # ê²©ì ìƒì„±
        self.dx = L / (nx - 1)
        self.x = np.linspace(0, L, nx)

        # ì•ˆì • ì¡°ê±´ì— ë”°ë¥¸ ì‹œê°„ ê°„ê²© ê²°ì •

        self.dt = T / self.nt  # ì •í™•íˆ Tì— ë„ë‹¬í•˜ë„ë¡ ì¡°ì •

        self.r = alpha * self.dt / self.dx**2

        print(f"FTCS ì—´ë°©ì •ì‹ ì„¤ì •")
        print(f"  dx = {self.dx:.4f}, dt = {self.dt:.6f}")
        print(f"  r = Î±Â·dt/dxÂ² = {self.r:.4f}")
        print(f"  ì‹œê°„ ìŠ¤í… ìˆ˜: {self.nt}")
        print(f"  ì•ˆì •ì„±: {'OK' if self.r <= 0.5 else 'WARNING!'}")

    def set_initial_condition(self, func):
        """ì´ˆê¸°ì¡°ê±´ ì„¤ì •"""
        self.u = func(self.x)
        self.u0 = self.u.copy()

    def set_boundary_conditions(self, left_type='dirichlet', left_value=0,
                                  right_type='dirichlet', right_value=0):
        """ê²½ê³„ì¡°ê±´ ì„¤ì •"""
        self.bc = {
            'left': {'type': left_type, 'value': left_value},
            'right': {'type': right_type, 'value': right_value}
        }

    def apply_bc(self, u):
        """ê²½ê³„ì¡°ê±´ ì ìš©"""
        # ì™¼ìª½ ê²½ê³„
        if self.bc['left']['type'] == 'dirichlet':
            u[0] = self.bc['left']['value']
        elif self.bc['left']['type'] == 'neumann':
            # âˆ‚u/âˆ‚x = flux => u[0] = u[1] - flux * dx
            u[0] = u[1] - self.bc['left']['value'] * self.dx

        # ì˜¤ë¥¸ìª½ ê²½ê³„
        if self.bc['right']['type'] == 'dirichlet':
            u[-1] = self.bc['right']['value']
        elif self.bc['right']['type'] == 'neumann':
            # âˆ‚u/âˆ‚x = flux => u[-1] = u[-2] + flux * dx
            u[-1] = u[-2] + self.bc['right']['value'] * self.dx

        return u

    def step(self):
        """í•œ ì‹œê°„ ìŠ¤í… ì§„í–‰ (FTCS)"""
        u_new = self.u.copy()

        # ë‚´ë¶€ì  ì—…ë°ì´íŠ¸
        u_new[1:-1] = self.u[1:-1] + self.r * (
            self.u[2:] - 2*self.u[1:-1] + self.u[:-2]
        )

        # ê²½ê³„ì¡°ê±´ ì ìš©
        u_new = self.apply_bc(u_new)

        self.u = u_new

    def solve(self, save_interval=None):
        """ì „ì²´ ì‹œê°„ êµ¬ê°„ í’€ì´"""
        if save_interval is None:
            save_interval = max(1, self.nt // 100)

        history = [self.u.copy()]
        times = [0]

        for n in range(self.nt):
            self.step()
            if (n + 1) % save_interval == 0:
                history.append(self.u.copy())
                times.append((n + 1) * self.dt)

        return np.array(times), np.array(history)

from scipy.sparse import diags
from scipy.sparse.linalg import spsolve

class HeatEquation1D_BTCS:
    """
    1D ì—´ë°©ì •ì‹ BTCS ìŒí•´ë²•

    ë¬´ì¡°ê±´ ì•ˆì • (unconditionally stable)
    """

    def __init__(self, L=1.0, alpha=0.01, nx=51, T=1.0, nt=100):
        """
        Parameters:
        -----------
        L : float - ì˜ì—­ ê¸¸ì´
        alpha : float - ì—´í™•ì‚°ê³„ìˆ˜
        nx : int - ê³µê°„ ê²©ìì  ìˆ˜
        T : float - ìµœì¢… ì‹œê°„
        nt : int - ì‹œê°„ ìŠ¤í… ìˆ˜
        """
        self.L = L
        self.alpha = alpha
        self.nx = nx
        self.T = T
        self.nt = nt

        # ê²©ì ìƒì„±
        self.dx = L / (nx - 1)
        self.dt = T / nt
        self.x = np.linspace(0, L, nx)

        self.r = alpha * self.dt / self.dx**2

        print(f"BTCS ì—´ë°©ì •ì‹ ì„¤ì •")
        print(f"  dx = {self.dx:.4f}, dt = {self.dt:.6f}")
        print(f"  r = Î±Â·dt/dxÂ² = {self.r:.4f}")
        print(f"  BTCSëŠ” ë¬´ì¡°ê±´ ì•ˆì • (rì— ì œí•œ ì—†ìŒ)")

        # í–‰ë ¬ A ìƒì„± (ë‚´ë¶€ì ë§Œ)
        self._build_matrix()

    def _build_matrix(self):
        """BTCS í–‰ë ¬ ìƒì„±"""
        n = self.nx - 2  # ë‚´ë¶€ì  ìˆ˜

        main_diag = (1 + 2*self.r) * np.ones(n)
        off_diag = -self.r * np.ones(n - 1)

        self.A = diags([off_diag, main_diag, off_diag], [-1, 0, 1], format='csr')

    def set_initial_condition(self, func):
        """ì´ˆê¸°ì¡°ê±´ ì„¤ì •"""
        self.u = func(self.x)
        self.u0 = self.u.copy()

    def set_boundary_conditions(self, left_value=0, right_value=0):
        """ë””ë¦¬í´ë ˆ ê²½ê³„ì¡°ê±´ ì„¤ì •"""
        self.u_left = left_value
        self.u_right = right_value

    def step(self):
        """í•œ ì‹œê°„ ìŠ¤í… ì§„í–‰ (BTCS)"""
        # ìš°ë³€ ë²¡í„° (ë‚´ë¶€ì )
        b = self.u[1:-1].copy()

        # ê²½ê³„ì¡°ê±´ ê¸°ì—¬
        b[0] += self.r * self.u_left
        b[-1] += self.r * self.u_right

        # ì„ í˜• ì‹œìŠ¤í…œ í’€ì´
        u_inner = spsolve(self.A, b)

        # ì „ì²´ í•´ ì—…ë°ì´íŠ¸
        self.u[1:-1] = u_inner
        self.u[0] = self.u_left
        self.u[-1] = self.u_right

    def solve(self, save_interval=None):
        """ì „ì²´ ì‹œê°„ êµ¬ê°„ í’€ì´"""
        if save_interval is None:
            save_interval = max(1, self.nt // 100)

        history = [self.u.copy()]
        times = [0]

        for n in range(self.nt):
            self.step()
            if (n + 1) % save_interval == 0:
                history.append(self.u.copy())
                times.append((n + 1) * self.dt)

        return np.array(times), np.array(history)

from scipy.sparse import diags
from scipy.sparse.linalg import spsolve

class HeatEquation1D_CrankNicolson:
    """
    1D ì—´ë°©ì •ì‹ Crank-Nicolson ë°©ë²•

    - ë¬´ì¡°ê±´ ì•ˆì •
    - 2ì°¨ ì •í™•ë„ (ì‹œê°„, ê³µê°„ ëª¨ë‘)
    """

    def __init__(self, L=1.0, alpha=0.01, nx=51, T=1.0, nt=100):
        """
        Parameters:
        -----------
        L : float - ì˜ì—­ ê¸¸ì´
        alpha : float - ì—´í™•ì‚°ê³„ìˆ˜
        nx : int - ê³µê°„ ê²©ìì  ìˆ˜
        T : float - ìµœì¢… ì‹œê°„
        nt : int - ì‹œê°„ ìŠ¤í… ìˆ˜
        """
        self.L = L
        self.alpha = alpha
        self.nx = nx
        self.T = T
        self.nt = nt

        # ê²©ì ìƒì„±
        self.dx = L / (nx - 1)
        self.dt = T / nt
        self.x = np.linspace(0, L, nx)

        self.r = alpha * self.dt / self.dx**2

        print(f"Crank-Nicolson ì—´ë°©ì •ì‹ ì„¤ì •")
        print(f"  dx = {self.dx:.4f}, dt = {self.dt:.6f}")
        print(f"  r = Î±Â·dt/dxÂ² = {self.r:.4f}")
        print(f"  2ì°¨ ì •í™•ë„ & ë¬´ì¡°ê±´ ì•ˆì •")

        # í–‰ë ¬ ìƒì„±
        self._build_matrices()

    def _build_matrices(self):
        """Crank-Nicolson í–‰ë ¬ A, B ìƒì„±"""
        n = self.nx - 2  # ë‚´ë¶€ì  ìˆ˜
        r = self.r

        # A í–‰ë ¬: ì¢Œë³€ (ì•”ì‹œì  ë¶€ë¶„)
        main_A = (1 + r) * np.ones(n)
        off_A = (-r/2) * np.ones(n - 1)
        self.A = diags([off_A, main_A, off_A], [-1, 0, 1], format='csr')

        # B í–‰ë ¬: ìš°ë³€ (ëª…ì‹œì  ë¶€ë¶„)
        main_B = (1 - r) * np.ones(n)
        off_B = (r/2) * np.ones(n - 1)
        self.B = diags([off_B, main_B, off_B], [-1, 0, 1], format='csr')

    def set_initial_condition(self, func):
        """ì´ˆê¸°ì¡°ê±´ ì„¤ì •"""
        self.u = func(self.x)
        self.u0 = self.u.copy()

    def set_boundary_conditions(self, left_value=0, right_value=0):
        """ë””ë¦¬í´ë ˆ ê²½ê³„ì¡°ê±´ ì„¤ì •"""
        self.u_left = left_value
        self.u_right = right_value

    def step(self):
        """í•œ ì‹œê°„ ìŠ¤í… ì§„í–‰ (Crank-Nicolson)"""
        r = self.r

        # ìš°ë³€: BÂ·u^n + ê²½ê³„ì¡°ê±´ ê¸°ì—¬
        b = self.B @ self.u[1:-1]

        # ê²½ê³„ì¡°ê±´ ê¸°ì—¬ (ì¢Œë³€ê³¼ ìš°ë³€ ëª¨ë‘ì—ì„œ)
        b[0] += (r/2) * (self.u_left + self.u_left)  # nê³¼ n+1 ì‹œì ì˜ BC
        b[-1] += (r/2) * (self.u_right + self.u_right)

        # ì„ í˜• ì‹œìŠ¤í…œ í’€ì´
        u_inner = spsolve(self.A, b)

        # ì „ì²´ í•´ ì—…ë°ì´íŠ¸
        self.u[1:-1] = u_inner
        self.u[0] = self.u_left
        self.u[-1] = self.u_right

    def solve(self, save_interval=None):
        """ì „ì²´ ì‹œê°„ êµ¬ê°„ í’€ì´"""
        if save_interval is None:
            save_interval = max(1, self.nt // 100)

        history = [self.u.copy()]
        times = [0]

        for n in range(self.nt):
            self.step()
            if (n + 1) % save_interval == 0:
                history.append(self.u.copy())
                times.append((n + 1) * self.dt)

        return np.array(times), np.array(history)

class HeatEquation2D_FTCS:
    """

    âˆ‚u/âˆ‚t = Î± Â· (âˆ‚Â²u/âˆ‚xÂ² + âˆ‚Â²u/âˆ‚yÂ²)
    """

    def __init__(self, Lx=1.0, Ly=1.0, alpha=0.01, nx=51, ny=51, T=0.5, safety=0.2):
        """
        Parameters:
        -----------
        Lx, Ly : float - ì˜ì—­ í¬ê¸°
        alpha : float - ì—´í™•ì‚°ê³„ìˆ˜
        nx, ny : int - ê²©ìì  ìˆ˜
        T : float - ìµœì¢… ì‹œê°„
        safety : float - CFL ì•ˆì „ê³„ìˆ˜
        """
        self.Lx = Lx
        self.Ly = Ly
        self.alpha = alpha
        self.nx = nx
        self.ny = ny
        self.T = T

        # ê²©ì ìƒì„±
        self.dx = Lx / (nx - 1)
        self.dy = Ly / (ny - 1)
        self.x = np.linspace(0, Lx, nx)
        self.y = np.linspace(0, Ly, ny)
        self.X, self.Y = np.meshgrid(self.x, self.y)

        # 2D CFL ì¡°ê±´: r_x + r_y â‰¤ 0.5
        dt_cfl = safety * 0.5 / (alpha * (1/self.dx**2 + 1/self.dy**2))
        self.dt = dt_cfl
        self.nt = int(np.ceil(T / self.dt))
        self.dt = T / self.nt

        self.rx = alpha * self.dt / self.dx**2
        self.ry = alpha * self.dt / self.dy**2

        print(f"2D FTCS ì—´ë°©ì •ì‹ ì„¤ì •")
        print(f"  ê²©ì: {nx} x {ny}")
        print(f"  dx = {self.dx:.4f}, dy = {self.dy:.4f}, dt = {self.dt:.6f}")
        print(f"  r_x = {self.rx:.4f}, r_y = {self.ry:.4f}")
        print(f"  r_x + r_y = {self.rx + self.ry:.4f} (â‰¤ 0.5ì´ì–´ì•¼ ì•ˆì •)")

    def set_initial_condition(self, func):
        """ì´ˆê¸°ì¡°ê±´ ì„¤ì •: u(x,y,0) = func(X, Y)"""
        self.u = func(self.X, self.Y)
        self.u0 = self.u.copy()

    def set_boundary_conditions(self, bc_value=0):
        """ë””ë¦¬í´ë ˆ ê²½ê³„ì¡°ê±´ (ëª¨ë“  ê²½ê³„ì—ì„œ ë™ì¼ ê°’)"""
        self.bc_value = bc_value

    def apply_bc(self, u):
        """ê²½ê³„ì¡°ê±´ ì ìš©"""
        u[0, :] = self.bc_value   # ì•„ë˜
        u[-1, :] = self.bc_value  # ìœ„
        u[:, 0] = self.bc_value   # ì™¼ìª½
        u[:, -1] = self.bc_value  # ì˜¤ë¥¸ìª½
        return u

    def step(self):
        """í•œ ì‹œê°„ ìŠ¤í… ì§„í–‰ (2D FTCS)"""
        u_new = self.u.copy()

        # ë‚´ë¶€ì  ì—…ë°ì´íŠ¸
        u_new[1:-1, 1:-1] = self.u[1:-1, 1:-1] + \
            self.rx * (self.u[1:-1, 2:] - 2*self.u[1:-1, 1:-1] + self.u[1:-1, :-2]) + \
            self.ry * (self.u[2:, 1:-1] - 2*self.u[1:-1, 1:-1] + self.u[:-2, 1:-1])

        # ê²½ê³„ì¡°ê±´
        u_new = self.apply_bc(u_new)

        self.u = u_new

    def solve(self, save_interval=None):
        """ì „ì²´ ì‹œê°„ êµ¬ê°„ í’€ì´"""
        if save_interval is None:
            save_interval = max(1, self.nt // 50)

        history = [self.u.copy()]
        times = [0]

        for n in range(self.nt):
            self.step()
            if (n + 1) % save_interval == 0:
                history.append(self.u.copy())
                times.append((n + 1) * self.dt)

        return np.array(times), history

from scipy.sparse import diags
from scipy.sparse.linalg import spsolve

class HeatEquation2D_ADI:
    """
    2D ì—´ë°©ì •ì‹ ADI (Alternating Direction Implicit) ë°©ë²•

    ê° ì‹œê°„ ìŠ¤í…ì„ ë‘ ë°˜ìŠ¤í…ìœ¼ë¡œ ë‚˜ëˆ”:
    1. x-ë°©í–¥ ì•”ì‹œì , y-ë°©í–¥ ëª…ì‹œì 
    2. y-ë°©í–¥ ì•”ì‹œì , x-ë°©í–¥ ëª…ì‹œì 

    ë¬´ì¡°ê±´ ì•ˆì • + 2ì°¨ ì •í™•ë„
    """

    def __init__(self, Lx=1.0, Ly=1.0, alpha=0.01, nx=51, ny=51, T=0.5, nt=100):
        self.Lx = Lx
        self.Ly = Ly
        self.alpha = alpha
        self.nx = nx
        self.ny = ny
        self.T = T
        self.nt = nt

        # ê²©ì ìƒì„±
        self.dx = Lx / (nx - 1)
        self.dy = Ly / (ny - 1)
        self.dt = T / nt
        self.x = np.linspace(0, Lx, nx)
        self.y = np.linspace(0, Ly, ny)
        self.X, self.Y = np.meshgrid(self.x, self.y)

        self.rx = alpha * self.dt / (2 * self.dx**2)
        self.ry = alpha * self.dt / (2 * self.dy**2)

        print(f"2D ADI ì—´ë°©ì •ì‹ ì„¤ì •")
        print(f"  ê²©ì: {nx} x {ny}")
        print(f"  r_x = {self.rx:.4f}, r_y = {self.ry:.4f}")

        # í–‰ë ¬ ìƒì„±
        self._build_matrices()

    def _build_matrices(self):
        """ADI ì‚¼ì¤‘ëŒ€ê° í–‰ë ¬ ìƒì„±"""
        # x-ë°©í–¥ (ê° yì— ëŒ€í•´)
        mx = self.nx - 2
        main_x = (1 + 2*self.rx) * np.ones(mx)
        off_x = -self.rx * np.ones(mx - 1)
        self.Ax = diags([off_x, main_x, off_x], [-1, 0, 1], format='csr')

        # y-ë°©í–¥ (ê° xì— ëŒ€í•´)
        my = self.ny - 2
        main_y = (1 + 2*self.ry) * np.ones(my)
        off_y = -self.ry * np.ones(my - 1)
        self.Ay = diags([off_y, main_y, off_y], [-1, 0, 1], format='csr')

    def set_initial_condition(self, func):
        """ì´ˆê¸°ì¡°ê±´ ì„¤ì •"""
        self.u = func(self.X, self.Y)
        self.u0 = self.u.copy()

    def set_boundary_conditions(self, bc_value=0):
        """ë””ë¦¬í´ë ˆ ê²½ê³„ì¡°ê±´"""
        self.bc_value = bc_value

    def step(self):
        """í•œ ì‹œê°„ ìŠ¤í… (ADI ë‘ ë°˜ìŠ¤í…)"""
        u = self.u
        bc = self.bc_value

        # ì¤‘ê°„ í•´ ë°°ì—´
        u_half = np.zeros_like(u)
        u_new = np.zeros_like(u)

        # ë°˜ìŠ¤í… 1: x-ì•”ì‹œì , y-ëª…ì‹œì 
        for j in range(1, self.ny - 1):
            # y-ëª…ì‹œì  ë¶€ë¶„ (ìš°ë³€)
            b = u[j, 1:-1] + self.ry * (u[j+1, 1:-1] - 2*u[j, 1:-1] + u[j-1, 1:-1])
            # ê²½ê³„ì¡°ê±´
            b[0] += self.rx * bc
            b[-1] += self.rx * bc
            # x-ì•”ì‹œì  í’€ì´
            u_half[j, 1:-1] = spsolve(self.Ax, b)

        # ê²½ê³„ì¡°ê±´ ì ìš©
        u_half[0, :] = bc
        u_half[-1, :] = bc
        u_half[:, 0] = bc
        u_half[:, -1] = bc

        # ë°˜ìŠ¤í… 2: y-ì•”ì‹œì , x-ëª…ì‹œì 
        for i in range(1, self.nx - 1):
            # x-ëª…ì‹œì  ë¶€ë¶„ (ìš°ë³€)
            b = u_half[1:-1, i] + self.rx * (u_half[1:-1, i+1] - 2*u_half[1:-1, i] + u_half[1:-1, i-1])
            # ê²½ê³„ì¡°ê±´
            b[0] += self.ry * bc
            b[-1] += self.ry * bc
            # y-ì•”ì‹œì  í’€ì´
            u_new[1:-1, i] = spsolve(self.Ay, b)

        # ê²½ê³„ì¡°ê±´ ì ìš©
        u_new[0, :] = bc
        u_new[-1, :] = bc
        u_new[:, 0] = bc
        u_new[:, -1] = bc

        self.u = u_new

        if save_interval is None:
            save_interval = max(1, self.nt // 50)

            self.step()
            if (n + 1) % save_interval == 0:
                history.append(self.u.copy())
                times.append((n + 1) * self.dt)

        return np.array(times), history

class HeatEquation1D_Neumann:
    """
    ë…¸ì´ë§Œ ê²½ê³„ì¡°ê±´ì„ ê°€ì§„ 1D ì—´ë°©ì •ì‹

    âˆ‚u/âˆ‚x|_{x=0} = flux_left
    âˆ‚u/âˆ‚x|_{x=L} = flux_right
    """

    def __init__(self, L=1.0, alpha=0.01, nx=51, T=1.0, safety=0.4):
        self.L = L
        self.alpha = alpha
        self.nx = nx
        self.T = T

        self.dx = L / (nx - 1)
        self.dt = safety * self.dx**2 / alpha
        self.nt = int(np.ceil(T / self.dt))
        self.dt = T / self.nt

        self.x = np.linspace(0, L, nx)
        self.r = alpha * self.dt / self.dx**2

        print(f"ë…¸ì´ë§Œ BC ì—´ë°©ì •ì‹: r = {self.r:.4f}")

    def set_initial_condition(self, func):
        self.u = func(self.x)

    def set_boundary_conditions(self, flux_left=0, flux_right=0):
        """ë…¸ì´ë§Œ ê²½ê³„ì¡°ê±´ ì„¤ì •"""
        self.flux_left = flux_left
        self.flux_right = flux_right

    def step(self):
        """í•œ ì‹œê°„ ìŠ¤í… (ë…¸ì´ë§Œ BC í¬í•¨)"""
        u_new = self.u.copy()

        # ë‚´ë¶€ì 
        u_new[1:-1] = self.u[1:-1] + self.r * (
            self.u[2:] - 2*self.u[1:-1] + self.u[:-2]
        )

        # ì™¼ìª½ ë…¸ì´ë§Œ BC: âˆ‚u/âˆ‚x = flux_left
        # ê³ ìŠ¤íŠ¸ ë…¸ë“œ ë°©ë²•: u[-1] = u[1] - 2*dx*flux_left
        # u_new[0] = u[0] + r*(u[1] - 2*u[0] + u[-1])
        #          = u[0] + r*(u[1] - 2*u[0] + u[1] - 2*dx*flux_left)
        #          = u[0] + r*(2*u[1] - 2*u[0] - 2*dx*flux_left)
        u_new[0] = self.u[0] + self.r * (
            2*self.u[1] - 2*self.u[0] - 2*self.dx*self.flux_left
        )

        # ì˜¤ë¥¸ìª½ ë…¸ì´ë§Œ BC: âˆ‚u/âˆ‚x = flux_right
        u_new[-1] = self.u[-1] + self.r * (
            2*self.u[-2] - 2*self.u[-1] + 2*self.dx*self.flux_right
        )

        self.u = u_new

    def solve(self):
        history = [self.u.copy()]
        times = [0]

        save_interval = max(1, self.nt // 100)

        for n in range(self.nt):
            self.step()
            if (n + 1) % save_interval == 0:
                history.append(self.u.copy())
                times.append((n + 1) * self.dt)

        return np.array(times), np.array(history)
