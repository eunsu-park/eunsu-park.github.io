2.1 FTCS ìŠ¤í‚´¶

(u_i^{n+1} - u_i^n) / Î”t + c Â· (u_{i+1}^n - u_{i-1}^n) / (2Î”x) = 0

class AdvectionFTCS:
    """
    ì´ë¥˜ë°©ì •ì‹ FTCS (ë¶ˆì•ˆì •)

    ì£¼ì˜: êµìœ¡ ëª©ì ìœ¼ë¡œë§Œ ì‚¬ìš©. ì‹¤ì œë¡œëŠ” ë¶ˆì•ˆì •!
    """

    def __init__(self, L=4.0, c=1.0, nx=101, T=1.0, courant=0.5):
        self.L = L
        self.c = c
        self.nx = nx
        self.T = T

        self.dx = L / (nx - 1)
        self.dt = courant * self.dx / abs(c)
        self.nt = int(np.ceil(T / self.dt))
        self.dt = T / self.nt

        self.x = np.linspace(0, L, nx)
        self.C = c * self.dt / self.dx

        print(f"FTCS ì´ë¥˜ë°©ì •ì‹ (ë¶ˆì•ˆì •)")
        print(f"  C = {self.C:.4f}")

    def set_initial_condition(self, func):
        self.u = func(self.x)
        self.u0 = self.u.copy()

    def step(self):
        """FTCS ìŠ¤í… (ë¶ˆì•ˆì •)"""
        u_new = self.u.copy()

        # ë‚´ë¶€ì 
        u_new[1:-1] = self.u[1:-1] - (self.C / 2) * (self.u[2:] - self.u[:-2])

        # ì£¼ê¸°ì  ê²½ê³„ì¡°ê±´
        u_new[0] = self.u[0] - (self.C / 2) * (self.u[1] - self.u[-2])
        u_new[-1] = u_new[0]

        self.u = u_new

    def solve(self, save_interval=1):
        history = [self.u0.copy()]
        times = [0]

        for n in range(self.nt):
            self.step()
            if (n + 1) % save_interval == 0:
                history.append(self.u.copy())
                times.append((n + 1) * self.dt)

        return np.array(times), np.array(history)

3.1 í’ìƒë²• ì›ë¦¬¶

    """
    ì´ë¥˜ë°©ì •ì‹ í’ìƒë²• (Upwind)

    ì¡°ê±´ë¶€ ì•ˆì •: 0 â‰¤ C â‰¤ 1
    1ì°¨ ì •í™•ë„
    """

    def __init__(self, L=4.0, c=1.0, nx=101, T=1.0, courant=0.8):
        self.L = L
        self.c = c
        self.nx = nx
        self.T = T

        self.dx = L / (nx - 1)
        self.dt = courant * self.dx / abs(c)
        self.nt = int(np.ceil(T / self.dt))
        self.dt = T / self.nt

        self.x = np.linspace(0, L, nx)
        self.C = c * self.dt / self.dx

        print(f"Upwind ì´ë¥˜ë°©ì •ì‹")
        print(f"  C = {self.C:.4f}")
        print(f"  ì•ˆì •ì„±: {'OK' if 0 <= self.C <= 1 else 'WARNING!'}")

    def set_initial_condition(self, func):
        self.u = func(self.x)
        self.u0 = self.u.copy()

    def step(self):
        """Upwind ìŠ¤í…"""
        u_new = self.u.copy()

        if self.c > 0:
            # í›„ë°©ì°¨ë¶„ (ì •ë³´ê°€ ì™¼ìª½ì—ì„œ ì˜´)
            u_new[1:] = self.u[1:] - self.C * (self.u[1:] - self.u[:-1])
            # ì™¼ìª½ ê²½ê³„: ìœ ì… ì¡°ê±´ (ì™¸ë¶€ì—ì„œ ë“¤ì–´ì˜¤ëŠ” ê°’)
            u_new[0] = self.u0[0]  # ë˜ëŠ” íŠ¹ì • ê°’
        else:
            # ì „ë°©ì°¨ë¶„ (ì •ë³´ê°€ ì˜¤ë¥¸ìª½ì—ì„œ ì˜´)
            u_new[:-1] = self.u[:-1] - self.C * (self.u[1:] - self.u[:-1])
            # ì˜¤ë¥¸ìª½ ê²½ê³„
            u_new[-1] = self.u0[-1]

        self.u = u_new

    def solve(self, save_interval=1):
        history = [self.u0.copy()]
        times = [0]

        for n in range(self.nt):
            self.step()
            if (n + 1) % save_interval == 0:
                history.append(self.u.copy())
                times.append((n + 1) * self.dt)

        return np.array(times), np.array(history)

4.1 ìŠ¤í‚´¶

FTCSë¥¼ ìˆ˜ì •í•˜ì—¬ ì•ˆì •í™”:

    """
    ì´ë¥˜ë°©ì •ì‹ Lax-Friedrichs ìŠ¤í‚´

    ì¡°ê±´ë¶€ ì•ˆì •: |C| â‰¤ 1
    1ì°¨ ì •í™•ë„
    """

    def __init__(self, L=4.0, c=1.0, nx=101, T=1.0, courant=0.8):
        self.L = L
        self.c = c
        self.nx = nx
        self.T = T

        self.dx = L / (nx - 1)
        self.dt = courant * self.dx / abs(c)
        self.nt = int(np.ceil(T / self.dt))
        self.dt = T / self.nt

        self.x = np.linspace(0, L, nx)
        self.C = c * self.dt / self.dx

        print(f"Lax-Friedrichs ì´ë¥˜ë°©ì •ì‹")
        print(f"  C = {self.C:.4f}")

    def set_initial_condition(self, func):
        self.u = func(self.x)
        self.u0 = self.u.copy()

    def step(self):

        u_new = np.zeros_like(self.u)

        # ë‚´ë¶€ì

        # ê²½ê³„: ì™¸ì‚½ ë˜ëŠ” ê³ ì •
        u_new[0] = self.u[0]
        u_new[-1] = self.u[-1]

        self.u = u_new

    def solve(self, save_interval=1):
        history = [self.u0.copy()]
        times = [0]

        for n in range(self.nt):
            self.step()
            if (n + 1) % save_interval == 0:
                history.append(self.u.copy())
                times.append((n + 1) * self.dt)

        return np.array(times), np.array(history)

class AdvectionLaxWendroff:
    """
    ì´ë¥˜ë°©ì •ì‹ Lax-Wendroff ìŠ¤í‚´

    ì¡°ê±´ë¶€ ì•ˆì •: |C| â‰¤ 1
    2ì°¨ ì •í™•ë„
    """

    def __init__(self, L=4.0, c=1.0, nx=101, T=1.0, courant=0.8):
        self.L = L
        self.c = c
        self.nx = nx
        self.T = T

        self.dx = L / (nx - 1)
        self.dt = courant * self.dx / abs(c)
        self.nt = int(np.ceil(T / self.dt))
        self.dt = T / self.nt

        self.x = np.linspace(0, L, nx)
        self.C = c * self.dt / self.dx
        self.C2 = self.C ** 2

        print(f"Lax-Wendroff ì´ë¥˜ë°©ì •ì‹")
        print(f"  C = {self.C:.4f}")

    def set_initial_condition(self, func):
        self.u = func(self.x)
        self.u0 = self.u.copy()

    def step(self):
        """Lax-Wendroff ìŠ¤í…"""
        u_new = np.zeros_like(self.u)

        # ë‚´ë¶€ì 
        u_new[1:-1] = (self.u[1:-1]
                       - (self.C / 2) * (self.u[2:] - self.u[:-2])
                       + (self.C2 / 2) * (self.u[2:] - 2*self.u[1:-1] + self.u[:-2]))

        # ê²½ê³„
        u_new[0] = self.u[0]
        u_new[-1] = self.u[-1]

        self.u = u_new

    def solve(self, save_interval=1):
        history = [self.u0.copy()]
        times = [0]

        for n in range(self.nt):
            self.step()
            if (n + 1) % save_interval == 0:
                history.append(self.u.copy())
                times.append((n + 1) * self.dt)

        return np.array(times), np.array(history)


def compare_all_schemes():
    """ëª¨ë“  ìŠ¤í‚´ ë¹„êµ"""
    L = 4.0
    c = 1.0
    T = 2.0
    C = 0.8

    def u0(x):
        return np.exp(-(x - 1.0)**2 / 0.08)

    schemes = {
        'Upwind': AdvectionUpwind(L, c, nx=101, T=T, courant=C),
        'Lax-Friedrichs': AdvectionLaxFriedrichs(L, c, nx=101, T=T, courant=C),
        'Lax-Wendroff': AdvectionLaxWendroff(L, c, nx=101, T=T, courant=C),
    }

    fig, axes = plt.subplots(1, 3, figsize=(15, 4))

    for idx, (name, solver) in enumerate(schemes.items()):
        solver.set_initial_condition(u0)
        times, history = solver.solve()

        ax = axes[idx]

        ax.plot(solver.x, u0(solver.x), 'k--', alpha=0.5, label='ì´ˆê¸°')
        ax.plot(solver.x, history[-1], 'b-', linewidth=2, label='ìˆ˜ì¹˜í•´')
        ax.plot(solver.x, u0(solver.x - c * T), 'r--', linewidth=2, label='í•´ì„í•´')

        ax.set_xlabel('x')
        ax.set_ylabel('u')

        ax.grid(True, alpha=0.3)

    plt.suptitle(f'ì´ë¥˜ ìŠ¤í‚´ ë¹„êµ (C = {C})', fontsize=12)

def dispersion_analysis():
    """ìˆ˜ì¹˜ ë¶„ì‚° ê´€ê³„ ë¶„ì„"""
    k_dx = np.linspace(0.01, np.pi, 200)

    fig, axes = plt.subplots(1, 2, figsize=(14, 5))

    C = 0.8

    # ì •í™•í•œ ë¶„ì‚° ê´€ê³„: Ï‰ = cÂ·k
    # ì •ê·œí™”: Ï‰Â·dt = CÂ·kÂ·dx
    omega_exact = C * k_dx

    # ê° ìŠ¤í‚´ì˜ ì¦í­ì¸ìë¡œë¶€í„° Ï‰ ê³„ì‚°

    # Upwind: G = 1 - C + CÂ·exp(-iÂ·kÂ·dx)
    G_upwind = 1 - C + C * np.exp(-1j * k_dx)
    omega_upwind = -np.angle(G_upwind)

    # Lax-Friedrichs: G = cos(kÂ·dx) - iÂ·CÂ·sin(kÂ·dx)
    G_lf = np.cos(k_dx) - 1j * C * np.sin(k_dx)
    omega_lf = -np.angle(G_lf)

    # Lax-Wendroff: G = 1 - iÂ·CÂ·sin(kÂ·dx) - CÂ²Â·(1 - cos(kÂ·dx))
    G_lw = 1 - 1j * C * np.sin(k_dx) - C**2 * (1 - np.cos(k_dx))
    omega_lw = -np.angle(G_lw)

    # ìœ„ìƒì†ë„ ë¹„êµ
    ax1 = axes[0]
    ax1.plot(k_dx, omega_exact / k_dx / C, 'k-', linewidth=2, label='ì •í™•')
    ax1.plot(k_dx, omega_upwind / k_dx / C, 'b-', linewidth=2, label='Upwind')
    ax1.plot(k_dx, omega_lf / k_dx / C, 'g-', linewidth=2, label='Lax-Friedrichs')
    ax1.plot(k_dx, omega_lw / k_dx / C, 'r-', linewidth=2, label='Lax-Wendroff')

    ax1.set_xlabel('kÎ”x')
    ax1.set_ylabel('c_num / c')
    ax1.set_title(f'ìœ„ìƒì†ë„ ì˜¤ì°¨ (C = {C})')
    ax1.legend()
    ax1.grid(True, alpha=0.3)
    ax1.set_xlim(0, np.pi)
    ax1.set_ylim(0, 1.2)

    # ì§„í­ ê°ì‡  (ì¦í­ì¸ì í¬ê¸°)
    ax2 = axes[1]
    ax2.axhline(y=1, color='k', linestyle='-', linewidth=2, label='ì •í™•')
    ax2.plot(k_dx, np.abs(G_upwind), 'b-', linewidth=2, label='Upwind')
    ax2.plot(k_dx, np.abs(G_lf), 'g-', linewidth=2, label='Lax-Friedrichs')
    ax2.plot(k_dx, np.abs(G_lw), 'r-', linewidth=2, label='Lax-Wendroff')

    ax2.set_xlabel('kÎ”x')
    ax2.set_ylabel('|G|')
    ax2.set_title(f'ì¦í­ì¸ì í¬ê¸° (ì§„í­ ê°ì‡ )')
    ax2.legend()
    ax2.grid(True, alpha=0.3)
    ax2.set_xlim(0, np.pi)
    ax2.set_ylim(0, 1.2)

    plt.tight_layout()
    plt.savefig('advection_dispersion.png', dpi=150, bbox_inches='tight')
    plt.show()

    print("ê´€ì°°:")
    print("1. Upwind, Lax-Friedrichs: í° ê°ì‡  (ìˆ˜ì¹˜ í™•ì‚°)")
    print("2. Lax-Wendroff: ì‘ì€ ê°ì‡ , ìœ„ìƒ ì˜¤ì°¨ (ìˆ˜ì¹˜ ë¶„ì‚°)")
    print("3. ëª¨ë“  ìŠ¤í‚´: ì§§ì€ íŒŒì¥(í° k)ì—ì„œ ì˜¤ì°¨ ì¦ê°€")

def courant_number_study():
    """Courant ìˆ˜ì— ë”°ë¥¸ ì•ˆì •ì„±ê³¼ ì •í™•ë„"""
    L = 4.0
    c = 1.0
    T = 1.0

    def u0(x):
        return np.exp(-(x - 1.0)**2 / 0.08)

    courant_values = [0.2, 0.5, 0.8, 0.95, 1.0, 1.05]
    errors = {'Upwind': [], 'Lax-Wendroff': []}
    stable = {'Upwind': [], 'Lax-Wendroff': []}

    for C in courant_values:
        try:
            # Upwind
            solver1 = AdvectionUpwind(L, c, nx=101, T=T, courant=C)
            solver1.set_initial_condition(u0)
            _, hist1 = solver1.solve()
            err1 = np.max(np.abs(hist1[-1] - u0(solver1.x - c * T)))
            errors['Upwind'].append(err1)
            stable['Upwind'].append(err1 < 10)

            # Lax-Wendroff
            solver2 = AdvectionLaxWendroff(L, c, nx=101, T=T, courant=C)
            solver2.set_initial_condition(u0)
            _, hist2 = solver2.solve()
            err2 = np.max(np.abs(hist2[-1] - u0(solver2.x - c * T)))
            errors['Lax-Wendroff'].append(err2)
            stable['Lax-Wendroff'].append(err2 < 10)

        except:
            errors['Upwind'].append(np.nan)
            errors['Lax-Wendroff'].append(np.nan)
            stable['Upwind'].append(False)
            stable['Lax-Wendroff'].append(False)

    # ì‹œê°í™”
    fig, ax = plt.subplots(figsize=(10, 6))

    x_pos = np.arange(len(courant_values))
    width = 0.35

    bars1 = ax.bar(x_pos - width/2, errors['Upwind'], width, label='Upwind', alpha=0.8)
    bars2 = ax.bar(x_pos + width/2, errors['Lax-Wendroff'], width, label='Lax-Wendroff', alpha=0.8)

    # ë¶ˆì•ˆì • í‘œì‹œ
    for i, C in enumerate(courant_values):
        if not stable['Upwind'][i]:
            bars1[i].set_color('red')
        if not stable['Lax-Wendroff'][i]:
            bars2[i].set_color('red')

    ax.set_xlabel('Courant ìˆ˜ C')
    ax.set_ylabel('ìµœëŒ€ ì˜¤ì°¨')
    ax.set_title('Courant ìˆ˜ì— ë”°ë¥¸ ì •í™•ë„ (ë¹¨ê°„ìƒ‰ = ë¶ˆì•ˆì •)')
    ax.set_xticks(x_pos)
    ax.set_xticklabels(courant_values)
    ax.legend()
    ax.grid(True, alpha=0.3, axis='y')

    # ì•ˆì • ì˜ì—­ í‘œì‹œ
    ax.axvline(x=4.5, color='red', linestyle='--', alpha=0.5)
    ax.text(4.7, ax.get_ylim()[1]*0.9, 'C > 1\në¶ˆì•ˆì •', color='red', fontsize=10)

    plt.tight_layout()
    plt.savefig('courant_study.png', dpi=150, bbox_inches='tight')
    plt.show()

    print("\nê²°ê³¼ ìš”ì•½:")
    for C, e1, e2 in zip(courant_values, errors['Upwind'], errors['Lax-Wendroff']):
        status = 'UNSTABLE' if C > 1 else ''
        print(f"C = {C}: Upwind = {e1:.4f}, L-W = {e2:.4f} {status}")
