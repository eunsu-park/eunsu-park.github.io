def electromagnetic_wave_1d():

    # ê³µê°„/ì‹œê°„ ì„¤ì •
    L = 10.0  # ë„ë©”ì¸ ê¸¸ì´ [m]
    T = 3 * L / c0  # ì‹œë®¬ë ˆì´ì…˜ ì‹œê°„

    x0 = L / 4

    k = 2 * np.pi / wavelength

    fig, axes = plt.subplots(2, 2, figsize=(14, 10))

    for idx, t in enumerate(times):
        ax = axes[idx // 2, idx % 2]

        # ì „ê¸°ì¥ Ey (ê°€ìš°ì‹œì•ˆ ë³€ì¡°ëœ ì‚¬ì¸íŒŒ)

        ax.axhline(y=0, color='gray', linestyle='-', alpha=0.3)

        ax.set_ylabel('Field amplitude')
        ax.set_title(f't = {t*1e9:.2f} ns')

1D FDTD ì—…ë°ì´íŠ¸ ë°©ì •ì‹:

1. H ì—…ë°ì´íŠ¸ (n â†’ n+1/2):
   Hz^(n+1/2)[i] = Hz^(n-1/2)[i] - (Î”t/Î¼â‚€Î”x)(Ey^n[i+1] - Ey^n[i])

2. E ì—…ë°ì´íŠ¸ (n+1/2 â†’ n+1):
   Ey^(n+1)[i] = Ey^n[i] - (Î”t/Îµâ‚€Î”x)(Hz^(n+1/2)[i] - Hz^(n+1/2)[i-1])

ë§¤ê°œë³€ìˆ˜í™”:
C_a = Î”t/(Î¼â‚€Î”x)  (H ê³„ìˆ˜)
C_b = Î”t/(Îµâ‚€Î”x)  (E ê³„ìˆ˜)

Hz^(n+1/2)[i] = Hz^(n-1/2)[i] - C_a (Ey^n[i+1] - Ey^n[i])

def simple_1d_fdtd():
    """ê°„ë‹¨í•œ 1D FDTD ì‹œë®¬ë ˆì´ì…˜"""

    # ê²©ì ì„¤ì •
    Nx = 200
    dx = 1e-3  # 1 mm

    # ì‹œê°„ ì„¤ì •
    dt = dx / (2 * c0)  # Courant ì¡°ê±´ ë§Œì¡±
    n_steps = 500

    # ë°°ì—´ ì´ˆê¸°í™”
    Ey = np.zeros(Nx)
    Hz = np.zeros(Nx)

    # ê³„ìˆ˜
    Ca = dt / (mu0 * dx)
    Cb = dt / (eps0 * dx)

    # Courant ìˆ˜
    S = c0 * dt / dx
    print(f"Courant ìˆ˜ S = {S:.4f}")

    # ì†ŒìŠ¤ ìœ„ì¹˜
    source_pos = Nx // 4

    # ê¸°ë¡ìš©
    Ey_history = []
    times_to_record = [0, 100, 200, 300, 400]

    # ë©”ì¸ ë£¨í”„
    for n in range(n_steps):
        # H ì—…ë°ì´íŠ¸
        Hz[:-1] = Hz[:-1] - Ca * (Ey[1:] - Ey[:-1])

        # ì†ŒìŠ¤ (ê°€ìš°ì‹œì•ˆ í„ìŠ¤)
        t = n * dt
        t0 = 30 * dt
        tau = 10 * dt
        source = np.exp(-((t - t0) / tau) ** 2)
        Ey[source_pos] += source

        # E ì—…ë°ì´íŠ¸
        Ey[1:] = Ey[1:] - Cb * (Hz[1:] - Hz[:-1])

        # ê²½ê³„ì¡°ê±´ (ê°„ë‹¨í•œ í¡ìˆ˜)
        Ey[0] = 0
        Ey[-1] = 0

        # ê¸°ë¡
        if n in times_to_record:
            Ey_history.append((n, Ey.copy()))

    # ì‹œê°í™”
    fig, axes = plt.subplots(2, 3, figsize=(15, 8))

    x = np.arange(Nx) * dx * 1000  # mm

    for idx, (n, Ey_snap) in enumerate(Ey_history):
        ax = axes[idx // 3, idx % 3]
        ax.plot(x, Ey_snap, 'b-', linewidth=1.5)
        ax.axvline(x=source_pos * dx * 1000, color='red', linestyle='--', alpha=0.5)
        ax.set_xlabel('x [mm]')
        ax.set_ylabel('Ey')
        ax.set_title(f'Step n = {n}')
        ax.grid(True, alpha=0.3)
        ax.set_ylim(-1.2, 1.2)

    # ë§ˆì§€ë§‰ subplotì— ì„¤ëª…
    ax = axes[1, 2]
    ax.text(0.5, 0.8, 'FDTD íŒŒë¼ë¯¸í„°:', fontsize=12, ha='center', transform=ax.transAxes)
    ax.text(0.5, 0.6, f'Nx = {Nx}', fontsize=10, ha='center', transform=ax.transAxes)
    ax.text(0.5, 0.5, f'dx = {dx*1000:.2f} mm', fontsize=10, ha='center', transform=ax.transAxes)
    ax.text(0.5, 0.4, f'dt = {dt*1e12:.2f} ps', fontsize=10, ha='center', transform=ax.transAxes)
    ax.text(0.5, 0.3, f'Courant S = {S:.2f}', fontsize=10, ha='center', transform=ax.transAxes)
    ax.axis('off')

    plt.suptitle('1D FDTD ì‹œë®¬ë ˆì´ì…˜ - ê°€ìš°ì‹œì•ˆ í„ìŠ¤ ì „íŒŒ', fontsize=14)
    plt.tight_layout()
    plt.savefig('fdtd_1d_simple.png', dpi=150, bbox_inches='tight')
    plt.show()

    return Ey, Hz

def fdtd_2d_tm_mode():
    """2D FDTD TM ëª¨ë“œ ì‹œë®¬ë ˆì´ì…˜"""

    # ê²©ì ì„¤ì •
    Nx, Ny = 100, 100
    dx = dy = 1e-3  # 1 mm

    # ì‹œê°„ ì„¤ì •
    dt = 1 / (c0 * np.sqrt(1/dx**2 + 1/dy**2)) * 0.99  # Courant
    n_steps = 300

    # ë¬¼ì„±ì¹˜
    eps = eps0 * np.ones((Ny, Nx))  # ìœ ì „ìœ¨
    mu = mu0 * np.ones((Ny, Nx))    # íˆ¬ììœ¨
    sigma = np.zeros((Ny, Nx))      # ì „ë„ë„

    # ë°°ì—´ ì´ˆê¸°í™”
    Ez = np.zeros((Ny, Nx))
    Hx = np.zeros((Ny, Nx))
    Hy = np.zeros((Ny, Nx))

    # ê³„ìˆ˜
    Ca = (1 - sigma * dt / (2 * eps)) / (1 + sigma * dt / (2 * eps))
    Cb = (dt / eps) / (1 + sigma * dt / (2 * eps))

    # ì†ŒìŠ¤ ìœ„ì¹˜
    source_x, source_y = Nx // 2, Ny // 2

    # Courant ìˆ˜
    S = c0 * dt * np.sqrt(1/dx**2 + 1/dy**2)
    print(f"2D Courant ìˆ˜ S = {S:.4f}")

    # ìŠ¤ëƒ…ìƒ· ê¸°ë¡
    snapshots = []
    record_steps = [50, 100, 150, 200, 250]

    # ë©”ì¸ ë£¨í”„
    for n in range(n_steps):
        # H ì—…ë°ì´íŠ¸
        Hx[:, :-1] = Hx[:, :-1] - dt / (mu[:, :-1] * dy) * (Ez[:, 1:] - Ez[:, :-1])
        Hy[:-1, :] = Hy[:-1, :] + dt / (mu[:-1, :] * dx) * (Ez[1:, :] - Ez[:-1, :])

        # ì†ŒìŠ¤ (ê°€ìš°ì‹œì•ˆ í„ìŠ¤)
        t = n * dt
        t0 = 50 * dt
        tau = 20 * dt
        source = np.exp(-((t - t0) / tau) ** 2)

        # E ì—…ë°ì´íŠ¸
        Ez[1:-1, 1:-1] = (Ca[1:-1, 1:-1] * Ez[1:-1, 1:-1] +
                         Cb[1:-1, 1:-1] * (
                             (Hy[1:-1, 1:-1] - Hy[:-2, 1:-1]) / dx -
                             (Hx[1:-1, 1:-1] - Hx[1:-1, :-2]) / dy
                         ))

        # ì†ŒìŠ¤ ì£¼ì…
        Ez[source_y, source_x] += source

        # ìŠ¤ëƒ…ìƒ· ê¸°ë¡
        if n in record_steps:
            snapshots.append((n, Ez.copy()))

    # ì‹œê°í™”
    fig, axes = plt.subplots(2, 3, figsize=(15, 10))

    x = np.arange(Nx) * dx * 1000
    y = np.arange(Ny) * dy * 1000
    X, Y = np.meshgrid(x, y)

    for idx, (step, Ez_snap) in enumerate(snapshots):
        ax = axes[idx // 3, idx % 3]
        vmax = np.max(np.abs(Ez_snap)) * 0.8
        if vmax == 0:
            vmax = 1

        im = ax.pcolormesh(X, Y, Ez_snap, cmap='RdBu_r', shading='auto',
                          vmin=-vmax, vmax=vmax)
        ax.plot(source_x * dx * 1000, source_y * dy * 1000, 'k*', markersize=10)
        ax.set_xlabel('x [mm]')
        ax.set_ylabel('y [mm]')
        ax.set_title(f'Step n = {step}')
        ax.set_aspect('equal')
        plt.colorbar(im, ax=ax, label='Ez')

    # ë§ˆì§€ë§‰ subplot
    ax = axes[1, 2]
    ax.text(0.5, 0.7, '2D FDTD TM ëª¨ë“œ', fontsize=14, ha='center', transform=ax.transAxes)
    ax.text(0.5, 0.5, f'ê²©ì: {Nx} x {Ny}', fontsize=11, ha='center', transform=ax.transAxes)

def courant_condition_analysis():
    """Courant ì¡°ê±´ ë¶„ì„"""

    fig, axes = plt.subplots(1, 2, figsize=(14, 5))

    # (1) Courant ìˆ˜ì— ë”°ë¥¸ ì•ˆì •ì„±
    ax1 = axes[0]

    # 1D FDTD ì‹œë®¬ë ˆì´ì…˜ (ë‹¤ì–‘í•œ Courant ìˆ˜)
    def run_1d_fdtd(S, n_steps=200):
        Nx = 100
        dx = 1e-3
        dt = S * dx / c0

        Ey = np.zeros(Nx)
        Hz = np.zeros(Nx)

        Ca = dt / (mu0 * dx)
        Cb = dt / (eps0 * dx)

        max_values = []

        for n in range(n_steps):
            Hz[:-1] = Hz[:-1] - Ca * (Ey[1:] - Ey[:-1])

            if n == 0:
                Ey[Nx//4] = 1.0  # ì´ˆê¸° í„ìŠ¤

            Ey[1:] = Ey[1:] - Cb * (Hz[1:] - Hz[:-1])
            Ey[0] = Ey[-1] = 0

            max_values.append(np.max(np.abs(Ey)))

        return max_values

    courant_numbers = [0.5, 0.9, 1.0, 1.01, 1.1]
    colors = ['green', 'blue', 'orange', 'red', 'darkred']

    for S, color in zip(courant_numbers, colors):
        max_vals = run_1d_fdtd(S)
        label = f'S = {S}' + (' (ì•ˆì •)' if S <= 1 else ' (ë¶ˆì•ˆì •)')
        ax1.semilogy(max_vals, color=color, linewidth=1.5, label=label)

    ax1.axhline(y=1, color='gray', linestyle='--', alpha=0.5)
    ax1.set_xlabel('Time step')
    ax1.set_ylabel('max|Ey|')
    ax1.set_title('1D FDTD: Courant ìˆ˜ì— ë”°ë¥¸ ì•ˆì •ì„±')
    ax1.legend()
    ax1.grid(True, alpha=0.3)
    ax1.set_ylim(1e-3, 1e10)

    # (2) ë¶„ì‚° ê´€ê³„
    ax2 = axes[1]

    # FDTD ìˆ˜ì¹˜ ë¶„ì‚° ê´€ê³„
    # Ï‰_num = (2/Î”t) arcsin(S sin(kÎ”x/2))
    # í•´ì„í•´: Ï‰ = ck

    k_norm = np.linspace(0, np.pi, 100)  # kÎ”x

    for S in [0.5, 0.8, 1.0]:
        omega_exact = k_norm / S  # ì •ê·œí™”ëœ Ï‰Î”t
        omega_fdtd = 2 * np.arcsin(S * np.sin(k_norm / 2))

        # ìœ„ìƒ ì†ë„ ë¹„ìœ¨
        vp_ratio = omega_fdtd / omega_exact

        ax2.plot(k_norm / np.pi, vp_ratio, linewidth=2, label=f'S = {S}')

    ax2.axhline(y=1, color='black', linestyle='--', alpha=0.5, label='Exact')
    ax2.set_xlabel(r'$k\Delta x / \pi$')
    ax2.set_ylabel(r'$v_p^{num} / c$')
    ax2.set_title('FDTD ìˆ˜ì¹˜ ë¶„ì‚° (ìœ„ìƒ ì†ë„ ë¹„)')
    ax2.legend()
    ax2.grid(True, alpha=0.3)
    ax2.set_xlim(0, 1)
    ax2.set_ylim(0.8, 1.05)

    plt.tight_layout()
    plt.savefig('courant_condition.png', dpi=150, bbox_inches='tight')
    plt.show()
