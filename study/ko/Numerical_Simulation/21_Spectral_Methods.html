def fft_performance_test():
    """
    ë‚˜ì´ë¸Œ DFT vs FFT ì„±ëŠ¥ ë¹„êµ.
    """

            for j in range(N):
                u_hat[k] += u[j] * np.exp(-2j * np.pi * k * j / N)

    sizes = [32, 64, 128, 256, 512]

    fft_times = []

        # ë‚˜ì´ë¸Œ DFT

    plt.ylabel('Time (s)')
    plt.title('DFT vs FFT Performance')
    plt.legend()
    plt.grid(True, which='both', alpha=0.3)
    plt.tight_layout()
    plt.savefig('fft_performance.png', dpi=150)
    plt.close()

def chebyshev_diff_matrix(N):
    """
    ì²´ë¹„ì…°í”„ ë¯¸ë¶„ í–‰ë ¬ D ê³„ì‚°.

    u'(xâ±¼) â‰ˆ Î£ Dâ±¼â‚– u(xâ‚–)
    """
    x = chebyshev_points(N)
    D = np.zeros((N+1, N+1))

    c = np.ones(N+1)
    c[0] = 2
    c[N] = 2

    for i in range(N+1):
        for j in range(N+1):
            if i != j:
                D[i, j] = (c[i] / c[j]) * ((-1)**(i+j)) / (x[i] - x[j])

    # ëŒ€ê° ì›ì†Œ
    for i in range(N+1):
        D[i, i] = -np.sum(D[i, :])

    return D, x

def dealias_product_3_2_rule(u, v):
    """
    3/2 ê·œì¹™ì„ ì‚¬ìš©í•˜ì—¬ ë””ì—ì¼ë¦¬ì–´ìŠ¤ëœ ê³± w = u*v ê³„ì‚°.

    Parameters:
    -----------
    u, v : ê¸¸ì´ Nì˜ ë°°ì—´

    Returns:
    --------
    w : ë””ì—ì¼ë¦¬ì–´ìŠ¤ëœ ê³± (ê¸¸ì´ N)
    """
    N = len(u)
    M = 3 * N // 2  # í™•ì¥ ê²©ì

    # í‘¸ë¦¬ì— ë³€í™˜
    u_hat = fft(u)
    v_hat = fft(v)

    # 3N/2ë¡œ ì œë¡œ íŒ¨ë”©
    u_hat_padded = np.zeros(M, dtype=complex)
    v_hat_padded = np.zeros(M, dtype=complex)

    # ì €ì£¼íŒŒ
    u_hat_padded[:N//2] = u_hat[:N//2]
    u_hat_padded[-N//2:] = u_hat[-N//2:]

    v_hat_padded[:N//2] = v_hat[:N//2]
    v_hat_padded[-N//2:] = v_hat[-N//2:]

    # í™•ì¥ ê²©ìë¡œ ì—­ë³€í™˜
    u_extended = ifft(u_hat_padded)
    v_extended = ifft(v_hat_padded)

    # ë¬¼ë¦¬ ê³µê°„ì—ì„œ ê³±ì…ˆ
    w_extended = u_extended * v_extended

    # ë‹¤ì‹œ ë³€í™˜
    w_hat_extended = fft(w_extended)

    # ì›ë˜ í•´ìƒë„ë¡œ ì˜ë¼ë‚´ê¸°
    w_hat = np.zeros(N, dtype=complex)
    w_hat[:N//2] = w_hat_extended[:N//2]
    w_hat[-N//2:] = w_hat_extended[-N//2:]

    # ìŠ¤ì¼€ì¼ (í™•ì¥ ê²©ì ê³ ë ¤)
    w_hat *= M / N

    w = ifft(w_hat)
    return np.real(w)

def burgers_spectral(nu=0.01, T=2.0, N=128, dt=0.001):
    """
    ë””ì—ì¼ë¦¬ì–´ì‹±ì„ ì‚¬ìš©í•œ í‘¸ë¦¬ì— ìŠ¤í™íŠ¸ëŸ¼ ë°©ë²•ìœ¼ë¡œ ë²„ê±°ìŠ¤ ë°©ì •ì‹ í’€ê¸°.

    âˆ‚u/âˆ‚t + u âˆ‚u/âˆ‚x = Î½ âˆ‚Â²u/âˆ‚xÂ²
    """
    # ê²©ì
    L = 2 * np.pi
    x = np.linspace(0, L, N, endpoint=False)
    k = fftfreq(N, L/N) * 2 * np.pi

    # ì´ˆê¸° ì¡°ê±´
    u = np.sin(x)

    # ì‹œê°„ ì ë¶„ (RK4)
    nt = int(T / dt)
    time = 0.0

    # ì„ íƒí•œ ì‹œê°„ì— í•´ ì €ì¥
    t_save = [0, 0.5, 1.0, 1.5, 2.0]
    u_save = []

    def rhs(u):
        """ìŠ¤í™íŠ¸ëŸ¼ ê³µê°„ì—ì„œ ë²„ê±°ìŠ¤ ë°©ì •ì‹ì˜ ìš°ë³€."""
        u_hat = fft(u)

        # ì„ í˜• í•­: Î½ âˆ‚Â²u/âˆ‚xÂ²
        linear = -nu * k**2 * u_hat

        # ë¹„ì„ í˜• í•­: -u âˆ‚u/âˆ‚x (ë””ì—ì¼ë¦¬ì–´ì‹±ìœ¼ë¡œ ë¬¼ë¦¬ ê³µê°„ì—ì„œ ê³„ì‚°)
        ux = np.real(ifft(1j * k * u_hat))
        nonlinear_phys = -u * ux
        nonlinear_phys = dealias_product_3_2_rule(u, ux)
        nonlinear = fft(nonlinear_phys)

        du_dt_hat = linear + nonlinear
        return np.real(ifft(du_dt_hat))

    for n in range(nt):
        # RK4 ì‹œê°„ ì§„ì „
        k1 = rhs(u)
        k2 = rhs(u + 0.5*dt*k1)
        k3 = rhs(u + 0.5*dt*k2)
        k4 = rhs(u + dt*k3)

        u = u + (dt/6) * (k1 + 2*k2 + 2*k3 + k4)
        time += dt

        # í•´ ì €ì¥
        if any(abs(time - t) < dt/2 for t in t_save):
            u_save.append(u.copy())

    # í”Œë¡¯
    plt.figure(figsize=(10, 6))
    for i, t in enumerate(t_save[:len(u_save)]):
        plt.plot(x, u_save[i], label=f't = {t:.1f}', linewidth=2)
    plt.xlabel('x')
    plt.ylabel('u(x,t)')
    plt.title(f'Burgers Equation (Î½={nu})')
    plt.legend()
    plt.grid(True)
    plt.tight_layout()
    plt.savefig('burgers_spectral.png', dpi=150)
    plt.close()

    return x, u_save
