class ReadWriteLock:
    def __init__(self):
        self.readers = 0
        self.lock = RLock()
        self.write_lock = RLock()

    def acquire_read(self):
        with self.lock:

            if self.readers == 1:

    def release_write(self):
        self.write_lock.release()
