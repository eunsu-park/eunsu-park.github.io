import numpy as np

class EpsilonGreedy:
    """
    Îµ-íƒìš• ì „ëµ: ê°€ì¥ ê¸°ë³¸ì ì¸ íƒí—˜-í™œìš© ê· í˜• ë°©ë²•
    """
    def __init__(self, n_actions, epsilon=0.1):
        self.n_actions = n_actions
        self.epsilon = epsilon
        self.q_values = np.zeros(n_actions)  # ê° í–‰ë™ì˜ ì¶”ì • ê°€ì¹˜
        self.action_counts = np.zeros(n_actions)

    def select_action(self):
        """
        í™•ë¥  Îµë¡œ ëœë¤ í–‰ë™ (íƒí—˜)
        í™•ë¥  1-Îµë¡œ ìµœì„  í–‰ë™ (í™œìš©)
        """
        if np.random.random() < self.epsilon:
            # íƒí—˜: ëœë¤ í–‰ë™
            return np.random.randint(self.n_actions)
        else:
            # í™œìš©: ìµœê³  ê°€ì¹˜ í–‰ë™
            return np.argmax(self.q_values)

    def update(self, action, reward):
        """í–‰ë™ ê°€ì¹˜ ì—…ë°ì´íŠ¸ (ì ì§„ì  í‰ê· )"""
        self.action_counts[action] += 1
        n = self.action_counts[action]
        self.q_values[action] += (reward - self.q_values[action]) / n

class ExplorationStrategies:

        if np.random.random() < epsilon:
            return np.random.randint(len(q_values))
        return np.argmax(q_values)

    @staticmethod
    def softmax(q_values, temperature=1.0):
        """
        Softmax (Boltzmann) íƒí—˜
        - temperature ë†’ìŒ: ë” ë§ì€ íƒí—˜
        - temperature ë‚®ìŒ: ë” ë§ì€ í™œìš©

        return np.random.choice(len(q_values), p=probs)

    @staticmethod
