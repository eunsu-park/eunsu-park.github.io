import numpy as np
from typing import Dict, List, Tuple

def policy_evaluation(mdp, policy: Dict, gamma: float = 0.9,
                      theta: float = 1e-6) -> Dict:
    """
    ì •ì±… í‰ê°€: ì£¼ì–´ì§„ ì •ì±…ì˜ ê°€ì¹˜ í•¨ìˆ˜ ê³„ì‚°

    Args:
        mdp: MDP í™˜ê²½
        policy: ì •ì±… {state: {action: probability}}
        gamma: í• ì¸ìœ¨
        theta: ìˆ˜ë ´ ì„ê³„ê°’

    Returns:
        V: ìƒíƒœ ê°€ì¹˜ í•¨ìˆ˜ {state: value}
    """
    # ê°€ì¹˜ í•¨ìˆ˜ ì´ˆê¸°í™”
    V = {s: 0.0 for s in mdp.get_states()}

    iteration = 0
    while True:
        delta = 0  # ìµœëŒ€ ë³€í™”ëŸ‰ ì¶”ì 
        iteration += 1

        # ëª¨ë“  ìƒíƒœì— ëŒ€í•´ ì—…ë°ì´íŠ¸
        for s in mdp.get_states():
            if mdp.is_terminal(s):
                continue

            v = V[s]  # ì´ì „ ê°’ ì €ì¥
            new_v = 0

            # ë²¨ë§Œ ê¸°ëŒ€ ë°©ì •ì‹ ì ìš©
            for a in mdp.actions:
                action_prob = policy[s].get(a, 0)

                for prob, next_s, reward, done in mdp.get_transitions(s, a):
                    if done:
                        new_v += action_prob * prob * reward
                    else:
                        new_v += action_prob * prob * (reward + gamma * V[next_s])

            V[s] = new_v
            delta = max(delta, abs(v - new_v))

        # ìˆ˜ë ´ ì²´í¬
        if delta < theta:
            print(f"ì •ì±… í‰ê°€ ìˆ˜ë ´: {iteration} iterations")
            break

    return V

class SimpleGridWorld:
    """ê°„ë‹¨í•œ ê·¸ë¦¬ë“œ ì›”ë“œ"""
    def __init__(self, size=4):
        self.size = size
        self.actions = ['up', 'down', 'left', 'right']

    def get_states(self):
        return [(i, j) for i in range(self.size) for j in range(self.size)]

    def is_terminal(self, state):
        return state == (0, 0) or state == (self.size-1, self.size-1)

    def get_transitions(self, state, action):
        if self.is_terminal(state):
            return [(1.0, state, 0, True)]

        deltas = {'up': (-1, 0), 'down': (1, 0), 'left': (0, -1), 'right': (0, 1)}
        delta = deltas[action]

        new_row = max(0, min(self.size-1, state[0] + delta[0]))
        new_col = max(0, min(self.size-1, state[1] + delta[1]))
        next_state = (new_row, new_col)

        return [(1.0, next_state, -1, self.is_terminal(next_state))]

def create_uniform_policy(mdp):
    policy = {}
    for s in mdp.get_states():
        policy[s] = {a: 1/len(mdp.actions) for a in mdp.actions}
    return policy

def policy_improvement(mdp, V: Dict, gamma: float = 0.9) -> Tuple[Dict, bool]:
    """
    ì •ì±… ê°œì„ : Vë¥¼ ê¸°ë°˜ìœ¼ë¡œ íƒìš•ì  ì •ì±… ìƒì„±

    Args:
        mdp: MDP í™˜ê²½
        V: í˜„ì¬ ê°€ì¹˜ í•¨ìˆ˜
        gamma: í• ì¸ìœ¨

    Returns:
        new_policy: ê°œì„ ëœ ì •ì±…
        stable: ì •ì±…ì´ ë³€í•˜ì§€ ì•Šì•˜ìœ¼ë©´ True
    """
    new_policy = {}
    policy_stable = True

    for s in mdp.get_states():
        if mdp.is_terminal(s):
            new_policy[s] = {a: 1/len(mdp.actions) for a in mdp.actions}
            continue

        # ê° í–‰ë™ì˜ Q ê°’ ê³„ì‚°
        q_values = {}
        for a in mdp.actions:
            q = 0
            for prob, next_s, reward, done in mdp.get_transitions(s, a):
                if done:
                    q += prob * reward
                else:
                    q += prob * (reward + gamma * V[next_s])
            q_values[a] = q

        # ìµœì  í–‰ë™ ì°¾ê¸°
        best_action = max(q_values, key=q_values.get)
        best_q = q_values[best_action]

        # ë™ë¥ ì¸ í–‰ë™ë“¤ ì°¾ê¸° (ìˆ˜ì¹˜ ì˜¤ì°¨ ê³ ë ¤)
        best_actions = [a for a, q in q_values.items()
                        if abs(q - best_q) < 1e-8]

        # ê²°ì •ì  ì •ì±… ìƒì„± (ë˜ëŠ” ë™ë¥  ì‹œ ê· ë“± ë¶„ë°°)
        new_policy[s] = {a: 0.0 for a in mdp.actions}
        for a in best_actions:
            new_policy[s][a] = 1.0 / len(best_actions)

    return new_policy, policy_stable


def get_greedy_action(mdp, V, state, gamma):
    """ê°€ì¹˜ í•¨ìˆ˜ì—ì„œ íƒìš•ì  í–‰ë™ ì„ íƒ"""

def policy_iteration(mdp, gamma: float = 0.9, theta: float = 1e-6):
    """
    ì •ì±… ë°˜ë³µ ì•Œê³ ë¦¬ì¦˜

    Returns:
        V: ìµœì  ê°€ì¹˜ í•¨ìˆ˜
        policy: ìµœì  ì •ì±…
    """

    iteration = 0

        iteration += 1
        print(f"\n=== ì •ì±… ë°˜ë³µ {iteration} ===")

        # 1. ì •ì±… í‰ê°€
        V = policy_evaluation(mdp, policy, gamma, theta)

        # 2. ì •ì±… ê°œì„

                continue

            # Q ê°’ ê³„ì‚°
            q_values = {}
            for a in mdp.actions:
                q = 0
                for prob, next_s, reward, done in mdp.get_transitions(s, a):

                        q += prob * reward
                    else:
                        q += prob * (reward + gamma * V[next_s])
                q_values[a] = q

            best_action = max(q_values, key=q_values.get)

        if policy_stable:

def value_iteration(mdp, gamma: float = 0.9, theta: float = 1e-6):
    """
    ê°€ì¹˜ ë°˜ë³µ ì•Œê³ ë¦¬ì¦˜

    Returns:
        V: ìµœì  ê°€ì¹˜ í•¨ìˆ˜
        policy: ìµœì  ì •ì±…
    """
    # ê°€ì¹˜ í•¨ìˆ˜ ì´ˆê¸°í™”
    V = {s: 0.0 for s in mdp.get_states()}

    iteration = 0
    while True:
        delta = 0
        iteration += 1

        for s in mdp.get_states():
            if mdp.is_terminal(s):
                continue

            v = V[s]

            # ë²¨ë§Œ ìµœì ì„± ë°©ì •ì‹: max over actions
            q_values = []
            for a in mdp.actions:
                q = 0
                for prob, next_s, reward, done in mdp.get_transitions(s, a):
                    if done:
                        q += prob * reward
                    else:
                        q += prob * (reward + gamma * V[next_s])
                q_values.append(q)

            V[s] = max(q_values)
            delta = max(delta, abs(v - V[s]))

        if iteration % 10 == 0:
            print(f"ë°˜ë³µ {iteration}: delta = {delta:.6f}")

        if delta < theta:
            print(f"\nê°€ì¹˜ ë°˜ë³µ ìˆ˜ë ´: {iteration} iterations")
            break

    # ìµœì  ì •ì±… ì¶”ì¶œ
    policy = {}
    for s in mdp.get_states():
        if mdp.is_terminal(s):
            policy[s] = {a: 1/len(mdp.actions) for a in mdp.actions}
            continue

        q_values = {}
        for a in mdp.actions:
            q = 0
            for prob, next_s, reward, done in mdp.get_transitions(s, a):
                if done:
                    q += prob * reward
                else:
                    q += prob * (reward + gamma * V[next_s])
            q_values[a] = q

        best_action = max(q_values, key=q_values.get)
        policy[s] = {a: 0.0 for a in mdp.actions}
        policy[s][best_action] = 1.0

    return V, policy

def async_value_iteration(mdp, gamma=0.9, n_updates=10000):
    """ë¹„ë™ê¸° ê°€ì¹˜ ë°˜ë³µ"""
    V = {s: 0.0 for s in mdp.get_states()}
    states = [s for s in mdp.get_states() if not mdp.is_terminal(s)]

    for i in range(n_updates):
        # ë¬´ì‘ìœ„ë¡œ ìƒíƒœ ì„ íƒ
        s = np.random.choice(len(states))
        s = states[s]

        # í•´ë‹¹ ìƒíƒœë§Œ ì—…ë°ì´íŠ¸
        q_values = []
        for a in mdp.actions:
            q = 0
            for prob, next_s, reward, done in mdp.get_transitions(s, a):
                if done:
                    q += prob * reward
                else:
                    q += prob * (reward + gamma * V[next_s])
            q_values.append(q)

        V[s] = max(q_values)

    return V

import gymnasium as gym
import numpy as np

def dp_frozen_lake():
    """Frozen Lake í™˜ê²½ì—ì„œ DP ì ìš©"""

    # í™˜ê²½ ìƒì„± (ë¯¸ë„ëŸ¬ì§€ì§€ ì•ŠëŠ” ë²„ì „)
    env = gym.make('FrozenLake-v1', is_slippery=False)

    n_states = env.observation_space.n
    n_actions = env.action_space.n
    gamma = 0.99
    theta = 1e-8

    # P[s][a] = [(prob, next_state, reward, done), ...]
    P = env.unwrapped.P

    # ê°€ì¹˜ ë°˜ë³µ
    V = np.zeros(n_states)

    for iteration in range(1000):
        delta = 0

        for s in range(n_states):
            v = V[s]

            # ê° í–‰ë™ì˜ ê°€ì¹˜ ê³„ì‚°
            q_values = []
            for a in range(n_actions):
                q = sum(prob * (reward + gamma * V[next_s] * (not done))
                        for prob, next_s, reward, done in P[s][a])
                q_values.append(q)

            V[s] = max(q_values)
            delta = max(delta, abs(v - V[s]))

        if delta < theta:
            print(f"ìˆ˜ë ´: {iteration + 1} iterations")
            break

    # ìµœì  ì •ì±… ì¶”ì¶œ
    policy = np.zeros(n_states, dtype=int)
    for s in range(n_states):
        q_values = []
        for a in range(n_actions):
            q = sum(prob * (reward + gamma * V[next_s] * (not done))
                    for prob, next_s, reward, done in P[s][a])
            q_values.append(q)
        policy[s] = np.argmax(q_values)

    # ê²°ê³¼ ì‹œê°í™”
    action_names = ['â†', 'â†“', 'â†’', 'â†‘']
    print("\nìµœì  ì •ì±… (4x4 ê·¸ë¦¬ë“œ):")
    for i in range(4):
        row = ""
        for j in range(4):
            s = i * 4 + j
            if s in [5, 7, 11, 12]:  # êµ¬ë©
                row += "  H  "
            elif s == 15:  # ëª©í‘œ
                row += "  G  "
            else:
                row += f"  {action_names[policy[s]]}  "
        print(row)

    print("\nê°€ì¹˜ í•¨ìˆ˜:")
    print(V.reshape(4, 4).round(3))

    # ì •ì±… í…ŒìŠ¤íŠ¸
    env = gym.make('FrozenLake-v1', is_slippery=False)
    success = 0
    n_tests = 100

    for _ in range(n_tests):
        state, _ = env.reset()
        done = False

        while not done:
            action = policy[state]
            state, reward, terminated, truncated, _ = env.step(action)
            done = terminated or truncated

        if reward > 0:
            success += 1

    print(f"\nì„±ê³µë¥ : {success}/{n_tests} = {success/n_tests*100:.1f}%")

    env.close()
    return V, policy
