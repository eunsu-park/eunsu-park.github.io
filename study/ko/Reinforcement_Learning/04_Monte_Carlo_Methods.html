def calculate_returns(episode, gamma=0.99):
    """
    ì—í”¼ì†Œë“œì—ì„œ ê° ìƒíƒœì˜ ë¦¬í„´ ê³„ì‚°

    Args:
        episode: [(state, action, reward), ...] í˜•íƒœì˜ ë¦¬ìŠ¤íŠ¸
        gamma: í• ì¸ìœ¨

    Returns:
        returns: {state: return} ë”•ì…”ë„ˆë¦¬
    """

        G = reward + gamma * G  # í• ì¸ëœ ë¦¬í„´
        returns[t] = (state, G)

    return returns

import numpy as np
from collections import defaultdict

def first_visit_mc_prediction(env, policy, n_episodes=10000, gamma=0.99):
    """
    First-visit MC ì •ì±… í‰ê°€

    Args:
        env: Gymnasium í™˜ê²½
        policy: ì •ì±… í•¨ìˆ˜ policy(state) -> action
        n_episodes: ì—í”¼ì†Œë“œ ìˆ˜
        gamma: í• ì¸ìœ¨

    Returns:
        V: ìƒíƒœ ê°€ì¹˜ í•¨ìˆ˜
    """
    # ê° ìƒíƒœì˜ ë¦¬í„´ í•©ê³¼ ë°©ë¬¸ íšŸìˆ˜
    returns_sum = defaultdict(float)
    returns_count = defaultdict(int)

        # ì—í”¼ì†Œë“œ ìƒì„±
        episode = []
        state, _ = env.reset()
        done = False

        while not done:

            next_state, reward, terminated, truncated, _ = env.step(action)
            episode.append((state, action, reward))
            state = next_state
            done = terminated or truncated

        if (episode_num + 1) % 1000 == 0:
            print(f"Episode {episode_num + 1}/{n_episodes}")

    return dict(V)

ëª¨ë“  ìƒíƒœ-í–‰ë™ ìŒì—ì„œ ì—í”¼ì†Œë“œê°€ ì‹œì‘í•  ìˆ˜ ìˆë‹¤ê³  ê°€ì •í•©ë‹ˆë‹¤.

def mc_exploring_starts(env, n_episodes=100000, gamma=0.99):
    """
    MC with Exploring Starts

    ëª¨ë“  (s, a) ìŒì´ ì‹œì‘ì ì´ ë  ìˆ˜ ìˆìŒì„ ê°€ì •
    """
    n_states = env.observation_space.n
    n_actions = env.action_space.n

    # Q í•¨ìˆ˜ì™€ ë°©ë¬¸ íšŸìˆ˜ ì´ˆê¸°í™”
    Q = defaultdict(lambda: np.zeros(n_actions))
    returns_sum = defaultdict(lambda: np.zeros(n_actions))
    returns_count = defaultdict(lambda: np.zeros(n_actions))

    # ì •ì±… (íƒìš•ì )
    policy = defaultdict(lambda: 0)

    for episode_num in range(n_episodes):
        # Exploring Starts: ëœë¤í•œ ìƒíƒœì™€ í–‰ë™ìœ¼ë¡œ ì‹œì‘
        start_state = env.observation_space.sample()
        start_action = env.action_space.sample()

        # ì—í”¼ì†Œë“œ ìƒì„±
        episode = []
        state = start_state
        action = start_action

        # ì²« ìŠ¤í…
        env.reset()
        env.unwrapped.s = state  # ìƒíƒœ ê°•ì œ ì„¤ì • (í™˜ê²½ì— ë”°ë¼ ë‹¤ë¦„)
        next_state, reward, terminated, truncated, _ = env.step(action)
        episode.append((state, action, reward))
        state = next_state
        done = terminated or truncated

        # ë‚˜ë¨¸ì§€ ì—í”¼ì†Œë“œ
        while not done:
            action = policy[state]
            next_state, reward, terminated, truncated, _ = env.step(action)
            episode.append((state, action, reward))
            state = next_state
            done = terminated or truncated

        # ë¦¬í„´ ê³„ì‚° ë° Q ì—…ë°ì´íŠ¸
        G = 0
        visited = set()

        for t in range(len(episode) - 1, -1, -1):
            state_t, action_t, reward_t = episode[t]
            G = gamma * G + reward_t

            if (state_t, action_t) not in visited:
                visited.add((state_t, action_t))
                returns_sum[state_t][action_t] += G
                returns_count[state_t][action_t] += 1
                Q[state_t][action_t] = (returns_sum[state_t][action_t] /
                                        returns_count[state_t][action_t])

                # ì •ì±… ê°œì„  (íƒìš•ì )
                policy[state_t] = np.argmax(Q[state_t])

    return Q, policy

4.2 Îµ-greedy ì •ì±…¶

Exploring StartsëŠ” í˜„ì‹¤ì ì´ì§€ ì•Šìœ¼ë¯€ë¡œ, Îµ-greedy ì •ì±…ì„ ì‚¬ìš©í•©ë‹ˆë‹¤.

$$\pi(a|s) = \begin{cases} 1 - \epsilon + \frac{\epsilon}{|A|} & \text{if } a = \arg\max_{a'} Q(s, a') \\ \frac{\epsilon}{|A|} & \text{otherwise} \end{cases}$$

def epsilon_greedy_policy(Q, state, n_actions, epsilon=0.1):
    """
    Îµ-íƒìš• í–‰ë™ ì„ íƒ

    Args:
        Q: í–‰ë™ ê°€ì¹˜ í•¨ìˆ˜
        state: í˜„ì¬ ìƒíƒœ
        n_actions: í–‰ë™ ìˆ˜
        epsilon: íƒí—˜ í™•ë¥ 

    Returns:
        action: ì„ íƒëœ í–‰ë™
    """
    if np.random.random() < epsilon:
        # íƒí—˜: ëœë¤ í–‰ë™
        return np.random.randint(n_actions)
    else:
        # í™œìš©: ìµœì„ ì˜ í–‰ë™
        return np.argmax(Q[state])

def mc_on_policy_control(env, n_episodes=100000, gamma=0.99,
                         epsilon=0.1, epsilon_decay=0.9999):
    """
    On-policy MC ì œì–´ (Îµ-greedy)

    Args:
        env: Gymnasium í™˜ê²½
        n_episodes: ì—í”¼ì†Œë“œ ìˆ˜
        gamma: í• ì¸ìœ¨
        epsilon: íƒí—˜ìœ¨
        epsilon_decay: epsilon ê°ì†Œìœ¨

    Returns:
        Q: í–‰ë™ ê°€ì¹˜ í•¨ìˆ˜
        policy: í•™ìŠµëœ ì •ì±…
    """
    n_actions = env.action_space.n

    Q = defaultdict(lambda: np.zeros(n_actions))
    returns_sum = defaultdict(lambda: np.zeros(n_actions))
    returns_count = defaultdict(lambda: np.zeros(n_actions))

    episode_rewards = []

    for episode_num in range(n_episodes):
        episode = []
        state, _ = env.reset()
        done = False
        total_reward = 0

        # Îµ-greedy ì •ì±…ìœ¼ë¡œ ì—í”¼ì†Œë“œ ìƒì„±
        while not done:
            action = epsilon_greedy_policy(Q, state, n_actions, epsilon)
            next_state, reward, terminated, truncated, _ = env.step(action)

            episode.append((state, action, reward))
            total_reward += reward

            state = next_state
            done = terminated or truncated

        episode_rewards.append(total_reward)

        # Q ì—…ë°ì´íŠ¸ (First-visit)
        G = 0
        visited = set()

        for t in range(len(episode) - 1, -1, -1):
            state_t, action_t, reward_t = episode[t]
            G = gamma * G + reward_t

            if (state_t, action_t) not in visited:
                visited.add((state_t, action_t))
                returns_sum[state_t][action_t] += G
                returns_count[state_t][action_t] += 1
                Q[state_t][action_t] = (returns_sum[state_t][action_t] /
                                        returns_count[state_t][action_t])

        # epsilon ê°ì†Œ
        epsilon = max(0.01, epsilon * epsilon_decay)

        if (episode_num + 1) % 10000 == 0:
            avg_reward = np.mean(episode_rewards[-1000:])
            print(f"Episode {episode_num + 1}: avg_reward = {avg_reward:.3f}, "
                  f"epsilon = {epsilon:.4f}")

    # ìµœì¢… íƒìš•ì  ì •ì±…
    policy = {}
    for state in Q:
        policy[state] = np.argmax(Q[state])

    return dict(Q), policy, episode_rewards

5.1 Off-policy í•™ìŠµì´ë€?¶

  • On-policy: í–‰ë™ ì •ì±… = ëª©í‘œ ì •ì±… (ê°™ì€ ì •ì±…ìœ¼ë¡œ íƒí—˜í•˜ê³  í•™ìŠµ)
  • Off-policy: í–‰ë™ ì •ì±… â‰  ëª©í‘œ ì •ì±… (ë‹¤ë¥¸ ì •ì±…ìœ¼ë¡œ íƒí—˜í•˜ê³  í•™ìŠµ)

5.2 ì¤‘ìš”ë„ ìƒ˜í”Œë§ (Importance Sampling)¶

$$\mathbb{E}_b[X] = \sum_x x \cdot b(x) = \sum_x x \cdot \frac{\pi(x)}{b(x)} \cdot b(x) = \mathbb{E}_b\left[\frac{\pi(X)}{b(X)} X\right]$$

ì¤‘ìš”ë„ ìƒ˜í”Œë§ ë¹„ìœ¨: $$\rho_{t:T-1} = \prod_{k=t}^{T-1} \frac{\pi(A_k | S_k)}{b(A_k | S_k)}$$

def importance_sampling_ratio(episode, target_policy, behavior_policy, t):
    """
    ì¤‘ìš”ë„ ìƒ˜í”Œë§ ë¹„ìœ¨ ê³„ì‚°

    Ï = Ï€(aâ‚€|sâ‚€)/b(aâ‚€|sâ‚€) Ã— Ï€(aâ‚|sâ‚)/b(aâ‚|sâ‚) Ã— ...
    """
    rho = 1.0

    for k in range(t, len(episode)):
        state, action, _ = episode[k]
        pi_prob = target_policy(state, action)  # Ï€(a|s)
        b_prob = behavior_policy(state, action)  # b(a|s)

        if b_prob == 0:
            return 0  # í–‰ë™ ì •ì±…ì—ì„œ ë¶ˆê°€ëŠ¥í•œ í–‰ë™

        rho *= pi_prob / b_prob

    return rho

def mc_off_policy_prediction(env, target_policy, behavior_policy,
                              n_episodes=100000, gamma=0.99):
    """
    Off-policy MC ì •ì±… í‰ê°€ (Weighted Importance Sampling)

    Args:
        target_policy: ëª©í‘œ ì •ì±… (ê²°ì •ì ) - Ï€(s) -> a
        behavior_policy: í–‰ë™ ì •ì±… - b(s) -> a (Îµ-greedy ë“±)
    """
    Q = defaultdict(lambda: np.zeros(env.action_space.n))
    C = defaultdict(lambda: np.zeros(env.action_space.n))  # ê°€ì¤‘ì¹˜ í•©

    for episode_num in range(n_episodes):
        # í–‰ë™ ì •ì±…ìœ¼ë¡œ ì—í”¼ì†Œë“œ ìƒì„±
        episode = []
        state, _ = env.reset()
        done = False

        while not done:
            action = behavior_policy(state)
            next_state, reward, terminated, truncated, _ = env.step(action)
            episode.append((state, action, reward))
            state = next_state
            done = terminated or truncated

        G = 0
        W = 1.0  # ì¤‘ìš”ë„ ìƒ˜í”Œë§ ê°€ì¤‘ì¹˜

        # ì—­ìˆœ ì²˜ë¦¬
        for t in range(len(episode) - 1, -1, -1):
            state_t, action_t, reward_t = episode[t]
            G = gamma * G + reward_t

            # ê°€ì¤‘ ì¤‘ìš”ë„ ìƒ˜í”Œë§ ì—…ë°ì´íŠ¸
            C[state_t][action_t] += W
            Q[state_t][action_t] += W / C[state_t][action_t] * (G - Q[state_t][action_t])

            # ëª©í‘œ ì •ì±…ì—ì„œì˜ í–‰ë™
            target_action = target_policy(state_t)

            # í–‰ë™ì´ ëª©í‘œ ì •ì±…ê³¼ ë‹¤ë¥´ë©´ ì¤‘ë‹¨ (ê²°ì •ì  ì •ì±…ì˜ ê²½ìš°)
            if action_t != target_action:
                break

            # ì¤‘ìš”ë„ ë¹„ìœ¨ ì—…ë°ì´íŠ¸
            # Ï€(a|s) = 1 (ê²°ì •ì ), b(a|s) = í–‰ë™ ì •ì±… í™•ë¥ 
            b_prob = behavior_policy_prob(state_t, action_t)  # êµ¬í˜„ í•„ìš”
            W = W * 1.0 / b_prob

    return dict(Q)

def learn_blackjack(n_episodes=500000, gamma=1.0, epsilon=0.1):
    """ë¸”ë™ì­ì—ì„œ MC ì œì–´"""

    env = gym.make('Blackjack-v1', sab=True)

    Q = defaultdict(lambda: np.zeros(2))
    returns_sum = defaultdict(lambda: np.zeros(2))
    returns_count = defaultdict(lambda: np.zeros(2))

    def get_action(state, Q, epsilon):
        if np.random.random() < epsilon:
            return env.action_space.sample()
        return np.argmax(Q[state])

    wins = 0
    for ep in range(n_episodes):
        episode = []
        state, _ = env.reset()
        done = False

        while not done:
            action = get_action(state, Q, epsilon)
            next_state, reward, terminated, truncated, _ = env.step(action)
            episode.append((state, action, reward))
            state = next_state
            done = terminated or truncated

        if episode[-1][2] == 1:  # ìŠ¹ë¦¬
            wins += 1

        # Q ì—…ë°ì´íŠ¸
        G = 0
        visited = set()

        for t in range(len(episode) - 1, -1, -1):
            state_t, action_t, reward_t = episode[t]
            G = gamma * G + reward_t

            if (state_t, action_t) not in visited:
                visited.add((state_t, action_t))
                returns_sum[state_t][action_t] += G
                returns_count[state_t][action_t] += 1
                Q[state_t][action_t] = (returns_sum[state_t][action_t] /
                                        returns_count[state_t][action_t])

        if (ep + 1) % 50000 == 0:
            win_rate = wins / (ep + 1)
            print(f"Episode {ep + 1}: ìŠ¹ë¥  = {win_rate:.3f}")

    env.close()
    return Q
