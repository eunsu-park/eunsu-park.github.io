2.1 Q-Learning ì•Œê³ ë¦¬ì¦˜¶

Off-Policy: í–‰ë™ ì •ì±…(behavior policy)ê³¼ íƒ€ê²Ÿ ì •ì±…(target policy)ì´ ë‹¤ë¦„

Q(s, a) â† Q(s, a) + Î±[r + Î³ max_a' Q(s', a') - Q(s, a)]

class QLearning:
    def __init__(self, n_states, n_actions, alpha=0.1, gamma=0.99, epsilon=0.1):
        self.q_table = np.zeros((n_states, n_actions))
        self.alpha = alpha      # í•™ìŠµë¥ 
        self.gamma = gamma      # í• ì¸ìœ¨
        self.epsilon = epsilon  # íƒìƒ‰ë¥ 
        self.n_actions = n_actions

    def choose_action(self, state):
        """Îµ-greedy ì •ì±…ìœ¼ë¡œ í–‰ë™ ì„ íƒ"""
        if np.random.random() < self.epsilon:
            return np.random.randint(self.n_actions)  # íƒìƒ‰
        return np.argmax(self.q_table[state])          # í™œìš©

    def update(self, state, action, reward, next_state, done):
        """Q-Learning ì—…ë°ì´íŠ¸"""
        if done:
            target = reward
        else:
            # Off-policy: ë‹¤ìŒ ìƒíƒœì—ì„œ ìµœëŒ€ Qê°’ ì‚¬ìš©
            target = reward + self.gamma * np.max(self.q_table[next_state])

        # TD ì—…ë°ì´íŠ¸
        td_error = target - self.q_table[state, action]
        self.q_table[state, action] += self.alpha * td_error

        return td_error

def train_qlearning(env, agent, n_episodes=1000):
    rewards_history = []

    for episode in range(n_episodes):
        state = env.reset()
        total_reward = 0
        done = False

        while not done:
            # Îµ-greedyë¡œ í–‰ë™ ì„ íƒ
            action = agent.choose_action(state)

            # í™˜ê²½ì—ì„œ í•œ ìŠ¤í… ì§„í–‰
            next_state, reward, done, _ = env.step(action)

            # Q í…Œì´ë¸” ì—…ë°ì´íŠ¸ (ë‹¤ìŒ í–‰ë™ê³¼ ë¬´ê´€)
            agent.update(state, action, reward, next_state, done)

            state = next_state
            total_reward += reward

        rewards_history.append(total_reward)

        # Epsilon decay (ì„ íƒì )
        agent.epsilon = max(0.01, agent.epsilon * 0.995)

    return rewards_history

class SARSA:
    def __init__(self, n_states, n_actions, alpha=0.1, gamma=0.99, epsilon=0.1):
        self.q_table = np.zeros((n_states, n_actions))
        self.alpha = alpha
        self.gamma = gamma
        self.epsilon = epsilon
        self.n_actions = n_actions

    def choose_action(self, state):
        """Îµ-greedy ì •ì±…"""
        if np.random.random() < self.epsilon:
            return np.random.randint(self.n_actions)
        return np.argmax(self.q_table[state])

    def update(self, state, action, reward, next_state, next_action, done):
        """SARSA ì—…ë°ì´íŠ¸"""
        if done:
            target = reward
        else:
            # On-policy: ì‹¤ì œ ë‹¤ìŒ í–‰ë™ì˜ Qê°’ ì‚¬ìš©
            target = reward + self.gamma * self.q_table[next_state, next_action]

        td_error = target - self.q_table[state, action]
        self.q_table[state, action] += self.alpha * td_error

        return td_error

def train_sarsa(env, agent, n_episodes=1000):
    rewards_history = []

    for episode in range(n_episodes):
        state = env.reset()
        action = agent.choose_action(state)  # ì´ˆê¸° í–‰ë™ ì„ íƒ
        total_reward = 0
        done = False

        while not done:
            # í™˜ê²½ì—ì„œ í•œ ìŠ¤í… ì§„í–‰
            next_state, reward, done, _ = env.step(action)

            # ë‹¤ìŒ í–‰ë™ ì„ íƒ (ì—…ë°ì´íŠ¸ ì „ì—)
            next_action = agent.choose_action(next_state)

            # SARSA ì—…ë°ì´íŠ¸ (ë‹¤ìŒ í–‰ë™ í•„ìš”)
            agent.update(state, action, reward, next_state, next_action, done)

            state = next_state
            action = next_action
            total_reward += reward

        rewards_history.append(total_reward)

    return rewards_history

def epsilon_greedy(q_values, epsilon):
    if np.random.random() < epsilon:
        return np.random.randint(len(q_values))
    return np.argmax(q_values)

# Epsilon decay ìŠ¤ì¼€ì¤„
def get_epsilon(episode, min_eps=0.01, max_eps=1.0, decay=0.995):
    return max(min_eps, max_eps * (decay ** episode))

def softmax_action(q_values, temperature=1.0):
    """ì˜¨ë„ì— ë”°ë¥¸ í™•ë¥ ì  í–‰ë™ ì„ íƒ"""
    exp_q = np.exp(q_values / temperature)
    probs = exp_q / np.sum(exp_q)
    return np.random.choice(len(q_values), p=probs)

5.3 UCB (Upper Confidence Bound)¶

class UCBAgent:
    def __init__(self, n_states, n_actions, c=2.0):
        self.q_table = np.zeros((n_states, n_actions))
        self.n_visits = np.zeros((n_states, n_actions))
        self.total_visits = np.zeros(n_states)
        self.c = c

    def choose_action(self, state):
        self.total_visits[state] += 1

        # ë°©ë¬¸í•˜ì§€ ì•Šì€ í–‰ë™ì´ ìˆìœ¼ë©´ ì„ íƒ
        if 0 in self.n_visits[state]:
            return np.argmin(self.n_visits[state])

        # UCB ê°’ ê³„ì‚°
        ucb_values = self.q_table[state] + self.c * np.sqrt(
            np.log(self.total_visits[state]) / self.n_visits[state]
        )
        return np.argmax(ucb_values)

class ExpectedSARSA:
    def __init__(self, n_states, n_actions, alpha=0.1, gamma=0.99, epsilon=0.1):
        self.q_table = np.zeros((n_states, n_actions))
        self.alpha = alpha
        self.gamma = gamma
        self.epsilon = epsilon
        self.n_actions = n_actions

    def get_policy_probs(self, state):
        """Îµ-greedy ì •ì±…ì˜ í™•ë¥  ë¶„í¬"""
        probs = np.ones(self.n_actions) * self.epsilon / self.n_actions
        best_action = np.argmax(self.q_table[state])
        probs[best_action] += 1 - self.epsilon
        return probs

    def update(self, state, action, reward, next_state, done):
        if done:
            target = reward
        else:
            # ê¸°ëŒ€ê°’ ê³„ì‚°
            probs = self.get_policy_probs(next_state)
            expected_q = np.sum(probs * self.q_table[next_state])
            target = reward + self.gamma * expected_q

        td_error = target - self.q_table[state, action]
        self.q_table[state, action] += self.alpha * td_error

8. ì‹¤ìŠµ: FrozenLake¶

import gymnasium as gym

def train_frozen_lake():
    env = gym.make('FrozenLake-v1', is_slippery=True)

    agent = QLearning(
        n_states=env.observation_space.n,
        n_actions=env.action_space.n,
        alpha=0.1,
        gamma=0.99,
        epsilon=1.0
    )

    n_episodes = 10000
    rewards = []

    for episode in range(n_episodes):
        state, _ = env.reset()
        total_reward = 0
        done = False

        while not done:
            action = agent.choose_action(state)
            next_state, reward, terminated, truncated, _ = env.step(action)
            done = terminated or truncated

            agent.update(state, action, reward, next_state, done)
            state = next_state
            total_reward += reward

        rewards.append(total_reward)
        agent.epsilon = max(0.01, agent.epsilon * 0.9995)

        if (episode + 1) % 1000 == 0:
            avg_reward = np.mean(rewards[-100:])
            print(f"Episode {episode + 1}, Avg Reward: {avg_reward:.3f}")

    return agent, rewards

if __name__ == "__main__":
    agent, rewards = train_frozen_lake()
